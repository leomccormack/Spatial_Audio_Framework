74 #if defined(SAF_USE_INTEL_MKL)
75 /*
76  * Using Intel's Math Kernal Library (MKL)

78  */

81 
82 #elif defined(SAF_USE_OPEN_BLAS_AND_LAPACKE)
83 /*
84  * Using OpenBLAS and the LAPACKE interface
85  * (A good option for both x86 and ARM based architectures)
86  */
87 # define VECLIB_USE_LAPACKE_INTERFACE
88 # include "cblas.h"
89 # include "lapacke.h"
90 
91 #elif defined(SAF_USE_ATLAS)
92 /*
93  * Using the Automatically Tuned Linear Algebra Software (ATLAS) library
94  * (Not recommended, since some saf_veclib functions do not work with ATLAS)
95  */
96 # define VECLIB_USE_CLAPACK_INTERFACE
97 # include "cblas-atlas.h"
98 # include "clapack.h"
99 # warning Note: CLAPACK does not include all LAPACK routines!
100 
101 #elif defined(__APPLE__) && defined(SAF_USE_APPLE_ACCELERATE)
102 /*
103  * Using Apple's Accelerate library (vDSP)
104  * (Solid choice for both x86 and ARM, but only works under MacOSX and is not as
105  * fast as Intel MKL for x86 systems)
106  */
107 # define VECLIB_USE_LAPACK_FORTRAN_INTERFACE
108 # include "Accelerate/Accelerate.h"
109 
110 #else
111 /*
112  * Note that if you would like to use some other CBLAS/LAPACK supporting library
113  * then please get in touch! :-)
114  */
115 # error SAF requires a library (or libraries) which supports CBLAS and LAPACK
116 #endif
