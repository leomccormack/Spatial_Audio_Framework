45  int i, j, k, retval,dimid[6], *dimids, ndimsp,ncid, varid, is0_360;
46  size_t dimlength[6], IR_dims[3], SourcePosition_dims[2];
47  size_t hrir_dims[3];
48  char dimname[6];
49  //const char* errorMessage;
50  double* IR, *SourcePosition, IR_fs;
51 
52  /* free any existing memory */
53  if(*hrirs!=NULL)
54  free(*hrirs);
55  if(*hrir_dirs_deg!=NULL)
56  free(*hrir_dirs_deg);
57 
58  /* open sofa file */
59  /* (retval is set to error value if sofa_filepath==NULL (intentional), or if the file
60  * path/name was not found (unintentional). */
61  ncid = -1;
62  if(sofa_filepath!=NULL){
63  if ((retval = nc_open(sofa_filepath, NC_NOWRITE, &ncid)))
64  nc_strerror(retval);
65  }
66  else
67  retval = NC_FATAL;
68 
69  /* if error: */
70  if(retval!=NC_NOERR){
71  is0_360 = 0;
72  /* return default HRIR data */
73  (*N_hrir_dirs) = __default_N_hrir_dirs;
74  (*hrir_len) = __default_hrir_len;
75  (*hrir_fs) = __default_hrir_fs;
76  (*hrirs) = malloc1d((*N_hrir_dirs) * 2 * (*hrir_len)*sizeof(float));
77  memcpy((*hrirs), (float*)__default_hrirs, (*N_hrir_dirs) * 2 * (*hrir_len)*sizeof(float));
78  (*hrir_dirs_deg) = malloc1d((*N_hrir_dirs) * 2 * sizeof(float));
79  memcpy((*hrir_dirs_deg), (float*)__default_hrir_dirs_deg, (*N_hrir_dirs) * 2 * sizeof(float));
80  for(i=0; i<(*N_hrir_dirs); i++){
81  if((*hrir_dirs_deg)[2*i+0]>=181.0f)
82  is0_360 = 1;
83  }
84  /* convert to -180..180, if azi is 0..360 */
85  if(is0_360)
86  for(i=0; i<(*N_hrir_dirs); i++)
87  (*hrir_dirs_deg)[2*i+0] = (*hrir_dirs_deg)[2*i+0]>180.0f ? (*hrir_dirs_deg)[2*i+0] -360.0f : (*hrir_dirs_deg)[2*i+0];
88 
89 #ifndef NDEBUG
90  /* also output warning message, if encountering this error value was
91  * unintentional (i.e. sofa_filepath!=NULL) */
92  if(sofa_filepath!=NULL)
93  saf_error_print(SAF_WARNING__SOFA_FILE_NOT_FOUND);
94 #endif
95  return;
96  }
97  assert(ncid!=-1);
98 
99  /* Determine dimension IDs and lengths */
100  /* Note: there are 6 possible dimension lengths in the sofa standard */
101  for (i=0; i<6; i++){
102  retval = nc_inq_dim(ncid, i, &dimname[i], &dimlength[i]);
103  nc_strerror(retval);
104  retval = nc_inq_dimid(ncid, &dimname[i], &dimid[i]);
105  nc_strerror(retval);
106  }
107 
108  /* Extract IR data */
109  if ((retval = nc_inq_varid(ncid, "Data.IR", &varid)))
110  nc_strerror(retval);
111  if ((retval = nc_inq_varndims (ncid, varid, &ndimsp)))
112  nc_strerror(retval);
113  dimids = malloc1d(ndimsp*sizeof(int));
114  if ((retval = nc_inq_vardimid(ncid, varid, dimids)))
115  nc_strerror(retval);
116  for(i=0; i<3; i++)
117  IR_dims[i] = dimlength[dimid[dimids[i]]];
118  free(dimids);
119  IR = malloc1d(IR_dims[0]*IR_dims[1]*IR_dims[2]*sizeof(double));
120  if ((retval = nc_get_var(ncid, varid, IR)))
121  nc_strerror(retval);
122  if ((retval = nc_inq_varid(ncid, "Data.SamplingRate", &varid)))
123  nc_strerror(retval);
124  if ((retval = nc_get_var(ncid, varid, &IR_fs)))
125  nc_strerror(retval);
126 
127  /* Extract positional data */
128  if ((retval = nc_inq_varid(ncid, "SourcePosition", &varid)))
129  nc_strerror(retval);
130  if ((retval = nc_inq_varndims (ncid, varid, &ndimsp)))
131  nc_strerror(retval);
132  dimids = malloc1d(ndimsp*sizeof(int));
133  if ((retval = nc_inq_vardimid(ncid, varid, dimids)))
134  nc_strerror(retval);
135  for(i=0; i<2; i++)
136  SourcePosition_dims[i] = dimlength[dimid[dimids[i]]];
137  free(dimids);
138  SourcePosition = malloc1d(SourcePosition_dims[0]*SourcePosition_dims[1]*sizeof(double));
139  if ((retval = nc_get_var(ncid, varid, SourcePosition)))
140  nc_strerror(retval);
141 
142  /* Close the file, freeing all resources. */
143  if ((retval = nc_close(ncid)))
144  nc_strerror(retval);
145 
146  /* Allocate memory */
147  (*hrir_len) = (int)IR_dims[2];
148  (*hrirs) = malloc1d(IR_dims[0]*IR_dims[1]*(*hrir_len) *sizeof(float));
149  (*hrir_dirs_deg) = malloc1d(SourcePosition_dims[0]*2*sizeof(float));
150 
151  /* Store relevent info in handle */
152  (*hrir_fs) = (int)(IR_fs+0.5);
153  hrir_dims[0] = IR_dims[0];
154  hrir_dims[1] = IR_dims[1];
155  hrir_dims[2] = (size_t)(*hrir_len);
156  (*N_hrir_dirs) = (int)hrir_dims[0];
157  for(i=0; i<(int)hrir_dims[0]; i++){
158  for(j=0; j<(int)hrir_dims[1]; j++){
159  for(k=0; k<(int)hrir_dims[2]; k++){
160  /* trunctate IRs and store in floating point precision */
161  (*hrirs)[i*hrir_dims[1]*hrir_dims[2] + j*hrir_dims[2] + k] = (float)IR[i*IR_dims[1]*IR_dims[2] + j*IR_dims[2] + k];
162  }
163  }
164  }
165 
166  /* store in floating point precision */
167  is0_360 = 0;
168  for(i=0; i<(int)SourcePosition_dims[0]; i++){
169  (*hrir_dirs_deg)[2*i+0] = (float)SourcePosition[i*SourcePosition_dims[1]+0];
170  (*hrir_dirs_deg)[2*i+1] = (float)SourcePosition[i*SourcePosition_dims[1]+1];
171  if((*hrir_dirs_deg)[2*i+0]>=181.0f)
172  is0_360 = 1;
173  }
174 
175  /* convert to -180..180, if azi is 0..360 */
176  if(is0_360)
177  for(i=0; i<(int)SourcePosition_dims[0]; i++)
178  (*hrir_dirs_deg)[2*i+0] = (*hrir_dirs_deg)[2*i+0]>180.0f ? (*hrir_dirs_deg)[2*i+0] -360.0f : (*hrir_dirs_deg)[2*i+0];
179 
180  free(IR);
181  free(SourcePosition);
182 }
