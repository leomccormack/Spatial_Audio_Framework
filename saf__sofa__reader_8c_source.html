50 SAF_SOFA_ERROR_CODES saf_sofa_open
51 (
52  saf_sofa_container* h,

54 )
55 {
56  int varid, attnum, i, j, ncid, retval, ndimsp, nvarsp, nattsp, unlimdimidp, varnattsp, counter;
57  size_t tmp_size, lenp;
58  char varname[NC_MAX_NAME+1], attname[NC_MAX_NAME+1];

60  double* tmp_data;
61  int* dimids, *dimid;
62  nc_type typep;
63  size_t* dimlength;

65  /* Default variables */
66  h->nSources = h->nReceivers = h->DataLengthIR = -1;
67  h->DataSamplingRate = 0.0f;
68  h->nEmitters = h->nListeners = -1;
69  h->DataIR = h->SourcePosition = h->ReceiverPosition = h->ListenerPosition =
70  h->ListenerUp = h->ListenerView = h->EmitterPosition = NULL;
71  h->DataDelay = NULL;

74  h->ListenerPositionType = h->ListenerPositionUnits = h->ReceiverPositionType

76  = h->EmitterPositionType = h->EmitterPositionUnits = h->DataSamplingRateUnits
77  = h->ListenerViewType = h->ListenerViewUnits = NULL;
78 
79  /* Default global attributes */
80  h->Conventions = h->Version = h->SOFAConventions = h->SOFAConventionsVersion
81  = h->APIName = h->APIVersion = h->ApplicationName = h->ApplicationVersion
82  = h->AuthorContact = h->Comment = h->DataType = h->History = h->License
83  = h->Organisation = h->References = h->RoomType = h->Origin = h->DateCreated
84  = h->DateModified = h->Title = h->DatabaseName = h->ListenerShortName = NULL;

86  /* NetCDF is not thread safe! */
87  counter = 0;
88  while (__SAF_NETCDF_IN_USE){
89  /* WARNING: this work around will only work with SAF-built programs.
90  * If another program tries to use netcdf while a SAF-built program is
91  * using it, then it is quite likely there will be a segfault or
92  * unexpected behaviour.
93  * More details on the status of this issue can be found here:
94  * https://github.com/Unidata/netcdf-c/issues/1373
95  */
96  /* Pause until currently active netcdf instance has been closed */
97  SAF_SLEEP(10);
98  counter++;
99  if(counter>1000) /* Give up if waiting too long... */
100  return SAF_SOFA_ERROR_NETCDF_IN_USE;
101  }
102  __saf_netcdf_setflag(1);

104  /* Open NetCDF file */
105  if ((retval = nc_open(sofa_filepath, NC_NOWRITE, &ncid)))
106  nc_strerror(retval);
107  if(retval!=NC_NOERR){/* if error: */
108  __saf_netcdf_setflag(0);
109  return SAF_SOFA_ERROR_INVALID_FILE_OR_FILE_PATH;

111  nc_inq(ncid, &ndimsp, &nvarsp, &nattsp, &unlimdimidp); /* find number of possible dimensions, variables, and attributes */
112 
113  /* Find dimension IDs and lengths */
114  dimid = malloc1d(ndimsp*sizeof(int));
115  dimlength = malloc1d(ndimsp*sizeof(size_t));
116  dimname = malloc1d(ndimsp*(NC_MAX_NAME+1)*sizeof(char)); /* +1 as NC_MAX_NAME does not include null-termination */
117  for (i=0; i<ndimsp; i++){
118  nc_inq_dim(ncid, i, &dimname[i*(NC_MAX_NAME+1)], &dimlength[i]);
119  nc_inq_dimid(ncid, &dimname[i*(NC_MAX_NAME+1)], &dimid[i]);
120  }

122  /* Loop over the variables and pull the data accordingly */
123  dimids = NULL;
124  tmp_data = NULL;
125  for(varid=0; varid<nvarsp; varid++){
126  nc_inq_var(ncid, varid, (char*)varname, NULL, NULL, NULL, NULL); /* Variable name */
127  nc_inq_varndims(ncid, varid, &ndimsp); /* Variable dimensionality */
128  dimids = realloc1d(dimids, ndimsp*sizeof(int));
129  nc_inq_vardimid(ncid, varid, dimids); /* Variable dimension IDs */
130  nc_inq_vartype(ncid, varid, &typep); /* Variable data type */
131  nc_inq_varnatts(ncid, varid, &varnattsp); /* Variable number of associated attributes */
132 
133  if (!strcmp((char*)varname,"Data.IR")){
134  /* Checks */
135  if(ndimsp!=3) { __saf_netcdf_setflag(0); return SAF_SOFA_ERROR_DIMENSIONS_UNEXPECTED; }
136  if(h->nReceivers!=-1 && (int)dimlength[dimid[dimids[1]]] != h->nReceivers) { __saf_netcdf_setflag(0); return SAF_SOFA_ERROR_DIMENSIONS_UNEXPECTED; }
137  if(typep!=NC_DOUBLE) { __saf_netcdf_setflag(0); return SAF_SOFA_ERROR_FORMAT_UNEXPECTED; }
138 
139  /* Pull data */
140  h->nSources = (int)dimlength[dimid[dimids[0]]];
141  h->nReceivers = (int)dimlength[dimid[dimids[1]]];
142  h->DataLengthIR = (int)dimlength[dimid[dimids[2]]];
143  tmp_size = dimlength[dimid[dimids[0]]] * dimlength[dimid[dimids[1]]] * dimlength[dimid[dimids[2]]];
144  tmp_data = realloc1d(tmp_data, tmp_size*sizeof(double));
145  nc_get_var(ncid, varid, tmp_data);
146  h->DataIR = malloc1d(tmp_size*sizeof(float));
147  for(i=0; i<(int)tmp_size; i++)
148  h->DataIR[i] = (float)tmp_data[i];
149  }
150  else if(!strcmp((char*)varname,"Data.SamplingRate")){
151  /* Checks */
152  if(!(ndimsp==1 && dimlength[dimid[dimids[0]]] == 1)) { __saf_netcdf_setflag(0); return SAF_SOFA_ERROR_DIMENSIONS_UNEXPECTED; }
153  if(typep!=NC_DOUBLE) { __saf_netcdf_setflag(0); return SAF_SOFA_ERROR_FORMAT_UNEXPECTED; }
154 
155  /* Pull data */
156  tmp_data = realloc1d(tmp_data, 1*sizeof(double));
157  nc_get_var(ncid, varid, tmp_data);
158  h->DataSamplingRate = (float)tmp_data[0];
159 
160  /* Pull associated attributes */
161  for(attnum=0; attnum<varnattsp; attnum++){
162  nc_inq_attname(ncid, varid, attnum, attname);
163  nc_inq_attlen(ncid, varid, attname, &lenp);
164  if (!strcmp((char*)attname,"Units")){
165  h->DataSamplingRateUnits = calloc1d(SAF_MAX((NC_MAX_NAME+1),lenp), sizeof(char));
166  nc_get_att(ncid, varid, attname, h->DataSamplingRateUnits);
167  }
168  }
169  }
170  else if (!strcmp((char*)varname,"Data.Delay")){
171  /* Checks */
172  if(!(ndimsp==2 || ndimsp==3)) { return SAF_SOFA_ERROR_DIMENSIONS_UNEXPECTED; }
173  if(h->nReceivers!=-1 && !((int)dimlength[dimid[dimids[1]]] == h->nReceivers || (int)dimlength[dimid[dimids[0]]] == h->nReceivers)) { __saf_netcdf_setflag(0); return SAF_SOFA_ERROR_DIMENSIONS_UNEXPECTED; }
174  if((int)dimlength[dimid[dimids[0]]] != 1 && (int)dimlength[dimid[dimids[1]]] != 1) { __saf_netcdf_setflag(0); return SAF_SOFA_ERROR_DIMENSIONS_UNEXPECTED; }
175  if(typep!=NC_DOUBLE) { __saf_netcdf_setflag(0); return SAF_SOFA_ERROR_FORMAT_UNEXPECTED; }
176

179  tmp_data = realloc1d(tmp_data, tmp_size*sizeof(double));
180  nc_get_var(ncid, varid, tmp_data);
181  h->DataDelay = malloc1d(tmp_size*sizeof(int));
182  for(i=0; i<(int)tmp_size; i++)
183  h->DataDelay[i] = (int)tmp_data[i];
184  }

189  if((int)dimlength[dimid[dimids[1]]] != 3) { __saf_netcdf_setflag(0); return SAF_SOFA_ERROR_DIMENSIONS_UNEXPECTED; }

205  if (!strcmp((char*)attname,"Type")){
206  h->SourcePositionType = calloc1d(SAF_MAX((NC_MAX_NAME+1),lenp), sizeof(char));

242  if((int)dimlength[dimid[dimids[1]]] != 3) { __saf_netcdf_setflag(0); return SAF_SOFA_ERROR_DIMENSIONS_UNEXPECTED; }

570 void loadSofaFile
571 (
572  char* sofa_filepath,
573  float** hrirs,
574  float** hrir_dirs_deg,
575  int* N_hrir_dirs,
576  int* hrir_len,
577  int* hrir_fs
578 )
579 {
580  int i, j, k, retval,dimid[6], *dimids, ndimsp,ncid, varid, is0_360;
581  size_t dimlength[6], IR_dims[3], SourcePosition_dims[2];
582  size_t hrir_dims[3];
583  char dimname[6];
584  //const char* errorMessage;
585  double* IR, *SourcePosition, IR_fs;
586 
587  /* free any existing memory */
588  if(*hrirs!=NULL)
589  free(*hrirs);
590  if(*hrir_dirs_deg!=NULL)
591  free(*hrir_dirs_deg);
592 
593  /* open sofa file */
594  /* (retval is set to error value if sofa_filepath==NULL (intentional), or if the file
595  * path/name was not found (unintentional). */
596  ncid = -1;
597  if(sofa_filepath!=NULL){
598  if ((retval = nc_open(sofa_filepath, NC_NOWRITE, &ncid)))
599  nc_strerror(retval);
600  }
601  else
602  retval = NC_FATAL;
603 
604  /* if error: */
605  if(retval!=NC_NOERR){
606  is0_360 = 0;
607  /* return default HRIR data */
608  (*N_hrir_dirs) = __default_N_hrir_dirs;
609  (*hrir_len) = __default_hrir_len;
610  (*hrir_fs) = __default_hrir_fs;
611  (*hrirs) = malloc1d((*N_hrir_dirs) * 2 * (*hrir_len)*sizeof(float));
612  memcpy((*hrirs), (float*)__default_hrirs, (*N_hrir_dirs) * 2 * (*hrir_len)*sizeof(float));
613  (*hrir_dirs_deg) = malloc1d((*N_hrir_dirs) * 2 * sizeof(float));
614  memcpy((*hrir_dirs_deg), (float*)__default_hrir_dirs_deg, (*N_hrir_dirs) * 2 * sizeof(float));
615  for(i=0; i<(*N_hrir_dirs); i++){
616  if((*hrir_dirs_deg)[2*i+0]>=181.0f)
617  is0_360 = 1;
618  }
619  /* convert to -180..180, if azi is 0..360 */
620  if(is0_360)
621  for(i=0; i<(*N_hrir_dirs); i++)
622  (*hrir_dirs_deg)[2*i+0] = (*hrir_dirs_deg)[2*i+0]>180.0f ? (*hrir_dirs_deg)[2*i+0] -360.0f : (*hrir_dirs_deg)[2*i+0];
623 
624 #ifndef NDEBUG
625  /* also output warning message, if encountering this error value was
626  * unintentional (i.e. sofa_filepath!=NULL) */
627  if(sofa_filepath!=NULL)
628  saf_print_warning("Unable to load the specified SOFA file. The default HRIR data has been loaded instead.");
629 #endif
630  return;
631  }
632  saf_assert(ncid!=-1, "Something very bad happened");
633 
634  /* Determine dimension IDs and lengths */
635  /* Note: there are 6 possible dimension lengths in the sofa standard */
636  for (i=0; i<6; i++){
637  retval = nc_inq_dim(ncid, i, &dimname[i], &dimlength[i]);
638  nc_strerror(retval);
639  retval = nc_inq_dimid(ncid, &dimname[i], &dimid[i]);
640  nc_strerror(retval);
641  }
642 
643  /* Extract IR data */
644  if ((retval = nc_inq_varid(ncid, "Data.IR", &varid)))
645  nc_strerror(retval);
646  if ((retval = nc_inq_varndims (ncid, varid, &ndimsp)))
647  nc_strerror(retval);
648  dimids = malloc1d(ndimsp*sizeof(int));
649  if ((retval = nc_inq_vardimid(ncid, varid, dimids)))
650  nc_strerror(retval);
651  for(i=0; i<3; i++)
652  IR_dims[i] = dimlength[dimid[dimids[i]]];
653  free(dimids);
654  IR = malloc1d(IR_dims[0]*IR_dims[1]*IR_dims[2]*sizeof(double));
655  if ((retval = nc_get_var(ncid, varid, IR)))
656  nc_strerror(retval);
657  if ((retval = nc_inq_varid(ncid, "Data.SamplingRate", &varid)))
658  nc_strerror(retval);
659  if ((retval = nc_get_var(ncid, varid, &IR_fs)))
660  nc_strerror(retval);
661 
662  /* Extract positional data */
663  if ((retval = nc_inq_varid(ncid, "SourcePosition", &varid)))
664  nc_strerror(retval);
665  if ((retval = nc_inq_varndims (ncid, varid, &ndimsp)))
666  nc_strerror(retval);
667  dimids = malloc1d(ndimsp*sizeof(int));
668  if ((retval = nc_inq_vardimid(ncid, varid, dimids)))
669  nc_strerror(retval);
670  for(i=0; i<2; i++)
671  SourcePosition_dims[i] = dimlength[dimid[dimids[i]]];
672  free(dimids);
673  SourcePosition = malloc1d(SourcePosition_dims[0]*SourcePosition_dims[1]*sizeof(double));
674  if ((retval = nc_get_var(ncid, varid, SourcePosition)))
675  nc_strerror(retval);
676 
677  /* Close the file, freeing all resources. */
678  if ((retval = nc_close(ncid)))
679  nc_strerror(retval);
680 
681  /* Allocate memory */
682  (*hrir_len) = (int)IR_dims[2];
683  (*hrirs) = malloc1d(IR_dims[0]*IR_dims[1]*(*hrir_len) *sizeof(float));
684  (*hrir_dirs_deg) = malloc1d(SourcePosition_dims[0]*2*sizeof(float));
685 
686  /* Store relevent info in handle */
687  (*hrir_fs) = (int)(IR_fs+0.5);
688  hrir_dims[0] = IR_dims[0];
689  hrir_dims[1] = IR_dims[1];
690  hrir_dims[2] = (size_t)(*hrir_len);
691  (*N_hrir_dirs) = (int)hrir_dims[0];
692  for(i=0; i<(int)hrir_dims[0]; i++){
693  for(j=0; j<(int)hrir_dims[1]; j++){
694  for(k=0; k<(int)hrir_dims[2]; k++){
695  /* trunctate IRs and store in floating point precision */
696  (*hrirs)[i*hrir_dims[1]*hrir_dims[2] + j*hrir_dims[2] + k] = (float)IR[i*IR_dims[1]*IR_dims[2] + j*IR_dims[2] + k];
697  }
698  }
699  }
700 
701  /* store in floating point precision */
702  is0_360 = 0;
703  for(i=0; i<(int)SourcePosition_dims[0]; i++){
704  (*hrir_dirs_deg)[2*i+0] = (float)SourcePosition[i*SourcePosition_dims[1]+0];
705  (*hrir_dirs_deg)[2*i+1] = (float)SourcePosition[i*SourcePosition_dims[1]+1];
706  if((*hrir_dirs_deg)[2*i+0]>=181.0f)
707  is0_360 = 1;
708  }
709 
710  /* convert to -180..180, if azi is 0..360 */
711  if(is0_360)
712  for(i=0; i<(int)SourcePosition_dims[0]; i++)
713  (*hrir_dirs_deg)[2*i+0] = (*hrir_dirs_deg)[2*i+0]>180.0f ? (*hrir_dirs_deg)[2*i+0] -360.0f : (*hrir_dirs_deg)[2*i+0];
714 
715  free(IR);
716  free(SourcePosition);
717 }
