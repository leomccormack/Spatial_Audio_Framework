29void pitch_shifter_create
30(
31 void ** const phPS
32)
33{
34 pitch_shifter_data* pData = (pitch_shifter_data*)malloc1d(sizeof(pitch_shifter_data));
35 *phPS = (void*)pData;
36
37 /* Default user parameters */
38 pData->new_nChannels = pData->nChannels = 1;
39 pData->pitchShift_factor = 1.0f;
40 pData->osamp_option = PITCH_SHIFTER_OSAMP_4;
41 pData->fftsize_option = PITCH_SHIFTER_FFTSIZE_4096;
42
43 /* internals */
44 pData->hSmb = NULL;
45 pData->progressBar0_1 = 0.0f;
46 pData->progressBarText = malloc1d(PROGRESSBARTEXT_CHAR_LENGTH*sizeof(char));
47 strcpy(pData->progressBarText,"");
48 pData->sampleRate = 0.0f;
49 pData->fftFrameSize = 4096; /* Not important to get right, as it will be overwritten upon init */
50 pData->stepsize = 1024; /* same here */
51
52 /* flags */
53 pData->procStatus = PROC_STATUS_NOT_ONGOING;
54 pData->codecStatus = CODEC_STATUS_NOT_INITIALISED;
55
56 /* set FIFO buffers */
57 pData->FIFO_idx = 0;
58 memset(pData->inFIFO, 0, MAX_NUM_CHANNELS*PITCH_SHIFTER_FRAME_SIZE*sizeof(float));
59 memset(pData->outFIFO, 0, MAX_NUM_CHANNELS*PITCH_SHIFTER_FRAME_SIZE*sizeof(float));
60}

62void pitch_shifter_destroy
63(
64 void ** const phPS
65)
66{
67 pitch_shifter_data *pData = (pitch_shifter_data*)(*phPS);
68
69 if (pData != NULL) {
70 /* not safe to free memory during intialisation/processing loop */
71 while (pData->codecStatus == CODEC_STATUS_INITIALISING ||
72 pData->procStatus == PROC_STATUS_ONGOING){
73 SAF_SLEEP(10);
74 }
75
76 if (pData->hSmb != NULL)
77 smb_pitchShift_destroy(&(pData->hSmb));
78 free(pData);
79 pData = NULL;
80 }
81}

97void pitch_shifter_initCodec
98(
99 void* const hPS
100)
101{
102 pitch_shifter_data *pData = (pitch_shifter_data*)(hPS);
103 int nChannels, fftSize, osamp;
104
105 if (pData->codecStatus != CODEC_STATUS_NOT_INITIALISED)
106 return; /* re-init not required, or already happening */
107 while (pData->procStatus == PROC_STATUS_ONGOING){
108 /* re-init required, but we need to wait for the current processing loop to end */
109 pData->codecStatus = CODEC_STATUS_INITIALISING; /* indicate that we want to init */
110 SAF_SLEEP(10);
111 }
112
113 /* for progress bar */
114 pData->codecStatus = CODEC_STATUS_INITIALISING;
115 strcpy(pData->progressBarText,"Initialising pitch shifter");
116 pData->progressBar0_1 = 0.0f;
117
118 nChannels = pData->new_nChannels;
119
120 /* destroy current handle*/
121 if (pData->hSmb != NULL)
122 smb_pitchShift_destroy(&(pData->hSmb));
123
124 /* Config */
125 switch(pData->osamp_option){
126 default:
127 /* fall through */
128 case PITCH_SHIFTER_OSAMP_2: osamp = 2; break;
129 case PITCH_SHIFTER_OSAMP_4: osamp = 4; break;
130 case PITCH_SHIFTER_OSAMP_8: osamp = 8; break;
131 case PITCH_SHIFTER_OSAMP_16: osamp = 16; break;
132 case PITCH_SHIFTER_OSAMP_32: osamp = 32; break;
133 }
134 switch(pData->fftsize_option){
135 default:
136 /* fall through */
137 case PITCH_SHIFTER_FFTSIZE_512: fftSize = 512; break;
138 case PITCH_SHIFTER_FFTSIZE_1024: fftSize = 1024; break;
139 case PITCH_SHIFTER_FFTSIZE_2048: fftSize = 2048; break;
140 case PITCH_SHIFTER_FFTSIZE_4096: fftSize = 4096; break;
141 case PITCH_SHIFTER_FFTSIZE_8192: fftSize = 8192; break;
142 case PITCH_SHIFTER_FFTSIZE_16384: fftSize = 16384; break;
143 }
144 pData->fftFrameSize = fftSize;
145 pData->stepsize = fftSize/osamp;
146
147 /* Create new handle */
148 smb_pitchShift_create(&(pData->hSmb), nChannels, fftSize, osamp, pData->sampleRate);
149 pData->nChannels = nChannels;
150
151 /* done! */
152 strcpy(pData->progressBarText,"Done!");
153 pData->progressBar0_1 = 1.0f;
154 pData->codecStatus = CODEC_STATUS_INITIALISED;
155}

157void pitch_shifter_process
158(
159 void * const hPS,
160 const float *const * inputs,
161 float ** const outputs,
162 int nInputs,
163 int nOutputs,
164 int nSamples
165)
166{
167 pitch_shifter_data *pData = (pitch_shifter_data*)(hPS);
168 int s, ch, nChannels;
169 nChannels = pData->nChannels;
170
171 /* Loop over all samples */
172 for(s=0; s<nSamples; s++){
173 /* Load input signals into inFIFO buffer */
174 for(ch=0; ch<SAF_MIN(nInputs,nChannels); ch++)
175 pData->inFIFO[ch][pData->FIFO_idx] = inputs[ch][s];
176 for(; ch<nChannels; ch++) /* Zero any channels that were not given */
177 pData->inFIFO[ch][pData->FIFO_idx] = 0.0f;
178
179 /* Pull output signals from outFIFO buffer */
180 for(ch=0; ch<SAF_MIN(nOutputs, nChannels); ch++)
181 outputs[ch][s] = pData->outFIFO[ch][pData->FIFO_idx];
182 for(; ch<nOutputs; ch++) /* Zero any extra channels */
183 outputs[ch][s] = 0.0f;
184
185 /* Increment buffer index */
186 pData->FIFO_idx++;
187
188 /* Process frame if inFIFO is full and codec is ready for it */
189 if (pData->FIFO_idx >= PITCH_SHIFTER_FRAME_SIZE && (pData->codecStatus == CODEC_STATUS_INITIALISED) ) {
190 pData->FIFO_idx = 0;
191 pData->procStatus = PROC_STATUS_ONGOING;
192
193 /* load input */
194 for(ch=0; ch<nChannels; ch++)
195 memcpy(pData->inputFrame[ch], pData->inFIFO[ch], PITCH_SHIFTER_FRAME_SIZE*sizeof(float));
196
197 /* Apply pitch shifting */
198 smb_pitchShift_apply(pData->hSmb, pData->pitchShift_factor, PITCH_SHIFTER_FRAME_SIZE, (float*)pData->inputFrame, (float*)pData->outputFrame);
199
200 /* Copy to output */
201 for(ch=0; ch<nChannels; ch++)
202 memcpy(pData->outFIFO[ch], pData->outputFrame[ch], PITCH_SHIFTER_FRAME_SIZE*sizeof(float));
203 }
204 else if(pData->FIFO_idx >= PITCH_SHIFTER_FRAME_SIZE){
205 /* clear outFIFO if codec was not ready */
206 pData->FIFO_idx = 0;
207 memset(pData->outFIFO, 0, MAX_NUM_CHANNELS*PITCH_SHIFTER_FRAME_SIZE*sizeof(float));
208 }
209 }
210
211 pData->procStatus = PROC_STATUS_NOT_ONGOING;
212}
