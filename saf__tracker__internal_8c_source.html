200 void tracker3d_predict
201 (
202  void* const hT3d,
203  int Tinc
204 )
205 {
206  tracker3d_data *pData = (tracker3d_data*)(hT3d);
207  int i, j, k, n, nDead, isDead, ind;
208  int* dead;
209  float dt0, dt1, p_death, rand01, distance_diff;
210  MCS_data* S;
211  tracker3d_config* tpars = &(pData->tpars);
212 #ifdef TRACKER_VERY_VERBOSE
213  char c_event[256], tmp[256];
214  printf("%s\n", "Prediction step");
215 #endif
216 
217  dead = NULL;
218 
219  /* Loop over particles */
220  for (i=0; i<tpars->Np; i++){
221  S = (MCS_data*)pData->SS[i];
222 
223  /* prep */
224  nDead = 0;
225  free(dead);
226  dead = NULL;
227 #ifdef TRACKER_VERY_VERBOSE
228  memset(c_event, 0, 256*sizeof(char));
229 #endif
230 
231  /* Loop over targets */
232  for (j=0; j<S->nTargets; j++){
233 
234  /* No target has died yet or multiple targets are allowed to die in
235  * one prediction step */
236  if (nDead==0 || tpars->ALLOW_MULTI_DEATH){
237  /* Probability of death */
238  dt0 = (float)S->Tcount[j] * S->dt;
239  dt1 = dt0 + S->dt * (float)Tinc;
240  if (dt0 == 0)
241  p_death = gamma_cdf(dt1, tpars->alpha_death, tpars->beta_death, 0.0f);
242  else
243  p_death = 1.0f - (1.0f-gamma_cdf(dt1, tpars->alpha_death,tpars->beta_death, 0.0f)) /
244  (1.0f-gamma_cdf(dt0,tpars->alpha_death,tpars->beta_death, 0.0f));
245 
246  /* Force probability of death to 1, if this target is too close
247  another target that has been alive longer. */
248  if (tpars->FORCE_KILL_TARGETS){
249  for(k=0; k<S->nTargets; k++){
250  if (k!=j){
251  distance_diff = (S->M[j].m0 - S->M[k].m0) * (S->M[j].m0 - S->M[k].m0) +
252  (S->M[j].m1 - S->M[k].m1) * (S->M[j].m1 - S->M[k].m1) +
253  (S->M[j].m2 - S->M[k].m2) * (S->M[j].m2 - S->M[k].m2);
254  distance_diff = sqrtf(distance_diff);
255  if (distance_diff < tpars->forceKillDistance && S->Tcount[j] <= S->Tcount[k])
256  p_death = 1.0f;
257  }
258  }
259  }
260 
261  /* Decide whether target should die */
262  rand_0_1(&rand01, 1);
263  if (rand01 < p_death){
264  nDead++; /* Target dies */
265  dead = realloc1d(dead, nDead*sizeof(float));
266  dead[nDead-1] = j;
267  }
268  }
269 
270  /* Prediction step if target is alive */
271  if( tpars->ALLOW_MULTI_DEATH ){
272  isDead = 0;
273  for(n=0; n<nDead; n++)
274  if(j==dead[n])
275  isDead = 1;
276 
277  /* Kalman Filter prediction for the target if alive */
278  if(!isDead)
279  kf_predict6(S->M[j].M, S->P[j].P, pData->A, pData->Q);
280  }
281  else {
282  /* Kalman Filter prediction for the target if alive */
283  if ( (nDead==0) || (j != dead[0]) )
284  kf_predict6(S->M[j].M, S->P[j].P, pData->A, pData->Q);
285  }
286  }
287 
288  /* Remove the dead target */
289  if (tpars->ALLOW_MULTI_DEATH){
290  for(j=0; j<nDead; j++){
291  /* Find index of the target to remove */
292  ind = -1;
293  for(k=0; k<S->nTargets; k++)
294  if(dead[j]==k)
295  ind = k;
296  saf_assert(ind != -1, "Ugly error");
297 
298  /* Shimy target data down by 1... overriding the dead target */
299  S->nTargets--;
300  if(ind!=S->nTargets){
301  memmove(&S->M[ind], &S->M[ind+1], (S->nTargets-ind)*sizeof(M6));
302  memmove(&S->P[ind], &S->P[ind+1], (S->nTargets-ind)*sizeof(P66));
303  memmove(&S->Tcount[ind], &S->Tcount[ind+1], (S->nTargets-ind)*sizeof(int));
304  memmove(&S->targetIDs[ind], &S->targetIDs[ind+1], (S->nTargets-ind)*sizeof(int));
305  }
306 
307  /* Remove dead index for next iteration */
308  for(k=0; k<nDead; k++)
309  dead[k]--;
310 
311 #ifdef TRACKER_VERY_VERBOSE
312  sprintf(tmp, " Target %d died ", ind);
313  strcat(c_event, tmp);
314 #endif
315  }
316  }
317  else {
318  if (nDead==1){
319  /* Find index of the target to remove */
320  ind = -1;
321  for(k=0; k<S->nTargets; k++)
322  if(dead[0]==k)
323  ind = k;
324  saf_assert(ind != -1, "Ugly error");
325 
326  S->nTargets--;
327  if(ind!=S->nTargets){
328  memmove(&S->M[ind], &S->M[ind+1], (S->nTargets-ind)*sizeof(M6));
329  memmove(&S->P[ind], &S->P[ind+1], (S->nTargets-ind)*sizeof(P66));
330  memmove(&S->Tcount[ind], &S->Tcount[ind+1], (S->nTargets-ind)*sizeof(int));
331  memmove(&S->targetIDs[ind], &S->targetIDs[ind+1], (S->nTargets-ind)*sizeof(int));
332  }
333 
334 #ifdef TRACKER_VERY_VERBOSE
335  sprintf(tmp, ", Target %d died ", ind);
336  strcpy(c_event, tmp);
337 #endif
338  }
339  }
340 
341  /* Print particle state */
342 #ifdef TRACKER_VERY_VERBOSE
343  sprintf(S->evstr, "MCS: %d, W: %.7f, IDs: [", i, S->W);
344  for (j=0; j<S->nTargets; j++){
345  sprintf(tmp, "%d ", S->targetIDs[j]);
346  strcat(S->evstr, tmp);
347  }
348  strcat(S->evstr, "] ");
349  strcat(S->evstr, c_event);
350  printf("%s\n", S->evstr);
351 #endif
352  }
353 }

355 void tracker3d_update
356 (
357  void* const hT3d,
358  float* Y,
359  int Tinc
360 )
361 {
362  tracker3d_data *pData = (tracker3d_data*)(hT3d);
363  int i, j, k, ss, n_events, count, cidx, unique, j_new, ev;
364  float TP0, LH, norm;
365  float M[6], P[6][6];
366  MCS_data* S, *S_event;
367  tracker3d_config* tpars = &(pData->tpars);
368 #ifdef TRACKER_VERY_VERBOSE
369  char tmp[256];
370  printf("%s\n", "Update step");
371 #endif
372 
373  /* Loop over particles */
374  for (i=0; i<tpars->Np; i++){
375  S = (MCS_data*)pData->SS[i];
376 
377  /* Association priors to targets */
378  TP0 = (1.0f-tpars->noiseLikelihood)/(S->nTargets+2.23e-10f);
379 
380  /* Number of possible events: */
381  n_events = S->nTargets + 1; /* clutter (+1) or 1 of the targets is active */
382  if( S->nTargets < tpars->maxNactiveTargets)
383  n_events++; /* Also a chance of a new target */
384  saf_assert(n_events<=TRACKER3D_MAX_NUM_EVENTS, "Number of hypotheses/events exceeded the maximum");
385 
386  /* Prep */
387 #ifdef TRACKER_VERBOSE
388  memset(pData->evt, 0, n_events*256*sizeof(char)); /* Event descriptions */
389 #endif
390  count = 0; /* Event counter */
391  cidx = 0; /* current index */
392 
393  /* Association to clutter */
394  count++;
395 #ifdef TRACKER_VERBOSE
396  strcpy(pData->evt[cidx], "Clutter");
397 #endif
398  pData->evta[cidx] = -1;
399  pData->evp[cidx] = (1.0f-tpars->init_birth)*tpars->noiseLikelihood;
400  pData->evl[cidx] = tpars->cd;
401  tracker3d_particleCopy(pData->SS[i], pData->str[cidx]);
402  cidx++;
403 
404  /* Loop over associations to targets */
405  for (j=0; j<S->nTargets; j++){
406  /* Compute update result and likelihood for association to signal j */
407  kf_update6(pData->hKF6, S->M[j].M, S->P[j].P, Y, pData->H, pData->R, M, P, &LH);
408  if(pData->tpars.ARE_UNIT_VECTORS)
409  cblas_sscal(3, 1.0f/L2_norm3(M), M, 1);
410 
411  /* Assocation to target j */
412  count++;
413 #ifdef TRACKER_VERBOSE
414  sprintf(pData->evt[cidx], "Target %d ", S->targetIDs[j]);
415 #endif
416  pData->evta[cidx] = S->targetIDs[j];
417  pData->evp[cidx] = (1.0f-tpars->init_birth)*TP0;
418  pData->evl[cidx] = LH;
419  tracker3d_particleCopy(pData->SS[i], pData->str[cidx]);
420  S_event = (MCS_data*)pData->str[cidx];
421  cblas_scopy(6, (float*)M, 1, (float*)S_event->M[j].M, 1);
422  cblas_scopy(6*6, (float*)P, 1, (float*)S_event->P[j].P, 1);
423  for(k=0; k<S->nTargets; k++)
424  S_event->Tcount[k] += Tinc;
425  cidx++;
426  }
427 
428  /* Association to new target */
429  if (S->nTargets < tpars->maxNactiveTargets && S->nTargets < TRACKER3D_MAX_NUM_TARGETS){
430  /* Initialization of new target */
431  kf_update6(pData->hKF6, tpars->M0, tpars->P0, Y, pData->H, pData->R, M, P, &LH);
432  if(pData->tpars.ARE_UNIT_VECTORS)
433  cblas_sscal(3, 1.0f/L2_norm3(M), M, 1);
434 
435  /* find an untaken ID */
436  j_new = 0;
437  for (ss = 0; ss<tpars->maxNactiveTargets; ss++){
438  unique = 1;
439  for(j=0; j<S->nTargets; j++){
440  if(ss == S->targetIDs[j]){
441  unique = 0;
442  }
443  };
444  if(unique){
445  j_new = ss;
446  break;
447  }
448  }
449 
450  count++;
451  j = S->nTargets;
452 #ifdef TRACKER_VERBOSE
453  sprintf(pData->evt[cidx], "New Target %d ", j);
454 #endif
455  pData->evta[cidx] = j;
456  pData->evp[cidx] = tpars->init_birth;
457  pData->evl[cidx] = LH;
458  tracker3d_particleCopy(pData->SS[i], pData->str[cidx]);
459  S_event = (MCS_data*)pData->str[cidx];
460  S_event->nTargets = j+1;
461  cblas_scopy(6, (float*)M, 1, (float*)S_event->M[j].M, 1);
462  cblas_scopy(6*6, (float*)P, 1, (float*)S_event->P[j].P, 1);
463  S_event->Tcount[j] = 0;
464  S_event->targetIDs[j] = j_new;
465  cidx++;
466  }
467 
468  /* Draw sample from importance distribution */
469  norm = 1.0f/sumf(pData->evp, count);
470  cblas_sscal(count, norm, pData->evp, 1);
471  utility_svvmul(pData->evp, pData->evl, count, pData->imp);
472  norm = 1.0f/sumf(pData->imp, count);
473  cblas_sscal(count, norm, pData->imp, 1);
474  ev = categ_rnd(pData->imp, count); /* Event index */
475  saf_assert(ev!=-1, "Falied to randomly select an event");
476 
477  /* Update particle */
478  tracker3d_particleCopy(pData->str[ev], pData->SS[i]);
479  S->W *= (pData->evl[ev] * pData->evp[ev]/ pData->imp[ev]);
480 
481  /* Print particle state */
482 #ifdef TRACKER_VERY_VERBOSE
483  sprintf(S->evstr, "MCS: %d, W: %.7f, IDs: [", i, S->W);
484  for (j=0; j<S->nTargets; j++){
485  sprintf(tmp, "%d ", S->targetIDs[j]);
486  strcat(S->evstr, tmp);
487  }
488  strcat(S->evstr, "] ");
489  strcat(S->evstr, pData->evt[ev]);
490  printf("%s\n", S->evstr);
491 #endif
492  }
493 
494  normalise_weights(pData->SS, tpars->Np);
495 }

524 void resampstr
525 (
526  voidPtr* SS,
527  int NP,
528  int* s
529 )
530 {
531  int i, j, k, a;
532  float c;
533  float pn[TRACKER3D_MAX_NUM_PARTICLES], r[TRACKER3D_MAX_NUM_PARTICLES];
534 
535  for (i=0; i<NP; i++)
536  pn[i] = ((MCS_data*)SS[i])->W*(float)NP;
537  memset(s, 0, NP*sizeof(int));
538  rand_0_1(r, NP);
539  k=0;
540  c=0.0f;
541  for (i=0; i<NP; i++){
542  c+=pn[i];
543  if (c>=1.0f) {
544  a = (int)floorf(c);
545  c = c-a;
546  for(j=0; j<a; j++)
547  s[k+j]=i;
548  k=k+a;
549  }
550  if (k<NP && c>=r[k]){
551  c=c-1.0f;
552  s[k]=i;
553  k++;
554  }
555  }
556 }
