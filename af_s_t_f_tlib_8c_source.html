78 static void afAnalyse
79 (
80  float* inTD/* nSamplesTD x nCH */,
81  int nSamplesTD,
82  int nCH,
83  int hopSize,
84  int LDmode,
85  int hybridmode,
86  float_complex* outTF /* out_nBands x nTimeslots x nCH */
87 )
88 {
89  int t, ch, sample, band;
90  void* hSTFT;
91  float_complex*** FrameTF;
92  float** tempFrameTD;
93  int nTimeSlots, nBands;
94 
95  nBands = hopSize + (hybridmode ? 5 : 1);
96  nTimeSlots = nSamplesTD/hopSize;
97 
98  /* allocate memory */
99  afSTFT_create(&(hSTFT), nCH, 1, hopSize, LDmode, hybridmode, AFSTFT_TIME_CH_BANDS);
100  FrameTF = (float_complex***)malloc3d(nTimeSlots, nCH, nBands, sizeof(float_complex));
101  tempFrameTD = (float**)calloc2d(nCH, nTimeSlots*hopSize, sizeof(float));
102 
103  /* perform TF transform */
104  for(ch=0; ch<nCH; ch++)
105  for(sample=0; sample<nSamplesTD; sample++)
106  tempFrameTD[ch][sample] = inTD[sample* nCH + ch];
107  afSTFT_forward(hSTFT, tempFrameTD, nTimeSlots*hopSize, FrameTF);
108 
109  /* save result to output */
110  for (band = 0; band < nBands; band++)
111  for (t = 0; t < nTimeSlots; t++)
112  for (ch = 0; ch < nCH; ch++)
113  outTF[band * nTimeSlots * nCH + t * nCH + ch] = FrameTF[t][ch][band];
114 
115  /* clean-up */
116  afSTFT_destroy(&hSTFT);
117  free(FrameTF);
118  free(tempFrameTD);
119 }

143 void afSTFT_create
144 (
145  void ** const phSTFT,
146  int nCHin,
147  int nCHout,
148  int hopsize,
149  int lowDelayMode,
150  int hybridmode,
151  AFSTFT_FDDATA_FORMAT format
152 )
153 {
154  *phSTFT = malloc1d(sizeof(afSTFT_data));
155  afSTFT_data *h = (afSTFT_data*)(*phSTFT);
156  int ch;
157 
158  if(hybridmode)
159  assert(hopsize==64 || hopsize==128 || hopsize==256);
160  assert(1024 % hopsize == 0 );
161 
162  h->nCHin = nCHin;
163  h->nCHout = nCHout;
164  h->hopsize = hopsize;
165  h->hybridmode = hybridmode;
166  h->nBands = hybridmode ? hopsize+5 : hopsize+1; /* hybrid mode incurs an additional 4 bands */
167  if(lowDelayMode)
168  h->afSTFTdelay = hybridmode ? 7*hopsize : 4*hopsize; /* hybrid mode incurs 3 additional hops of latency */
169  else
170  h->afSTFTdelay = hybridmode ? 12*hopsize : 9*hopsize; /* hybrid mode incurs 3 additional hops of latency */
171  h->format = format;
172 
173  /* init afSTFT core */
174  afSTFTlib_init(&(h->hInt), hopsize, nCHin, nCHout, lowDelayMode, hybridmode);
175 
176  /* temp buffers */
177  if(nCHout>0){
178  h->STFTOutputFrameTF = malloc1d(nCHout * sizeof(complexVector));
179  for(ch=0; ch < nCHout; ch++) {
180  h->STFTOutputFrameTF[ch].re = (float*)calloc1d(h->nBands, sizeof(float));
181  h->STFTOutputFrameTF[ch].im = (float*)calloc1d(h->nBands, sizeof(float));
182  }
183  }
184  else
185  h->STFTOutputFrameTF = NULL;
186  if(nCHout > 0 || nCHin > 0)
187  h->tempHopFrameTD = (float**)malloc2d( SAF_MAX(nCHin, nCHout), hopsize, sizeof(float));
188  if(nCHin>0){
189  h->STFTInputFrameTF = malloc1d(nCHin * sizeof(complexVector));
190  for(ch=0; ch < nCHin; ch++) {
191  h->STFTInputFrameTF[ch].re = (float*)calloc1d(h->nBands, sizeof(float));
192  h->STFTInputFrameTF[ch].im = (float*)calloc1d(h->nBands, sizeof(float));
193  }
194  }
195  else
196  h->STFTInputFrameTF = NULL;
197 }

230 void afSTFT_forward
231 (
232  void * const hSTFT,
233  float** dataTD,
234  int framesize,
235  float_complex*** dataFD
236 )
237 {
238  afSTFT_data *h = (afSTFT_data*)(hSTFT);
239  int ch, t, nHops, band;
240 
241  assert(framesize % h->hopsize == 0); /* framesize must be multiple of hopsize */
242  nHops = framesize/h->hopsize;
243 
244  /* Loop over hops */
245  for(t=0; t < nHops; t++) {
246  /* forward transform */
247  for(ch = 0; ch < h->nCHin; ch++)
248  utility_svvcopy(&(dataTD[ch][t*(h->hopsize)]), (h->hopsize), h->tempHopFrameTD[ch]);
249  afSTFTlib_forward(h->hInt, h->tempHopFrameTD, h->STFTInputFrameTF);
250 
251  /* store */
252  switch(h->format){
253  case AFSTFT_BANDS_CH_TIME:
254  for(band=0; band<h->nBands; band++)
255  for(ch=0; ch < h->nCHin; ch++)
256  dataFD[band][ch][t] = cmplxf(h->STFTInputFrameTF[ch].re[band], h->STFTInputFrameTF[ch].im[band]);
257  break;
258  case AFSTFT_TIME_CH_BANDS:
259  for(ch=0; ch < h->nCHin; ch++){
260  cblas_scopy(h->nBands, h->STFTInputFrameTF[ch].re, 1, (float*)dataFD[t][ch], 2);
261  cblas_scopy(h->nBands, h->STFTInputFrameTF[ch].im, 1, &((float*)dataFD[t][ch])[1], 2);
262  }
263  break;
264  }
265  }
266 }

547 (

550  int nBands,

552 )

562  }

575  getUniformFreqVector(h->hopsize*2, fs, centreFreq_tmp);
576 
577  /* convert first 5 to hybrid centre frequencies */

579  (const float*)__stft2hybCentreFreq, 5,

581  freqVector, 1);
582 
583  /* Remaining centre frequencies are then the uniform centre frequencies 6:end */

593 void afSTFT_FIRtoFilterbankCoeffs
594 (
595  float* hIR /*N_dirs x nCH x ir_len*/,
596  int N_dirs,
597  int nCH,
598  int ir_len,
599  int hopSize,
600  int LDmode,
601  int hybridmode,
602  float_complex* hFB /* nBands x nCH x N_dirs */
603 )
604 {
605  int i, j, t, nd, nm, nTimeSlots, ir_pad, nBands;
606  int* maxIdx;
607  float maxVal, idxDel, irFB_energy, irFB_gain, phase;
608  float* centerImpulse, *centerImpulseFB_energy, *ir;
609  float_complex cross;
610  float_complex* centerImpulseFB, *irFB;
611 
612  nBands = hopSize + (hybridmode ? 5 : 1);
613  ir_pad = 1024;//+512;
614  nTimeSlots = (SAF_MAX(ir_len,hopSize)+ir_pad)/hopSize;
615  maxIdx = calloc1d(nCH,sizeof(int));
616  centerImpulse = calloc1d(SAF_MAX(ir_len,hopSize)+ir_pad, sizeof(float));
617 
618  /* pick a direction to estimate the center of FIR delays */
619  for(j=0; j<nCH; j++){
620  maxVal = 2.23e-13f;
621  for(i=0; i<ir_len; i++){
622  if(hIR[j*ir_len + i] > maxVal){
623  maxVal = hIR[j*ir_len + i];
624  maxIdx[j] = i;
625  }
626  }
627  }
628  idxDel = 0.0f;
629  for(j=0; j<nCH; j++)
630  idxDel += (float)maxIdx[j];
631  idxDel /= (float)nCH;
632  idxDel = (idxDel + 1.5f);
633 
634  /* ideal impulse at mean delay */
635  centerImpulse[(int)idxDel] = 1.0f;
636 
637  /* analyse impulse with the filterbank */
638  centerImpulseFB = malloc1d(nBands*nTimeSlots*nCH*sizeof(float_complex));
639  afAnalyse(centerImpulse, SAF_MAX(ir_len,hopSize)+ir_pad, 1, hopSize, LDmode, hybridmode, centerImpulseFB);
640  centerImpulseFB_energy = calloc1d(nBands, sizeof(float));
641  for(i=0; i<nBands; i++)
642  for(t=0; t<nTimeSlots; t++)
643  centerImpulseFB_energy[i] += powf(cabsf(centerImpulseFB[i*nTimeSlots + t]), 2.0f);
644 
645  /* initialise FB coefficients */
646  ir = calloc1d( (SAF_MAX(ir_len,hopSize)+ir_pad) * nCH, sizeof(float));
647  irFB = malloc1d(nBands*nCH*nTimeSlots*sizeof(float_complex));
648  for(nd=0; nd<N_dirs; nd++){
649  for(j=0; j<ir_len; j++)
650  for(i=0; i<nCH; i++)
651  ir[j*nCH+i] = hIR[nd*nCH*ir_len + i*ir_len + j];
652  afAnalyse(ir, SAF_MAX(ir_len,hopSize)+ir_pad, nCH, hopSize, LDmode, hybridmode, irFB);
653  for(nm=0; nm<nCH; nm++){
654  for(i=0; i<nBands; i++){
655  irFB_energy = 0;
656  for(t=0; t<nTimeSlots; t++)
657  irFB_energy += powf(cabsf(irFB[i*nTimeSlots*nCH + t*nCH + nm]), 2.0f); /* out_nBands x nTimeslots x nCH */
658  irFB_gain = sqrtf(irFB_energy/SAF_MAX(centerImpulseFB_energy[i], 2.23e-8f));
659  cross = cmplxf(0.0f,0.0f);
660  for(t=0; t<nTimeSlots; t++)
661  cross = ccaddf(cross, ccmulf(irFB[i*nTimeSlots*nCH + t*nCH + nm], conjf(centerImpulseFB[i*nTimeSlots + t])));
662  phase = atan2f(cimagf(cross), crealf(cross));
663  hFB[i*nCH*N_dirs + nm*N_dirs + nd] = crmulf( cexpf(cmplxf(0.0f, phase)), irFB_gain);
664  }
665  }
666  }
667 
668  /* clean-up */
669  free(maxIdx);
670  free(centerImpulse);
671  free(centerImpulseFB_energy);
672  free(centerImpulseFB);
673  free(ir);
674  free(irFB);
675 }
