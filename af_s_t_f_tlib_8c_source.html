104  int nCHin,
105  int nCHout,
106  int hopsize,
107  int lowDelayMode,
108  int hybridmode,
109  AFSTFT_FDDATA_FORMAT format
110 )
111 {

113  afSTFT_data *h = (afSTFT_data*)(*phSTFT);
114  int ch;
115 
116  if(hybridmode)

118  assert(1024 % hopsize == 0 );
119 
120  h->nCHin = nCHin;
121  h->nCHout = nCHout;
122  h->hopsize = hopsize;
123  h->hybridmode = hybridmode;
124  h->nBands = hybridmode ? hopsize+5 : hopsize+1; /* hybrid mode incurs an additional 4 bands */
125  h->afSTFTdelay = hybridmode ? 12*hopsize : 9*hopsize; /* hybrid mode incurs 3 additional hops of latency */

131  /* temp buffers */

134  for(ch=0; ch < nCHout; ch++) {

137  }
138  }

141  if(nCHout > 0 || nCHin > 0)
142  h->tempHopFrameTD = (float**)malloc2d( MAX(nCHin, nCHout), hopsize, sizeof(float));
143  if(nCHin>0){

185 void afSTFT_forward
186 (
187  void * const hSTFT,
188  float** dataTD,
189  int framesize,
190  float_complex*** dataFD
191 )
192 {
193  afSTFT_data *h = (afSTFT_data*)(hSTFT);
194  int ch, t, nHops, band;
195 
196  assert(framesize % h->hopsize == 0); /* framesize must be multiple of hopsize */
197  nHops = framesize/h->hopsize;
198 
199  /* Loop over hops */
200  for(t=0; t < nHops; t++) {
201  /* forward transform */
202  for(ch = 0; ch < h->nCHin; ch++)
203  utility_svvcopy(&(dataTD[ch][t*(h->hopsize)]), (h->hopsize), h->tempHopFrameTD[ch]);
204  afSTFTlib_forward(h->hInt, h->tempHopFrameTD, h->STFTInputFrameTF);
205 
206  /* store */
207  switch(h->format){
208  case AFSTFT_BANDS_CH_TIME:
209  for(band=0; band<h->nBands; band++)
210  for(ch=0; ch < h->nCHin; ch++)
211  dataFD[band][ch][t] = cmplxf(h->STFTInputFrameTF[ch].re[band], h->STFTInputFrameTF[ch].im[band]);
212  break;
213  case AFSTFT_TIME_CH_BANDS:
214  for(band=0; band<h->nBands; band++)
215  for(ch=0; ch < h->nCHin; ch++)
216  dataFD[t][ch][band] = cmplxf(h->STFTInputFrameTF[ch].re[band], h->STFTInputFrameTF[ch].im[band]);
217  break;
218  }
219  }
220 }

271  nHops = framesize/h->hopsize;
272

275  /* backward transform */

414 void afSTFT_getCentreFreqs
415 (
416  void * const hSTFT,
417  float fs,
418  int nBands,
419  float* freqVector
420 )
421 {
422  int band;
423  if(hSTFT==NULL){
424  assert(nBands>=133);
425  for(band=0; band < nBands; band++){
426  if(fs==44100.0f)
427  freqVector[band] = (float)__afCenterFreq44100[band];
428  else /* assume 48e3 */
429  freqVector[band] = (float)__afCenterFreq48e3[band];
430  }
431  return;
432  }
433 
434  afSTFT_data *h = (afSTFT_data*)(hSTFT);
435  float* centreFreq_tmp;
436  int i, j;
437 
438  assert(nBands >= h->nBands); /* just to check that "centreFreq" is of sufficient length */
439 
440  if(h->hybridmode){
441  /* uniform frequency vector: */
442  centreFreq_tmp = malloc1d((h->hopsize+1)*sizeof(float));
443  getUniformFreqVector(h->hopsize*2, fs, centreFreq_tmp);
444 
445  /* convert first 5 to hybrid centre frequencies */
446  cblas_sgemm(CblasRowMajor, CblasNoTrans, CblasNoTrans, 9, 1, 5, 1.0f,
447  (const float*)__stft2hybCentreFreq, 5,
448  centreFreq_tmp, 1, 0.0f,
449  freqVector, 1);
450 
451  /* Remaining centre frequencies are then the uniform centre frequencies 6:end */
452  for(i=9, j=5; i<h->nBands; i++, j++)
453  freqVector[i] = centreFreq_tmp[j];
454 
455  free(centreFreq_tmp);
456  }
457  else
458  getUniformFreqVector(h->hopsize*2, fs, freqVector);
459 }
