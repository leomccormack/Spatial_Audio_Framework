89 void quaternion2rotationMatrix
90 (
91  quaternion_data* Q,
92  float R[3][3]
93 )
94 {
95  R[0][0] = 2.0f * (Q->w * Q->w + Q->z * Q->z) - 1.0f;
96  R[0][1] = 2.0f * (Q->z * Q->y - Q->w * Q->x);
97  R[0][2] = 2.0f * (Q->z * Q->x + Q->w * Q->y);
98  R[1][0] = 2.0f * (Q->z * Q->y + Q->w * Q->x);
99  R[1][1] = 2.0f * (Q->w * Q->w + Q->y * Q->y) - 1.0f;
100  R[1][2] = 2.0f * (Q->y * Q->x - Q->w * Q->z);
101  R[2][0] = 2.0f * (Q->z * Q->x - Q->w * Q->y);
102  R[2][1] = 2.0f * (Q->y * Q->x + Q->w * Q->z);
103  R[2][2] = 2.0f * (Q->w * Q->w + Q->x * Q->x) - 1.0f;
104 }

106 /* Adapted from: https://www.euclideanspace.com/maths/geometry/rotations/conversions/matrixToQuaternion/forum.htm */
107 void rotationMatrix2quaternion
108 (
109  float R[3][3],
110  quaternion_data* Q
111 )
112 {
113  Q->w = sqrtf( SAF_MAX( 0.0f, 1.0f + R[0][0] + R[1][1] + R[2][2] ) ) / 2.0f;
114  Q->z = sqrtf( SAF_MAX( 0.0f, 1.0f + R[0][0] - R[1][1] - R[2][2] ) ) / 2.0f;
115  Q->y = sqrtf( SAF_MAX( 0.0f, 1.0f - R[0][0] + R[1][1] - R[2][2] ) ) / 2.0f;
116  Q->x = sqrtf( SAF_MAX( 0.0f, 1.0f - R[0][0] - R[1][1] + R[2][2] ) ) / 2.0f;
117  Q->z = copysignf( Q->z, R[2][1] - R[1][2] );
118  Q->y = copysignf( Q->y, R[0][2] - R[2][0] );
119  Q->x = copysignf( Q->x, R[1][0] - R[0][1] );
120 }

122 /* Adapted from (ISC License): https://github.com/MartinWeigel/Quaternion */
123 void euler2Quaternion
124 (
125  float alpha,
126  float beta,
127  float gamma,
128  int degreesFlag,
129  EULER_ROTATION_CONVENTIONS convention,
130  quaternion_data* Q
131 )
132 {
133  float cy, sy, cr, sr, cp, sp;
134 
135  cy = sy = cr = sr = cp = sp = 0.0f;
136  switch(convention){
137  case EULER_ROTATION_Y_CONVENTION: /* fall through*/
138  case EULER_ROTATION_X_CONVENTION: saf_print_error("This convention is not supported"); return;
139  case EULER_ROTATION_YAW_PITCH_ROLL:
140  cy = cosf((degreesFlag ? alpha*SAF_PI/180.0f : alpha) * 0.5f); /* x */
141  sy = sinf((degreesFlag ? alpha*SAF_PI/180.0f : alpha) * 0.5f); /* x */
142  cp = cosf((degreesFlag ? beta*SAF_PI/180.0f : beta) * 0.5f); /* y */
143  sp = sinf((degreesFlag ? beta*SAF_PI/180.0f : beta) * 0.5f); /* y */
144  cr = cosf((degreesFlag ? gamma*SAF_PI/180.0f : gamma) * 0.5f); /* z */
145  sr = sinf((degreesFlag ? gamma*SAF_PI/180.0f : gamma) * 0.5f); /* z */
146  break;
147  case EULER_ROTATION_ROLL_PITCH_YAW:
148  cy = cosf((degreesFlag ? gamma*SAF_PI/180.0f : gamma) * 0.5f); /* x */
149  sy = sinf((degreesFlag ? gamma*SAF_PI/180.0f : gamma) * 0.5f); /* x */
150  cp = cosf((degreesFlag ? beta*SAF_PI/180.0f : beta) * 0.5f); /* y */
151  sp = sinf((degreesFlag ? beta*SAF_PI/180.0f : beta) * 0.5f); /* y */
152  cr = cosf((degreesFlag ? alpha*SAF_PI/180.0f : alpha) * 0.5f); /* z */
153  sr = sinf((degreesFlag ? alpha*SAF_PI/180.0f : alpha) * 0.5f); /* z */
154  break;
155  }
156  Q->w = cy * cr * cp + sy * sr * sp;
157  Q->x = cy * sr * cp - sy * cr * sp;
158  Q->y = cy * cr * sp + sy * sr * cp;
159  Q->z = sy * cr * cp - cy * sr * sp;
160 }

162 /* Adapted from (ISC License): https://github.com/MartinWeigel/Quaternion */
163 void quaternion2euler
164 (
165  quaternion_data* Q,
166  int degreesFlag,
167  EULER_ROTATION_CONVENTIONS convention,
168  float* alpha,
169  float* beta,
170  float* gamma
171 )
172 {
173  float sinr_cosp, cosr_cosp, sinp, siny_cosp, cosy_cosp;
174 
175  sinr_cosp = 2.0f * (Q->w * Q->x + Q->y * Q->z);
176  cosr_cosp = 1.0f - 2.0f * (Q->x * Q->x + Q->y * Q->y);
177  sinp = 2.0f * (Q->w * Q->y - Q->z * Q->x);
178  siny_cosp = 2.0f * (Q->w * Q->z + Q->x * Q->y);
179  cosy_cosp = 1.0f - 2.0f * (Q->y * Q->y + Q->z * Q->z);
180  switch(convention){
181  case EULER_ROTATION_Y_CONVENTION: /* fall through */
182  case EULER_ROTATION_X_CONVENTION: saf_print_error("This convention is not supported"); break;
183  case EULER_ROTATION_YAW_PITCH_ROLL:
184  /* Yaw (z-axis rotation) */
185  (*gamma) = atan2f(sinr_cosp, cosr_cosp);
186  /* Pitch (y-axis rotation) */
187  if (fabsf(sinp) >= 1.0f)
188  (*beta) = copysignf(SAF_PI / 2.0f, sinp); /* use 90 degrees if out of range */
189  else
190  (*beta) = asinf(sinp);
191  /* Roll (x-axis rotation) */
192  (*alpha) = atan2f(siny_cosp, cosy_cosp);
193  break;
194  case EULER_ROTATION_ROLL_PITCH_YAW:
195  /* Roll (x-axis rotation) */
196  (*alpha) = atan2f(sinr_cosp, cosr_cosp);
197  /* Pitch (y-axis rotation) */
198  if (fabs(sinp) >= 1.0f)
199  (*beta) = copysignf(SAF_PI / 2.0f, sinp); /* use 90 degrees if out of range */
200  else
201  (*beta) = asinf(sinp);
202  /* Yaw (z-axis rotation) */
203  (*gamma) = atan2f(siny_cosp, cosy_cosp);
204  break;
205  }
206  if(degreesFlag){
207  (*alpha) *= 180.0f/SAF_PI;
208  (*beta) *= 180.0f/SAF_PI;
209  (*gamma) *= 180.0f/SAF_PI;
210  }
211 }

213 void euler2rotationMatrix
214 (
215  float alpha,
216  float beta,
217  float gamma,
218  int degreesFlag,
219  EULER_ROTATION_CONVENTIONS convention,
220  float R[3][3]
221 )
222 {
223  float R1[3][3], R2[3][3], R3[3][3], Rtmp[3][3];
224 
225  switch(convention){
226  case EULER_ROTATION_Y_CONVENTION:
227  getRz(degreesFlag ? alpha*SAF_PI/180.0f : alpha, R1);
228  getRy(degreesFlag ? beta*SAF_PI/180.0f : beta, R2);
229  getRz(degreesFlag ? gamma*SAF_PI/180.0f : gamma, R3);
230  break;
231  case EULER_ROTATION_X_CONVENTION:
232  getRz(degreesFlag ? alpha*SAF_PI/180.0f : alpha, R1);
233  getRx(degreesFlag ? beta*SAF_PI/180.0f : beta, R2);
234  getRz(degreesFlag ? gamma*SAF_PI/180.0f : gamma, R3);
235  break;
236  case EULER_ROTATION_YAW_PITCH_ROLL:
237  getRz(degreesFlag ? alpha*SAF_PI/180.0f : alpha, R1);
238  getRy(degreesFlag ? beta*SAF_PI/180.0f : beta, R2);
239  getRx(degreesFlag ? gamma*SAF_PI/180.0f : gamma, R3);
240  break;
241  case EULER_ROTATION_ROLL_PITCH_YAW:
242  getRx(degreesFlag ? alpha*SAF_PI/180.0f : alpha, R1);
243  getRy(degreesFlag ? beta*SAF_PI/180.0f : beta, R2);
244  getRz(degreesFlag ? gamma*SAF_PI/180.0f : gamma, R3);
245  break;
246  }
247  cblas_sgemm(CblasRowMajor, CblasNoTrans, CblasNoTrans, 3, 3, 3, 1.0f,
248  (float*)R2, 3,
249  (float*)R1, 3, 0.0f,
250  (float*)Rtmp, 3);
251  cblas_sgemm(CblasRowMajor, CblasNoTrans, CblasNoTrans, 3, 3, 3, 1.0f,
252  (float*)R3, 3,
253  (float*)Rtmp, 3, 0.0f,
254  (float*)R, 3);
255 }

273 (
274  float* dirs,
275  int nDirs,
276  int anglesInDegreesFLAG,
277  float* dirs_xyz
278 )
279 {
280  int i;
281  float tmp_rad[2];
282 
283  if(anglesInDegreesFLAG){
284  for(i=0; i<nDirs; i++){
285  tmp_rad[0] = dirs[i*2] * SAF_PI/180.0f;
286  tmp_rad[1] = dirs[i*2+1] * SAF_PI/180.0f;
287  dirs_xyz[i*3] = cosf(tmp_rad[1]) * cosf(tmp_rad[0]);

290  }
291  }
292  else { /* Angles given in radians */
293  for(i=0; i<nDirs; i++){
294  dirs_xyz[i*3] = cosf(dirs[i*2+1]) * cosf(dirs[i*2]);
295  dirs_xyz[i*3+1] = cosf(dirs[i*2+1]) * sinf(dirs[i*2]);
296  dirs_xyz[i*3+2] = sinf(dirs[i*2+1]);
297  }
298  }

378 float getDistBetweenPointAndLine
379 (
380  float point[3],
381  float v1[3],
382  float v2[3]
383 )
384 {
385  float a[3], b[3], cross_a_ab[3];
386  a[0] = v1[0] - v2[0];
387  a[1] = v1[1] - v2[1];
388  a[2] = v1[2] - v2[2];
389  b[0] = point[0] - v2[0];
390  b[1] = point[1] - v2[1];
391  b[2] = point[2] - v2[2];
392  crossProduct3(a, b, cross_a_ab);
393  return L2_norm3(cross_a_ab)/(L2_norm3(a)+2.3e-9f);
394 }

474 void delaunaynd
475 (
476  const float* points,
477  const int nPoints,
478  const int nd,
479  int** DT,
480  int* nDT
481 )
482 {
483  int i, j, k, nHullFaces, maxW_idx, nVisible;
484  int* hullfaces;
485  CH_FLOAT w0, w_optimal, w_optimal2;
486  CH_FLOAT* projpoints, *cf, *df, *p0, *p, *visible;
487 
488  /* Project the N-dimensional points onto a N+1-dimensional paraboloid */
489  projpoints = malloc1d(nPoints*(nd+1)*sizeof(CH_FLOAT));
490  for(i = 0; i < nPoints; i++) {
491  projpoints[i*(nd+1)+nd] = 0.0;
492  for(j=0; j<nd; j++){
493  projpoints[i*(nd+1)+j] = (CH_FLOAT)points[i*nd+j] + 0.0000001*(CH_FLOAT)rand()/(CH_FLOAT)RAND_MAX;
494  projpoints[i*(nd+1)+nd] += (projpoints[i*(nd+1)+j]*projpoints[i*(nd+1)+j]); /* w vector */
495  }
496  }
497 
498  /* The N-dimensional delaunay triangulation requires first computing the convex hull of this N+1-dimensional paraboloid */
499  hullfaces = NULL;
500  cf = df = NULL;
501  convhull_nd_build(projpoints, nPoints, nd+1, &hullfaces, &cf, &df, &nHullFaces);
502 
503  /* Find the coordinates of the point with the maximum (N+1 dimension) coordinate (i.e. the w vector) */
504  if(sizeof(CH_FLOAT)==sizeof(double))
505  maxW_idx = (int)cblas_idamax(nPoints, (double*)&projpoints[nd], nd+1);
506  else
507  maxW_idx = (int)cblas_isamax(nPoints, (float*)&projpoints[nd], nd+1);
508  w0 = projpoints[maxW_idx*(nd+1)+nd];
509  p0 = malloc1d(nd*sizeof(CH_FLOAT));
510  for(j=0; j<nd; j++)
511  p0[j] = projpoints[maxW_idx*(nd+1)+j];
512 
513  /* Find the point where the plane tangent to the point (p0,w0) on the paraboloid crosses the w axis.
514  * This is the point that can see the entire lower hull. */
515  w_optimal = 0.0;
516  for(j=0; j<nd; j++)
517  w_optimal += (2.0*pow(p0[j], 2.0));
518  w_optimal = w0-w_optimal;
519 
520  /* Subtract 1000 times the absolute value of w_optimal to ensure that the point where the tangent plane
521  * crosses the w axis will see all points on the lower hull. This avoids numerical roundoff errors. */
522  w_optimal2=w_optimal-1000.0*fabs(w_optimal);
523 
524  /* Set the point where the tangent plane crosses the w axis */
525  p = calloc1d((nd+1),sizeof(CH_FLOAT));
526  p[nd] = w_optimal2;
527 
528  /* Find all faces that are visible from this point */
529  visible = malloc1d(nHullFaces*sizeof(CH_FLOAT));
530  if(sizeof(CH_FLOAT)==sizeof(double)){
531  cblas_dgemm(CblasRowMajor, CblasNoTrans, CblasNoTrans, nHullFaces, 1, nd+1, 1.0,
532  (double*)cf, nd+1,
533  (double*)p, 1, 0.0,
534  (double*)visible, 1);
535  }
536  else{
537  cblas_sgemm(CblasRowMajor, CblasNoTrans, CblasNoTrans, nHullFaces, 1, nd+1, 1.0f,
538  (float*)cf, nd+1,
539  (float*)p, 1, 0.0f,
540  (float*)visible, 1);
541  }
542  nVisible = 0;
543  for(j=0; j<nHullFaces; j++){
544  visible[j] += df[j];
545  if(visible[j]>0.0)
546  nVisible++;
547  }
548 
549  /* Output */
550  (*nDT) = nVisible;
551  if(nVisible>0){
552  (*DT) = malloc1d(nVisible*(nd+1)*sizeof(int));
553  for(i=0, j=0; i<nHullFaces; i++){
554  if(visible[i]>0.0){
555  for(k=0; k<nd+1; k++)
556  (*DT)[j*(nd+1)+k] = hullfaces[i*(nd+1)+k];
557  j++;
558  }
559  }
560  saf_assert(j==nVisible, "Ugly error");
561  }
562 
563  /* clean up */
564  free(projpoints);
565  free(hullfaces);
566  free(cf);
567  free(df);

607 void sphVoronoi
608 (
609  int* faces,
610  int nFaces,
611  float* vertices,
612  int nDirs,
613  voronoi_data* voronoi
614 )
615 {
616  int n, m;
617  int* duplicates;
618  float r_12[3], r_13[3], r_normal[3];
619  float norm;
620 
621  /* prep */
622  voronoi->nVert = nFaces;
623  voronoi->nFaces = nDirs;
624  voronoi->vert = (float**)malloc2d(voronoi->nVert, 3, sizeof(float));
625  voronoi->faces = (int**)malloc1d(voronoi->nFaces*sizeof(int*));
626  voronoi->nPointsPerFace = malloc1d(voronoi->nFaces*sizeof(int));
627 
628  /* Calculate the voronoi vertices for each triangle, which for the unit
629  * sphere are given by the unit normal vector of the triangle */
630  for(n = 0; n<nFaces; n++){
631  r_12[0] = vertices[faces[n*3+1]*3] - vertices[faces[n*3]*3];
632  r_12[1] = vertices[faces[n*3+1]*3+1] - vertices[faces[n*3]*3+1];
633  r_12[2] = vertices[faces[n*3+1]*3+2] - vertices[faces[n*3]*3+2];
634  r_13[0] = vertices[faces[n*3+2]*3] - vertices[faces[n*3]*3];
635  r_13[1] = vertices[faces[n*3+2]*3+1] - vertices[faces[n*3]*3+1];
636  r_13[2] = vertices[faces[n*3+2]*3+2] - vertices[faces[n*3]*3+2];
637  crossProduct3(r_12, r_13, r_normal);
638 
639  norm = 1.0f/L2_norm3(r_normal);
640  utility_svsmul(r_normal, &norm, 3, r_normal);
641  memcpy(voronoi->vert[n], r_normal, 3*sizeof(float));
642  }
643 
644  /* Find duplicate vertices if any, due to two triangles sharing the same
645  * circumscribed circle */
646  duplicates = calloc1d(voronoi->nVert, sizeof(int));
647  for(n = 0; n<voronoi->nVert; n++){
648  if (duplicates[n] == 0 ){
649  for (m = 0; m<voronoi->nVert; m++){
650  if (n != m){
651  if (fabsf(voronoi->vert[n][0] - voronoi->vert[m][0]) < 1.0e-5f &&
652  fabsf(voronoi->vert[n][1] - voronoi->vert[m][1]) < 1.0e-5f &&
653  fabsf(voronoi->vert[n][2] - voronoi->vert[m][2]) < 1.0e-5f ){
654  duplicates[m] = n;
655  }
656  }
657  }
658  }
659  }
660 
661  int NOT_SORTED;
662  int i, j, k, l, nFaceIdx, currentfaceIdx, currentvertIdx, currentvert, nSorted;
663  int currentface[3];
664  int* faceIdx, *sorted, *tempfacelist;
665  faceIdx = NULL;
666  sorted = NULL;
667  tempfacelist = NULL;
668 
669  /* Calculate the voronoi polygons
670  *
671  * find the an ordered sequence of the triangles around each vertex and get
672  * the proper sequence of the voronoi vertices that constitute a polygon */
673  for (n = 0; n<voronoi->nFaces; n++){
674  nFaceIdx=0;
675  for(m=0; m<voronoi->nVert; m++)
676  if(faces[m*3+0]==n || faces[m*3+1]==n || faces[m*3+2]==n)
677  nFaceIdx++;
678  faceIdx = realloc1d(faceIdx, nFaceIdx*sizeof(int));
679 
680  /* list of triangles that contain this specific vertex */
681  i=0;
682  for(m=0; m<voronoi->nVert; m++){
683  if(faces[m*3+0]==n || faces[m*3+1]==n || faces[m*3+2]==n){
684  faceIdx[i] = m;
685  i++;
686  }
687  }
688 
689  /* Each triangle from the list contain the common vertex and two other
690  * vertices - each triangle has two common vertices with each other. One
691  * (brute) way of sorting the sequence is to pick one triangle, find the
692  * neighbour triangle by finding their common
693  * vertex, move to that triangle and iterate till all the number of
694  * triangles have been checked. */
695  k = 0;
696  currentfaceIdx = faceIdx[k]; /* pick-up one of the triangles in the list */
697  memcpy(currentface, &faces[currentfaceIdx*3], 3*sizeof(int)); /* the triangle's vertex indices */
698 
699  /* pick-up one of the vertices that is not the central one */
700  currentvertIdx = -1;
701  for(j=0; j<3; j++){
702  if(currentface[j] != n){
703  currentvertIdx = j;
704  break;
705  }
706  }
707  saf_assert(currentvertIdx!=-1, "Ugly error");
708  currentvert = currentface[currentvertIdx];
709 
710  /* Prep */
711  nSorted = 1;
712  sorted = realloc1d(sorted, nSorted*sizeof(int));
713  sorted[0] = faceIdx[k]; /* this is the list that keeps the the ordered triangles */
714  NOT_SORTED = 1;
715  while(NOT_SORTED){
716  /* exclude the current triangle from the temporary list */
717  tempfacelist = realloc1d(tempfacelist, (nFaceIdx-1)*sizeof(int));
718  l=0;
719  for(i=0; i<nFaceIdx; i++){
720  if(faceIdx[i]!=currentfaceIdx){
721  tempfacelist[l] = faceIdx[i];
722  l++;
723  }
724  }
725  saf_assert(l==nFaceIdx-1, "Ugly error");
726 
727  for (l = 0; l<nFaceIdx-1; l++){
728  currentfaceIdx = tempfacelist[l];
729  memcpy(currentface, &faces[currentfaceIdx*3], 3*sizeof(int));
730 
731  /* if the currentvert exists in the current triangles vertices... */
732  if (currentface[0] == currentvert || currentface[1] == currentvert || currentface[2] == currentvert){
733  /* then it's the neighbour triangle, so store its index */
734  nSorted++;
735  sorted = realloc1d(sorted, nSorted*sizeof(int));
736  sorted[nSorted-1] = currentfaceIdx;
737 
738  /* if the sorted list has the length of faceIdx, then we're done */
739  if (nSorted == nFaceIdx){
740  NOT_SORTED = 0;
741  break;
742  }
743 
744  /* find the next vertex from current triangle that excludes the central one and the previous one */
745  for(j=0; j<3; j++)
746  if(currentface[j] != n && currentface[j] != currentvert)
747  currentvertIdx = j;
748  currentvert = currentface[currentvertIdx];
749  break;
750  }
751  }
752  }
753 
754  /* remove the duplicate vertices from the list */
755  for (i=0; i<nSorted; i++)
756  if (duplicates[sorted[i]] != 0)
757  sorted[i] = duplicates[sorted[i]];
758 
759 
760  /* Identify unique IDs, and sort them in accending order */
761  int nUnique;
762  int* uniqueIDs;
763  uniqueIDs = NULL;
764  unique_i(sorted, nSorted, NULL, &uniqueIDs, &nUnique);
765  sorti(uniqueIDs, uniqueIDs, NULL, nUnique, 0);
766 
767  /* Save */
768  voronoi->faces[n] = malloc1d(nUnique*sizeof(int));
769  for(i=0; i<nUnique; i++)
770  voronoi->faces[n][i] = sorted[uniqueIDs[i]];
771  voronoi->nPointsPerFace[n] = nUnique;
772 
773  /* clean-up */
774  free(uniqueIDs);
775  }
776 
777  /* clean-up */
778  free(faceIdx);
779  free(sorted);
780  free(tempfacelist);
781 }

783 void sphVoronoiAreas
784 (
785  voronoi_data* voronoi,
786  float* areas
787 )
788 {
789  int i, m, n, N_poly, tmp_i;
790  int* face;
791  float tmp, r_21_norm, r_23_norm;
792  float *theta;
793  float r_01[3], r_02[3], r_2x1[3], r_21[3], r_03[3], r_2x3[3], r_23[3];
794 
795  face = NULL;
796  theta = NULL;
797  for(m=0; m<voronoi->nFaces; m++){
798  N_poly = voronoi->nPointsPerFace[m]; /* number of vertices in the polygon */
799  face = realloc1d(face, N_poly*sizeof(int));
800  theta = realloc1d(theta, N_poly*sizeof(float));
801  memcpy(face, voronoi->faces[m], N_poly*sizeof(int)); /* current face */
802 
803  for(n=0; n<N_poly; n++){
804  /* find vector between vertex origin and vertices 1 & 2 */
805  memcpy(r_01, voronoi->vert[face[0]], 3*sizeof(float));
806  memcpy(r_02, voronoi->vert[face[1]], 3*sizeof(float));
807 
808  /* find normal vector to the great circle of 1 & 2 */
809  crossProduct3(r_02, r_01, r_2x1);
810 
811  /* find tangent vector to great circle at 2 */
812  crossProduct3(r_2x1, r_02, r_21);
813 
814  /* find vector between vertex origin and vertex 3 */
815  memcpy(r_03, voronoi->vert[face[2]], 3*sizeof(float));
816 
817  /* find normal vector to the great circle of 2 & 3 */
818  crossProduct3(r_02, r_03, r_2x3);
819 
820  /* find tangent vector to great circle at 2 */
821  crossProduct3(r_2x3, r_02, r_23);
822 
823  /* normalise tangent vectors */
824  r_21_norm = 1.0f/L2_norm3(r_21);
825  utility_svsmul(r_21, &r_21_norm, 3, r_21);
826  r_23_norm = 1.0f/L2_norm3(r_23);
827  utility_svsmul(r_23, &r_23_norm, 3, r_23);
828 
829  /* get angle between the normals */
830  utility_svvdot(r_21, r_23, 3, &tmp);
831  theta[n] = acosf(tmp);
832 
833  /* shift the vertex list by one position and repeat */
834  tmp_i = face[0];
835  for(i=1; i<N_poly; i++)
836  face[i-1] = face[i];
837  face[N_poly-1] = tmp_i;
838  }
839 
840  /* compute area of spherical polygon by spherical excess */
841  tmp = 0.0f;
842  for(i=0; i<N_poly; i++)
843  tmp += theta[i];
844  areas[m] = tmp - ((float)N_poly-2.0f)*SAF_PI;
845  }
846 }
