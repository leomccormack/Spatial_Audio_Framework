85 void quaternion2rotationMatrix
86 (
87  quaternion_data* Q,
88  float R[3][3]
89 )
90 {
91  R[0][0] = 2.0f * (Q->w * Q->w + Q->z * Q->z) - 1.0f;
92  R[0][1] = 2.0f * (Q->z * Q->y - Q->w * Q->x);
93  R[0][2] = 2.0f * (Q->z * Q->x + Q->w * Q->y);
94  R[1][0] = 2.0f * (Q->z * Q->y + Q->w * Q->x);
95  R[1][1] = 2.0f * (Q->w * Q->w + Q->y * Q->y) - 1.0f;
96  R[1][2] = 2.0f * (Q->y * Q->x - Q->w * Q->z);
97  R[2][0] = 2.0f * (Q->z * Q->x - Q->w * Q->y);
98  R[2][1] = 2.0f * (Q->y * Q->x + Q->w * Q->z);
99  R[2][2] = 2.0f * (Q->w * Q->w + Q->x * Q->x) - 1.0f;
100 }

102 /* Adapted from: https://www.euclideanspace.com/maths/geometry/rotations/conversions/matrixToQuaternion/forum.htm */
103 void rotationMatrix2quaternion
104 (
105  float R[3][3],
106  quaternion_data* Q
107 )
108 {
109  Q->w = sqrtf( MAX( 0.0f, 1.0f + R[0][0] + R[1][1] + R[2][2] ) ) / 2.0f;
110  Q->z = sqrtf( MAX( 0.0f, 1.0f + R[0][0] - R[1][1] - R[2][2] ) ) / 2.0f;
111  Q->y = sqrtf( MAX( 0.0f, 1.0f - R[0][0] + R[1][1] - R[2][2] ) ) / 2.0f;
112  Q->x = sqrtf( MAX( 0.0f, 1.0f - R[0][0] - R[1][1] + R[2][2] ) ) / 2.0f;
113  Q->z = copysignf( Q->z, R[2][1] - R[1][2] );
114  Q->y = copysignf( Q->y, R[0][2] - R[2][0] );
115  Q->x = copysignf( Q->x, R[1][0] - R[0][1] );
116 }

118 /* Adapted from (ISC License): https://github.com/MartinWeigel/Quaternion */
119 void euler2Quaternion
120 (
121  float alpha,
122  float beta,
123  float gamma,
124  int degreesFlag,
125  EULER_ROTATION_CONVENTIONS convention,
126  quaternion_data* Q
127 )
128 {
129  float cy, sy, cr, sr, cp, sp;
130 
131  cy = sy = cr = sr = cp = sp = 0.0f;
132  switch(convention){
133  case EULER_ROTATION_Y_CONVENTION: /* fall through*/
134  case EULER_ROTATION_X_CONVENTION: assert(0); /* Not supported */; return;
135  case EULER_ROTATION_YAW_PITCH_ROLL:
136  cy = cosf((degreesFlag ? alpha*SAF_PI/180.0f : alpha) * 0.5f); /* x */
137  sy = sinf((degreesFlag ? alpha*SAF_PI/180.0f : alpha) * 0.5f); /* x */
138  cp = cosf((degreesFlag ? beta*SAF_PI/180.0f : beta) * 0.5f); /* y */
139  sp = sinf((degreesFlag ? beta*SAF_PI/180.0f : beta) * 0.5f); /* y */
140  cr = cosf((degreesFlag ? gamma*SAF_PI/180.0f : gamma) * 0.5f); /* z */
141  sr = sinf((degreesFlag ? gamma*SAF_PI/180.0f : gamma) * 0.5f); /* z */
142  break;
143  case EULER_ROTATION_ROLL_PITCH_YAW:
144  cy = cosf((degreesFlag ? gamma*SAF_PI/180.0f : gamma) * 0.5f); /* x */
145  sy = sinf((degreesFlag ? gamma*SAF_PI/180.0f : gamma) * 0.5f); /* x */
146  cp = cosf((degreesFlag ? beta*SAF_PI/180.0f : beta) * 0.5f); /* y */
147  sp = sinf((degreesFlag ? beta*SAF_PI/180.0f : beta) * 0.5f); /* y */
148  cr = cosf((degreesFlag ? alpha*SAF_PI/180.0f : alpha) * 0.5f); /* z */
149  sr = sinf((degreesFlag ? alpha*SAF_PI/180.0f : alpha) * 0.5f); /* z */
150  break;
151  }
152  Q->w = cy * cr * cp + sy * sr * sp;
153  Q->x = cy * sr * cp - sy * cr * sp;
154  Q->y = cy * cr * sp + sy * sr * cp;
155  Q->z = sy * cr * cp - cy * sr * sp;
156 }

158 /* Adapted from (ISC License): https://github.com/MartinWeigel/Quaternion */
159 void quaternion2euler
160 (
161  quaternion_data* Q,
162  int degreesFlag,
163  EULER_ROTATION_CONVENTIONS convention,
164  float* alpha,
165  float* beta,
166  float* gamma
167 )
168 {
169  float sinr_cosp, cosr_cosp, sinp, siny_cosp, cosy_cosp;
170 
171  sinr_cosp = 2.0f * (Q->w * Q->x + Q->y * Q->z);
172  cosr_cosp = 1.0f - 2.0f * (Q->x * Q->x + Q->y * Q->y);
173  sinp = 2.0f * (Q->w * Q->y - Q->z * Q->x);
174  siny_cosp = 2.0f * (Q->w * Q->z + Q->x * Q->y);
175  cosy_cosp = 1.0f - 2.0f * (Q->y * Q->y + Q->z * Q->z);
176  switch(convention){
177  case EULER_ROTATION_Y_CONVENTION: assert(0) /* Not supported */; break;
178  case EULER_ROTATION_X_CONVENTION: assert(0) /* Not supported */; break;
179  case EULER_ROTATION_YAW_PITCH_ROLL:
180  /* Yaw (z-axis rotation) */
181  (*gamma) = atan2f(sinr_cosp, cosr_cosp);
182  /* Pitch (y-axis rotation) */
183  if (fabsf(sinp) >= 1.0f)
184  (*beta) = copysignf(SAF_PI / 2.0f, sinp); /* use 90 degrees if out of range */
185  else
186  (*beta) = asinf(sinp);
187  /* Roll (x-axis rotation) */
188  (*alpha) = atan2f(siny_cosp, cosy_cosp);
189  break;
190  case EULER_ROTATION_ROLL_PITCH_YAW:
191  /* Roll (x-axis rotation) */
192  (*alpha) = atan2f(sinr_cosp, cosr_cosp);
193  /* Pitch (y-axis rotation) */
194  if (fabs(sinp) >= 1.0f)
195  (*beta) = copysignf(SAF_PI / 2.0f, sinp); /* use 90 degrees if out of range */
196  else
197  (*beta) = asinf(sinp);
198  /* Yaw (z-axis rotation) */
199  (*gamma) = atan2f(siny_cosp, cosy_cosp);
200  break;
201  }
202  if(degreesFlag){
203  (*alpha) *= 180.0f/SAF_PI;
204  (*beta) *= 180.0f/SAF_PI;
205  (*gamma) *= 180.0f/SAF_PI;
206  }
207 }

209 void euler2rotationMatrix
210 (
211  float alpha,
212  float beta,
213  float gamma,
214  int degreesFlag,
215  EULER_ROTATION_CONVENTIONS convention,
216  float R[3][3]
217 )
218 {
219  float R1[3][3], R2[3][3], R3[3][3], Rtmp[3][3];
220 
221  switch(convention){
222  case EULER_ROTATION_Y_CONVENTION:
223  getRz(degreesFlag ? alpha*SAF_PI/180.0f : alpha, R1);
224  getRy(degreesFlag ? beta*SAF_PI/180.0f : beta, R2);
225  getRz(degreesFlag ? gamma*SAF_PI/180.0f : gamma, R3);
226  break;
227  case EULER_ROTATION_X_CONVENTION:
228  getRz(degreesFlag ? alpha*SAF_PI/180.0f : alpha, R1);
229  getRx(degreesFlag ? beta*SAF_PI/180.0f : beta, R2);
230  getRz(degreesFlag ? gamma*SAF_PI/180.0f : gamma, R3);
231  break;
232  case EULER_ROTATION_YAW_PITCH_ROLL:
233  getRz(degreesFlag ? alpha*SAF_PI/180.0f : alpha, R1);
234  getRy(degreesFlag ? beta*SAF_PI/180.0f : beta, R2);
235  getRx(degreesFlag ? gamma*SAF_PI/180.0f : gamma, R3);
236  break;
237  case EULER_ROTATION_ROLL_PITCH_YAW:
238  getRx(degreesFlag ? alpha*SAF_PI/180.0f : alpha, R1);
239  getRy(degreesFlag ? beta*SAF_PI/180.0f : beta, R2);
240  getRz(degreesFlag ? gamma*SAF_PI/180.0f : gamma, R3);
241  break;
242  }
243  cblas_sgemm(CblasRowMajor, CblasNoTrans, CblasNoTrans, 3, 3, 3, 1.0f,
244  (float*)R2, 3,
245  (float*)R1, 3, 0.0f,
246  (float*)Rtmp, 3);
247  cblas_sgemm(CblasRowMajor, CblasNoTrans, CblasNoTrans, 3, 3, 3, 1.0f,
248  (float*)R3, 3,
249  (float*)Rtmp, 3, 0.0f,
250  (float*)R, 3);
251 }

269 (
270  float* dirs,
271  int nDirs,
272  int anglesInDegreesFLAG,
273  float* dirs_xyz
274 )
275 {
276  int i;
277  float tmp_rad[2];
278

291  dirs_xyz[i*3+1] = cosf(dirs[i*2+1]) * sinf(dirs[i*2]);
292  dirs_xyz[i*3+2] = sinf(dirs[i*2+1]);

404 (
405  const float* dirs_deg,
406  const int nDirs,
407  int** faces,
408  int* nFaces,
409  float* vertices
410 )

437 void sphVoronoi
438 (
439  int* faces,
440  int nFaces,
441  float* vertices,
442  int nDirs,
443  voronoi_data* voronoi
444 )
445 {
446  int n, m;
447  int* duplicates;
448  float r_12[3], r_13[3], r_normal[3];
449  float norm;
450 
451  /* prep */
452  voronoi->nVert = nFaces;
453  voronoi->nFaces = nDirs;
454  voronoi->vert = (float**)malloc2d(voronoi->nVert, 3, sizeof(float));
455  voronoi->faces = (int**)malloc1d(voronoi->nFaces*sizeof(int*));
456  voronoi->nPointsPerFace = malloc1d(voronoi->nFaces*sizeof(int));
457 
458  /* Calculate the voronoi vertices for each triangle, which for the unit
459  * sphere are given by the unit normal vector of the triangle */
460  for(n = 0; n<nFaces; n++){
461  r_12[0] = vertices[faces[n*3+1]*3] - vertices[faces[n*3]*3];
462  r_12[1] = vertices[faces[n*3+1]*3+1] - vertices[faces[n*3]*3+1];
463  r_12[2] = vertices[faces[n*3+1]*3+2] - vertices[faces[n*3]*3+2];
464  r_13[0] = vertices[faces[n*3+2]*3] - vertices[faces[n*3]*3];
465  r_13[1] = vertices[faces[n*3+2]*3+1] - vertices[faces[n*3]*3+1];
466  r_13[2] = vertices[faces[n*3+2]*3+2] - vertices[faces[n*3]*3+2];
467  crossProduct3(r_12, r_13, r_normal);
468 
469  norm = 1.0f/L2_norm3(r_normal);
470  utility_svsmul(r_normal, &norm, 3, r_normal);
471  memcpy(voronoi->vert[n], r_normal, 3*sizeof(float));
472  }
473 
474  /* Find duplicate vertices if any, due to two triangles sharing the same
475  * circumscribed circle */
476  duplicates = calloc1d(voronoi->nVert, sizeof(int));
477  for(n = 0; n<voronoi->nVert; n++){
478  if (duplicates[n] == 0 ){
479  for (m = 0; m<voronoi->nVert; m++){
480  if (n != m){
481  if (fabsf(voronoi->vert[n][0] - voronoi->vert[m][0]) < 1.0e-5f &&
482  fabsf(voronoi->vert[n][1] - voronoi->vert[m][1]) < 1.0e-5f &&
483  fabsf(voronoi->vert[n][2] - voronoi->vert[m][2]) < 1.0e-5f ){
484  duplicates[m] = n;
485  }
486  }
487  }
488  }
489  }
490 
491  int NOT_SORTED;
492  int i, j, k, l, nFaceIdx, currentfaceIdx, currentvertIdx, currentvert, nSorted;
493  int currentface[3];
494  int* faceIdx, *sorted, *tempfacelist;
495  faceIdx = NULL;
496  sorted = NULL;
497  tempfacelist = NULL;
498 
499  /* Calculate the voronoi polygons
500  *
501  * find the an ordered sequence of the triangles around each vertex and get
502  * the proper sequence of the voronoi vertices that constitute a polygon */
503  for (n = 0; n<voronoi->nFaces; n++){
504  nFaceIdx=0;
505  for(m=0; m<voronoi->nVert; m++)
506  if(faces[m*3+0]==n || faces[m*3+1]==n || faces[m*3+2]==n)
507  nFaceIdx++;
508  faceIdx = realloc1d(faceIdx, nFaceIdx*sizeof(int));
509 
510  /* list of triangles that contain this specific vertex */
511  i=0;
512  for(m=0; m<voronoi->nVert; m++){
513  if(faces[m*3+0]==n || faces[m*3+1]==n || faces[m*3+2]==n){
514  faceIdx[i] = m;
515  i++;
516  }
517  }
518 
519  /* Each triangle from the list contain the common vertex and two other
520  * vertices - each triangle has two common vertices with each other. One
521  * (brute) way of sorting the sequence is to pick one triangle, find the
522  * neighbour triangle by finding their common
523  * vertex, move to that triangle and iterate till all the number of
524  * triangles have been checked. */
525  k = 0;
526  currentfaceIdx = faceIdx[k]; /* pick-up one of the triangles in the list */
527  memcpy(currentface, &faces[currentfaceIdx*3], 3*sizeof(int)); /* the triangle's vertex indices */
528 
529  /* pick-up one of the vertices that is not the central one */
530  currentvertIdx = -1;
531  for(j=0; j<3; j++){
532  if(currentface[j] != n){
533  currentvertIdx = j;
534  break;
535  }
536  }
537  assert(currentvertIdx!=-1);
538  currentvert = currentface[currentvertIdx];
539 
540  /* Prep */
541  nSorted = 1;
542  sorted = realloc1d(sorted, nSorted*sizeof(int));
543  sorted[0] = faceIdx[k]; /* this is the list that keeps the the ordered triangles */
544  NOT_SORTED = 1;
545  while(NOT_SORTED){
546  /* exclude the current triangle from the temporary list */
547  tempfacelist = realloc1d(tempfacelist, (nFaceIdx-1)*sizeof(int));
548  l=0;
549  for(i=0; i<nFaceIdx; i++){
550  if(faceIdx[i]!=currentfaceIdx){
551  tempfacelist[l] = faceIdx[i];
552  l++;
553  }
554  }
555  assert(l==nFaceIdx-1);
556 
557  for (l = 0; l<nFaceIdx-1; l++){
558  currentfaceIdx = tempfacelist[l];
559  memcpy(currentface, &faces[currentfaceIdx*3], 3*sizeof(int));
560 
561  /* if the currentvert exists in the current triangles vertices... */
562  if (currentface[0] == currentvert || currentface[1] == currentvert || currentface[2] == currentvert){
563  /* then it's the neighbour triangle, so store its index */
564  nSorted++;
565  sorted = realloc1d(sorted, nSorted*sizeof(int));
566  sorted[nSorted-1] = currentfaceIdx;
567 
568  /* if the sorted list has the length of faceIdx, then we're done */
569  if (nSorted == nFaceIdx){
570  NOT_SORTED = 0;
571  break;
572  }
573 
574  /* find the next vertex from current triangle that excludes the central one and the previous one */
575  for(j=0; j<3; j++)
576  if(currentface[j] != n && currentface[j] != currentvert)
577  currentvertIdx = j;
578  currentvert = currentface[currentvertIdx];
579  break;
580  }
581  }
582  }
583 
584  /* remove the duplicate vertices from the list */
585  for (i=0; i<nSorted; i++)
586  if (duplicates[sorted[i]] != 0)
587  sorted[i] = duplicates[sorted[i]];
588 
589 
590  /* Identify unique IDs, and sort them in accending order */
591  int nUnique;
592  int* uniqueIDs;
593  uniqueIDs = NULL;
594  unique_i(sorted, nSorted, NULL, &uniqueIDs, &nUnique);
595  sorti(uniqueIDs, uniqueIDs, NULL, nUnique, 0);
596 
597  /* Save */
598  voronoi->faces[n] = malloc1d(nUnique*sizeof(int));
599  for(i=0; i<nUnique; i++)
600  voronoi->faces[n][i] = sorted[uniqueIDs[i]];
601  voronoi->nPointsPerFace[n] = nUnique;
602 
603  /* clean-up */
604  free(uniqueIDs);
605  }
606 
607  /* clean-up */
608  free(faceIdx);
609  free(sorted);
610  free(tempfacelist);
611 }

613 void sphVoronoiAreas
614 (
615  voronoi_data* voronoi,
616  float* areas
617 )
618 {
619  int i, m, n, N_poly, tmp_i;
620  int* face;
621  float tmp, r_21_norm, r_23_norm;
622  float *theta;
623  float r_01[3], r_02[3], r_2x1[3], r_21[3], r_03[3], r_2x3[3], r_23[3];
624 
625  face = NULL;
626  theta = NULL;
627  for(m=0; m<voronoi->nFaces; m++){
628  N_poly = voronoi->nPointsPerFace[m]; /* number of vertices in the polygon */
629  face = realloc1d(face, N_poly*sizeof(int));
630  theta = realloc1d(theta, N_poly*sizeof(float));
631  memcpy(face, voronoi->faces[m], N_poly*sizeof(int)); /* current face */
632 
633  for(n=0; n<N_poly; n++){
634  /* find vector between vertex origin and vertices 1 & 2 */
635  memcpy(r_01, voronoi->vert[face[0]], 3*sizeof(float));
636  memcpy(r_02, voronoi->vert[face[1]], 3*sizeof(float));
637 
638  /* find normal vector to the great circle of 1 & 2 */
639  crossProduct3(r_02, r_01, r_2x1);
640 
641  /* find tangent vector to great circle at 2 */
642  crossProduct3(r_2x1, r_02, r_21);
643 
644  /* find vector between vertex origin and vertex 3 */
645  memcpy(r_03, voronoi->vert[face[2]], 3*sizeof(float));
646 
647  /* find normal vector to the great circle of 2 & 3 */
648  crossProduct3(r_02, r_03, r_2x3);
649 
650  /* find tangent vector to great circle at 2 */
651  crossProduct3(r_2x3, r_02, r_23);
652 
653  /* normalise tangent vectors */
654  r_21_norm = 1.0f/L2_norm3(r_21);
655  utility_svsmul(r_21, &r_21_norm, 3, r_21);
656  r_23_norm = 1.0f/L2_norm3(r_23);
657  utility_svsmul(r_23, &r_23_norm, 3, r_23);
658 
659  /* get angle between the normals */
660  utility_svvdot(r_21, r_23, 3, &tmp);
661  theta[n] = acosf(tmp);
662 
663  /* shift the vertex list by one position and repeat */
664  tmp_i = face[0];
665  for(i=1; i<N_poly; i++)
666  face[i-1] = face[i];
667  face[N_poly-1] = tmp_i;
668  }
669 
670  /* compute area of spherical polygon by spherical excess */
671  tmp = 0.0f;
672  for(i=0; i<N_poly; i++)
673  tmp += theta[i];
674  areas[m] = tmp - ((float)N_poly-2.0f)*SAF_PI;
675  }
676 }

679 (
680  float* dirs_deg,
681  int nDirs,
682  int diagFLAG,
683  float* weights
684 )
685 {
686  int i, nFaces;
687  int* faces;

689  voronoi_data voronoi;
690 
691  /* Perform delaunay triangulation */
692  faces = NULL;

695

697  sphVoronoi(faces, nFaces, vertices, nDirs, &voronoi);
698

700  areas = malloc1d(voronoi.nFaces * sizeof(float));
701  sphVoronoiAreas(&voronoi, areas);
702 
703  /* Store weights */
704  if(diagFLAG){
705  /* along the diagonal... */
706  memset(weights, 0, nDirs*nDirs*sizeof(float));
707  for(i=0; i<nDirs; i++)
708  weights[i*nDirs+i] = areas[i];
709  }
710  else
711  memcpy(weights, areas, nDirs*sizeof(float));
712 
713  /* clean-up */
714  free(faces);
715  free(vertices);
716  free(areas);
717  for(i=0; i<voronoi.nFaces; i++)
718  free(voronoi.faces[i]);
719  free(voronoi.vert);
720  free(voronoi.nPointsPerFace);
721 }
