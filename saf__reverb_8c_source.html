35 void ims_shoebox_create
36 (
37  void** phIms,
38  float length,
39  float width,
40  float height,
41  float* abs_wall,
42  float lowestOctaveBand,
43  int nOctBands,
44  float c_ms,
45  float fs
46 )
47 {
48  *phIms = malloc1d(sizeof(ims_scene_data));
49  ims_scene_data *sc = (ims_scene_data*)(*phIms);
50  int i,j,band,wall;
51 
52  /* Shoebox dimensions */
53  sc->room_dims[0] = length;
54  sc->room_dims[1] = width;
55  sc->room_dims[2] = height;
56  sc->c_ms = c_ms;
57 
58  /* Octave band centre frequencies */
59  if(nOctBands>1){
60  sc->nBands = nOctBands;
61  sc->band_centerfreqs = malloc1d(nOctBands*sizeof(float));
62  sc->band_centerfreqs[0] = lowestOctaveBand;
63  for(band=1; band<nOctBands; band++)
64  sc->band_centerfreqs[band] = sc->band_centerfreqs[band-1]*2.0f;
65  sc->band_cutofffreqs = malloc1d((sc->nBands-1)*sizeof(float));
66  getOctaveBandCutoffFreqs(sc->band_centerfreqs, sc->nBands, sc->band_cutofffreqs);
67  }
68  else { /* Broad-band operation */
69  sc->nBands = 1;
70  sc->band_centerfreqs = NULL;
71  sc->band_cutofffreqs = NULL;
72  }
73 
74  /* Samplerate */
75  sc->fs = fs;
76 
77  /* Absorption coeffients per wall and octave band */
78  sc->abs_wall = (float**)malloc2d(sc->nBands, IMS_NUM_WALLS_SHOEBOX, sizeof(float));
79  for(band=0; band<sc->nBands; band++)
80  for(wall=0; wall<IMS_NUM_WALLS_SHOEBOX; wall++)
81  sc->abs_wall[band][wall] = abs_wall[band*IMS_NUM_WALLS_SHOEBOX+wall];
82 
83  /* Default is no sources or receivers in the room */
84  for(i=0; i<IMS_MAX_NUM_SOURCES; i++)
85  sc->srcs[i].ID = -1; /* -1 indicates not in use */
86  for(i=0; i<IMS_MAX_NUM_RECEIVERS; i++)
87  sc->recs[i].ID = -1;
88  sc->nSources = 0;
89  sc->nReceivers = 0;
90 
91  /* ims_core_workspace per source / receiver combination */
92  sc->hCoreWrkSpc = (voidPtr**)malloc2d(IMS_MAX_NUM_RECEIVERS, IMS_MAX_NUM_SOURCES, sizeof(voidPtr));
93  for(i=0; i<IMS_MAX_NUM_RECEIVERS; i++)
94  for(j=0; j<IMS_MAX_NUM_SOURCES; j++)
95  sc->hCoreWrkSpc[i][j] = NULL;
96 
97  /* FIR Fiterbank */
98  sc->H_filt = NULL;
99 
100  /* RIRs per source / receiver combination */
101  sc->rirs = (ims_rir**)malloc2d(IMS_MAX_NUM_RECEIVERS, IMS_MAX_NUM_SOURCES, sizeof(ims_rir));
102  for(i=0; i<IMS_MAX_NUM_RECEIVERS; i++){
103  for(j=0; j<IMS_MAX_NUM_SOURCES; j++){
104  sc->rirs[i][j].data = NULL;
105  sc->rirs[i][j].length = sc->rirs[i][j].nChannels = 0;
106  }
107  }
108 
109  /* Circular buffers (only used/allocated when applyEchogramTD() function is called for the first time) */
110  sc->wIdx[0] = 0;
111  sc->wIdx[1] = 0;
112  sc->circ_buffer[0] = NULL;
113  sc->circ_buffer[1] = NULL;
114 
115  /* IIR Filterbank per source (only used/allocated when applyEchogramTD() function is called for the first time) */
116  sc->hFaFbank = malloc1d(IMS_MAX_NUM_SOURCES*sizeof(voidPtr));
117  sc->src_sigs_bands = malloc1d(IMS_MAX_NUM_SOURCES*sizeof(float**));
118  for(j=0; j<IMS_MAX_NUM_SOURCES; j++){
119  sc->hFaFbank[j] = NULL;
120  sc->src_sigs_bands[j] = NULL;
121  }
122 
123  /* Temp buffers for cross-fading (only used/allocated when applyEchogramTD() function is called for the first time) */
124  sc->rec_sig_tmp = malloc1d(IMS_MAX_NUM_RECEIVERS*sizeof(float**));
125  for(j=0; j<IMS_MAX_NUM_RECEIVERS; j++)
126  sc->rec_sig_tmp[j] = NULL;
127  memset(sc->applyCrossFadeFLAG, 0, IMS_MAX_NUM_RECEIVERS*IMS_MAX_NUM_SOURCES*sizeof(int));
128  sc->interpolator_fIn = NULL;
129  sc->interpolator_fOut = NULL;
130  sc->framesize = -1;
131 
132  /* Lagrange interpolator look-up table */
133  for(i=0; i<IMS_LAGRANGE_LOOKUP_TABLE_SIZE; i++)
134  sc->lookup_fractions[i] = 1.0f/IMS_LAGRANGE_LOOKUP_TABLE_SIZE * (float)i;
135  lagrangeWeights(IMS_LAGRANGE_ORDER, sc->lookup_fractions, IMS_LAGRANGE_LOOKUP_TABLE_SIZE, (float*)sc->lookup_H_frac);
136 }

138 void ims_shoebox_destroy
139 (
140  void** phIms
141 )
142 {
143  ims_scene_data *sc = (ims_scene_data*)(*phIms);
144  int i,j;
145 
146  if(sc!=NULL){
147  free(sc->band_centerfreqs);
148  free(sc->band_cutofffreqs);
149  free(sc->abs_wall);
150  for(i=0; i<IMS_MAX_NUM_RECEIVERS; i++)
151  for(j=0; j<IMS_MAX_NUM_SOURCES; j++)
152  ims_shoebox_coreWorkspaceDestroy(&(sc->hCoreWrkSpc[i][j]));
153  free(sc->hCoreWrkSpc);
154  free(sc->H_filt);
155  for(i=0; i<IMS_MAX_NUM_RECEIVERS; i++)
156  for(j=0; j<IMS_MAX_NUM_SOURCES; j++)
157  free(sc->rirs[i][j].data);
158  free(sc->rirs);
159  free(sc->circ_buffer[0]);
160  free(sc->circ_buffer[1]);
161  for(j=0; j<IMS_MAX_NUM_SOURCES; j++){
162  faf_IIRFilterbank_destroy(&(sc->hFaFbank[j]));
163  free(sc->src_sigs_bands[j]);
164  }
165  free(sc->hFaFbank);
166  free(sc->src_sigs_bands);
167  for(i=0; i<IMS_MAX_NUM_RECEIVERS; i++)
168  free(sc->rec_sig_tmp[i]);
169  free(sc->rec_sig_tmp);
170  free(sc->interpolator_fIn);
171  free(sc->interpolator_fOut);
172  free(sc);
173  sc=NULL;
174  *phIms = NULL;
175  }

178 void ims_shoebox_computeEchograms
179 (
180  void* hIms,
181  float maxTime_ms
182 )
183 {
184  ims_scene_data *sc = (ims_scene_data*)(hIms);
185  ims_core_workspace* workspace;
186  ims_pos_xyz src2, rec2;
187  int src_idx, rec_idx, band;
188 
189  /* Compute echograms for active source/receiver combinations */
190  for(rec_idx = 0; rec_idx < IMS_MAX_NUM_RECEIVERS; rec_idx++){
191  for(src_idx = 0; src_idx < IMS_MAX_NUM_SOURCES; src_idx++){
192  if( (sc->srcs[src_idx].ID != -1) && (sc->recs[rec_idx].ID != -1) ){
193  /* Change y coord for Receiver and Source to match convention
194  * used inside the coreInit function */
195  rec2.x = sc->recs[rec_idx].pos.x;
196  rec2.y = sc->room_dims[1] - sc->recs[rec_idx].pos.y;
197  rec2.z = sc->recs[rec_idx].pos.z;
198  src2.x = sc->srcs[src_idx].pos.x;
199  src2.y = sc->room_dims[1] - sc->srcs[src_idx].pos.y;
200  src2.z = sc->srcs[src_idx].pos.z;
201 
202  /* Workspace handle for this source/receiver combination */
203  workspace = sc->hCoreWrkSpc[rec_idx][src_idx];
204 
205  /* Copy previous echograms */
206  for(band=0; band<workspace->nBands; band++)
207  ims_shoebox_echogramCopy(workspace->hEchogram_abs[band], workspace->hPrevEchogram_abs[band]);
208 
209  /* Only update if it is required */
210  if(workspace->refreshEchogramFLAG){
211  /* Compute echogram due to pure propagation (frequency-independent, omni-directional) */
212  ims_shoebox_coreInit(workspace, sc->room_dims, src2, rec2, maxTime_ms, sc->c_ms);
213 
214  /* Apply receiver directivities */
215  switch(sc->recs[rec_idx].type){
216  case RECEIVER_SH:
217  ims_shoebox_coreRecModuleSH(workspace, NSH2ORDER(sc->recs[rec_idx].nChannels));
218  break;
219  }
220 
221  /* Apply boundary absorption per frequency band */
222  ims_shoebox_coreAbsorptionModule(workspace, sc->abs_wall);
223 
224  /* Indicate that the echogram is now up to date, and that the RIR should be updated */
225  workspace->refreshEchogramFLAG = 0;
226  workspace->refreshRIRFLAG = 1;
227 
228  /* Also indicate that applyTD() should cross-fade the next frame to void clicks */
229  sc->applyCrossFadeFLAG[src_idx][rec_idx] = 1;
230  }
231  }
232  }
233  }

236 void ims_shoebox_renderRIRs
237 (
238  void* hIms,
239  int fractionalDelayFLAG
240 )
241 {
242  ims_scene_data *sc = (ims_scene_data*)(hIms);
243  ims_core_workspace* wrk;
244  int src_idx, rec_idx;
245 
246  /* Compute FIR Filterbank coefficients (if this is the first time this
247  * function is being called) */
248  if(sc->H_filt==NULL){
249  sc->H_filt = (float**)realloc2d((void**)sc->H_filt, sc->nBands, (IMS_FIR_FILTERBANK_ORDER+1), sizeof(float));
250  FIRFilterbank(IMS_FIR_FILTERBANK_ORDER, sc->band_cutofffreqs, sc->nBands-1,
251  sc->fs, WINDOWING_FUNCTION_HAMMING, 1, FLATTEN2D(sc->H_filt));
252  }
253 
254  /* Render RIRs for all active source/receiver combinations */
255  for(rec_idx = 0; rec_idx < IMS_MAX_NUM_RECEIVERS; rec_idx++){
256  for(src_idx = 0; src_idx < IMS_MAX_NUM_SOURCES; src_idx++){
257  if( (sc->srcs[src_idx].ID!=-1) && (sc->recs[rec_idx].ID!=-1) ){
258 
259  /* Workspace handle for this source/receiver combination */
260  wrk = sc->hCoreWrkSpc[rec_idx][src_idx];
261 
262  /* Only update if it is required */
263  if(wrk->refreshRIRFLAG){
264  /* Render the RIRs for each band */
265  ims_shoebox_renderRIR(wrk, fractionalDelayFLAG, sc->fs, sc->H_filt, &(sc->rirs[rec_idx][src_idx]));
266 
267  wrk->refreshRIRFLAG = 0;
268  }
269  }
270  }
271  }

274 void ims_shoebox_applyEchogramTD
275 (
276  void* hIms,
277  long receiverID,
278  int nSamples,
279  int fractionalDelaysFLAG
280 )
281 {
282  ims_scene_data *sc = (ims_scene_data*)(hIms);
283  ims_core_workspace* wrk;
284  echogram_data *echogram_abs, *echogram_abs_0;
285  int k, i, n, im, band, ch, rec_idx, src_idx, time_samples, wIdx_n;
286  unsigned long rIdx;
287 
288  assert(fractionalDelaysFLAG==0); /* Not tested yet... */
289  assert(nSamples <= IMS_MAX_NSAMPLES_PER_FRAME);
290 
291  /* Allocate circular buffers (if this is the first time this function is being called) */
292  if(sc->circ_buffer[0] == NULL)
293  sc->circ_buffer[0] = (float***)calloc3d(IMS_MAX_NUM_SOURCES, sc->nBands, IMS_CIRC_BUFFER_LENGTH, sizeof(float));
294  if(sc->circ_buffer[1] == NULL)
295  sc->circ_buffer[1] = (float***)calloc3d(IMS_MAX_NUM_SOURCES, sc->nBands, IMS_CIRC_BUFFER_LENGTH, sizeof(float));
296 
297  /* Also allocate signal buffers and filterbank handles (if this is the first time this function is being called) */
298  for(src_idx = 0; src_idx < IMS_MAX_NUM_SOURCES; src_idx++){
299  /* only allocate if source is active... */
300  if( (sc->srcs[src_idx].ID != -1) && (sc->src_sigs_bands[src_idx] == NULL) ){
301  if(sc->nBands>1){ /* ... and only create the filterbank if there is more than one band: */
302  faf_IIRFilterbank_create(&(sc->hFaFbank[src_idx]), IMS_IIR_FILTERBANK_ORDER, sc->band_cutofffreqs,
303  sc->nBands-1, sc->fs, IMS_MAX_NSAMPLES_PER_FRAME);
304  }
305  sc->src_sigs_bands[src_idx] = (float**)malloc2d(sc->nBands, IMS_MAX_NSAMPLES_PER_FRAME, sizeof(float));
306  }
307  }
308 
309  /* Find index corresponding to this receiver ID */
310  rec_idx = -1;
311  for(i=0; i<IMS_MAX_NUM_RECEIVERS; i++){
312  if(sc->recs[i].ID == receiverID){
313  rec_idx = i;
314  break;
315  }
316  }
317  assert(rec_idx != -1);
318 
319  /* Allocate temporary buffer (if this is the first time this function is being called) */
320  if( (sc->recs[rec_idx].ID != -1) && (sc->rec_sig_tmp[rec_idx] == NULL) )
321  sc->rec_sig_tmp[rec_idx] = (float**)malloc2d(sc->recs[rec_idx].nChannels, IMS_MAX_NSAMPLES_PER_FRAME, sizeof(float));
322  if(sc->framesize!=nSamples){
323  sc->framesize = nSamples;
324  sc->interpolator_fIn = realloc1d(sc->interpolator_fIn, nSamples*sizeof(float));
325  sc->interpolator_fOut = realloc1d(sc->interpolator_fOut, nSamples*sizeof(float));
326  for(i=0; i<nSamples; i++){
327  sc->interpolator_fIn[i] = (i+1)*1.0f/(float)nSamples;
328  sc->interpolator_fOut[i] = 1.0f-sc->interpolator_fIn[i];
329  }
330  }
331 
332  /* Initialise all receiver channels with zeros */
333  for(ch=0; ch<sc->recs[rec_idx].nChannels; ch++) /* (looping over channels since this array is not guaranteed to be contiguous) */
334  memset(sc->recs[rec_idx].sigs[ch], 0, nSamples * sizeof(float));
335 
336  /* Process all active sources for this specific receiver, directly in the time-domain */
337  for(src_idx = 0; src_idx < IMS_MAX_NUM_SOURCES; src_idx++){
338  if( (sc->srcs[src_idx].ID!=-1) && (sc->recs[rec_idx].ID!=-1) ){
339 
340  /* Broad-band operation */
341  if(sc->nBands==1)
342  memcpy(sc->src_sigs_bands[src_idx][0], sc->srcs[src_idx].sig, nSamples*sizeof(float));
343  else /* OR: Pass source signal through the Favrot & Faller (power-complementary) IIR filterbank */
344  faf_IIRFilterbank_apply(sc->hFaFbank[src_idx], sc->srcs[src_idx].sig, sc->src_sigs_bands[src_idx], nSamples);
345 
346  /* Workspace handle for this source/receiver combination */
347  wrk = sc->hCoreWrkSpc[rec_idx][src_idx];
348 
349  /* k=0 is for the current echogram,
350  * k=1 (when applyCrossFadeFLAG is enabled) is for the previous echogram */
351  for(k=sc->applyCrossFadeFLAG[src_idx][rec_idx]; k>=0; k--){
352 
353  /* Loop over samples */
354  for(n=0; n<nSamples; n++){
355 
356  /* Determine write index */
357  wIdx_n = sc->wIdx[k] & IMS_CIRC_BUFFER_LENGTH_MASK;
358 
359  /* Since the time vector is the same across bands, it makes sense to determine the read-indices only once... */
360  if(k==1)
361  echogram_abs_0 = (echogram_data*)wrk->hPrevEchogram_abs[0];
362  else
363  echogram_abs_0 = (echogram_data*)wrk->hEchogram_abs[0];
364 
365  /* Handle the special case of an empty echogram */
366  if(echogram_abs_0->numImageSources==0){
367  /* Set output to 0... */
368  if(k==1)
369  for(ch=0; ch<sc->recs[rec_idx].nChannels; ch++)
370  sc->rec_sig_tmp[rec_idx][ch][n] = 0.0f;
371  else
372  for(ch=0; ch<sc->recs[rec_idx].nChannels; ch++)
373  sc->recs[rec_idx].sigs[ch][n] = 0.0f;
374 
375  /* Store current sample (per band) into the circular buffer */
376  for(band=0; band < sc->nBands; band++){
377  sc->circ_buffer[k][src_idx][band][wIdx_n] = sc->src_sigs_bands[src_idx][band][n];
378  if(sc->applyCrossFadeFLAG[src_idx][rec_idx]==0)
379  sc->circ_buffer[1][src_idx][band][wIdx_n] = sc->src_sigs_bands[src_idx][band][n];
380  }
381 
382  /* Increment write index */
383  sc->wIdx[k]++;
384  if(sc->applyCrossFadeFLAG[src_idx][rec_idx]==0)
385  sc->wIdx[1]++;
386 
387  continue; /* to next sample... */
388  }
389 
390  /* Convert time from seconds to number of samples */
391  memset(echogram_abs_0->tmp1, 0, echogram_abs_0->numImageSources*sizeof(float));
392  cblas_saxpy(echogram_abs_0->numImageSources, (sc->fs), echogram_abs_0->time, 1, echogram_abs_0->tmp1, 1);
393 
394  /* Determine read-indices and (optionally) also the interpolation weights */
395  if(fractionalDelaysFLAG){
396  /* Loop over all image sources, and determine the circular buffer read indices */
397  for(im=0; im <echogram_abs_0->numImageSources; im++){
398  /* Base read-index */
399  time_samples = (int)(echogram_abs_0->tmp1[im]); /* FLOOR */
400  rIdx = IMS_CIRC_BUFFER_LENGTH-time_samples + sc->wIdx[k] /* read index for this image source */
401  + (IMS_LAGRANGE_ORDER/2); /* in order to correctly centre the filter */
402  echogram_abs_0->rIdx[im] = rIdx & IMS_CIRC_BUFFER_LENGTH_MASK; /* wrap-around if needed */
403  }
404 
405  /* Find fractional parts */
406  utility_svmod(echogram_abs_0->tmp1, echogram_abs_0->ones_dummy, echogram_abs_0->numImageSources, echogram_abs_0->tmp2);
407 
408  /* Find read-indices for interpolator */
409  for(im=0; im <echogram_abs_0->numImageSources; im++){
410  /* Centre the filter */
411  echogram_abs_0->tmp2[im] += (float)(IMS_LAGRANGE_ORDER/2);
412 
413  /* Read-indices for lagrange interpolation */
414  for(i=1; i<IMS_LAGRANGE_ORDER; i++){
415  echogram_abs_0->rIdx_frac[i-1][im] = echogram_abs_0->rIdx[im] - i;
416  /* Wrap around if needed */
417  if(echogram_abs_0->rIdx_frac[i-1][im]<0)
418  echogram_abs_0->rIdx_frac[i-1][im] += IMS_CIRC_BUFFER_LENGTH;
419  }
420  }
421 
422  /* Compute interpolation weights */ // TODO: This line is around 50% CPU usage of the whole function, a look-up table would be faster...
423  lagrangeWeights(IMS_LAGRANGE_ORDER, echogram_abs_0->tmp2, echogram_abs_0->numImageSources, FLATTEN2D(echogram_abs_0->h_frac));
424  }
425  else{
426  /* Loop over all image sources, and determine the circular buffer read indices based on the nearest sample */
427  for(im=0; im <echogram_abs_0->numImageSources; im++){
428  time_samples = (int)(echogram_abs_0->tmp1[im] + 0.5f); /* ROUND to nearest sample */
429  rIdx = IMS_CIRC_BUFFER_LENGTH-time_samples + sc->wIdx[k]; /* read index for this image source */
430  echogram_abs_0->rIdx[im] = rIdx & IMS_CIRC_BUFFER_LENGTH_MASK; /* wrap-around if needed */
431  }
432  }
433 
434  /* Loop over octave bands */
435  for(band=0; band < sc->nBands; band++){
436  /* Echogram for this source/receiver at this band */
437  if(k==1)
438  echogram_abs = (echogram_data*)wrk->hPrevEchogram_abs[band];
439  else
440  echogram_abs = (echogram_data*)wrk->hEchogram_abs[band];
441  assert(echogram_abs_0->numImageSources == echogram_abs->numImageSources);
442 
443  /* Pull values from the circular buffer corresponding to these read indices, and store this sparse vector as a "compressed" vector */
444  utility_ssv2cv_inds(sc->circ_buffer[k][src_idx][band], echogram_abs_0->rIdx, echogram_abs->numImageSources, echogram_abs->cb_vals[0]);
445 
446  /* Apply interpolation if enabled */
447  if(fractionalDelaysFLAG){
448  /* Apply the weights corresponding to the first tap in the filter */
449  utility_svvmul(echogram_abs->cb_vals[0], echogram_abs_0->h_frac[0], echogram_abs->numImageSources, echogram_abs->tmp1);
450 
451  /* Step through the rest of the filter */
452  for(i=1; i<IMS_LAGRANGE_ORDER; i++){
453  /* Pull values from the circular buffer corresponding to the read-indices for the current tap in the filter */
454  utility_ssv2cv_inds(sc->circ_buffer[k][src_idx][band], echogram_abs_0->rIdx_frac[i-1], echogram_abs->numImageSources, echogram_abs->cb_vals[0]);
455 
456  /* Apply interpolation weights */
457  utility_svvmul(echogram_abs->cb_vals[0], echogram_abs_0->h_frac[i], echogram_abs->numImageSources, echogram_abs->cb_vals[0]);
458 
459  /* Sum */
460  utility_svvadd(echogram_abs->tmp1, echogram_abs->cb_vals[0], echogram_abs->numImageSources, echogram_abs->tmp1);
461  }
462 
463  /* Copy result */
464  cblas_scopy(echogram_abs->numImageSources, echogram_abs->tmp1, 1, echogram_abs->cb_vals[0], 1);
465  }
466 
467  /* Replicate these circular buffer values for all output channels */
468  for(ch=1; ch<sc->recs[rec_idx].nChannels; ch++)
469  cblas_scopy(echogram_abs->numImageSources, echogram_abs->cb_vals[0], 1, echogram_abs->cb_vals[ch], 1);
470 
471  /* Apply the echogram scalings to each image source - for all channels */
472  utility_svvmul(FLATTEN2D(echogram_abs->value), FLATTEN2D(echogram_abs->cb_vals),
473  (sc->recs[rec_idx].nChannels)*echogram_abs->numImageSources, FLATTEN2D(echogram_abs->contrib));
474 
475  /* EITHER: We are rendering the current frame based on the previous echogram... In which case, we store it in the tmp buffer */
476  if(k==1){
477  for(ch=0; ch<sc->recs[rec_idx].nChannels; ch++)
478  sc->rec_sig_tmp[rec_idx][ch][n] = cblas_sdot(echogram_abs->numImageSources, echogram_abs->ones_dummy,
479  1, echogram_abs->contrib[ch], 1);
480  }
481  /* OR: We are rendering the current frame based on the current echogram... In which case, we store it in the current rec buffer */
482  else{
483  /* Sum all scaled image source values per channel */
484  for(ch=0; ch<sc->recs[rec_idx].nChannels; ch++)
485  sc->recs[rec_idx].sigs[ch][n] = cblas_sdot(echogram_abs->numImageSources, echogram_abs->ones_dummy,
486  1, echogram_abs->contrib[ch], 1);
487  }
488 
489  /* Store current sample into the circular buffer */
490  sc->circ_buffer[k][src_idx][band][wIdx_n] = sc->src_sigs_bands[src_idx][band][n];
491  if(sc->applyCrossFadeFLAG[src_idx][rec_idx]==0)
492  sc->circ_buffer[1][src_idx][band][wIdx_n] = sc->src_sigs_bands[src_idx][band][n];
493  }
494 
495  /* Increment write index */
496  sc->wIdx[k]++;
497  if(sc->applyCrossFadeFLAG[src_idx][rec_idx]==0)
498  sc->wIdx[1]++;
499 
500  } /* Loop over samples */
501  } /* Loop over slots */
502 
503  /* Cross-fade between the buffers rendered using the previous and current echograms */
504  if(sc->applyCrossFadeFLAG[src_idx][rec_idx]){
505  for(ch=0; ch<sc->recs[rec_idx].nChannels; ch++){
506  /* Apply linear interpolator to fade in with the new echogram and fade out with the previous echogram */
507  utility_svvmul(sc->recs[rec_idx].sigs[ch], sc->interpolator_fIn, nSamples, sc->recs[rec_idx].sigs[ch]);
508  utility_svvmul(sc->rec_sig_tmp[rec_idx][ch], sc->interpolator_fOut, nSamples, sc->rec_sig_tmp[rec_idx][ch]);
509 
510  /* Sum the result */
511  utility_svvadd(sc->recs[rec_idx].sigs[ch], sc->rec_sig_tmp[rec_idx][ch], nSamples, sc->recs[rec_idx].sigs[ch]);
512  }
513 
514  /* No longer need to cross-fade for future frames (unless the echograms change again that is...) */
515  sc->applyCrossFadeFLAG[src_idx][rec_idx] = 0;
516  }
517 
518  } /* If source active */
519  } /* Loop over sources */
520 }
