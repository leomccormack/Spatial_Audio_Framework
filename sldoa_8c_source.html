50void sldoa_create
51(
52 void ** const phSld
53)
54{
55 sldoa_data* pData = (sldoa_data*)malloc1d(sizeof(sldoa_data));
56 *phSld = (void*)pData;
57 int i, j, band;
58
59 /* Default user parameters */
60 pData->new_masterOrder = pData->masterOrder = 1;
61 for(band=0; band<HYBRID_BANDS; band++){
62 pData->analysisOrderPerBand[band] = pData->masterOrder;
63 pData->nSectorsPerBand[band] = ORDER2NUMSECTORS(pData->analysisOrderPerBand[band]);
64 }
65 pData->minFreq = 500.0f;
66 pData->maxFreq = 5e3f;
67 pData->avg_ms = 500.0f;
68 pData->chOrdering = CH_ACN;
69 pData->norm = NORM_SN3D;
70
71 /* TFT */
72 afSTFT_create(&(pData->hSTFT), MAX_NUM_SH_SIGNALS, 0, HOP_SIZE, 0, 1, AFSTFT_BANDS_CH_TIME);
73 pData->SHframeTD = (float**)malloc2d(MAX_NUM_SH_SIGNALS, SLDOA_FRAME_SIZE, sizeof(float));
74 pData->SHframeTF = (float_complex***)malloc3d(HYBRID_BANDS, MAX_NUM_SH_SIGNALS, TIME_SLOTS, sizeof(float_complex));
75
76 /* internal */
77 pData->progressBar0_1 = 0.0f;
78 pData->progressBarText = malloc1d(PROGRESSBARTEXT_CHAR_LENGTH*sizeof(char));
79 strcpy(pData->progressBarText,"");
80 pData->codecStatus = CODEC_STATUS_NOT_INITIALISED;
81 pData->procStatus = PROC_STATUS_NOT_ONGOING;
82 for(i=0; i<MAX_SH_ORDER-1; i++)
83 pData->secCoeffs[i] = NULL;
84 for(i=0; i<64; i++)
85 for(j=0; j<NUM_GRID_DIRS; j++)
86 pData->grid_Y[i][j] = (float)__grid_Y[i][j] * sqrtf(4.0f*SAF_PI);
87 for(i=0; i<3; i++)
88 for(j=0; j<NUM_GRID_DIRS; j++)
89 pData->grid_Y_dipoles_norm[i][j] = pData->grid_Y[i+1][j]/sqrtf(3); /* scale to [0..1] */
90 for(i=0; i<NUM_GRID_DIRS; i++)
91 for(j=0; j<2; j++)
92 pData->grid_dirs_deg[i][j] = (float)__grid_dirs_deg[i][j];
93
94 /* display */
95 for(i=0; i<NUM_DISP_SLOTS; i++){
96 pData->azi_deg[i] = malloc1d(HYBRID_BANDS*MAX_NUM_SECTORS * sizeof(float));
97 pData->elev_deg[i] = malloc1d(HYBRID_BANDS*MAX_NUM_SECTORS * sizeof(float));
98 pData->colourScale[i] = malloc1d(HYBRID_BANDS*MAX_NUM_SECTORS * sizeof(float));
99 pData->alphaScale[i] = malloc1d(HYBRID_BANDS*MAX_NUM_SECTORS * sizeof(float));
100 }
101
102 /* set FIFO buffer */
103 pData->FIFO_idx = 0;
104 memset(pData->inFIFO, 0, MAX_NUM_SH_SIGNALS*SLDOA_FRAME_SIZE*sizeof(float));
105}

107void sldoa_destroy
108(
109 void ** const phSld
110)
111{
112 sldoa_data *pData = (sldoa_data*)(*phSld);
113 int i;
114
115 if (pData != NULL) {
116 /* not safe to free memory during intialisation/processing loop */
117 while (pData->codecStatus == CODEC_STATUS_INITIALISING ||
118 pData->procStatus == PROC_STATUS_ONGOING){
119 SAF_SLEEP(10);
120 }
121
122 /* free afSTFT and buffers */
123 afSTFT_destroy(&(pData->hSTFT));
124 free(pData->SHframeTD);
125 free(pData->SHframeTF);
126
127 for(i=0; i<NUM_DISP_SLOTS; i++){
128 free(pData->azi_deg[i]);
129 free(pData->elev_deg[i]);
130 free(pData->colourScale[i]);
131 free(pData->alphaScale[i]);
132 }
133 free(pData->progressBarText);
134 free(pData);
135 pData = NULL;
136 }
137}

139void sldoa_init
140(
141 void * const hSld,
142 float sampleRate
143)
144{
145 sldoa_data *pData = (sldoa_data*)(hSld);
146 int i;
147
148 pData->fs = sampleRate;
149
150 /* specify frequency vector and determine the number of bands */
151 afSTFT_getCentreFreqs(pData->hSTFT, sampleRate, HYBRID_BANDS, pData->freqVector);
152
153 /* intialise display parameters */
154 pData->current_disp_idx = 0;
155 memset(pData->doa_rad, 0, HYBRID_BANDS*MAX_NUM_SECTORS*2* sizeof(float));
156 memset(pData->energy, 0, HYBRID_BANDS*MAX_NUM_SECTORS* sizeof(float));
157 for(i=0; i<NUM_DISP_SLOTS; i++){
158 memset(pData->azi_deg[i], 0, HYBRID_BANDS*MAX_NUM_SECTORS* sizeof(float));
159 memset(pData->elev_deg[i], 0, HYBRID_BANDS*MAX_NUM_SECTORS * sizeof(float));
160 memset(pData->colourScale[i], 0, HYBRID_BANDS*MAX_NUM_SECTORS * sizeof(float));
161 memset(pData->alphaScale[i], 0, HYBRID_BANDS*MAX_NUM_SECTORS * sizeof(float));
162 }
163}

165void sldoa_initCodec
166(
167 void* const hSld
168)
169{
170 sldoa_data *pData = (sldoa_data*)(hSld);
171
172 if (pData->codecStatus != CODEC_STATUS_NOT_INITIALISED)
173 return; /* re-init not required, or already happening */
174 while (pData->procStatus == PROC_STATUS_ONGOING){
175 /* re-init required, but we need to wait for the current processing loop to end */
176 pData->codecStatus = CODEC_STATUS_INITIALISING; /* indicate that we want to init */
177 SAF_SLEEP(10);
178 }
179
180 /* for progress bar */
181 pData->codecStatus = CODEC_STATUS_INITIALISING;
182 strcpy(pData->progressBarText,"Initialising");
183 pData->progressBar0_1 = 0.0f;
184
185 sldoa_initTFT(hSld);
186 sldoa_initAna(hSld);
187
188 /* done! */
189 strcpy(pData->progressBarText,"Done!");
190 pData->progressBar0_1 = 1.0f;
191 pData->codecStatus = CODEC_STATUS_INITIALISED;
192}

194void sldoa_analysis
195(
196 void * const hSld,
197 const float *const * inputs,
198 int nInputs,
199 int nSamples,
200 int isPlaying
201)
202{
203 sldoa_data *pData = (sldoa_data*)(hSld);
204 int s, i, j, t, ch, band, nSectors, min_band, numAnalysisBands, current_disp_idx;
205 float avgCoeff, max_en[HYBRID_BANDS], min_en[HYBRID_BANDS];
206 float new_doa[MAX_NUM_SECTORS][TIME_SLOTS][2], new_doa_xyz[3], doa_xyz[3], avg_xyz[3];
207 float new_energy[MAX_NUM_SECTORS][TIME_SLOTS];
208
209 /* local parameters */
210 int nSH, masterOrder;
211 int analysisOrderPerBand[HYBRID_BANDS];
212 int nSectorsPerBand[HYBRID_BANDS];
213 float minFreq, maxFreq, avg_ms;
214 CH_ORDER chOrdering;
215 NORM_TYPES norm;
216 memcpy(analysisOrderPerBand, pData->analysisOrderPerBand, HYBRID_BANDS*sizeof(int));
217 memcpy(nSectorsPerBand, pData->nSectorsPerBand, HYBRID_BANDS*sizeof(int));
218 minFreq = pData->minFreq;
219 maxFreq = pData->maxFreq;
220 avg_ms = pData->avg_ms;
221 chOrdering = pData->chOrdering;
222 norm = pData->norm;
223 masterOrder = pData->masterOrder;
224 nSH = ORDER2NSH(masterOrder);
225
226 /* Loop over all samples */
227 for(s=0; s<nSamples; s++){
228 /* Load input signals into inFIFO buffer */
229 for(ch=0; ch<SAF_MIN(nInputs,nSH); ch++)
230 pData->inFIFO[ch][pData->FIFO_idx] = inputs[ch][s];
231 for(; ch<nSH; ch++) /* Zero any channels that were not given */
232 pData->inFIFO[ch][pData->FIFO_idx] = 0.0f;
233
234 /* Increment buffer index */
235 pData->FIFO_idx++;
236
237 /* Process frame if inFIFO is full and codec is ready for it */
238 if (pData->FIFO_idx >= SLDOA_FRAME_SIZE && (pData->codecStatus == CODEC_STATUS_INITIALISED) && isPlaying) {
239 pData->FIFO_idx = 0;
240 pData->procStatus = PROC_STATUS_ONGOING;
241 current_disp_idx = pData->current_disp_idx;
242
243 /* Load time-domain data */
244 for(ch=0; ch<nSH; ch++)
245 memcpy(pData->SHframeTD[ch],pData->inFIFO[ch], SLDOA_FRAME_SIZE*sizeof(float));
246
247 /* account for input channel order */
248 switch(chOrdering){
249 case CH_ACN: /* already ACN */ break; /* Otherwise, convert to ACN... */
250 case CH_FUMA: convertHOAChannelConvention(FLATTEN2D(pData->SHframeTD), masterOrder, SLDOA_FRAME_SIZE, HOA_CH_ORDER_FUMA, HOA_CH_ORDER_ACN); break;
251 }
252
253 /* account for input normalisation scheme */
254 switch(norm){
255 case NORM_N3D: /* already in N3D, do nothing */ break; /* Otherwise, convert to N3D... */
256 case NORM_SN3D: convertHOANormConvention(FLATTEN2D(pData->SHframeTD), masterOrder, SLDOA_FRAME_SIZE, HOA_NORM_SN3D, HOA_NORM_N3D); break;
257 case NORM_FUMA: convertHOANormConvention(FLATTEN2D(pData->SHframeTD), masterOrder, SLDOA_FRAME_SIZE, HOA_NORM_FUMA, HOA_NORM_N3D); break;
258 }
259
260 /* apply the time-frequency transform */
261 afSTFT_forward_knownDimensions(pData->hSTFT, pData->SHframeTD, SLDOA_FRAME_SIZE, MAX_NUM_SH_SIGNALS, TIME_SLOTS, pData->SHframeTF);
262
263 /* apply sector-based, frequency-dependent DOA analysis */
264 numAnalysisBands = 0;
265 min_band = 0;
266 for(band=1/* ignore DC */; band<HYBRID_BANDS; band++){
267 if(pData->freqVector[band] <= minFreq)
268 min_band = band;
269 if(pData->freqVector[band] >= minFreq && pData->freqVector[band]<=maxFreq){
270 nSectors = nSectorsPerBand[band];
271 avgCoeff = avg_ms < 10.0f ? 1.0f : 1.0f / ((avg_ms/1e3f) / (1.0f/(float)HOP_SIZE) + 2.23e-9f);
272 avgCoeff = SAF_MAX(SAF_MIN(avgCoeff, 0.99999f), 0.0f); /* ensures stability */
273 sldoa_estimateDoA(pData->SHframeTF[band],
274 analysisOrderPerBand[band],
275 pData->secCoeffs[analysisOrderPerBand[band]-2], /* -2, as first order is skipped */
276 new_doa,
277 new_energy);
278
279 /* average the raw data over time */
280 for(i=0; i<nSectors; i++){
281 for( t = 0; t<TIME_SLOTS; t++){
282 /* avg doa estimate */
283 unitSph2cart((float*)new_doa[i][t], 1, 0, (float*)new_doa_xyz);
284 unitSph2cart((float*)pData->doa_rad[band][i], 1, 0, (float*)doa_xyz);
285 for(j=0; j<3; j++)
286 avg_xyz[j] = new_doa_xyz[j]*avgCoeff + doa_xyz[j] * (1.0f-avgCoeff);
287 unitCart2sph((float*)avg_xyz, 1, 0, (float*)pData->doa_rad[band][i]);
288
289 /* avg energy */
290 pData->energy[band][i] = new_energy[i][t]*avgCoeff + pData->energy[band][i] * (1.0f-avgCoeff);
291 }
292 }
293 numAnalysisBands++;
294 }
295 }
296
297 /* determine the minimum and maximum sector energies per frequency (to scale them 0..1) */
298 for(band=1/* ignore DC */; band<HYBRID_BANDS; band++){
299 if(pData->freqVector[band] >= minFreq && pData->freqVector[band]<=maxFreq){
300 nSectors = nSectorsPerBand[band];
301 max_en[band] = 2.3e-13f; min_en[band] = 2.3e13f; /* starting values */
302 for(i=0; i<nSectors; i++){
303 max_en[band] = pData->energy[band][i] > max_en[band] ? pData->energy[band][i] : max_en[band];
304 min_en[band] = pData->energy[band][i] < min_en[band] ? pData->energy[band][i] : min_en[band];
305 }
306 }
307 }
308
309 /* prep data for plotting */
310 for(band=1/* ignore DC */; band<HYBRID_BANDS; band++){
311 if(pData->freqVector[band] >= minFreq && pData->freqVector[band]<=maxFreq){
312 nSectors = nSectorsPerBand[band];
313 /* store averaged values */
314 for(i=0; i<nSectors; i++){
315 pData->azi_deg [current_disp_idx][band*MAX_NUM_SECTORS + i] = pData->doa_rad[band][i][0]*180.0f/SAF_PI;
316 pData->elev_deg[current_disp_idx][band*MAX_NUM_SECTORS + i] = pData->doa_rad[band][i][1]*180.0f/SAF_PI;
317
318 /* colour should indicate the different frequencies */
319 pData->colourScale[current_disp_idx][band*MAX_NUM_SECTORS + i] = (float)(band-min_band)/(float)(numAnalysisBands+1);
320
321 /* transparancy should indicate the energy of the sector for each DoA estimate, for each frequency */
322 if( analysisOrderPerBand[band]==1 )
323 pData->alphaScale[current_disp_idx][band*MAX_NUM_SECTORS + i] = 1.0f;
324 else
325 pData->alphaScale[current_disp_idx][band*MAX_NUM_SECTORS + i] = SAF_MIN(SAF_MAX((pData->energy[band][i]-min_en[band])/(max_en[band]-min_en[band]+2.3e-10f), 0.05f),1.0f);
326 }
327 }
328 else{
329 memset(&(pData->azi_deg [current_disp_idx][band*MAX_NUM_SECTORS]), 0, MAX_NUM_SECTORS*sizeof(float));
330 memset(&(pData->elev_deg [current_disp_idx][band*MAX_NUM_SECTORS]), 0, MAX_NUM_SECTORS*sizeof(float));
331 memset(&(pData->colourScale [current_disp_idx][band*MAX_NUM_SECTORS]), 0, MAX_NUM_SECTORS*sizeof(float));
332 memset(&(pData->alphaScale [current_disp_idx][band*MAX_NUM_SECTORS]), 0, MAX_NUM_SECTORS*sizeof(float));
333 }
334 }
335 }
336 else if(pData->FIFO_idx >= SLDOA_FRAME_SIZE){
337 /* reset FIFO_idx if codec was not ready */
338 pData->FIFO_idx = 0;
339 }
340 }
341
342 pData->procStatus = PROC_STATUS_NOT_ONGOING;
343}

348{
349 sldoa_data *pData = (sldoa_data*)(hSld);

356 pData->chOrdering = CH_ACN;
357 if(pData->new_masterOrder!=SH_ORDER_FIRST && pData->norm == NORM_FUMA)
358 pData->norm = NORM_SN3D;

390void sldoa_setSourcePreset(void* const hSld, int newPresetID)
391{
392 sldoa_data *pData = (sldoa_data*)(hSld);
393 int band, rangeIdx, curOrder, reverse;
394
395 rangeIdx = 0;
396 curOrder = 1;
397 reverse = 0;
398 switch(newPresetID){
399 case MIC_PRESET_IDEAL:
400 for(band=0; band<HYBRID_BANDS; band++)
401 pData->analysisOrderPerBand[band] = pData->new_masterOrder;
402 break;
403
404 case MIC_PRESET_ZYLIA:
405 for(band=0; band<HYBRID_BANDS; band++){
406 if(rangeIdx<2*(__Zylia_maxOrder-1)){
407 if(pData->freqVector[band]>__Zylia_freqRange[rangeIdx]){
408 if(!reverse)
409 curOrder++;
410 else
411 curOrder--;
412 reverse = (curOrder == __Zylia_maxOrder) || (reverse) ? 1 : 0;
413 rangeIdx++;
414 }
415 }
416 pData->analysisOrderPerBand[band] = SAF_MIN(pData->new_masterOrder,curOrder);
417 }
418 pData->maxFreq = __Zylia_freqRange[(__Zylia_maxOrder-1)*2-1];
419 break;
420
421 case MIC_PRESET_EIGENMIKE32:
422 for(band=0; band<HYBRID_BANDS; band++){
423 if(rangeIdx<2*(__Eigenmike32_maxOrder-1)){
424 if(pData->freqVector[band]>__Eigenmike32_freqRange[rangeIdx]){
425 if(!reverse)
426 curOrder++;
427 else
428 curOrder--;
429 reverse = (curOrder == __Eigenmike32_maxOrder) || (reverse) ? 1 : 0;
430 rangeIdx++;
431 }
432 }
433 pData->analysisOrderPerBand[band] = SAF_MIN(pData->new_masterOrder,curOrder);
434 }
435 pData->maxFreq = __Eigenmike32_freqRange[(__Eigenmike32_maxOrder-1)*2-1];
436 break;
437
438 case MIC_PRESET_DTU_MIC:
439 for(band=0; band<HYBRID_BANDS; band++){
440 if(rangeIdx<2*(__DTU_mic_maxOrder-1)){
441 if(pData->freqVector[band]>__DTU_mic_freqRange[rangeIdx]){
442 if(!reverse)
443 curOrder++;
444 else
445 curOrder--;
446 reverse = (curOrder == __DTU_mic_maxOrder) || (reverse) ? 1 : 0;
447 rangeIdx++;
448 }
449 }
450 pData->analysisOrderPerBand[band] = SAF_MIN(pData->new_masterOrder,curOrder);
451 }
452 pData->maxFreq = __DTU_mic_freqRange[(__DTU_mic_maxOrder-1)*2-1];
453 break;
454 }
455 for(band=0; band<HYBRID_BANDS; band++)
456 pData->nSectorsPerBand[band] = ORDER2NUMSECTORS(pData->analysisOrderPerBand[band]);
457}

547/* Not very elegent, but does the job */
548void sldoa_getDisplayData
549(
550 void * const hSld,
551 float** azi_deg,
552 float** elev_deg,
553 float** colourScale,
554 float** alphaScale,
555 int** pNsectorsPerBand,
556 int* maxNumSectors,
557 int* startBand,
558 int* endBand
559)
560{
561 sldoa_data *pData = (sldoa_data*)(hSld);
562 int i;
563
564 (*azi_deg) = pData->azi_deg[pData->current_disp_idx];
565 (*elev_deg) = pData->elev_deg[pData->current_disp_idx];
566 (*colourScale) = pData->colourScale[pData->current_disp_idx];
567 (*alphaScale) = pData->alphaScale[pData->current_disp_idx];
568 (*pNsectorsPerBand) = pData->nSectorsPerBand;
569 (*maxNumSectors) = MAX_NUM_SECTORS;
570 (*startBand) =1;
