51{

53 if(newStatus==CODEC_STATUS_NOT_INITIALISED){
54 /* Pause until current initialisation is complete */

57 }

61void sldoa_initAna(void* const hSld)
62{
63 sldoa_data *pData = (sldoa_data*)(hSld);
64 int i, n, j, k, order, nSectors, nSH, grid_N_vbap_gtable, grid_nGroups, maxOrder;
65 float* sec_dirs_deg, *grid_vbap_gtable, *w_SG, *pinv_Y, *grid_vbap_gtable_T;
66 float secPatterns[4][NUM_GRID_DIRS];
67
68 maxOrder = pData->new_masterOrder;
69
70 grid_vbap_gtable_T = malloc1d(ORDER2NUMSECTORS(maxOrder) * NUM_GRID_DIRS * sizeof(float));
71
72 for(i=0, order=2; order<=maxOrder; i++,order++){
73 nSectors = ORDER2NUMSECTORS(order);
74 nSH = (order+1)*(order+1);
75
76 /* define sector directions */
77 sec_dirs_deg = malloc1d(nSectors*2*sizeof(float));
78 memcpy(sec_dirs_deg, __HANDLES_SphCovering_dirs_deg[nSectors-1], nSectors*2*sizeof(float));
79
80 /* generate VBAP gain table */
81 generateVBAPgainTable3D_srcs((float*)pData->grid_dirs_deg, NUM_GRID_DIRS, sec_dirs_deg, nSectors, 0, 0, 0.0f,
82 &(grid_vbap_gtable), &(grid_N_vbap_gtable), &(grid_nGroups));
83
84 /* convert to amplitude preserving gains */
85 VBAPgainTable2InterpTable(grid_vbap_gtable, NUM_GRID_DIRS, nSectors);
86
87 /* transpose */
88 for(n=0; n<nSectors; n++)
89 for(j=0; j<NUM_GRID_DIRS; j++)
90 grid_vbap_gtable_T[n*NUM_GRID_DIRS+j] = grid_vbap_gtable[j*nSectors+n];
91
92 /* generate sector coefficients */
93 if(pData->secCoeffs[i]!=NULL)
94 free(pData->secCoeffs[i]);
95 pData->secCoeffs[i] = malloc1d(4 * (nSH*nSectors) * sizeof(float_complex));
96 w_SG = malloc1d(4 * (nSH) * sizeof(float));
97 pinv_Y = malloc1d(NUM_GRID_DIRS*nSH*sizeof(float));
98 for(n=0; n<nSectors; n++){
99 utility_svvmul(&(grid_vbap_gtable_T[n*NUM_GRID_DIRS]), pData->grid_Y[0], NUM_GRID_DIRS, secPatterns[0]);
100 for(j=0; j<3; j++)
101 utility_svvmul(&(grid_vbap_gtable_T[n*NUM_GRID_DIRS]), pData->grid_Y_dipoles_norm[j], NUM_GRID_DIRS, secPatterns[j+1]);
102 utility_spinv(NULL, &(pData->grid_Y[0][0]), nSH, NUM_GRID_DIRS, pinv_Y);
103 cblas_sgemm(CblasRowMajor, CblasNoTrans, CblasNoTrans, 4, nSH, NUM_GRID_DIRS, 1.0f,
104 &(secPatterns[0][0]), NUM_GRID_DIRS,
105 pinv_Y, nSH, 0.0f,
106 w_SG, nSH);
107
108 /* stack the sector coefficients */
109 for(j=0; j<4; j++)
110 for(k=0; k<nSH; k++)
111 pData->secCoeffs[i][j*(nSectors*nSH)+n*nSH+k] = cmplxf(w_SG[j*nSH+k], 0.0f);
112 }
113 free(w_SG);
114 free(pinv_Y);
115 free(grid_vbap_gtable);
116 free(sec_dirs_deg);
117 }
118
119 free(grid_vbap_gtable_T);
120
121 pData->masterOrder = maxOrder;
122}

142void sldoa_estimateDoA
143(
144 float_complex** SHframeTF,
145 int anaOrder,
146 float_complex* secCoeffs,
147 float doa[MAX_NUM_SECTORS][TIME_SLOTS][2],
148 float energy[MAX_NUM_SECTORS][TIME_SLOTS]
149)
150{
151 int n, ch, i, j, nSectors, analysisOrder, nSH;
152 float_complex secSig[4][TIME_SLOTS];
153 float_complex* sec_c;
154 float secEnergy[TIME_SLOTS], secIntensity[3][TIME_SLOTS], secAzi[TIME_SLOTS], secElev[TIME_SLOTS];
155 const float_complex calpha = cmplxf(1.0f, 0.0f); const float_complex cbeta = cmplxf(0.0f, 0.0f);
156
157 /* prep */
158 memset(doa,0,MAX_NUM_SECTORS*TIME_SLOTS*2*sizeof(float));
159 memset(energy,0,MAX_NUM_SECTORS*TIME_SLOTS*sizeof(float));
160 analysisOrder = SAF_MAX(SAF_MIN(MAX_SH_ORDER, anaOrder),1);
161 nSectors = ORDER2NUMSECTORS(analysisOrder);
162 nSH = (analysisOrder+1)*(analysisOrder+1);
163 sec_c = malloc1d(4*nSH*sizeof(float_complex));
164
165 /* calculate energy and DoA for each sector */
166 for( n=0; n<nSectors; n++){
167 if(anaOrder==1 || secCoeffs == NULL) /* standard first order active-intensity based DoA estimation */
168 for (i=0; i<4; i++)
169 memcpy(secSig[i], SHframeTF[i], TIME_SLOTS * sizeof(float_complex));
170 else{ /* spatially localised active-intensity based DoA estimation */
171 for (i=0; i<4; i++)
172 for (j=0; j<nSH; j++)
173 sec_c[i*nSH+j] = secCoeffs[i*(nSectors*nSH)+n*nSH+j];
174 cblas_cgemm(CblasRowMajor, CblasNoTrans, CblasNoTrans, 4, TIME_SLOTS, nSH, &calpha,
175 sec_c, nSH,
176 FLATTEN2D(SHframeTF), TIME_SLOTS, &cbeta,
177 secSig, TIME_SLOTS);
178 }
179
180 /* convert N3D to SN3D */
181 for (ch = 1; ch<4; ch++)
182 for(i = 0; i<TIME_SLOTS; i++)
183 secSig[ch][i] = crmulf(secSig[ch][i], 1.0f/sqrtf(3.0f));
184
185 /* calculate sector energy and intensity vector */
186 memset(secEnergy, 0, TIME_SLOTS*sizeof(float));
187 for (i=0; i<4; i++)
188 for (j=0; j<TIME_SLOTS; j++)
189 secEnergy[j] += 0.5f*powf(cabsf(secSig[i][j]), 2.0f);
190 for (i=0; i<3; i++)
191 for (j=0; j<TIME_SLOTS; j++)
192 secIntensity[i][j] = crealf(ccmulf(conjf(secSig[0][j]), secSig[1+i][j]));
193
194 /* extract DoA */
195 for (j=0; j<TIME_SLOTS; j++){
196 secAzi[j] = atan2f( secIntensity[0][j], secIntensity[2][j] );
197 secElev[j] = atan2f( secIntensity[1][j], sqrtf( powf(secIntensity[2][j], 2.0f) + powf(secIntensity[0][j], 2.0f)) );
198 }
199
200 /* store energy and DoA estimate */
201 for (j=0; j<TIME_SLOTS; j++){
202 doa[n][j][0] = secAzi[j];
203 doa[n][j][1] = secElev[j];
204 energy[n][j] = secEnergy[j]*1e6f;
205 }
206 }
207
208 free(sec_c);
209}
