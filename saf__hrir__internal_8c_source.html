39 static void afAnalyse
40 (
41  float* inTD/* nSamplesTD x nCH */,
42  int nSamplesTD,
43  int nCH,
44  int hopSize,
45  int hybridmode,
46  float_complex* outTF /* out_nBands x nTimeslots x nCH */
47 )
48 {
49  int t, ch, sample, band;
50  void* hSTFT;
51  float_complex*** FrameTF;
52  float** tempFrameTD;
53  int nTimeSlots, nBands;
54 
55  nBands = hopSize + (hybridmode ? 5 : 1);
56  nTimeSlots = nSamplesTD/hopSize;
57 
58  /* allocate memory */
59  afSTFT_create(&(hSTFT), nCH, 1, hopSize, 0, hybridmode, AFSTFT_TIME_CH_BANDS);
60  FrameTF = (float_complex***)malloc3d(nTimeSlots, nCH, nBands, sizeof(float_complex));
61  tempFrameTD = (float**)malloc2d(nCH, nSamplesTD, sizeof(float));
62 
63  /* perform TF transform */
64  for(ch=0; ch<nCH; ch++)
65  for(sample=0; sample<nSamplesTD; sample++)
66  tempFrameTD[ch][sample] = inTD[sample* nCH + ch];
67  afSTFT_forward(hSTFT, tempFrameTD, nSamplesTD, FrameTF);
68 
69  /* save result to output */
70  for (band = 0; band < nBands; band++)
71  for (t = 0; t < nTimeSlots; t++)
72  for (ch = 0; ch < nCH; ch++)
73  outTF[band * nTimeSlots * nCH + t * nCH + ch] = FrameTF[t][ch][band];
74 
75  /* clean-up */
76  afSTFT_destroy(&hSTFT);
77  free(FrameTF);
78  free(tempFrameTD);
79 }

126 void FIRtoFilterbankCoeffs_afSTFT
127 (
128  float* hIR /*N_dirs x nCH x ir_len*/,
129  int N_dirs,
130  int nCH,
131  int ir_len,
132  int hopSize,
133  int hybridmode,
134  float_complex* hFB /* nBands x nCH x N_dirs */
135 )
136 {
137  int i, j, t, nd, nm, nTimeSlots, ir_pad, nBands;
138  int* maxIdx;
139  float maxVal, idxDel, irFB_energy, irFB_gain, phase;
140  float* centerImpulse, *centerImpulseFB_energy, *ir;
141  float_complex cross;
142  float_complex* centerImpulseFB, *irFB;
143 
144  nBands = hopSize + (hybridmode ? 5 : 1);
145  ir_pad = 1024;//+512;
146  nTimeSlots = (ir_len+ir_pad)/hopSize;
147  maxIdx = calloc1d(nCH,sizeof(int));
148  centerImpulse = calloc1d(ir_len+ir_pad, sizeof(float));
149 
150  /* pick a direction to estimate the center of FIR delays */
151  for(j=0; j<nCH; j++){
152  maxVal = 2.23e-13f;
153  for(i=0; i<ir_len; i++){
154  if(hIR[j*ir_len + i] > maxVal){
155  maxVal = hIR[j*ir_len + i];
156  maxIdx[j] = i;
157  }
158  }
159  }
160  idxDel = 0.0f;
161  for(j=0; j<nCH; j++)
162  idxDel += (float)maxIdx[j];
163  idxDel /= (float)nCH;
164  idxDel = (idxDel + 1.5f);
165 
166  /* ideal impulse at mean delay */
167  centerImpulse[(int)idxDel] = 1.0f;
168 
169  /* analyse impulse with the filterbank */
170  centerImpulseFB = malloc1d(nBands*nTimeSlots*nCH*sizeof(float_complex));
171  afAnalyse(centerImpulse, ir_len+ir_pad, 1, hopSize, hybridmode, centerImpulseFB);
172  centerImpulseFB_energy = calloc1d(nBands, sizeof(float));
173  for(i=0; i<nBands; i++)
174  for(t=0; t<nTimeSlots; t++)
175  centerImpulseFB_energy[i] += powf(cabsf(centerImpulseFB[i*nTimeSlots + t]), 2.0f);
176 
177  /* initialise FB coefficients */
178  ir = calloc1d( (ir_len+ir_pad) * nCH, sizeof(float));
179  irFB = malloc1d(nBands*nCH*nTimeSlots*sizeof(float_complex));
180  for(nd=0; nd<N_dirs; nd++){
181  for(j=0; j<ir_len; j++)
182  for(i=0; i<nCH; i++)
183  ir[j*nCH+i] = hIR[nd*nCH*ir_len + i*ir_len + j];
184  afAnalyse(ir, ir_len+ir_pad, nCH, hopSize, hybridmode, irFB);
185  for(nm=0; nm<nCH; nm++){
186  for(i=0; i<nBands; i++){
187  irFB_energy = 0;
188  for(t=0; t<nTimeSlots; t++)
189  irFB_energy += powf(cabsf(irFB[i*nTimeSlots*nCH + t*nCH + nm]), 2.0f); /* out_nBands x nTimeslots x nCH */
190  irFB_gain = sqrtf(irFB_energy/MAX(centerImpulseFB_energy[i], 2.23e-8f));
191  cross = cmplxf(0.0f,0.0f);
192  for(t=0; t<nTimeSlots; t++)
193  cross = ccaddf(cross, ccmulf(irFB[i*nTimeSlots*nCH + t*nCH + nm], conjf(centerImpulseFB[i*nTimeSlots + t])));
194  phase = atan2f(cimagf(cross), crealf(cross));
195  hFB[i*nCH*N_dirs + nm*N_dirs + nd] = crmulf( cexpf(cmplxf(0.0f, phase)), irFB_gain);
196  }
197  }
198  }
199 
200  /* clean-up */
201  free(maxIdx);
202  free(centerImpulse);
203  free(centerImpulseFB_energy);
204  free(centerImpulseFB);
205  free(ir);
206  free(irFB);
207 }
