28void spreader_create
29(
30 void ** const phSpr
31)
32{
33 spreader_data* pData = (spreader_data*)malloc1d(sizeof(spreader_data));
34 *phSpr = (void*)pData;
35 int band, t, src;
36
37 /* user parameters */
38 pData->sofa_filepath = NULL;
39 pData->nSources = 1;
40 pData->procMode = SPREADER_MODE_OM;
41 pData->useDefaultHRIRsFLAG = 1;
42 pData->covAvgCoeff = 0.85f;
43 memset(pData->src_spread, 0, SPREADER_MAX_NUM_SOURCES*sizeof(float));
44 memset(pData->src_dirs_deg, 0, SPREADER_MAX_NUM_SOURCES*2*sizeof(float));
45
46 /* time-frequency transform + buffers */
47 pData->hSTFT = NULL;
48 pData->inputFrameTD = (float**)malloc2d(MAX_NUM_INPUTS, SPREADER_FRAME_SIZE, sizeof(float));
49 pData->outframeTD = (float**)malloc2d(MAX_NUM_OUTPUTS, SPREADER_FRAME_SIZE, sizeof(float));
50 pData->inputframeTF = (float_complex***)malloc3d(HYBRID_BANDS, MAX_NUM_INPUTS, TIME_SLOTS, sizeof(float_complex));
51 pData->protoframeTF = (float_complex***)malloc3d(HYBRID_BANDS, MAX_NUM_OUTPUTS, TIME_SLOTS, sizeof(float_complex));
52 pData->decorframeTF = (float_complex***)malloc3d(HYBRID_BANDS, MAX_NUM_OUTPUTS, TIME_SLOTS, sizeof(float_complex));
53 pData->spreadframeTF = (float_complex***)malloc3d(HYBRID_BANDS, MAX_NUM_OUTPUTS, TIME_SLOTS, sizeof(float_complex));
54 pData->outputframeTF = (float_complex***)malloc3d(HYBRID_BANDS, MAX_NUM_OUTPUTS, TIME_SLOTS, sizeof(float_complex));
55
56 /* Internal */
57 pData->Q = pData->nGrid = pData->h_len = 0;
58 pData->h_fs = 0.0f;
59 pData->h_grid = NULL;
60 pData->H_grid = NULL;
61 for(band=0; band<HYBRID_BANDS; band++)
62 pData->HHH[band] = NULL;
63 pData->grid_dirs_deg = NULL;
64 pData->grid_dirs_xyz = NULL;
65 pData->weights = NULL;
66 pData->angles = NULL;
67 for(src=0; src<SPREADER_MAX_NUM_SOURCES; src++){
68 pData->hDecor[src] = NULL;
69 pData->Cy[src] = NULL;
70 pData->Cproto[src] = NULL;
71 pData->prev_M[src] = NULL;
72 pData->prev_Mr[src] = NULL;
73 pData->dirActive[src] = NULL;
74 }
75 pData->new_M = NULL;
76 pData->new_Mr = NULL;
77 pData->interp_M = NULL;
78 pData->interp_Mr = NULL;
79 pData->interp_Mr_cmplx = NULL;
80 for(t=0; t<TIME_SLOTS; t++){
81 pData->interpolatorFadeIn[t] = ((float)t+1.0f)/(float)TIME_SLOTS;
82 pData->interpolatorFadeOut[t] = 1.0f - ((float)t+1.0f)/(float)TIME_SLOTS;
83 }
84
85 /* Optimal mixing */
86 pData->hCdf = NULL;
87 pData->hCdf_res = NULL;
88 pData->Qmix = NULL;
89 pData->Qmix_cmplx = NULL;
90 pData->Cr = NULL;
91 pData->Cr_cmplx = NULL;
92
93 /* flags/status */
94 pData->new_procMode = pData->procMode;
95 pData->new_nSources = pData->nSources;
96 pData->progressBar0_1 = 0.0f;
97 pData->progressBarText = malloc1d(PROGRESSBARTEXT_CHAR_LENGTH*sizeof(char));
98 strcpy(pData->progressBarText,"");
99 pData->codecStatus = CODEC_STATUS_NOT_INITIALISED;
100 pData->procStatus = PROC_STATUS_NOT_ONGOING;
101}

103void spreader_destroy
104(
105 void ** const phSpr
106)
107{
108 spreader_data *pData = (spreader_data*)(*phSpr);
109 int band, src;
110
111 if (pData != NULL) {
112 /* not safe to free memory during intialisation/processing loop */
113 while (pData->codecStatus == CODEC_STATUS_INITIALISING ||
114 pData->procStatus == PROC_STATUS_ONGOING){
115 SAF_SLEEP(10);
116 }
117
118 /* free afSTFT and buffers */
119 if(pData->hSTFT !=NULL)
120 afSTFT_destroy(&(pData->hSTFT));
121 free(pData->inputFrameTD);
122 free(pData->outframeTD);
123 free(pData->inputframeTF);
124 free(pData->decorframeTF);
125 free(pData->spreadframeTF);
126 free(pData->outputframeTF);
127
128 /* internal */
129 free(pData->h_grid);
130 free(pData->H_grid);
131 for(band=0; band<HYBRID_BANDS; band++)
132 free(pData->HHH[band]);
133 free(pData->grid_dirs_deg);
134 free(pData->grid_dirs_xyz);
135 free(pData->weights);
136 free(pData->angles);
137 for(src=0; src<SPREADER_MAX_NUM_SOURCES; src++){
138 latticeDecorrelator_destroy(&(pData->hDecor[src]));
139 free(pData->Cy[src]);
140 free(pData->Cproto[src]);
141 free(pData->prev_M[src]);
142 free(pData->prev_Mr[src]);
143 free(pData->dirActive[src]);
144 }
145 free(pData->new_M);
146 free(pData->new_Mr);
147 free(pData->interp_M);
148 free(pData->interp_Mr);
149 free(pData->interp_Mr_cmplx);
150
151 /* Optimal mixing */
152 cdf4sap_cmplx_destroy(&(pData->hCdf));
153 cdf4sap_destroy(&(pData->hCdf_res));
154 free(pData->Qmix);
155 free(pData->Qmix_cmplx);
156 free(pData->Cr);
157 free(pData->Cr_cmplx);
158
159 free(pData->progressBarText);
160
161 free(pData);
162 pData = NULL;
163 }
164}

179void spreader_initCodec
180(
181 void* const hSpr
182)
183{
184 spreader_data *pData = (spreader_data*)(hSpr);
185 int q, band, ng, nSources, src;
186 float_complex scaleC;
187#ifdef SAF_ENABLE_SOFA_READER_MODULE
188 saf_sofa_container sofa;
189 SAF_SOFA_ERROR_CODES error;
190#endif
191 SPREADER_PROC_MODES procMode;
192 float_complex H_tmp[MAX_NUM_CHANNELS];
193 const float_complex calpha = cmplxf(1.0f, 0.0f), cbeta = cmplxf(0.0f, 0.0f);
194
195 if (pData->codecStatus != CODEC_STATUS_NOT_INITIALISED)
196 return; /* re-init not required, or already happening */
197 while (pData->procStatus == PROC_STATUS_ONGOING){
198 /* re-init required, but we need to wait for the current processing loop to end */
199 pData->codecStatus = CODEC_STATUS_INITIALISING; /* indicate that we want to init */
200 SAF_SLEEP(10);
201 }
202
203 nSources = pData->new_nSources;
204 procMode = pData->new_procMode;
205
206 /* for progress bar */
207 pData->codecStatus = CODEC_STATUS_INITIALISING;
208 strcpy(pData->progressBarText,"Initialising");
209 pData->progressBar0_1 = 0.0f;
210
211 /* Load measurements (e.g. HRIRs, Microphone array IRs etc.) */
212#ifndef SAF_ENABLE_SOFA_READER_MODULE
213 pData->useDefaultHRIRsFLAG = 1;
214#endif
215 if(pData->useDefaultHRIRsFLAG){
216 /* Load default HRIR data */
217 pData->Q = NUM_EARS;
218 pData->nGrid = __default_N_hrir_dirs;
219 pData->h_len = __default_hrir_len;
220 pData->h_fs = (float)__default_hrir_fs;
221 pData->h_grid = realloc1d(pData->h_grid, pData->nGrid * (pData->Q) * (pData->h_len) * sizeof(float));
222 memcpy(pData->h_grid, (float*)__default_hrirs, pData->nGrid * (pData->Q) * (pData->h_len) * sizeof(float));
223 pData->grid_dirs_deg = realloc1d(pData->grid_dirs_deg, pData->nGrid * 2 * sizeof(float));
224 memcpy(pData->grid_dirs_deg, (float*)__default_hrir_dirs_deg, pData->nGrid * 2 * sizeof(float));
225 }
226#ifdef SAF_ENABLE_SOFA_READER_MODULE
227 else{
228 /* Use sofa loader */
229 error = saf_sofa_open(&sofa, pData->sofa_filepath);
230 if(error!=SAF_SOFA_OK){
231 /* if failed, then load default data instead */
232 pData->useDefaultHRIRsFLAG = 1;
233 saf_print_warning("Unable to load the specified SOFA file. Using default HRIR data instead");
234 spreader_initCodec(hSpr);
235 }
236 pData->h_fs = sofa.DataSamplingRate;
237 pData->h_len = sofa.DataLengthIR;
238 pData->nGrid = sofa.nSources;
239 pData->h_grid = realloc1d(pData->h_grid, pData->nGrid*(pData->Q)*(pData->h_len)*sizeof(float));
240 memcpy(pData->h_grid, sofa.DataIR, pData->nGrid*(pData->Q)*(pData->h_len)*sizeof(float));
241 pData->grid_dirs_deg = realloc1d(pData->grid_dirs_deg, pData->nGrid*2*sizeof(float));
242 cblas_scopy(pData->nGrid, sofa.SourcePosition, 3, pData->grid_dirs_deg, 2); /* azi */
243 cblas_scopy(pData->nGrid, &sofa.SourcePosition[1], 3, &pData->grid_dirs_deg[1], 2); /* elev */
244 saf_sofa_close(&sofa);
245 }
246#endif
247
248 /* Convert from the 0..360 convention, to -180..180, and pre-compute unit Cartesian vectors */
249 convert_0_360To_m180_180(pData->grid_dirs_deg, pData->nGrid);
250 pData->grid_dirs_xyz = realloc1d(pData->grid_dirs_xyz, pData->nGrid*3*sizeof(float));
251 unitSph2cart(pData->grid_dirs_deg, pData->nGrid, 1, pData->grid_dirs_xyz);
252
253 /* Initialise time-frequency transform and decorrelators */
254 afSTFT_destroy(&(pData->hSTFT));
255 afSTFT_create(&(pData->hSTFT), nSources, pData->Q, HOP_SIZE, 0, 1, AFSTFT_BANDS_CH_TIME);
256 int orders[4] = {20, 15, 6, 6}; /* 20th order up to 700Hz, 15th->2.4kHz, 6th->4kHz, 3rd->12kHz, NONE(only delays)->Nyquist */
257 //float freqCutoffs[4] = {600.0f, 2.6e3f, 4.5e3f, 12e3f};
258 float freqCutoffs[4] = {900.0f, 6.8e3f, 12e3f, 24e3f};
259 const int maxDelay = 12;
260 for(src=0; src<SPREADER_MAX_NUM_SOURCES; src++){
261 latticeDecorrelator_destroy(&(pData->hDecor[src]));
262 latticeDecorrelator_create(&(pData->hDecor[src]), (float)pData->fs, HOP_SIZE, pData->freqVector, HYBRID_BANDS, pData->Q, orders, freqCutoffs, 4, maxDelay, 0, 0.75f);
263 }
264
265 /* Convert to filterbank coefficients and pre-compute outer products */
266 pData->H_grid = realloc1d(pData->H_grid, HYBRID_BANDS*(pData->Q)*pData->nGrid*sizeof(float_complex));
267 afSTFT_FIRtoFilterbankCoeffs(pData->h_grid, pData->nGrid, pData->Q, pData->h_len, HOP_SIZE, 0, 1, pData->H_grid);
268 pData->weights = realloc1d(pData->weights, pData->nGrid*sizeof(float));
269 getVoronoiWeights(pData->grid_dirs_deg, pData->nGrid, 0, pData->weights);
270 cblas_sscal(pData->nGrid, 1.0f/FOURPI, pData->weights, 1);
271 for(band=0; band<HYBRID_BANDS; band++){
272 pData->HHH[band] = (float_complex**)realloc2d((void**)pData->HHH[band], pData->nGrid, pData->Q * (pData->Q), sizeof(float_complex));
273 for(ng=0; ng<pData->nGrid; ng++){
274 for(q=0; q<pData->Q; q++)
275 H_tmp[q] = pData->H_grid[band*(pData->Q)*pData->nGrid + q*pData->nGrid + ng];
276 cblas_cgemm(CblasRowMajor, CblasNoTrans, CblasConjTrans, pData->Q, pData->Q, 1, &calpha,
277 H_tmp, 1,
278 H_tmp, 1, &cbeta,
279 pData->HHH[band][ng], pData->Q);
280 scaleC = cmplxf(pData->weights[ng], 0.0f);
281 cblas_cscal(pData->Q * (pData->Q), &scaleC, pData->HHH[band][ng], 1);
282 }
283 }
284 pData->angles = realloc1d(pData->angles, pData->nGrid*sizeof(float));
285
286 /* OM structures */
287 cdf4sap_cmplx_destroy(&(pData->hCdf));
288 cdf4sap_cmplx_create(&(pData->hCdf), pData->Q, pData->Q);
289 cdf4sap_destroy(&(pData->hCdf_res));
290 cdf4sap_create(&(pData->hCdf_res), pData->Q, pData->Q);
291 pData->Qmix = realloc1d(pData->Qmix, pData->Q*(pData->Q)*sizeof(float));
292 memset(pData->Qmix, 0, pData->Q*(pData->Q)*sizeof(float));
293 pData->Qmix_cmplx = realloc1d(pData->Qmix_cmplx, pData->Q*(pData->Q)*sizeof(float_complex));
294 memset(pData->Qmix_cmplx, 0, pData->Q*(pData->Q)*sizeof(float_complex));
295 for(q=0; q<pData->Q; q++){
296 pData->Qmix[q*(pData->Q)+q] = 1.0f;
297 pData->Qmix_cmplx[q*(pData->Q)+q] = cmplxf(1.0f, 0.0f);
298 }
299 pData->Cr = realloc1d(pData->Cr, pData->Q*(pData->Q)*sizeof(float));
300 pData->Cr_cmplx = realloc1d(pData->Cr_cmplx, pData->Q*(pData->Q)*sizeof(float_complex));
301
302 /* mixing matrices and buffers */
303 for(src=0; src<SPREADER_MAX_NUM_SOURCES; src++){
304 pData->Cy[src] = (float_complex**)realloc2d((void**)pData->Cy[src], HYBRID_BANDS, (pData->Q)*(pData->Q), sizeof(float_complex));
305 memset(FLATTEN2D(pData->Cy[src]), 0, HYBRID_BANDS * (pData->Q)*(pData->Q) * sizeof(float_complex));
306 pData->Cproto[src] = (float_complex**)realloc2d((void**)pData->Cproto[src], HYBRID_BANDS, (pData->Q)*(pData->Q), sizeof(float_complex));
307 memset(FLATTEN2D(pData->Cproto[src]), 0, HYBRID_BANDS * (pData->Q)*(pData->Q) * sizeof(float_complex));
308 pData->prev_M[src] = (float_complex**)realloc2d((void**)pData->prev_M[src], HYBRID_BANDS, (pData->Q)*(pData->Q), sizeof(float_complex));
309 memset(FLATTEN2D(pData->prev_M[src]), 0, HYBRID_BANDS * (pData->Q)*(pData->Q) * sizeof(float_complex));
310 pData->prev_Mr[src] = (float**)realloc2d((void**)pData->prev_Mr[src], HYBRID_BANDS, (pData->Q)*(pData->Q), sizeof(float));
311 memset(FLATTEN2D(pData->prev_Mr[src]), 0, HYBRID_BANDS * (pData->Q)*(pData->Q) * sizeof(float));
312 pData->dirActive[src] = realloc1d(pData->dirActive[src], pData->nGrid * sizeof(int));
313 memset(pData->dirActive[src], 0, pData->nGrid*sizeof(int));
314 }
315 pData->new_M = (float_complex**)realloc2d((void**)pData->new_M, HYBRID_BANDS, (pData->Q)*(pData->Q), sizeof(float_complex));
316 pData->new_Mr = (float**)realloc2d((void**)pData->new_Mr, HYBRID_BANDS, (pData->Q)*(pData->Q), sizeof(float));
317 pData->interp_M = realloc1d(pData->interp_M, (pData->Q)*(pData->Q) * sizeof(float_complex));
318 pData->interp_Mr = realloc1d(pData->interp_Mr, (pData->Q)*(pData->Q) * sizeof(float));
319 pData->interp_Mr_cmplx = realloc1d(pData->interp_Mr_cmplx, (pData->Q)*(pData->Q) * sizeof(float_complex));
320 memset(pData->interp_Mr_cmplx, 0, (pData->Q)*(pData->Q) * sizeof(float_complex));
321
322 /* New config */
323 pData->nSources = nSources;
324 pData->procMode = procMode;
325
326 /* done! */
327 strcpy(pData->progressBarText,"Done!");
328 pData->progressBar0_1 = 1.0f;
329 pData->codecStatus = CODEC_STATUS_INITIALISED;
330}

332void spreader_process
333(
334 void * const hSpr,
335 const float *const * inputs,
336 float ** const outputs,
337 int nInputs,
338 int nOutputs,
339 int nSamples
340)
341{
342 spreader_data *pData = (spreader_data*)(hSpr);
343 int q, src, ng, ch, i, j, band, t, nSources, Q, centre_ind, nSpread;
344 float trace, Ey, Eproto, Gcomp;
345 float src_dirs_deg[SPREADER_MAX_NUM_SOURCES][2], src_dir_xyz[3], CprotoDiag[MAX_NUM_OUTPUTS*MAX_NUM_OUTPUTS], src_spread[MAX_NUM_OUTPUTS];
346 float_complex scaleC, tmp;
347 float_complex tmpFrame[MAX_NUM_CHANNELS][TIME_SLOTS], H_tmp[MAX_NUM_CHANNELS], Cy[MAX_NUM_CHANNELS*MAX_NUM_CHANNELS];
348 float_complex E_dir[MAX_NUM_CHANNELS*MAX_NUM_CHANNELS], V[MAX_NUM_OUTPUTS*MAX_NUM_OUTPUTS], D[MAX_NUM_OUTPUTS*MAX_NUM_OUTPUTS];
349 float_complex Cproto[MAX_NUM_OUTPUTS*MAX_NUM_OUTPUTS];
350#if 0
351 float_complex Cx[MAX_NUM_OUTPUTS*MAX_NUM_OUTPUTS];
352 float CxDiag[MAX_NUM_OUTPUTS*MAX_NUM_OUTPUTS];
353#endif
354 SPREADER_PROC_MODES procMode;
355 const float_complex calpha = cmplxf(1.0f, 0.0f), cbeta = cmplxf(0.0f, 0.0f);
356
357 /* copy user parameters to local variables */
358 procMode = pData->procMode;
359 nSources = pData->nSources;
360 Q = pData->Q;
361 memcpy((float*)src_dirs_deg, pData->src_dirs_deg, nSources*2*sizeof(float));
362 memcpy((float*)src_spread, pData->src_spread, nSources*sizeof(float));
363
364 /* apply binaural panner */
365 if ((nSamples == SPREADER_FRAME_SIZE) && (pData->codecStatus==CODEC_STATUS_INITIALISED) ){
366 pData->procStatus = PROC_STATUS_ONGOING;
367
368 /* Load time-domain data */
369 for(i=0; i < SAF_MIN(nSources,nInputs); i++)
370 utility_svvcopy(inputs[i], SPREADER_FRAME_SIZE, pData->inputFrameTD[i]);
371 for(; i<nSources; i++)
372 memset(pData->inputFrameTD[i], 0, SPREADER_FRAME_SIZE * sizeof(float));
373
374 /* Apply time-frequency transform (TFT) */
375 afSTFT_forward_knownDimensions(pData->hSTFT, pData->inputFrameTD, SPREADER_FRAME_SIZE, MAX_NUM_INPUTS, TIME_SLOTS, pData->inputframeTF);
376
377 /* Zero output buffer */
378 for(band=0; band<HYBRID_BANDS; band++)
379 memset(FLATTEN2D(pData->outputframeTF[band]), 0, Q*TIME_SLOTS*sizeof(float_complex));
380
381 /* Loop over sources */
382 for(src=0; src<nSources; src++){
383 /* Find the "spread" indices */
384 unitSph2cart(src_dirs_deg[src], 1, 1, src_dir_xyz);
385 cblas_sgemm(CblasRowMajor, CblasNoTrans, CblasNoTrans, pData->nGrid, 1, 3, 1.0f,
386 pData->grid_dirs_xyz, 3,
387 src_dir_xyz, 1, 0.0f,
388 pData->angles, 1);
389 for(i=0; i<pData->nGrid; i++)
390 pData->angles[i] = acosf(SAF_MIN(pData->angles[i], 0.9999999f))*180.0f/SAF_PI;
391 utility_siminv(pData->angles, pData->nGrid, &centre_ind);
392
393 /* Define Prototype signals */
394 switch(procMode){
395 case SPREADER_MODE_NAIVE: /* fall through */
396 case SPREADER_MODE_OM:
397 for(band=0; band<HYBRID_BANDS; band++){
398 if(pData->freqVector[band]<MAX_SPREAD_FREQ){
399 /* Loop over all angles, and sum the H_grid's within the spreading area */
400 memset(H_tmp, 0, Q*sizeof(float_complex));
401 for(ng=0,nSpread=0; ng<pData->nGrid; ng++){
402 if(pData->angles[ng] <= (src_spread[src]/2.0f)){
403 for(q=0; q<Q; q++)
404 H_tmp[q] = ccaddf(H_tmp[q], pData->H_grid[band*Q*pData->nGrid + q*pData->nGrid + ng]);
405 nSpread++;
406 pData->dirActive[src][ng] = 1;
407 }
408 else
409 pData->dirActive[src][ng] = 0;
410 }
411 }
412 else
413 nSpread = 0;
414
415 /* If no directions found in the spread area, then just include the nearest one */
416 if(nSpread==0){
417 for(q=0; q<Q; q++)
418 H_tmp[q] = pData->H_grid[band*Q*pData->nGrid + q*pData->nGrid + centre_ind];
419 nSpread=1;
420 }
421
422 /* Apply */
423 cblas_cgemm(CblasRowMajor, CblasNoTrans, CblasNoTrans, Q, TIME_SLOTS, 1, &calpha,
424 H_tmp, 1,
425 pData->inputframeTF[band][src], TIME_SLOTS, &cbeta,
426 FLATTEN2D(pData->protoframeTF[band]), TIME_SLOTS);
427
428 /* Scale by number of spreading directions */
429 cblas_sscal(/*re+im*/2*Q*TIME_SLOTS, 1.0f/(float)nSpread, (float*)FLATTEN2D(pData->protoframeTF[band]), 1);
430 }
431 break;
432#if 0
433 case SPREADER_MODE_OM:
434 /* Use the centre direction as the prototype */
435 for(band=0; band<HYBRID_BANDS; band++){
436 for(q=0; q<Q; q++)
437 H_tmp[q] = pData->H_grid[band*Q*pData->nGrid + q*pData->nGrid + centre_ind];
438 cblas_cgemm(CblasRowMajor, CblasNoTrans, CblasNoTrans, Q, TIME_SLOTS, 1, &calpha,
439 H_tmp, 1,
440 pData->inputframeTF[band][src], TIME_SLOTS, &cbeta,
441 FLATTEN2D(pData->protoframeTF[band]), TIME_SLOTS);
442 }
443 break;
444#endif
445
446 case SPREADER_MODE_EVD:
447 /* Replicate the mono signal for all Q channels */
448 for(band=0; band<HYBRID_BANDS; band++)
449 for(q=0; q<Q; q++)
450 memcpy(pData->protoframeTF[band][q], pData->inputframeTF[band][src], TIME_SLOTS*sizeof(float_complex));
451 break;
452 }
453
454 /* Main processing */
455 if(procMode==SPREADER_MODE_NAIVE) {
456 /* If naive mode, then we're already done... */
457 for(band=0; band<HYBRID_BANDS; band++)
458 memcpy(FLATTEN2D(pData->spreadframeTF[band]), FLATTEN2D(pData->protoframeTF[band]), Q*TIME_SLOTS*sizeof(float_complex));
459 }
460 else{
461 /* Apply decorrelation of prototype signals */
462 latticeDecorrelator_apply(pData->hDecor[src], pData->protoframeTF, TIME_SLOTS, pData->decorframeTF);
463
464 /* Compute prototype covariance matrix and average over time */
465 for(band=0; band<HYBRID_BANDS; band++){
466 cblas_cgemm(CblasRowMajor, CblasNoTrans, CblasConjTrans, Q, Q, TIME_SLOTS, &calpha,
467 FLATTEN2D(pData->protoframeTF[band]), TIME_SLOTS,
468 FLATTEN2D(pData->protoframeTF[band]), TIME_SLOTS, &cbeta,
469 Cproto, Q);
470 cblas_sscal(/*re+im*/2*Q*Q, pData->covAvgCoeff, (float*)pData->Cproto[src][band], 1);
471 cblas_saxpy(/*re+im*/2*Q*Q, 1.0f-pData->covAvgCoeff, (float*)Cproto, 1, (float*)pData->Cproto[src][band], 1);
472 }
473
474 /* Define target covariance matrices */
475 for(band=0; band<HYBRID_BANDS; band++){
476 /* Sum the H_array outer product matrices for the whole spreading area */
477 if(pData->freqVector[band]<MAX_SPREAD_FREQ){
478 memset(Cy, 0, Q*Q*sizeof(float_complex));
479 memset(H_tmp, 0, Q*sizeof(float_complex));
480 for(ng=0, nSpread=0; ng<pData->nGrid; ng++){
481 if(pData->angles[ng] <= (src_spread[src]/2.0f)){
482 cblas_caxpy(Q*Q, &calpha, pData->HHH[band][ng], 1, Cy, 1);
483 for(q=0; q<Q; q++)
484 H_tmp[q] = ccaddf(H_tmp[q], pData->H_grid[band*Q*pData->nGrid + q*pData->nGrid + ng]);
485 nSpread++;
486 pData->dirActive[src][ng] = 1;
487 }
488 else
489 pData->dirActive[src][ng] = 0;
490 }
491 }
492 else
493 nSpread = 0;
494
495 /* If no directions found in the spread area, then just include the nearest one */
496 if(nSpread==0) {
497 cblas_caxpy(Q*Q, &calpha, pData->HHH[band][centre_ind], 1, Cy, 1);
498 for(q=0; q<Q; q++)
499 H_tmp[q] = pData->H_grid[band*Q*pData->nGrid + q*pData->nGrid + centre_ind];
500 nSpread++;
501 }
502#if 1
503 /* Impose target energies too */
504 if (procMode == SPREADER_MODE_OM && pData->freqVector[band]<MAX_SPREAD_FREQ){
505 /* Normalise Cy */
506 trace = 0.0f;
507 for(q=0; q<Q; q++)
508 trace += crealf(Cy[q*Q+q]);
509 cblas_sscal(/*re+im*/2*Q*Q, 1.0f/(trace+2.23e-9f), (float*)Cy, 1);
510
511 /* Compute signals for the centre of the spread */
512 for(q=0; q<Q; q++)
513 H_tmp[q] = pData->H_grid[band*Q*pData->nGrid + q*pData->nGrid + centre_ind];
514 cblas_cgemm(CblasRowMajor, CblasNoTrans, CblasNoTrans, Q, TIME_SLOTS, 1, &calpha,
515 H_tmp, 1,
516 pData->inputframeTF[band][src], TIME_SLOTS, &cbeta,
517 tmpFrame, TIME_SLOTS);
518
519 /* Introduce their channel energies into the target */
520 cblas_cgemm(CblasRowMajor, CblasNoTrans, CblasConjTrans, Q, Q, TIME_SLOTS, &calpha,
521 tmpFrame, TIME_SLOTS,
522 tmpFrame, TIME_SLOTS, &cbeta,
523 E_dir, Q);
524 trace = 0.0f;
525 for(q=0; q<Q; q++)
526 trace += crealf(E_dir[q*Q+q]);
527 cblas_sscal(/*re+im*/2*Q*Q, trace, (float*)Cy, 1);
528 }
529#endif
530 /* Average over time */
531 cblas_sscal(/*re+im*/2*Q*Q, pData->covAvgCoeff, (float*)pData->Cy[src][band], 1);
532 cblas_saxpy(/*re+im*/2*Q*Q, 1.0f-pData->covAvgCoeff, (float*)Cy, 1, (float*)pData->Cy[src][band], 1);
533 }
534
535 /* Formulate mixing matrices */
536 switch(procMode){
537 case SPREADER_MODE_NAIVE: saf_print_error("Shouldn't have gotten this far?"); break;
538 case SPREADER_MODE_EVD:
539 /* For normalising the level of Cy */
540 Ey = Eproto = 0.0f;
541 for(band=0; band<HYBRID_BANDS; band++){
542 for(i=0; i<Q; i++){
543 Ey += crealf(pData->Cy[src][band][i*Q+i]);
544 Eproto += crealf(pData->Cproto[src][band][i*Q+i])+0.000001f;
545 }
546 }
547 Gcomp = sqrtf(Eproto/(Ey+2.23e-9f));
548
549 /* Compute mixing matrix per band */
550 for(band=0; band<HYBRID_BANDS; band++){
551 memcpy(Cy, pData->Cy[src][band], Q*Q*sizeof(float_complex));
552 cblas_sscal(/*re+im*/2*Q*Q, Gcomp, (float*)Cy, 1);
553 utility_cseig(NULL, Cy, Q, 1, V, D, NULL);
554 for(i=0; i<Q; i++)
555 for(j=0; j<Q; j++)
556 D[i*Q+j] = i==j ? csqrtf(D[i*Q+j]) : cmplxf(0.0f, 0.0f);
557 cblas_cgemm(CblasRowMajor, CblasNoTrans, CblasNoTrans, Q, Q, Q, &calpha,
558 V, Q,
559 D, Q, &cbeta,
560 pData->new_M[band], Q);
561 }
562 break;
563
564 case SPREADER_MODE_OM:
565 for(band=0; band<HYBRID_BANDS; band++){
566 if(pData->freqVector[band]<MAX_SPREAD_FREQ){
567#if 1
568 /* Diagonalise and diagonally load the Cproto matrices */
569 cblas_ccopy(Q*Q, pData->Cproto[src][band], 1, Cproto, 1);
570 for(i=0; i<Q; i++){
571 for(j=0; j<Q; j++){
572 if(i==j)
573 Cproto[i*Q+i] = craddf(Cproto[i*Q+i], 0.00001f);
574 CprotoDiag[i*Q+j] = i==j ? crealf(Cproto[i*Q+i]) : 0.0f;
575 }
576 }
577
578 /* Compute mixing matrices */
579 formulate_M_and_Cr_cmplx(pData->hCdf, Cproto, pData->Cy[src][band], pData->Qmix_cmplx, 0, 0.2f, pData->new_M[band], pData->Cr_cmplx);
580 for(i=0; i<Q*Q; i++)
581 pData->Cr[i] = crealf(pData->Cr_cmplx[i]);
582 formulate_M_and_Cr(pData->hCdf_res, CprotoDiag, pData->Cr, pData->Qmix, 0, 0.2f, pData->new_Mr[band], NULL);
583#else
584 for(q=0; q<Q; q++)
585 H_tmp[q] = pData->H_grid[band*Q*pData->nGrid + q*pData->nGrid + centre_ind];
586 cblas_cgemm(CblasRowMajor, CblasNoTrans, CblasConjTrans, Q, Q, 1, &calpha,
587 H_tmp, 1,
588 H_tmp, 1, &cbeta,
589 Cx, Q);
590 memset(CxDiag, 0, Q*Q*sizeof(float));
591 for(i=0; i<Q; i++)
592 CxDiag[i*Q+i] = crealf(Cx[i*Q+i]);
593
594 /* Compute mixing matrices */
595 formulate_M_and_Cr_cmplx(pData->hCdf, Cx, pData->Cy[src][band], pData->Qmix_cmplx, 0, 0.2f, pData->new_M[band], pData->Cr_cmplx);
596 for(i=0; i<Q*Q; i++)
597 pData->Cr[i] = crealf(pData->Cr_cmplx[i]);
598 formulate_M_and_Cr(pData->hCdf_res, CxDiag, pData->Cr, pData->Qmix, 0, 0.2f, pData->new_Mr[band], NULL);
599#endif
600 }
601 else{
602 memcpy(pData->new_M[band], pData->Qmix_cmplx, Q*Q*sizeof(float_complex));
603 memset(pData->new_Mr[band], 0, Q*Q*sizeof(float));
604 }
605 }
606 break;
607 }
608
609 /* Apply mixing matrices */
610 for(band=0; band<HYBRID_BANDS; band++){
611 for(t=0; t<TIME_SLOTS; t++){
612 scaleC = cmplxf(pData->interpolatorFadeIn[t], 0.0f);
613 utility_cvsmul(pData->new_M[band], &scaleC, Q*Q, pData->interp_M);
614 cblas_saxpy(/*re+im*/2*Q*Q, pData->interpolatorFadeOut[t], (float*)pData->prev_M[src][band], 1, (float*)pData->interp_M, 1);
615 for(i=0; i<Q; i++) {
616 cblas_cdotu_sub(Q, (float_complex*)(&(pData->interp_M[i*Q])), 1,
617 FLATTEN2D((procMode == SPREADER_MODE_EVD ? pData->decorframeTF[band] : pData->protoframeTF[band])) + t,
618 TIME_SLOTS, &(pData->spreadframeTF[band][i][t]));
619 }
620 }
621
622 /* Also mix in the residual part */
623 if(procMode == SPREADER_MODE_OM){
624 if(pData->freqVector[band]<MAX_SPREAD_FREQ){
625 for(t=0; t<TIME_SLOTS; t++){
626 utility_svsmul(pData->new_Mr[band], &(pData->interpolatorFadeIn[t]), Q*Q, pData->interp_Mr);
627 cblas_saxpy(Q*Q, pData->interpolatorFadeOut[t], pData->prev_Mr[src][band], 1, pData->interp_Mr, 1);
628 cblas_scopy(Q*Q, pData->interp_Mr, 1, (float*)pData->interp_Mr_cmplx, 2);
629 for(i=0; i<Q; i++){
630 cblas_cdotu_sub(Q, (float_complex*)(&(pData->interp_Mr_cmplx[i*Q])), 1, FLATTEN2D(pData->decorframeTF[band]) + t, TIME_SLOTS, &tmp);
631 pData->spreadframeTF[band][i][t] = ccaddf(pData->spreadframeTF[band][i][t], tmp);
632 }
633 }
634 }
635 }
636 }
637 }
638
639 /* Add the spread frame to the output frame, then move onto the next source... */
640 for(band=0; band<HYBRID_BANDS; band++)
641 cblas_saxpy(/*re+im*/2*Q*TIME_SLOTS, 1.0f, (float*)FLATTEN2D(pData->spreadframeTF[band]), 1, (float*)FLATTEN2D(pData->outputframeTF[band]), 1);
642
643 /* For next frame */
644 cblas_ccopy(HYBRID_BANDS*Q*Q, FLATTEN2D(pData->new_M), 1, FLATTEN2D(pData->prev_M[src]), 1);
645 cblas_scopy(HYBRID_BANDS*Q*Q, FLATTEN2D(pData->new_Mr), 1, FLATTEN2D(pData->prev_Mr[src]), 1);
646 }
647
648 /* inverse-TFT */
649 afSTFT_backward_knownDimensions(pData->hSTFT, pData->outputframeTF, SPREADER_FRAME_SIZE, MAX_NUM_OUTPUTS, TIME_SLOTS, pData->outframeTD);
650
651 /* Copy to output buffer */
652 for (ch = 0; ch < SAF_MIN(Q, nOutputs); ch++)
653 utility_svvcopy(pData->outframeTD[ch], SPREADER_FRAME_SIZE, outputs[ch]);
654 for (; ch < nOutputs; ch++)
655 memset(outputs[ch], 0, SPREADER_FRAME_SIZE*sizeof(float));
656 }
657 else{
658 for (ch=0; ch < nOutputs; ch++)
659 memset(outputs[ch],0, SPREADER_FRAME_SIZE*sizeof(float));
660 }
661
662 pData->procStatus = PROC_STATUS_NOT_ONGOING;
663}
