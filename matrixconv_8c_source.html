28void matrixconv_create
29(
30 void ** const phMCnv
31)
32{

34 *phMCnv = (void*)pData;
35

37 pData->nInputChannels = 1;

96void matrixconv_process
97(
98 void * const hMCnv,
99 const float *const * inputs,
100 float* const* const outputs,
101 int nInputs,
102 int nOutputs,
103 int nSamples
104)
105{
106 matrixconv_data *pData = (matrixconv_data*)(hMCnv);
107 int s, ch, i;
108 int numInputChannels, numOutputChannels;
109
110 matrixconv_checkReInit(hMCnv);
111
112 /* prep */
113 numInputChannels = pData->nInputChannels;
114 numOutputChannels = pData->nOutputChannels;
115
116 for(s=0; s<nSamples; s++){
117 /* Load input signals into inFIFO buffer */
118 for(ch=0; ch<SAF_MIN(SAF_MIN(nInputs,numInputChannels),MAX_NUM_CHANNELS); ch++)
119 pData->inFIFO[ch][pData->FIFO_idx] = inputs[ch][s];
120 for(; ch<numInputChannels; ch++) /* Zero any channels that were not given */
121 pData->inFIFO[ch][pData->FIFO_idx] = 0.0f;
122
123 /* Pull output signals from outFIFO buffer */
124 for(ch=0; ch<SAF_MIN(SAF_MIN(nOutputs, numOutputChannels),MAX_NUM_CHANNELS); ch++)
125 outputs[ch][s] = pData->outFIFO[ch][pData->FIFO_idx];
126 for(; ch<nOutputs; ch++) /* Zero any extra channels */
127 outputs[ch][s] = 0.0f;
128
129 /* Increment buffer index */
130 pData->FIFO_idx++;
131
132 /* Process frame if inFIFO is full and filters are loaded and saf_matrixConv_apply is ready for it */
133 if (pData->FIFO_idx >= pData->hostBlockSize_clamped && pData->reInitFilters == 0 ) {
134 pData->FIFO_idx = 0;
135
136 /* Load time-domain data */
137 for(i=0; i < numInputChannels; i++)
138 utility_svvcopy(pData->inFIFO[i], pData->hostBlockSize_clamped, pData->inputFrameTD[i]);
139
140 /* Apply matrix convolution */
141 if(pData->hMatrixConv != NULL && pData->filter_length>0)
142 saf_matrixConv_apply(pData->hMatrixConv, FLATTEN2D(pData->inputFrameTD), FLATTEN2D(pData->outputFrameTD));
143 /* if the matrix convolver handle has not been initialised yet (i.e. no filters have been loaded) then zero the output */
144 else
145 memset(FLATTEN2D(pData->outputFrameTD), 0, MAX_NUM_CHANNELS * (pData->hostBlockSize_clamped)*sizeof(float));
146
147 /* copy signals to output buffer */
148 for (i = 0; i < SAF_MIN(numOutputChannels, MAX_NUM_CHANNELS); i++)
149 utility_svvcopy(pData->outputFrameTD[i], pData->hostBlockSize_clamped, pData->outFIFO[i]);
150 }
151 else if(pData->FIFO_idx >= pData->hostBlockSize_clamped){
152 /* clear outFIFO if codec was not ready */
153 pData->FIFO_idx = 0;
154 memset(pData->outFIFO, 0, MAX_NUM_CHANNELS*MAX_FRAME_SIZE*sizeof(float));
155 }
156 }
157}

168void matrixconv_checkReInit(void* const hMCnv)
169{
170 matrixconv_data *pData = (matrixconv_data*)(hMCnv);
171
172 /* reinitialise if needed */
173 if ((pData->reInitFilters == 1) && (pData->filters != NULL)) {
174 pData->reInitFilters = 2;
175 saf_matrixConv_destroy(&(pData->hMatrixConv));
176 pData->hMatrixConv = NULL;
177
178 /* if length of the loaded wav file was not divisable by the specified number of inputs, then the handle remains NULL,
179 * and no convolution is applied */
180 pData->hostBlockSize_clamped = SAF_CLAMP(pData->hostBlockSize, MIN_FRAME_SIZE, MAX_FRAME_SIZE);
181 if(pData->filter_length>0){
182 saf_matrixConv_create(&(pData->hMatrixConv),
183 pData->hostBlockSize_clamped, /*pData->hostBlockSize,*/
184 pData->filters,
185 pData->filter_length,
186 pData->nInputChannels,
187 pData->nOutputChannels,
188 pData->enablePartitionedConv);
189 }
190
191 /* Resize buffers */
192 pData->inputFrameTD = (float**)realloc2d((void**)pData->inputFrameTD, MAX_NUM_CHANNELS, pData->hostBlockSize_clamped, sizeof(float));
193 pData->outputFrameTD = (float**)realloc2d((void**)pData->outputFrameTD, MAX_NUM_CHANNELS, pData->hostBlockSize_clamped, sizeof(float));
194 memset(FLATTEN2D(pData->inputFrameTD), 0, MAX_NUM_CHANNELS*(pData->hostBlockSize_clamped)*sizeof(float));
195
196 /* reset FIFO buffers */
197 pData->FIFO_idx = 0;
198 memset(pData->inFIFO, 0, MAX_NUM_CHANNELS*MAX_FRAME_SIZE*sizeof(float));
199 memset(pData->outFIFO, 0, MAX_NUM_CHANNELS*MAX_FRAME_SIZE*sizeof(float));
200
201 pData->reInitFilters = 0;
202 }

205void matrixconv_setFilters
206(
207 void* const hMCnv,
208 const float* const* H,
209 int numChannels,
210 int numSamples,
211 int sampleRate
212)
213{
214 matrixconv_data *pData = (matrixconv_data*)(hMCnv);
215 int i;
216 saf_assert(numChannels<=MAX_NUM_CHANNELS_FOR_WAV && numChannels > 0 && numSamples > 0, "WAV is limited to 1024 channels");
217
218 pData->nOutputChannels = SAF_MIN(numChannels, MAX_NUM_CHANNELS);
219 pData->input_wav_length = numSamples;
220 pData->nfilters = (pData->nOutputChannels) * (pData->nInputChannels);
221
222 /* store the loaded filters */
223 pData->filters = realloc1d(pData->filters, numChannels * numSamples * sizeof(float));
224 for(i=0; i<numChannels; i++)
225 memcpy(&(pData->filters[i*numSamples]), H[i], numSamples * sizeof(float));
226 pData->filter_fs = sampleRate;
227
228 /* if the number of samples in loaded data is not divisable by the currently specified number of
229 * inputs, then the filter length is set to 0 and no further processing is conducted. */
230 if(pData->input_wav_length % pData->nInputChannels == 0)
231 pData->filter_length = (pData->input_wav_length) / (pData->nInputChannels);
232 else
233 pData->filter_length = 0;
234
235 pData->reInitFilters = 1;
236}

248{

250 pData->nInputChannels = SAF_CLAMP(newValue, 1, MAX_NUM_CHANNELS);
251 pData->nfilters = (pData->nOutputChannels) * (pData->nInputChannels);
252 if((pData->nOutputChannels > 0) && (pData->input_wav_length % pData->nInputChannels == 0))
253 pData->filter_length = (pData->input_wav_length) / (pData->nInputChannels);
254 else
255 pData->filter_length = 0;
