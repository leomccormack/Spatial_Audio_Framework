38 void estimateITDs
39 (
40  float* hrirs /* N_dirs x NUM_EARS x hrir_len */,
41  int N_dirs,
42  int hrir_len,
43  int fs,
44  float* itds_s
45 )
46 {
47  int i, n, j, k, maxIdx, xcorr_len;
48  float maxVal, itd_bounds, fc, Q, K, KK, D, wn, Wz1[2], Wz2[2], b[3], a[3];
49  float* xcorr_LR, *ir_L, *ir_R, *hrir_lpf;
50 
51  /* calculate LPF coefficients, 2nd order IIR design equations from DAFX (2nd ed) p50 */
52  fc = 750.0f;
53  Q = 0.7071f;
54  K = tanf(SAF_PI * fc/(float)fs);
55  KK = K * K;
56  D = KK * Q + K + Q;
57  b[0] = (KK * Q) / D; b[1] = (2.0f * KK * Q) / D; b[2] = (KK * Q) / D;
58  a[0] = 1.0f; a[1] = (2.0f * Q * (KK - 1.0f)) / D; a[2] = (KK * Q - K + Q) / D;
59 
60  /* determine the ITD via the cross-correlation between the LPF'd left and right HRIR signals */
61  xcorr_len = 2*(hrir_len)-1;
62  itd_bounds = sqrtf(2.0f)/2e3f;
63  xcorr_LR = (float*)malloc1d(xcorr_len*sizeof(float));
64  ir_L = (float*)malloc1d(hrir_len*sizeof(float));
65  ir_R = (float*)malloc1d(hrir_len*sizeof(float));
66  hrir_lpf = (float*)malloc1d(hrir_len*2*sizeof(float));
67  for(i=0; i<N_dirs; i++){
68  /* apply lpf */
69  memset(Wz1, 0, 2*sizeof(float));
70  memset(Wz2, 0, 2*sizeof(float));
71  for (n=0; n<hrir_len; n++){
72  for(j=0; j<NUM_EARS; j++){
73  /* biquad difference equation (Direct form 2) */
74  wn = hrirs[i*NUM_EARS*hrir_len + j*hrir_len + n] - a[1] * Wz1[j] - a[2] * Wz2[j];
75  hrir_lpf[n*2+j] = b[0] * wn + b[1]*Wz1[j] + b[2]*Wz2[j];
76 
77  /* shuffle delays */
78  Wz2[j] = Wz1[j];
79  Wz1[j] = wn;
80  }
81  }
82 
83  /* xcorr between L and R */
84  for(k=0; k<hrir_len; k++){
85  ir_L[k] = hrir_lpf[k*2+0];
86  ir_R[k] = hrir_lpf[k*2+1];
87  }
88  cxcorr(ir_L, ir_R, xcorr_LR, hrir_len, hrir_len);
89  maxIdx = 0;
90  maxVal = 0.0f;
91  for(j=0; j<xcorr_len; j++){
92  if(xcorr_LR[j] > maxVal){
93  maxIdx = j;
94  maxVal = xcorr_LR[j];
95  }
96  }
97  itds_s[i] = ((float)hrir_len-(float)maxIdx-1.0f)/(float)fs;
98  itds_s[i] = itds_s[i] > itd_bounds ? itd_bounds : itds_s[i];
99  itds_s[i] = itds_s[i] < -itd_bounds ? -itd_bounds : itds_s[i];
100  }
101 
102  free(xcorr_LR);
103  free(ir_L);
104  free(ir_R);
105  free(hrir_lpf);
106 }

136 void HRIRs2HRTFs
137 (
138  float* hrirs, /* N_dirs x NUM_EARS x hrir_len */
139  int N_dirs,
140  int hrir_len,
141  int fftSize,
142  float_complex* hrtfs /* (fftSize/2+1) x NUM_EARS x N_dirs */
143 )
144 {
145  int i, j, k, nBins;
146  void* hSafFFT;
147  float* hrir_pad;
148  float_complex* hrtf;
149 
150  //TODO: if fftSize is shorter than hrir_len, maybe truncate based on the median peak?
151  /* Perform FFT */
152  nBins = fftSize/2 + 1;
153  saf_rfft_create(&hSafFFT, fftSize);
154  hrir_pad = calloc1d(fftSize, sizeof(float));
155  hrtf = malloc1d(nBins*sizeof(float_complex));
156  for(i=0; i<N_dirs; i++){
157  for(j=0; j<NUM_EARS; j++){
158  memcpy(hrir_pad, &hrirs[i*NUM_EARS*hrir_len+j*hrir_len], MIN(fftSize, hrir_len)*sizeof(float));
159  saf_rfft_forward(hSafFFT, hrir_pad, hrtf);
160  for(k=0; k<nBins; k++)
161  hrtfs[k*NUM_EARS*N_dirs + j*N_dirs + i] = hrtf[k];
162  }
163  }
164 
165  saf_rfft_destroy(&hSafFFT);
166  free(hrir_pad);
167  free(hrtf);
168 }

170 void diffuseFieldEqualiseHRTFs
171 (
172  int N_dirs,
173  float* itds_s,
174  float* centreFreq,
175  int N_bands,
176  float* weights,
177  int applyEQ,
178  int applyPhase,
179  float_complex* hrtfs /* N_bands x #NUM_EARS x N_dirs */
180 )
181 {
182  /* Anything to do at all? */
183  if(applyEQ + applyPhase)
184  {
185  int i, j, nd, band;
186  float* ipd, *hrtf_diff, *_weights;
187 
188  /* diffuse-field equalise */
189  if(applyEQ){
190  hrtf_diff = calloc1d(N_bands*NUM_EARS, sizeof(float));
191  if(weights == NULL){
192  _weights = malloc1d(N_dirs*sizeof(float));
193  for(int idx=0; idx < N_dirs; idx++)
194  _weights[idx] = 4.f*SAF_PI / (float)N_dirs;
195  }
196  else
197  _weights = weights;
198  for(band=0; band<N_bands; band++)
199  for(i=0; i<NUM_EARS; i++)
200  for(j=0; j<N_dirs; j++)
201  hrtf_diff[band*NUM_EARS + i] += _weights[j]/(4.f*SAF_PI) * powf(cabsf(hrtfs[band*NUM_EARS*N_dirs + i*N_dirs + j]), 2.0f);
202  for(band=0; band<N_bands; band++)
203  for(i=0; i<NUM_EARS; i++)
204  hrtf_diff[band*NUM_EARS + i] = sqrtf(hrtf_diff[band*NUM_EARS + i]);
205  for(band=0; band<N_bands; band++)
206  for(i=0; i<NUM_EARS; i++)
207  for(nd=0; nd<N_dirs; nd++)
208  hrtfs[band*NUM_EARS*N_dirs + i*N_dirs + nd] = ccdivf(hrtfs[band*NUM_EARS*N_dirs + i*N_dirs + nd], cmplxf(hrtf_diff[band*NUM_EARS + i] + 2.23e-8f, 0.0f));
209  free(hrtf_diff);
210  if(weights==NULL)
211  free(_weights);
212  }
213 
214  /* Create complex HRTFs by introducing the interaural phase differences
215  * (IPDs) to the HRTF magnitude responses */
216  if(applyPhase){
217  /* convert ITDs to phase differences -pi..pi */
218  ipd = malloc1d(N_bands*N_dirs*sizeof(float));
219  cblas_sgemm(CblasRowMajor, CblasNoTrans, CblasTrans, N_bands, N_dirs, 1, 1.0,
220  centreFreq, 1,
221  itds_s, 1, 0.0,
222  ipd, N_dirs);
223  for(i=0; i<N_bands; i++)
224  for(j=0; j<N_dirs; j++)
225  ipd[i*N_dirs+j] = (matlab_fmodf(2.0f*SAF_PI*ipd[i*N_dirs+j] + SAF_PI, 2.0f*SAF_PI) - SAF_PI)/2.0f; /* /2 here, not later */
226 
227  for(band=0; band<N_bands; band++){
228  for(nd=0; nd<N_dirs; nd++){
229  hrtfs[band*NUM_EARS*N_dirs + 0*N_dirs + nd] = crmulf( cexpf(cmplxf(0.0f, ipd[band*N_dirs + nd])), cabsf(hrtfs[band*NUM_EARS*N_dirs + 0*N_dirs + nd]) );
230  hrtfs[band*NUM_EARS*N_dirs + 1*N_dirs + nd] = crmulf( cexpf(cmplxf(0.0f,-ipd[band*N_dirs + nd])), cabsf(hrtfs[band*NUM_EARS*N_dirs + 1*N_dirs + nd]) );
231  }
232  }
233  free(ipd);
234  }
235  }

239 void interpHRTFs
240 (
241  float_complex* hrtfs, /* N_bands x 2 x N_hrtf_dirs */
242  float* itds,
243  float* freqVector,
244  float* vbap_gtable,
245  int N_hrtf_dirs,
246  int N_bands,
247  int N_interp_dirs,
248  float_complex* hrtfs_interp /* pre-alloc, N_bands x 2 x N_interp_dirs */
249 )
250 {
251  int i, band;
252  float* itd_interp, *mags_interp, *ipd_interp;
253  float** mags;
254 
255  mags = (float**)malloc1d(N_bands*sizeof(float*));
256  itd_interp = malloc1d(N_interp_dirs*sizeof(float));
257  mags_interp = malloc1d(N_interp_dirs*NUM_EARS*sizeof(float));
258  ipd_interp = malloc1d(N_interp_dirs*sizeof(float));
259 
260  /* calculate HRTF magnitudes */
261  for(band=0; band<N_bands; band++){
262  mags[band] = malloc1d(NUM_EARS * N_hrtf_dirs*sizeof(float));
263  for(i=0; i< NUM_EARS * N_hrtf_dirs ; i++)
264  mags[band][i] = cabsf(hrtfs[band*NUM_EARS * N_hrtf_dirs + i]);
265  }
266 
267  /* interpolate ITDs */
268  cblas_sgemm(CblasRowMajor, CblasNoTrans, CblasNoTrans, N_interp_dirs, 1, N_hrtf_dirs, 1.0f,
269  vbap_gtable, N_hrtf_dirs,
270  itds, 1, 0.0f,
271  itd_interp, 1);
272  for(band=0; band<N_bands; band++){
273  /* interpolate HRTF magnitudes */
274  cblas_sgemm(CblasRowMajor, CblasNoTrans, CblasTrans, N_interp_dirs, NUM_EARS, N_hrtf_dirs, 1.0f,
275  vbap_gtable, N_hrtf_dirs,
276  mags[band], N_hrtf_dirs, 0.0f,
277  mags_interp, NUM_EARS);
278 
279  /* convert ITDs to phase differences -pi..pi */
280  for(i=0; i<N_interp_dirs; i++)
281  ipd_interp[i] = (matlab_fmodf(2.0f*SAF_PI*freqVector[band]*itd_interp[i] + SAF_PI, 2.0f*SAF_PI) - SAF_PI)/2.0f; /* /2 here, not later */
282 
283  /* reintroduce the interaural phase differences (IPD) */
284  for(i=0; i<N_interp_dirs; i++){
285  hrtfs_interp[band*NUM_EARS*N_interp_dirs + 0* N_interp_dirs +i] = ccmulf( cmplxf(mags_interp[i*NUM_EARS+0],0.0f), cexpf(cmplxf(0.0f, ipd_interp[i])) );
286  hrtfs_interp[band*NUM_EARS*N_interp_dirs + 1* N_interp_dirs +i] = ccmulf( cmplxf(mags_interp[i*NUM_EARS+1],0.0f), cexpf(cmplxf(0.0f,-ipd_interp[i])) );
287  }
288  }
289 
290  free(itd_interp);
291  for(band=0; band<N_bands; band++)
292  free(mags[band]);
293  free(mags);
294  free(mags_interp);
295  free(ipd_interp);
296 }

298 void binauralDiffuseCoherence
299 (
300  float_complex* hrtfs, /* N_bands x 2 x N_hrtf_dirs */
301  float* itds,
302  float* freqVector,
303  int N_hrtf_dirs,
304  int N_bands,
305  float* HRTFcoh
306 )
307 {
308  int i, j;
309  float* ipd;
310  float_complex *hrtf_ipd_lr;
311 
312  /* convert ITDs to phase differences -pi..pi */
313  ipd = malloc1d(N_bands*N_hrtf_dirs*sizeof(float));
314  cblas_sgemm(CblasRowMajor, CblasNoTrans, CblasTrans, N_bands, N_hrtf_dirs, 1, 1.0,
315  freqVector, 1,
316  itds, 1, 0.0,
317  ipd, N_hrtf_dirs);
318  for(i=0; i<N_bands; i++)
319  for(j=0; j<N_hrtf_dirs; j++)
320  ipd[i*N_hrtf_dirs+j] = (matlab_fmodf(2.0f*SAF_PI*ipd[i*N_hrtf_dirs+j] + SAF_PI, 2.0f*SAF_PI) - SAF_PI);
321 
322  /* compute complex coherence */
323  hrtf_ipd_lr = calloc1d(N_bands, sizeof(float_complex));
324  for(i=0; i<N_bands; i++){
325  for(j=0; j<N_hrtf_dirs; j++)
326  hrtf_ipd_lr[i] = ccaddf(hrtf_ipd_lr[i], crmulf(cexpf(crmulf(cmplxf(0.0f, 1.0f), ipd[i*N_hrtf_dirs+j])),
327  cabsf(hrtfs[i*NUM_EARS*N_hrtf_dirs + 0*N_hrtf_dirs + j])*
328  cabsf(hrtfs[i*NUM_EARS*N_hrtf_dirs + 1*N_hrtf_dirs + j])));
329  hrtf_ipd_lr[i] = ccdivf(hrtf_ipd_lr[i], cmplxf((float)N_hrtf_dirs, 0.0f));
330  }
331 
332  /* due to almost axisymmetry of ITD, the coherence is almost real */
333  for(i=0; i<N_bands; i++)
334  HRTFcoh[i] = crealf(hrtf_ipd_lr[i]) < 0.0f ? 0.0f : crealf(hrtf_ipd_lr[i]);
335  HRTFcoh[0] = 1.0f; /* force 1 at DC */
336 
337  free(ipd);
338  free(hrtf_ipd_lr);
339 }
