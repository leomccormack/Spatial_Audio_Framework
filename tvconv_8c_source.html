27void tvconv_create
28(
29 void** const phTVCnv
30)
31{
32 tvconv_data* pData = (tvconv_data*)malloc1d(sizeof(tvconv_data));
33 *phTVCnv = (void*)pData;
34
35 /* Default user parameters */
36 pData->nInputChannels = 1;
37
38 /* internal values */
39 pData->hostBlockSize = -1; /* force initialisation */
40 pData->inputFrameTD = NULL;
41 pData->outputFrameTD = NULL;
42 pData->hTVConv = NULL;
43 pData->irs = NULL;
44 pData->reInitFilters = 1;
45 pData->nIrChannels = 0;
46 pData->ir_length = 0;
47 pData->ir_fs = 0;
48 pData->nOutputChannels = 0;
49 pData->sofa_filepath = NULL;
50
51 /* set FIFO buffers */
52 pData->FIFO_idx = 0;
53 memset(pData->inFIFO, 0, MAX_NUM_CHANNELS*MAX_FRAME_SIZE*sizeof(float));
54 memset(pData->outFIFO, 0, MAX_NUM_CHANNELS*MAX_FRAME_SIZE*sizeof(float));
55
56 /* positions */
57 pData->listenerPositions = NULL;
58 pData->nListenerPositions = 0;
59 pData->position_idx = 0;
60 for (int d = 0; d < NUM_DIMENSIONS; d++){
61 pData->sourcePosition[d] = 0.0f;
62 pData->targetPosition[d] = 0.0f;
63 pData->minDimensions[d] = 0.0f;
64 pData->maxDimensions[d] = 0.0f;
65 }
66
67 /* flags/status */
68 pData->progressBar0_1 = 0.0f;
69 pData->progressBarText = malloc1d(PROGRESSBARTEXT_CHAR_LENGTH*sizeof(char));
70 strcpy(pData->progressBarText,"");
71 pData->codecStatus = CODEC_STATUS_NOT_INITIALISED;
72 pData->procStatus = PROC_STATUS_NOT_ONGOING;
73}

75void tvconv_destroy
76(
77 void** const phTVCnv
78)
79{
80 tvconv_data* pData = (tvconv_data*)(*phTVCnv);
81
82 if (pData != NULL){
83 /* not safe to free memory during intialisation/processing loop */
84 while (pData->codecStatus == CODEC_STATUS_INITIALISING ||
85 pData->procStatus == PROC_STATUS_ONGOING){
86 SAF_SLEEP(10);
87 }
88
89 free(pData->inputFrameTD);
90 free(pData->outputFrameTD);
91 free(pData->irs);
92 free(pData->listenerPositions);
93 saf_TVConv_destroy(&(pData->hTVConv));
94 free(pData);
95 pData = NULL;
96 }
97}

100(

105{

119void tvconv_process
120(
121 void * const hTVCnv,
122 float ** const inputs,
123 float ** const outputs,
124 int nInputs,
125 int nOutputs,
126 int nSamples
127)
128{
129 tvconv_data *pData = (tvconv_data*)(hTVCnv);
130 int s, ch, i;
131 int numInputChannels, numOutputChannels;
132
133 tvconv_checkReInit(hTVCnv);
134 pData->procStatus = PROC_STATUS_ONGOING;
135
136 numInputChannels = pData->nInputChannels;
137 numOutputChannels = pData->nOutputChannels;
138
139 for(s=0; s<nSamples; s++){
140 /* Load input signals into inFIFO buffer */
141 for(ch=0; ch<SAF_MIN(SAF_MIN(nInputs,numInputChannels),MAX_NUM_CHANNELS); ch++)
142 pData->inFIFO[ch][pData->FIFO_idx] = inputs[ch][s];
143 for(; ch<numInputChannels; ch++) /* Zero any channels that were not given */
144 pData->inFIFO[ch][pData->FIFO_idx] = 0.0f;
145
146 /* Pull output signals from outFIFO buffer */
147 for(ch=0; ch<SAF_MIN(SAF_MIN(nOutputs, numOutputChannels),MAX_NUM_CHANNELS); ch++)
148 outputs[ch][s] = pData->outFIFO[ch][pData->FIFO_idx];
149 for(; ch<nOutputs; ch++) /* Zero any extra channels */
150 outputs[ch][s] = 0.0f;
151
152 /* Increment buffer index */
153 pData->FIFO_idx++;
154
155 /* Process frame if inFIFO is full and filters are loaded and saf_matrixConv_apply is ready for it */
156 if (pData->FIFO_idx >= pData->hostBlockSize_clamped && pData->reInitFilters == 0 &&
157 pData->codecStatus == CODEC_STATUS_INITIALISED) {
158 pData->FIFO_idx = 0;
159
160 /* Load time-domain data */
161 for(i=0; i < numInputChannels; i++)
162 utility_svvcopy(pData->inFIFO[i], pData->hostBlockSize_clamped, pData->inputFrameTD[i]);
163
164 if(pData->hTVConv != NULL && pData->ir_length>0){
165 saf_TVConv_apply(pData->hTVConv,
166 FLATTEN2D(pData->inputFrameTD),
167 FLATTEN2D(pData->outputFrameTD),
168 pData->position_idx);
169 }
170 /* if the matrix convolver handle has not been initialised yet (i.e. no filters have been loaded) then zero the output */
171 else
172 memset(FLATTEN2D(pData->outputFrameTD), 0, MAX_NUM_CHANNELS * (pData->hostBlockSize_clamped)*sizeof(float));
173
174 /* copy signals to output buffer */
175 for (i = 0; i < SAF_MIN(numOutputChannels, MAX_NUM_CHANNELS); i++)
176 utility_svvcopy(pData->outputFrameTD[i], pData->hostBlockSize_clamped, pData->outFIFO[i]);
177 }
178 else if(pData->FIFO_idx >= pData->hostBlockSize_clamped){
179 /* clear outFIFO if codec was not ready */
180 pData->FIFO_idx = 0;
181 memset(pData->outFIFO, 0, MAX_NUM_CHANNELS*MAX_FRAME_SIZE*sizeof(float));
182 }
183 }
184 pData->procStatus = PROC_STATUS_NOT_ONGOING;
185}

197void tvconv_checkReInit(void* const hTVCnv)
198{
199 tvconv_data *pData = (tvconv_data*)(hTVCnv);
200
201 while (pData->procStatus == CODEC_STATUS_INITIALISING){
202 SAF_SLEEP(10);
203 }
204 /* reinitialise if needed */
205 if ((pData->reInitFilters == 1) && (pData->irs != NULL)) {
206 pData->reInitFilters = 2;
207 saf_TVConv_destroy(&(pData->hTVConv));
208 pData->hTVConv = NULL;
209 /* if length of the loaded sofa file was not divisable by the specified number of inputs, then the handle remains NULL,
210 * and no convolution is applied */
211 pData->hostBlockSize_clamped = SAF_CLAMP(pData->hostBlockSize, MIN_FRAME_SIZE, MAX_FRAME_SIZE);
212 if(pData->ir_length>0){
213 saf_TVConv_create(&(pData->hTVConv),
214 pData->hostBlockSize_clamped,
215 pData->irs,
216 pData->ir_length,
217 pData->nListenerPositions,
218 pData->nOutputChannels,
219 pData->position_idx);
220 }
221
222 /* Resize buffers */
223 pData->inputFrameTD = (float**)realloc2d((void**)pData->inputFrameTD, MAX_NUM_CHANNELS, pData->hostBlockSize_clamped, sizeof(float));
224 pData->outputFrameTD = (float**)realloc2d((void**)pData->outputFrameTD, MAX_NUM_CHANNELS, pData->hostBlockSize_clamped, sizeof(float));
225 memset(FLATTEN2D(pData->inputFrameTD), 0, MAX_NUM_CHANNELS*(pData->hostBlockSize_clamped)*sizeof(float));
226
227 /* reset FIFO buffers */
228 pData->FIFO_idx = 0;
229 memset(pData->inFIFO, 0, MAX_NUM_CHANNELS*MAX_FRAME_SIZE*sizeof(float));
230 memset(pData->outFIFO, 0, MAX_NUM_CHANNELS*MAX_FRAME_SIZE*sizeof(float));
231
232 pData->reInitFilters = 0;
233 pData->codecStatus = CODEC_STATUS_INITIALISED;
234 }
235}

237void tvconv_setFiltersAndPositions
238(
239 void* const hTVCnv
240)
241{
242 tvconv_data* pData = (tvconv_data*) hTVCnv;
243#ifdef SAF_ENABLE_SOFA_READER_MODULE
244 int i;
245 vectorND tmp;
246 SAF_SOFA_ERROR_CODES error;
247 saf_sofa_container sofa;
248#endif
249
250 if (pData->codecStatus != CODEC_STATUS_NOT_INITIALISED)
251 return; /* re-init not required, or already happening */
252 while (pData->procStatus == PROC_STATUS_ONGOING){
253 /* re-init required, but we need to wait for the current processing loop to end */
254 pData->codecStatus = CODEC_STATUS_INITIALISING; /* indicate that we want to init */
255 SAF_SLEEP(10);
256 }
257
258 /* for progress bar */
259 pData->codecStatus = CODEC_STATUS_INITIALISING;
260 strcpy(pData->progressBarText,"Initialising");
261 pData->progressBar0_1 = 0.0f;
262
263#ifdef SAF_ENABLE_SOFA_READER_MODULE
264 if(pData->sofa_filepath!=NULL){
265 strcpy(pData->progressBarText,"Opening SOFA file");
266 pData->progressBar0_1 = 0.2f;
267 error = saf_sofa_open(&sofa, pData->sofa_filepath, SAF_SOFA_READER_OPTION_NETCDF);
268
269 if(error==SAF_SOFA_OK){
270 strcpy(pData->progressBarText,"Loading IRs");
271 pData->progressBar0_1 = 0.5f;
272
273 pData->ir_fs = (int)sofa.DataSamplingRate;
274 pData->ir_length = sofa.DataLengthIR;
275 pData->nIrChannels = sofa.nReceivers;
276 pData->nListenerPositions = sofa.nListeners;
277
278 /* copy only the first source position, because number of source positions might be incorrect in sofa */
279 if(!strcmp(sofa.SourcePositionType, "spherical")){
280 memcpy(tmp, sofa.SourcePosition, sizeof(vectorND));
281 unitSph2cart((float*)tmp, 1, 1, pData->sourcePosition);
282 }
283 else
284 memcpy(pData->sourcePosition, sofa.SourcePosition, sizeof(vectorND));
285
286
287 pData->irs = (float**)realloc2d((void**)pData->irs, pData->nListenerPositions, pData->nIrChannels*pData->ir_length, sizeof(float));
288 int tmp_length = pData->nIrChannels * pData->ir_length;
289 for(i=0; i<pData->nListenerPositions; i++){
290 memcpy(pData->irs[i], &(sofa.DataIR[i*tmp_length]), tmp_length*sizeof(float));
291 }
292
293 strcpy(pData->progressBarText,"Loading positions");
294 pData->progressBar0_1 = 0.8f;
295
296 pData->listenerPositions = (vectorND*)realloc1d((void*)pData->listenerPositions, pData->nListenerPositions*sizeof(vectorND));
297 memcpy(pData->listenerPositions, sofa.ListenerPosition, pData->nListenerPositions*sizeof(vectorND));
298 }
299 }
300 saf_sofa_close(&sofa);
301 pData->nOutputChannels = SAF_MIN(pData->nIrChannels, MAX_NUM_CHANNELS);
302 tvconv_setMinMaxDimensions(hTVCnv);
303#else
304 pData->ir_length = 0;
305 saf_print_warning("This example requires SAF_ENABLE_SOFA_READER_MODULE to do anything");
306#endif
307
308 pData->position_idx = 0;
309 pData->codecStatus = CODEC_STATUS_INITIALISED;
310 pData->reInitFilters = 1;
311
312 /* done! */
313 strcpy(pData->progressBarText,"Done!");
314 pData->progressBar0_1 = 1.0f;
315}
