66 void getDecorrelationDelays
67 (
68  int nChannels, /* number of channels */
69  float* freqs, /* centre frequencies */
70  int nFreqs, /* number of elements in frequency vector */
71  float fs, /* host fs */
72  int maxTFdelay, /* max number of time-slots to delay */
73  int hopSize, /* STFT hop size */
74  int* delayTF /* nFreq x nChannels */
75 )
76 {
77  int band, ch;
78  int* randperm_nCH;
79  float maxMilliseconds, nChannelsf;
80  float* delayRangeMax, *delayRangeMin, *tmp_delays, *delays;
81 
82  nChannelsf = (float)nChannels;
83  randperm_nCH = malloc1d(nChannels*sizeof(int));
84  delayRangeMax = malloc1d(nFreqs*sizeof(float));
85  delayRangeMin = malloc1d(nFreqs*sizeof(float));
86  tmp_delays = malloc1d(nChannels*sizeof(float));
87  delays = malloc1d(nFreqs*nChannels*sizeof(float));
88  maxMilliseconds = MIN(80.0f, (maxTFdelay-1.0f)*(float)hopSize/fs*1000.0f);
89  for(band=0; band<nFreqs; band++){
90  delayRangeMax[band] = MAX(7.0f, MIN(maxMilliseconds, 50.0f*1000.0f/(freqs[band]+2.23e-9f)));
91  delayRangeMin[band] = MAX(3.0f, MIN(20.0f, 10.0f*1000.0f/(freqs[band]+2.23e-9f)));
92  }
93  for(band=0; band<nFreqs; band++)
94  for(ch=0; ch<nChannels; ch++)
95  delays[band*nChannels+ch] = (float)ch/nChannelsf + ((float)rand()/(float)RAND_MAX)/nChannelsf;
96  for(band=0; band<nFreqs; band++){
97  randperm(nChannels, randperm_nCH);
98  memcpy(tmp_delays, &delays[band*nChannels], nChannels*sizeof(float));
99  for(ch=0; ch<nChannels; ch++)
100  delays[band*nChannels+ch] = tmp_delays[randperm_nCH[ch]];
101  }
102  for(band=0; band<nFreqs; band++){
103  for(ch=0; ch<nChannels; ch++){
104  delays[band*nChannels+ch] = delays[band*nChannels+ch]*(delayRangeMax[band]-delayRangeMin[band])+delayRangeMin[band];
105  delayTF[band*nChannels+ch] = MAX((int)(delays[band*nChannels+ch]/1000.0f*fs/(float)hopSize + 0.5f)-1,0);
106  }
107  }
108 
109  free(randperm_nCH);
110  free(delayRangeMax);
111  free(delayRangeMin);
112  free(tmp_delays);
113  free(delays);
114 }

116 void synthesiseNoiseReverb
117 (
118  int nCH,
119  float fs,
120  float* t60,
121  float* fcen_oct,
122  int nBands,
123  int flattenFLAG,
124  float** rir_filt,
125  int* rir_len
126 )
127 {
128  int i, j, k, rir_filt_len, rir_filt_lout, filterOrder;
129  float alpha, max_t60, t;
130  float *rir, *fcut, *h_filt, *rir_filt_tmp;
131 
132  filterOrder = 800;
133 
134  /* find rir length */
135  max_t60 = 0.0f;
136  for(i=0; i<nBands; i++)
137  max_t60 = max_t60 < t60[i] ? t60[i] : max_t60;
138  rir_filt_len = (int)(max_t60*fs+0.5f); /* length of RIRs */
139  rir_filt_lout = rir_filt_len + filterOrder/2; /* truncated output length */
140 
141  /* Generate noise and shape with exponentially decaying envelopes */
142  rir = calloc1d(nCH*nBands*rir_filt_lout, sizeof(float));
143  for(i=0; i<nCH; i++){
144  for(j=0; j<nBands; j++){
145  /* decay constants for t60 */
146  alpha = 3.0f*logf(10.0f)/t60[j];
147  for(k=0, t=0.0f; k<rir_filt_len; k++, t+=1.0f/fs)
148  rir[i*nBands*rir_filt_lout + j*rir_filt_lout + k] = expf(-t*alpha) * /* envelope */
149  2.0f * ((float)rand()/(float)RAND_MAX-0.5f); /* whitenoise */
150  }
151  }
152 
153  /* get bank of FIRs filters - octave bands */
154  fcut = malloc1d((nBands-1)*sizeof(float));
155  h_filt = malloc1d(nBands*(filterOrder+1)*sizeof(float));
156  getOctaveBandCutoffFreqs(fcen_oct, nBands, fcut);
157  FIRFilterbank(filterOrder, fcut, (nBands-1), fs, WINDOWING_FUNCTION_HAMMING, 1, h_filt);
158 
159  /* filter RIRs with filterbank */
160  (*rir_filt) = realloc1d((*rir_filt), nCH*rir_filt_lout*sizeof(float));
161  memset((*rir_filt), 0, nCH*rir_filt_lout*sizeof(float));
162  rir_filt_tmp = malloc1d(nBands*rir_filt_lout*sizeof(float));
163  for(i=0; i<nCH; i++){
164  fftfilt(&rir[i*nBands*rir_filt_lout], h_filt, rir_filt_lout, filterOrder+1, nBands, rir_filt_tmp);
165  /* sum over bands */
166  for(j=0; j<nBands; j++){
167  for(k=0; k<rir_filt_lout; k++)
168  (*rir_filt)[i*rir_filt_lout+k] += rir_filt_tmp[j*rir_filt_lout+k];
169  }
170  }
171 
172  /* equalise, to force flat magnitude response */
173  if(flattenFLAG)
174  for(i=0; i<nCH; i++)
175  flattenMinphase(&((*rir_filt)[i*rir_filt_lout]), rir_filt_lout);
176 
177  /* remove filterbank delay */
178  for(i=0; i<nCH; i++)
179  memcpy(&((*rir_filt)[i*rir_filt_len]), &((*rir_filt)[i*rir_filt_lout + filterOrder/2]), rir_filt_len*sizeof(float));
180  (*rir_len) = rir_filt_len;
181 
182  /* clean-up */
183  free(rir);
184  free(fcut);
185  free(h_filt);
186  free(rir_filt_tmp);
187 }

189 void latticeDecorrelator_create
190 (
191  void** phDecor,
192  int nCH,
193  int* orders,
194  float* freqCutoffs,
195  int* fixedDelays,
196  int nCutoffs,
197  float* freqVector,
198  int lookupOffset,
199  int nBands
200 )
201 {
202  *phDecor = malloc1d(sizeof(latticeDecor_data));
203  latticeDecor_data *h = (latticeDecor_data*)(*phDecor);
204  int i, band, ch, o, filterIdx, maxDelay;
205 
206  h->nCH = nCH;
207  h->nCutoffs = nCutoffs;
208  h->nBands = nBands;
209 
210  /* alloc */
211  h->orders = malloc1d(nCutoffs*sizeof(int));
212  memcpy(h->orders, orders, nCutoffs*sizeof(int));
213  h->TF_delays = malloc1d(nBands * sizeof(int));
214  h->lttc_apf = (latticeAPF**)malloc2d(nBands, nCH, sizeof(latticeAPF));
215 
216  /* set-up delays and allpass filters per band and channel */
217  maxDelay = 0;
218  for(band=0; band<nBands; band++){
219  filterIdx = -1;
220 
221  /* Find filter index for this band: */
222  for(o=0; o<nCutoffs; o++){
223  if(freqVector[band]<freqCutoffs[o]){
224  filterIdx = o;
225  break;
226  }
227  }
228 
229  /* Fixed frequency dependent delays */
230  if(filterIdx==-1)
231  h->TF_delays[band] = fixedDelays[nCutoffs]+1;
232  else
233  h->TF_delays[band] = fixedDelays[filterIdx]+1;
234 
235  /* keep track of maximum delay */
236  if(h->TF_delays[band]>maxDelay)
237  maxDelay = h->TF_delays[band];
238 
239  /* Pull lattice allpass filter coefficients from database */
240  for(ch=0; ch<nCH; ch++){
241  if(filterIdx == -1){
242  /* Not needed... */
243  h->lttc_apf[band][ch].order = -1;
244  h->lttc_apf[band][ch].filterLength = 0;
245  h->lttc_apf[band][ch].coeffs = NULL;
246  h->lttc_apf[band][ch].buffer = NULL;
247  }
248  else{
249  h->lttc_apf[band][ch].order = h->orders[filterIdx];
250  h->lttc_apf[band][ch].coeffs = (float**)malloc2d(2, h->orders[filterIdx], sizeof(float));
251  h->lttc_apf[band][ch].buffer = calloc1d(h->orders[filterIdx], sizeof(float_complex));
252 
253  /* numerator coefficients */
254  switch(orders[filterIdx]){
255  case 20: memcpy(h->lttc_apf[band][ch].coeffs[0], __lattice_coeffs_o20[ch+lookupOffset], (h->orders[filterIdx])*sizeof(float)); break;
256  case 18: memcpy(h->lttc_apf[band][ch].coeffs[0], __lattice_coeffs_o18[ch+lookupOffset], (h->orders[filterIdx])*sizeof(float)); break;
257  case 16: memcpy(h->lttc_apf[band][ch].coeffs[0], __lattice_coeffs_o16[ch+lookupOffset], (h->orders[filterIdx])*sizeof(float)); break;
258  case 15: memcpy(h->lttc_apf[band][ch].coeffs[0], __lattice_coeffs_o15[ch+lookupOffset], (h->orders[filterIdx])*sizeof(float)); break;
259  case 14: memcpy(h->lttc_apf[band][ch].coeffs[0], __lattice_coeffs_o14[ch+lookupOffset], (h->orders[filterIdx])*sizeof(float)); break;
260  case 12: memcpy(h->lttc_apf[band][ch].coeffs[0], __lattice_coeffs_o12[ch+lookupOffset], (h->orders[filterIdx])*sizeof(float)); break;
261  case 10: memcpy(h->lttc_apf[band][ch].coeffs[0], __lattice_coeffs_o10[ch+lookupOffset], (h->orders[filterIdx])*sizeof(float)); break;
262  case 8: memcpy(h->lttc_apf[band][ch].coeffs[0], __lattice_coeffs_o8[ch+lookupOffset], (h->orders[filterIdx])*sizeof(float)); break;
263  case 6: memcpy(h->lttc_apf[band][ch].coeffs[0], __lattice_coeffs_o6[ch+lookupOffset], (h->orders[filterIdx])*sizeof(float)); break;
264  case 4: memcpy(h->lttc_apf[band][ch].coeffs[0], __lattice_coeffs_o4[ch+lookupOffset], (h->orders[filterIdx])*sizeof(float)); break;
265  case 3: memcpy(h->lttc_apf[band][ch].coeffs[0], __lattice_coeffs_o3[ch+lookupOffset], (h->orders[filterIdx])*sizeof(float)); break;
266  case 2: memcpy(h->lttc_apf[band][ch].coeffs[0], __lattice_coeffs_o2[ch+lookupOffset], (h->orders[filterIdx])*sizeof(float)); break;
267  default: assert(0); /* Unsupported filter order specified */
268  }
269 
270  /* denominator coefficients */
271  for(i=0; i<orders[filterIdx]; i++)
272  h->lttc_apf[band][ch].coeffs[1][i] = h->lttc_apf[band][ch].coeffs[0][orders[filterIdx]-i-1];
273  }
274  }
275  }
276 
277  /* Run-time */
278  h->maxBufferLen = maxDelay;
279  h->delayBuffers = (float_complex***)calloc3d(nBands, nCH, h->maxBufferLen, sizeof(float_complex));
280  h->rIdx = malloc1d(nBands*sizeof(int));
281  for(band=0; band<nBands; band++)
282  h->rIdx[band] = h->TF_delays[band]-1;
283  h->wIdx = calloc1d(nBands,sizeof(int));
284 }

314 void latticeDecorrelator_apply
315 (
316  void* hDecor,
317  float_complex*** inFrame,
318  int nTimeSlots,
319  float_complex*** decorFrame
320 )
321 {
322  latticeDecor_data *h = (latticeDecor_data*)(hDecor);
323  int band, ch, t, i;
324  float_complex ytmp, xtmp;
325 
326  /* Apply fixed delay */
327  for(t=0; t<nTimeSlots; t++){
328  for(band=0; band <h->nBands; band++){
329  for(ch=0; ch<h->nCH; ch++){
330  h->delayBuffers[band][ch][h->wIdx[band]] = inFrame[band][ch][t];
331  decorFrame[band][ch][t] = h->delayBuffers[band][ch][h->rIdx[band]];
332  }
333 
334  /* increment and wrap-around as needed */
335  h->rIdx[band]++;
336  h->wIdx[band]++;
337  if( h->rIdx[band] >= h->TF_delays[band] )
338  h->rIdx[band] = 0;
339  if( h->wIdx[band] >= h->TF_delays[band] )
340  h->wIdx[band] = 0;
341  }
342  }
343 
344  /* Apply lattice allpass filters */
345 #if _MSC_VER >= 1900
346  for(band=0; band <h->nBands; band++){
347  for(ch=0; ch<h->nCH; ch++){
348  if(h->lttc_apf[band][ch].buffer!=NULL){ /* only if filter is defined... */
349  for(t=0; t<nTimeSlots; t++){
350  xtmp = decorFrame[band][ch][t];
351  ytmp = ccaddf(h->lttc_apf[band][ch].buffer[0], crmulf(xtmp, (h->lttc_apf[band][ch].coeffs[0][0])));
352  decorFrame[band][ch][t] = ytmp;
353 
354  /* propagate through the rest of the lattice filter structure */
355  for(i=0; i<h->lttc_apf[band][ch].order-1; i++){
356  h->lttc_apf[band][ch].buffer[i] = ccaddf(h->lttc_apf[band][ch].buffer[i+1],
357  ccsubf(crmulf(xtmp, h->lttc_apf[band][ch].coeffs[0][i+1]), /* numerator */
358  crmulf(ytmp, h->lttc_apf[band][ch].coeffs[1][i+1]))); /* denominator */
359  }
360  }
361  }
362  }
363  }
364 #else
365  for(band=0; band <h->nBands; band++){
366  for(ch=0; ch<h->nCH; ch++){
367  if(h->lttc_apf[band][ch].buffer!=NULL){ /* only if filter is defined... */
368  for(t=0; t<nTimeSlots; t++){
369  xtmp = decorFrame[band][ch][t];
370  ytmp = h->lttc_apf[band][ch].buffer[0] + xtmp * (h->lttc_apf[band][ch].coeffs[0][0]);
371  decorFrame[band][ch][t] = ytmp;
372 
373  /* propagate through the rest of the lattice filter structure */
374  for(i=0; i<h->lttc_apf[band][ch].order-1; i++){
375  h->lttc_apf[band][ch].buffer[i] = h->lttc_apf[band][ch].buffer[i+1] +
376  h->lttc_apf[band][ch].coeffs[0][i+1] * xtmp - /* numerator */
377  h->lttc_apf[band][ch].coeffs[1][i+1] * ytmp; /* denominator */
378  }
379  }
380  }
381  }
382  }
383 #endif
384 }

418 void transientDucker_apply
419 (
420  void* hDucker,
421  float_complex*** inFrame,
422  int nTimeSlots,
423  float alpha,
424  float beta,
425  float_complex*** outFrame
426 )
427 {
428  transientDucker_data *h = (transientDucker_data*)(hDucker);
429  int band, i, t;
430  float detectorEne, transientEQ;
431  //const float alpha = 0.95f;
432  //const float beta = 0.995f;
433 
434  for(band=0; band<h->nBands; band++){
435  for(i=0; i<h->nCH; i++){
436  for(t=0; t<nTimeSlots; t++){
437  detectorEne = cabsf(inFrame[band][i][t]);
438  detectorEne = detectorEne*detectorEne;
439  h->transientDetector1[band][i] *= alpha;
440  if(h->transientDetector1[band][i]<detectorEne)
441  h->transientDetector1[band][i] = detectorEne;
442  h->transientDetector2[band][i] = h->transientDetector2[band][i]*beta + (1.0f-beta)*(h->transientDetector1[band][i]);
443  if(h->transientDetector2[band][i] > h->transientDetector1[band][i])
444  h->transientDetector2[band][i] = h->transientDetector1[band][i];
445  transientEQ = MIN(1.0f, 4.0f * (h->transientDetector2[band][i])/(h->transientDetector1[band][i]+2.23e-9f));
446 #ifdef _MSC_VER
447  outFrame[band][i][t] = crmulf(inFrame[band][i][t], transientEQ);
448 #else
449  outFrame[band][i][t] = inFrame[band][i][t] * transientEQ;
450 #endif
451  }
452  }
453  }
