52void test__sofa_comparison(void){
53 SAF_SOFA_ERROR_CODES error;
54 saf_sofa_container sofa;
55 int err;
56 struct MYSOFA_HRTF *hrtf;
57
58 /* Load the same SOFA file */
59 error = saf_sofa_open(&sofa, SAF_TEST_SOFA_FILE_PATH);
60 hrtf = mysofa_load(SAF_TEST_SOFA_FILE_PATH, &err);
61
62 /* If both SOFA loaders were successful */
63 if(error==SAF_SOFA_OK && err==MYSOFA_OK){
64 /* Check that data is equivalent */
65 for(unsigned int i=0; i<hrtf->M * hrtf->R * hrtf->N; i++)
66 TEST_ASSERT_TRUE(fabsf(sofa.DataIR[i]- hrtf->DataIR.values[i])<0.000001f);
67 for(unsigned int i=0; i<hrtf->M * hrtf->C; i++)
68 TEST_ASSERT_TRUE(fabsf(sofa.SourcePosition[i]- hrtf->SourcePosition.values[i])<0.000001f);
69 for(unsigned int i=0; i<hrtf->M * hrtf->C; i++)
70 TEST_ASSERT_TRUE(fabsf(sofa.SourcePosition[i]- hrtf->SourcePosition.values[i])<0.000001f);
71 for(unsigned int i=0; i<hrtf->_I * hrtf->R; i++)
72 TEST_ASSERT_TRUE(fabsf(sofa.DataDelay[i]- hrtf->DataDelay.values[i])<0.000001f);
73 }
74
75 saf_sofa_close(&sofa);
76 mysofa_free(hrtf);
77}
