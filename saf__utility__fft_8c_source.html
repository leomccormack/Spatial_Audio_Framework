126 void fftconv
127 (
128  float* x,
129  float* h,
130  int x_len,
131  int h_len,
132  int nCH,
133  float* y
134 )
135 {
136  int i, y_len, fftSize, nBins;
137  float* h0, *x0, *y0;
138  float_complex* H, *X, *Y;
139  void* hfft;
140 
141  /* prep */
142  y_len = x_len + h_len - 1;
143  fftSize = (int)((float)nextpow2(y_len)+0.5f);
144  nBins = fftSize/2+1;
145  h0 = calloc1d(fftSize, sizeof(float));
146  x0 = calloc1d(fftSize, sizeof(float));
147  y0 = malloc1d(fftSize * sizeof(float));
148  H = malloc1d(nBins*sizeof(float_complex));
149  X = malloc1d(nBins*sizeof(float_complex));
150  Y = malloc1d(nBins*sizeof(float_complex));
151  saf_rfft_create(&hfft, fftSize);
152 
153  /* apply convolution per channel */
154  for(i=0; i<nCH; i++){
155  /* zero pad to avoid circular convolution artefacts, prior to fft */
156  memcpy(h0, &h[i*h_len], h_len*sizeof(float));
157  memcpy(x0, &x[i*x_len], x_len*sizeof(float));
158  saf_rfft_forward(hfft, x0, X);
159  saf_rfft_forward(hfft, h0, H);
160 
161  /* multiply the two spectra */
162  utility_cvvmul(X, H, nBins, Y);
163 
164  /* ifft, truncate and store to output */
165  saf_rfft_backward(hfft, Y, y0);
166  memcpy(&y[i*y_len], y0, y_len*sizeof(float));
167  }
168 
169  /* tidy up */
170  saf_rfft_destroy(&hfft);
171  free(h0);
172  free(x0);
173  free(y0);
174  free(H);
175  free(X);
176  free(Y);
177 }

199 void hilbert
200 (
201  float_complex* x,
202  int x_len,
203  float_complex* y
204 )
205 {
206  int i;
207  float_complex *xfft, *h, *xhfft;
208  void* hfft;
209 
210  saf_fft_create(&hfft, x_len);
211  xfft = malloc1d(x_len*sizeof(float_complex));
212  h = malloc1d(x_len*sizeof(float_complex));
213  xhfft = malloc1d(x_len*sizeof(float_complex));
214 
215  /* Forward fft */
216  saf_fft_forward(hfft, x, xfft);
217 
218  /* define vector h */
219  memset(h, 0, sizeof(float_complex)*x_len);
220  if(x_len % 2 == 0){
221  /* even */
222  h[0] = cmplxf(1.0f, 0.0f);
223  h[x_len/2] = cmplxf(1.0f, 0.0f);
224  for(i=1;i<x_len/2;i++)

226  }
227  else{
228  assert(0); // uneven lengths not actually supported by saf_fft
229  /* odd */
230  h[0] = cmplxf(1.0f, 0.0f);
231  for(i=1;i<(x_len+1)/2;i++)
232  h[i] = cmplxf(2.0f, 0.0f);
233  }
234 
235  /* apply h, and ifft */
236  utility_cvvmul(xfft, h, x_len, xhfft);
237  saf_fft_backward(hfft, xhfft, y);
238 
239  /* tidy up */
240  saf_fft_destroy(&hfft);
241  free(xfft);
242  free(h);
243  free(xhfft);

283 
284  /* Windowing function */

317 void saf_stft_forward
318 (
319  void * const hSTFT,
320  float** dataTD,
321  int framesize,
322  float_complex*** dataFD
323 
324 )
325 {
326  saf_stft_data *h = (saf_stft_data*)(hSTFT);
327  int ch, j, nHops, t, band, idx, hIdx;
328 
329  assert(framesize % h->hopsize == 0); /* framesize must be multiple of hopsize */
330  nHops = framesize/h->hopsize;
331 
332  /* For linear time-invariant (LTI) operation (i.e. no previous hops are
333  * required) */
334  if(h->winsize==h->hopsize){
335  for (t = 0; t<nHops; t++){
336  for(ch=0; ch < h->nCHin; ch++){
337  /* Window input signal (Rectangular) */
338  memcpy(h->insig_win, &dataTD[ch][t*(h->hopsize)], h->winsize*sizeof(float));
339 
340  /* Apply FFT and copy data to output dataFD buffer */
341  switch(h->FDformat){
342  case SAF_STFT_TIME_CH_BANDS:
343  saf_rfft_forward(h->hFFT, h->insig_win, dataFD[t][ch]);
344  break;
345 
346  case SAF_STFT_BANDS_CH_TIME:
347  saf_rfft_forward(h->hFFT, h->insig_win, h->tmp_fft);
348  for(band=0; band<h->nBands; band++)
349  dataFD[band][ch][t] = h->tmp_fft[band];
350  break;
351  }
352  }
353  }
354  }
355  /* For oversampled TF transforms */
356  else{
357  idx = 0;
358  for (t = 0; t<nHops; t++){
359  for(ch=0; ch < h->nCHin; ch++){
360  hIdx = 0;
361  /* Window input signal */
362  while (hIdx < h->winsize){
363  memcpy(&(h->insig_win[hIdx]), h->prev_inhops[0][ch], h->hopsize*sizeof(float));
364  for(j=0; j< h->nPrevHops-1; j++)
365  memcpy(h->prev_inhops[j][ch], h->prev_inhops[j+1][ch], h->hopsize*sizeof(float));
366  memcpy(h->prev_inhops[h->nPrevHops-1][ch], &dataTD[ch][idx], h->hopsize*sizeof(float));
367  hIdx += h->hopsize;
368  }
369  utility_svvmul(h->insig_win, h->window, h->winsize, h->insig_win);
370 
371  /* Apply FFT and copy data to output dataFD buffer */
372  switch(h->FDformat){
373  case SAF_STFT_TIME_CH_BANDS:
374  saf_rfft_forward(h->hFFT, h->insig_win, dataFD[t][ch]);
375  break;
376 
377  case SAF_STFT_BANDS_CH_TIME:
378  saf_rfft_forward(h->hFFT, h->insig_win, h->tmp_fft);
379  for(band=0; band<h->nBands; band++)
380  dataFD[band][ch][t] = h->tmp_fft[band];
381  break;
382  }
383  }
384  idx += h->hopsize;
385  }
386  }
387 }

389 void saf_stft_backward
390 (
391  void * const hSTFT,
392  float_complex*** dataFD,
393  int framesize,
394  float** dataTD
395 )
396 {
397  saf_stft_data *h = (saf_stft_data*)(hSTFT);
398  int t, ch, nHops, band;
399 
400  assert(framesize % h->hopsize == 0); /* framesize must be multiple of hopsize */
401  nHops = framesize/h->hopsize;
402 
403  for (t = 0; t<nHops; t++){
404  for(ch=0; ch < h->nCHout; ch++){
405  /* Shift data down */
406  memcpy(h->overlapAddBuffer[ch], &h->overlapAddBuffer[ch][h->hopsize], (h->numOvrlpAddBlocks-1)*(h->hopsize)*sizeof(float));
407 
408  /* Append with zeros */
409  memset(&h->overlapAddBuffer[ch][(h->numOvrlpAddBlocks-1)*(h->hopsize)], 0, h->hopsize*sizeof(float));
410 
411  /* Apply inverse FFT */
412  switch(h->FDformat){
413  case SAF_STFT_TIME_CH_BANDS:
414  saf_rfft_backward(h->hFFT, dataFD[t][ch], h->outsig_win);
415  break;
416  case SAF_STFT_BANDS_CH_TIME:
417  for(band=0; band<h->nBands; band++)
418  h->tmp_fft[band] = dataFD[band][ch][t];
419  saf_rfft_backward(h->hFFT, h->tmp_fft, h->outsig_win);
420  break;
421  }
422 
423  /* Overlap-Add and copy 1:hopsize to output buffer */
424  utility_svvadd(h->overlapAddBuffer[ch], h->outsig_win, h->fftsize, h->overlapAddBuffer[ch]);
425  memcpy(&dataTD[ch][t*(h->hopsize)], h->overlapAddBuffer[ch], h->hopsize*sizeof(float));
426  }
427  }
428 }

443  void * const hSTFT,
444  int new_nCHin,
445  int new_nCHout
446 )
447 {
448  saf_stft_data *h = (saf_stft_data*)(hSTFT);
449  int i, ch;
450

456  /* Zero new channels if new_nCHin > nCHin */
457  for(i=0; i< h->nPrevHops; i++)
458  for(ch=h->nCHin; ch<new_nCHin; ch++)
459  memset(h->prev_inhops[i][ch], 0, h->hopsize * sizeof(float));
460 
461  h->nCHin = new_nCHin;
462  }
463 
464  if(new_nCHout != h->nCHout){
465  /* Reallocate memory while retaining previous values (which will be
466  * truncated if new_nCHout < nCHout) */
467  h->overlapAddBuffer = (float**)realloc2d_r((void**)h->overlapAddBuffer, new_nCHout, h->bufferlength,
468  h->nCHout, h->bufferlength, sizeof(float));
469  /* Zero new channels if new_nCHout > nCHout */
470  for(ch=h->nCHout; ch<new_nCHout; ch++)
471  memset(h->overlapAddBuffer[ch], 0, h->bufferlength * sizeof(float));
472 
473  h->nCHout = new_nCHout;
474  }

482 void saf_rfft_create
483 (
484  void ** const phFFT,
485  int N
486 )
487 {
488  *phFFT = malloc1d(sizeof(saf_rfft_data));
489  saf_rfft_data *h = (saf_rfft_data*)(*phFFT);
490 
491  h->N = N;
492  h->Scale = 1.0f/(float)N; /* output scaling after ifft */
493  assert(N>=2 && ISEVEN(N)); /* only even (non zero) FFT sizes allowed */
494 #if defined(SAF_USE_INTEL_IPP)
495  h->useKissFFT_flag = 0;
496  /* Use ippsFFT if N is 2^x, otherwise, use ippsDFT */
497  if(ceilf(log2f(N)) == floorf(log2f(N))){
498  h->useIPPfft_FLAG = 1;
499  h->log2n = (int)(log2f((float)N)+0.1f);
500  ippsFFTGetSize_R_32f(h->log2n, IPP_FFT_DIV_INV_BY_N, ippAlgHintNone, &(h->specSize), &(h->specBufferSize), &(h->bufferSize));
501  h->hFFTspec = NULL;
502  h->memSpec = (Ipp8u*) ippMalloc(h->specSize);
503  h->buffer = (Ipp8u*) ippMalloc(h->bufferSize);
504  h->memInit = (Ipp8u*) ippMalloc(h->specBufferSize);
505  ippsFFTInit_R_32f(&(h->hFFTspec), h->log2n, IPP_FFT_DIV_INV_BY_N, ippAlgHintNone, h->memSpec, h->memInit);
506  }
507  else{
508  h->useIPPfft_FLAG = 0;
509  ippsDFTGetSize_R_32f(N, IPP_FFT_DIV_INV_BY_N, ippAlgHintNone, &(h->specSize), &(h->specBufferSize), &(h->bufferSize));
510  h->hDFTspec = (IppsDFTSpec_R_32f*) ippMalloc(h->specSize);
511  h->buffer = (Ipp8u*) ippMalloc(h->bufferSize);
512  h->memInit = (Ipp8u*) ippMalloc(h->specBufferSize);
513  ippsDFTInit_R_32f(N, IPP_FFT_DIV_INV_BY_N, ippAlgHintNone, h->hDFTspec, h->memInit);
514  }
515  if (h->memInit)
516  ippFree(h->memInit);
517 #elif defined(__ACCELERATE__)
518  if(ceilf(log2f(N)) == floorf(log2f(N))) /* true if N is 2 to the power of some integer number */
519  h->useKissFFT_flag = 0;
520  else
521  h->useKissFFT_flag = 1;
522  /* Apple Accelerate only supports 2^x FFT sizes */
523  if(!h->useKissFFT_flag){
524  h->log2n = (int)(log2f((float)N)+0.1f);
525  h->FFT = (void*)vDSP_create_fftsetup(h->log2n, FFT_RADIX2);
526  h->VDSP_split.realp = malloc1d((h->N/2)*sizeof(float));
527  h->VDSP_split.imagp = malloc1d((h->N/2)*sizeof(float));
528  }
529 #elif defined(INTEL_MKL_VERSION)
530  h->useKissFFT_flag = 0;
531  h->MKL_FFT_Handle = 0;
532  h->Status = DftiCreateDescriptor(&(h->MKL_FFT_Handle), DFTI_SINGLE, DFTI_REAL, 1, h->N); /* 1-D, single precision, real_input->fft->half_complex->ifft->real_output */
533  h->Status = DftiSetValue(h->MKL_FFT_Handle, DFTI_PLACEMENT, DFTI_NOT_INPLACE); /* Not inplace, i.e. output has its own dedicated memory */
534  /* specify output format as complex conjugate-symmetric data. This is the same as MatLab, except only the
535  * first N/2+1 elements are returned. The inverse transform will automatically symmetrically+conjugate
536  * replicate these elements, in order to get the required N elements internally. */
537  h->Status = DftiSetValue(h->MKL_FFT_Handle, DFTI_CONJUGATE_EVEN_STORAGE, DFTI_COMPLEX_COMPLEX);
538  /* Configuration parameters for backward-FFT */
539  h->Status = DftiSetValue(h->MKL_FFT_Handle, DFTI_BACKWARD_SCALE, h->Scale); /* scalar applied after ifft */
540  /* commit these chosen parameters */
541  h->Status = DftiCommitDescriptor(h->MKL_FFT_Handle);
542 #else
543  h->useKissFFT_flag = 1;
544 #endif
545  if(h->useKissFFT_flag){
546  h->kissFFThandle_fwd = kiss_fftr_alloc(h->N, 0, NULL, NULL);
547  h->kissFFThandle_bkw = kiss_fftr_alloc(h->N, 1, NULL, NULL);
548  }
549 }

597  if(h->useIPPfft_FLAG)
598  ippsFFTFwd_RToCCS_32f((Ipp32f*)inputTD, (Ipp32f*)outputFD, h->hFFTspec, h->buffer);
599  else
600  ippsDFTFwd_RToCCS_32f((Ipp32f*)inputTD, (Ipp32f*)outputFD, h->hDFTspec, h->buffer);
601 #elif defined(__ACCELERATE__)
602  int i;
603  if(!h->useKissFFT_flag){
604  vDSP_ctoz((DSPComplex*)inputTD, 2, &(h->VDSP_split), 1, (h->N)/2);
605  vDSP_fft_zrip((FFTSetup)(h->FFT),&(h->VDSP_split), 1, h->log2n, FFT_FORWARD);
606  /* DC */
607  outputFD[0] = cmplxf(h->VDSP_split.realp[0]/2.0f, 0.0f);
608  /* Note: the output is scaled by 2, because vDSP_fft automatically compensates for the loss of energy
609  * when removing the symmetric/conjugate (N/2+2:N) bins. However, this is dumb... so the 2x scaling
610  * is removed here; so it has parity with the other FFT implementations supported by SAF. */
611  for(i=1; i<h->N/2; i++)
612  outputFD[i] = cmplxf(h->VDSP_split.realp[i]/2.0f, h->VDSP_split.imagp[i]/2.0f);
613  /* the real part of the Nyquist value is the imaginary part of DC. */
614  outputFD[h->N/2] = cmplxf(h->VDSP_split.imagp[0]/2.0f, 0.0f);
615  /* https://stackoverflow.com/questions/43289265/implementing-an-fft-using-vdsp */
616  }
617 #elif defined(INTEL_MKL_VERSION)
618  h->Status = DftiComputeForward(h->MKL_FFT_Handle, inputTD, outputFD);
619 #endif
620  if(h->useKissFFT_flag)
621  kiss_fftr(h->kissFFThandle_fwd, inputTD, (kiss_fft_cpx*)outputFD);
622 }

624 void saf_rfft_backward
625 (
626  void * const hFFT,
627  float_complex* inputFD,
628  float* outputTD
629 )
630 {
631  saf_rfft_data *h = (saf_rfft_data*)(hFFT);
632  int i;
633 
634 #if defined(SAF_USE_INTEL_IPP)
635  if(h->useIPPfft_FLAG)
636  ippsFFTInv_CCSToR_32f((Ipp32f*)inputFD, (Ipp32f*)outputTD, h->hFFTspec, h->buffer);
637  else
638  ippsDFTInv_CCSToR_32f((Ipp32f*)inputFD, (Ipp32f*)outputTD, h->hDFTspec, h->buffer);
639 #elif defined(__ACCELERATE__)
640  if(!h->useKissFFT_flag){
641  h->VDSP_split.realp[0] = crealf(inputFD[0]);
642  h->VDSP_split.imagp[0] = crealf(inputFD[h->N/2]);
643  for(i=1; i<h->N/2; i++){
644  h->VDSP_split.realp[i] = crealf(inputFD[i]);
645  h->VDSP_split.imagp[i] = cimagf(inputFD[i]);
646  }
647  vDSP_fft_zrip(h->FFT, &(h->VDSP_split), 1, h->log2n, FFT_INVERSE);
648  vDSP_ztoc(&(h->VDSP_split), 1, (DSPComplex*)outputTD, 2, (h->N)/2);
649  vDSP_vsmul(outputTD, 1, &(h->Scale), outputTD, 1, h->N);
650  }
651 #elif defined(INTEL_MKL_VERSION)
652  h->Status = DftiComputeBackward(h->MKL_FFT_Handle, inputFD, outputTD);
653 #endif
654  if(h->useKissFFT_flag){
655  kiss_fftri(h->kissFFThandle_bkw, (kiss_fft_cpx*)inputFD, outputTD);
656  for(i=0; i<h->N; i++)
657  outputTD[i] /= (float)(h->N);
658  }

666 void saf_fft_create
667 (
668  void ** const phFFT,
669  int N
670 )
671 {
672  *phFFT = malloc1d(sizeof(saf_fft_data));
673  saf_fft_data *h = (saf_fft_data*)(*phFFT);
674 
675  h->N = N;
676  h->Scale = 1.0f/(float)N; /* output scaling after ifft */
677  assert(N>=2); /* only even (non zero) FFT sizes allowed */
678 #if defined(SAF_USE_INTEL_IPP)
679  h->useKissFFT_flag = 0;
680  /* Use ippsFFT if N is 2^x, otherwise, use ippsDFT */
681  if(ceilf(log2f(N)) == floorf(log2f(N))){
682  h->useIPPfft_FLAG = 1;
683  h->log2n = (int)(log2f((float)N)+0.1f);
684  ippsFFTGetSize_C_32f(h->log2n, IPP_FFT_DIV_INV_BY_N, ippAlgHintNone, &(h->specSize), &(h->specBufferSize), &(h->bufferSize));
685  h->hFFTspec = NULL;
686  h->memSpec = (Ipp8u*) ippsMalloc_8u(h->specSize);
687  h->buffer = (Ipp8u*) ippsMalloc_8u(h->bufferSize);
688  h->memInit = (Ipp8u*) ippsMalloc_8u(h->specBufferSize);
689  ippsFFTInit_C_32fc(&(h->hFFTspec), h->log2n, IPP_FFT_DIV_INV_BY_N, ippAlgHintNone, h->memSpec, h->memInit);
690  }
691  else{
692  h->useIPPfft_FLAG = 0;
693  ippsDFTGetSize_C_32f(N, IPP_FFT_DIV_INV_BY_N, ippAlgHintNone, &(h->specSize), &(h->specBufferSize), &(h->bufferSize));
694  h->hDFTspec = (IppsDFTSpec_C_32fc*) ippsMalloc_8u(h->specSize);
695  h->buffer = (Ipp8u*) ippsMalloc_8u(h->bufferSize);
696  h->memInit = (Ipp8u*) ippsMalloc_8u(h->specBufferSize);
697  ippsDFTInit_C_32fc(N, IPP_FFT_DIV_INV_BY_N, ippAlgHintNone, h->hDFTspec, h->memInit);
698  }
699  if (h->memInit)
700  ippFree(h->memInit);
701 #elif defined(__ACCELERATE__) && 0 /* NOT IMPLEMENTED YET */
702  if(ceilf(log2f(N)) == floorf(log2f(N))) /* true if N is 2 to the power of some integer number */
703  h->useKissFFT_flag = 0;
704  else
705  h->useKissFFT_flag = 1;
706  /* Apple Accelerate only supports 2^x FFT sizes */
707  if(!h->useKissFFT_flag){
708  h->log2n = (int)(log2f((float)N)+0.1f);
709  h->FFT = (void*)vDSP_create_fftsetup(h->log2n, FFT_RADIX2);
710  h->VDSP_split.realp = malloc1d((h->N/2)*sizeof(float));
711  h->VDSP_split.imagp = malloc1d((h->N/2)*sizeof(float));
712  }
713 #elif defined(INTEL_MKL_VERSION)
714  h->useKissFFT_flag = 0;
715  h->MKL_FFT_Handle = 0;
716  h->Status = DftiCreateDescriptor( &(h->MKL_FFT_Handle), DFTI_SINGLE,
717  DFTI_COMPLEX, 1, h->N); /* 1-D, single precision, complex_input_td->fft->complex_input_fd->ifft->complex_output_td */
718  h->Status = DftiSetValue(h->MKL_FFT_Handle, DFTI_PLACEMENT, DFTI_NOT_INPLACE); /* Not inplace, i.e. output has its own dedicated memory */
719  /* Configuration parameters for backward-FFT */
720  h->Status = DftiSetValue(h->MKL_FFT_Handle, DFTI_BACKWARD_SCALE, h->Scale); /* scalar applied after ifft */
721  /* commit these chosen parameters */
722  h->Status = DftiCommitDescriptor(h->MKL_FFT_Handle);
723 #else
724  h->useKissFFT_flag = 1;
725 #endif
726  if(h->useKissFFT_flag){
727  h->kissFFThandle_fwd = kiss_fft_alloc(h->N, 0, NULL, NULL);
728  h->kissFFThandle_bkw = kiss_fft_alloc(h->N, 1, NULL, NULL);
729  }
730 }
