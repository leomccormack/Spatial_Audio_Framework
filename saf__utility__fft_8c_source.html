156 void fftconv
157 (
158  float* x,
159  float* h,
160  int x_len,
161  int h_len,
162  int nCH,
163  float* y
164 )
165 {
166  int i, y_len, fftSize, nBins;
167  float* h0, *x0, *y0;
168  float_complex* H, *X, *Y;
169  void* hfft;
170 
171  /* prep */
172  y_len = x_len + h_len - 1;
173  fftSize = (int)((float)nextpow2(y_len)+0.5f);
174  nBins = fftSize/2+1;
175  h0 = calloc1d(fftSize, sizeof(float));
176  x0 = calloc1d(fftSize, sizeof(float));
177  y0 = malloc1d(fftSize * sizeof(float));
178  H = malloc1d(nBins*sizeof(float_complex));
179  X = malloc1d(nBins*sizeof(float_complex));
180  Y = malloc1d(nBins*sizeof(float_complex));
181  saf_rfft_create(&hfft, fftSize);
182 
183  /* apply convolution per channel */
184  for(i=0; i<nCH; i++){
185  /* zero pad to avoid circular convolution artefacts, prior to fft */
186  memcpy(h0, &h[i*h_len], h_len*sizeof(float));
187  memcpy(x0, &x[i*x_len], x_len*sizeof(float));
188  saf_rfft_forward(hfft, x0, X);
189  saf_rfft_forward(hfft, h0, H);
190 
191  /* multiply the two spectra */
192  utility_cvvmul(X, H, nBins, Y);
193 
194  /* ifft, truncate and store to output */
195  saf_rfft_backward(hfft, Y, y0);
196  memcpy(&y[i*y_len], y0, y_len*sizeof(float));
197  }
198 
199  /* tidy up */
200  saf_rfft_destroy(&hfft);
201  free(h0);
202  free(x0);
203  free(y0);
204  free(H);
205  free(X);
206  free(Y);
207 }

229 void hilbert
230 (
231  float_complex* x,
232  int x_len,
233  float_complex* y
234 )
235 {
236  int i;
237  float_complex *xfft, *h, *xhfft;
238  void* hfft;
239 
240  saf_fft_create(&hfft, x_len);
241  xfft = malloc1d(x_len*sizeof(float_complex));
242  h = malloc1d(x_len*sizeof(float_complex));
243  xhfft = malloc1d(x_len*sizeof(float_complex));
244 
245  /* Forward fft */
246  saf_fft_forward(hfft, x, xfft);
247 
248  /* define vector h */
249  memset(h, 0, sizeof(float_complex)*x_len);
250  if(x_len % 2 == 0){
251  /* even */
252  h[0] = cmplxf(1.0f, 0.0f);
253  h[x_len/2] = cmplxf(1.0f, 0.0f);
254  for(i=1;i<x_len/2;i++)
255  h[i] = cmplxf(2.0f, 0.0f);
256  }
257  else{
258  saf_assert(x_len % 2 == 0, "Uneven lengths are not supported by saf_fft");
259  /* odd */
260  h[0] = cmplxf(1.0f, 0.0f);
261  for(i=1;i<(x_len+1)/2;i++)
262  h[i] = cmplxf(2.0f, 0.0f);
263  }
264 
265  /* apply h, and ifft */
266  utility_cvvmul(xfft, h, x_len, xhfft);
267  saf_fft_backward(hfft, xhfft, y);
268 
269  /* tidy up */
270  saf_fft_destroy(&hfft);
271  free(xfft);
272  free(h);
273  free(xhfft);
274 }

280 void saf_stft_create
281 (
282  void ** const phSTFT,
283  int winsize,
284  int hopsize,
285  int nCHin,
286  int nCHout,
287  SAF_STFT_FDDATA_FORMAT FDformat
288 )
289 {
290  *phSTFT = malloc1d(sizeof(saf_stft_data));
291  saf_stft_data *h = (saf_stft_data*)(*phSTFT);
292 
293  h->nCHin = nCHin;
294  h->nCHout = nCHout;
295  h->winsize = winsize;
296  h->hopsize = hopsize;
297  h->nBands = winsize+1;
298  h->FDformat = FDformat;
299 
300  /* set-up FFT */
301  h->fftsize = 2*winsize;
302  saf_rfft_create(&(h->hFFT), h->fftsize);
303  h->insig_rect_win = calloc1d(h->fftsize, sizeof(float));
304  h->insig_win = calloc1d(h->fftsize, sizeof(float));
305 
306  /* Intermediate buffers */
307  h->tmp_fft = malloc1d(h->nBands * sizeof(float_complex));
308  h->outsig_win = malloc1d(h->fftsize*sizeof(float));
309  h->nPrevHops = winsize/hopsize-1;
310  if (h->nPrevHops>0)
311  h->prev_inhops = (float***)calloc3d(h->nPrevHops, nCHin, hopsize, sizeof(float));
312  else
313  h->prev_inhops = NULL;
314 
315  /* Windowing function */
316  if(winsize==hopsize)
317  h->window = NULL;
318  else{
319  h->window = malloc1d(winsize*sizeof(float));
320  getWindowingFunction(WINDOWING_FUNCTION_HANN, winsize, h->window);
321  }
322 
323  /* Overlap-add buffer */
324  h->numOvrlpAddBlocks = winsize/hopsize;
325  h->bufferlength = h->numOvrlpAddBlocks * (h->fftsize);
326  h->overlapAddBuffer = (float**)calloc2d(nCHout, h->bufferlength, sizeof(float));
327 }

349 void saf_stft_forward
350 (
351  void * const hSTFT,
352  float** dataTD,
353  int framesize,
354  float_complex*** dataFD
355 
356 )
357 {
358  saf_stft_data *h = (saf_stft_data*)(hSTFT);
359  int ch, j, nHops, t, band, idx, hIdx;
360 
361  saf_assert(framesize % h->hopsize == 0, "framesize must be multiple of hopsize");
362  nHops = framesize/h->hopsize;
363 
364  /* For linear time-invariant (LTI) operation (i.e. no previous hops are
365  * required) */
366  if(h->winsize==h->hopsize){
367  for (t = 0; t<nHops; t++){
368  for(ch=0; ch < h->nCHin; ch++){
369  /* Window input signal (Rectangular) */
370  memcpy(h->insig_rect_win, &dataTD[ch][t*(h->hopsize)], h->winsize*sizeof(float));
371 
372  /* Apply FFT and copy data to output dataFD buffer */
373  switch(h->FDformat){
374  case SAF_STFT_TIME_CH_BANDS:
375  saf_rfft_forward(h->hFFT, h->insig_rect_win, dataFD[t][ch]);
376  break;
377 
378  case SAF_STFT_BANDS_CH_TIME:
379  saf_rfft_forward(h->hFFT, h->insig_rect_win, h->tmp_fft);
380  for(band=0; band<h->nBands; band++)
381  dataFD[band][ch][t] = h->tmp_fft[band];
382  break;
383  }
384  }
385  }
386  }
387  /* For oversampled TF transforms */
388  else{
389  idx = 0;
390  for (t = 0; t<nHops; t++){
391  for(ch=0; ch < h->nCHin; ch++){
392  hIdx = 0;
393  /* Window input signal */
394  while (hIdx < h->winsize){
395  memcpy(&(h->insig_rect_win[hIdx]), h->prev_inhops[0][ch], h->hopsize*sizeof(float));
396  for(j=0; j< h->nPrevHops-1; j++)
397  memcpy(h->prev_inhops[j][ch], h->prev_inhops[j+1][ch], h->hopsize*sizeof(float));
398  memcpy(h->prev_inhops[h->nPrevHops-1][ch], &dataTD[ch][idx], h->hopsize*sizeof(float));
399  hIdx += h->hopsize;
400  }
401  utility_svvmul(h->insig_rect_win, h->window, h->winsize, h->insig_win);
402 
403  /* Apply FFT and copy data to output dataFD buffer */
404  switch(h->FDformat){
405  case SAF_STFT_TIME_CH_BANDS:
406  saf_rfft_forward(h->hFFT, h->insig_win, dataFD[t][ch]);
407  break;
408 
409  case SAF_STFT_BANDS_CH_TIME:
410  saf_rfft_forward(h->hFFT, h->insig_win, h->tmp_fft);
411  for(band=0; band<h->nBands; band++)
412  dataFD[band][ch][t] = h->tmp_fft[band];
413  break;
414  }
415  }
416  idx += h->hopsize;
417  }
418  }
419 }

421 void saf_stft_backward
422 (
423  void * const hSTFT,
424  float_complex*** dataFD,
425  int framesize,
426  float** dataTD
427 )
428 {
429  saf_stft_data *h = (saf_stft_data*)(hSTFT);
430  int t, ch, nHops, band;
431 
432  saf_assert(framesize % h->hopsize == 0, "framesize must be multiple of hopsize");
433  nHops = framesize/h->hopsize;
434 
435  for (t = 0; t<nHops; t++){
436  for(ch=0; ch < h->nCHout; ch++){
437  /* Shift data down */
438  memcpy(h->overlapAddBuffer[ch], h->overlapAddBuffer[ch] + h->hopsize, (h->numOvrlpAddBlocks-1)*(h->hopsize)*sizeof(float));
439 
440  /* Append with zeros */
441  memset(h->overlapAddBuffer[ch] + (h->numOvrlpAddBlocks-1)*(h->hopsize), 0, h->hopsize*sizeof(float));
442 
443  /* Apply inverse FFT */
444  switch(h->FDformat){
445  case SAF_STFT_TIME_CH_BANDS:
446  saf_rfft_backward(h->hFFT, dataFD[t][ch], h->outsig_win);
447  break;
448  case SAF_STFT_BANDS_CH_TIME:
449  for(band=0; band<h->nBands; band++)
450  h->tmp_fft[band] = dataFD[band][ch][t];
451  saf_rfft_backward(h->hFFT, h->tmp_fft, h->outsig_win);
452  break;
453  }
454 
455  /* Overlap-Add 1:hopsize to output buffer */
456  cblas_saxpy(h->fftsize, 1.0f, h->outsig_win, 1, h->overlapAddBuffer[ch], 1);
457  memcpy(dataTD[ch] + t*(h->hopsize), h->overlapAddBuffer[ch], h->hopsize*sizeof(float));
458  }
459  }
460 }

473 void saf_stft_channelChange
474 (
475  void * const hSTFT,
476  int new_nCHin,
477  int new_nCHout
478 )
479 {
480  saf_stft_data *h = (saf_stft_data*)(hSTFT);
481  int i, ch;
482 
483  if(new_nCHin != h->nCHin && h->nPrevHops > 0){
484  /* Reallocate memory while retaining previous values (which will be
485  * truncated if new_nCHin < nCHin) */
486  h->prev_inhops = (float***)realloc3d_r((void***)h->prev_inhops, h->nPrevHops, new_nCHin, h->hopsize,
487  h->nPrevHops, h->nCHin, h->hopsize, sizeof(float));
488  /* Zero new channels if new_nCHin > nCHin */
489  for(i=0; i< h->nPrevHops; i++)
490  for(ch=h->nCHin; ch<new_nCHin; ch++)
491  memset(h->prev_inhops[i][ch], 0, h->hopsize * sizeof(float));
492 
493  h->nCHin = new_nCHin;
494  }
495 
496  if(new_nCHout != h->nCHout){
497  /* Reallocate memory while retaining previous values (which will be
498  * truncated if new_nCHout < nCHout) */
499  h->overlapAddBuffer = (float**)realloc2d_r((void**)h->overlapAddBuffer, new_nCHout, h->bufferlength,
500  h->nCHout, h->bufferlength, sizeof(float));
501  /* Zero new channels if new_nCHout > nCHout */
502  for(ch=h->nCHout; ch<new_nCHout; ch++)
503  memset(h->overlapAddBuffer[ch], 0, h->bufferlength * sizeof(float));
504 
505  h->nCHout = new_nCHout;
506  }
507 }

514 void saf_rfft_create
515 (
516  void ** const phFFT,
517  int N
518 )
519 {
520  *phFFT = malloc1d(sizeof(saf_rfft_data));
521  saf_rfft_data *h = (saf_rfft_data*)(*phFFT);
522 
523  h->N = N;
524  h->Scale = 1.0f/(float)N; /* output scaling after ifft */
525  saf_assert(N>=2 && ISEVEN(N), "Only even (non zero) FFT sizes are supported");
526  h->useKissFFT_FLAG = 0;
527 #if defined(SAF_USE_FFTW)
528  h->fwd_bufferTD = malloc1d(h->N*sizeof(float));
529  h->bwd_bufferTD = malloc1d(h->N*sizeof(float));
530  h->fwd_bufferFD = malloc1d((h->N/2+1)*sizeof(fftwf_complex));
531  h->bwd_bufferFD = malloc1d((h->N/2+1)*sizeof(fftwf_complex));
532  h->p_fwd = fftwf_plan_dft_r2c_1d(h->N, h->fwd_bufferTD, h->fwd_bufferFD, FFTW_ESTIMATE);
533  h->p_bwd = fftwf_plan_dft_c2r_1d(h->N, h->bwd_bufferFD, h->bwd_bufferTD, FFTW_ESTIMATE);
534 #elif defined(SAF_USE_INTEL_IPP)
535  /* Use ippsFFT if N is 2^x, otherwise, use ippsDFT */
536  if(ceilf(log2f(N)) == floorf(log2f(N))){
537  h->useIPPfft_FLAG = 1;
538  h->log2n = (int)(log2f((float)N)+0.1f);
539  ippsFFTGetSize_R_32f(h->log2n, IPP_FFT_DIV_INV_BY_N, ippAlgHintNone, &(h->specSize), &(h->specBufferSize), &(h->bufferSize));
540  h->hFFTspec = NULL;
541  h->memSpec = (Ipp8u*) ippMalloc(h->specSize);
542  h->buffer = (Ipp8u*) ippMalloc(h->bufferSize);
543  h->memInit = (Ipp8u*) ippMalloc(h->specBufferSize);
544  ippsFFTInit_R_32f(&(h->hFFTspec), h->log2n, IPP_FFT_DIV_INV_BY_N, ippAlgHintNone, h->memSpec, h->memInit);
545  }
546  else{
547  h->useIPPfft_FLAG = 0;
548  ippsDFTGetSize_R_32f(N, IPP_FFT_DIV_INV_BY_N, ippAlgHintNone, &(h->specSize), &(h->specBufferSize), &(h->bufferSize));
549  h->hDFTspec = (IppsDFTSpec_R_32f*) ippMalloc(h->specSize);
550  h->buffer = (Ipp8u*) ippMalloc(h->bufferSize);
551  h->memInit = (Ipp8u*) ippMalloc(h->specBufferSize);
552  ippsDFTInit_R_32f(N, IPP_FFT_DIV_INV_BY_N, ippAlgHintNone, h->hDFTspec, h->memInit);
553  }
554  if (h->memInit)
555  ippFree(h->memInit);
556 #elif defined(SAF_USE_APPLE_ACCELERATE)
557 # ifdef SAF_USE_INTERLEAVED_VDSP
558  h->DFT_fwd = vDSP_DFT_Interleaved_CreateSetup(0, N, vDSP_DFT_FORWARD, vDSP_DFT_Interleaved_RealtoComplex);
559  h->DFT_bwd = vDSP_DFT_Interleaved_CreateSetup(0, N, vDSP_DFT_INVERSE, vDSP_DFT_Interleaved_RealtoComplex);
560 # else
561  h->DFT_fwd = vDSP_DFT_zrop_CreateSetup(0, N, vDSP_DFT_FORWARD);
562  h->DFT_bwd = vDSP_DFT_zrop_CreateSetup(0, N, vDSP_DFT_INVERSE);
563 # endif
564  if(h->DFT_fwd==0 || h->DFT_bwd==0) /* specified N not supported by vDSP, so must use the default */
565  h->useKissFFT_FLAG = 1;
566  else{
567  /* Note that DFT lengths must satisfy: f * 2.^g, where f is 1, 3, 5, or 15, and g >=4 */
568  saf_assert(h->DFT_fwd!=0 && h->DFT_bwd!=0, "Failed to create vDSP DFT");
569 # ifndef SAF_USE_INTERLEAVED_VDSP
570  h->VDSP_split_tmp.realp = malloc1d((h->N/2)*sizeof(float));

651 void saf_rfft_forward
652 (
653  void * const hFFT,
654  float* inputTD,
655  float_complex* outputFD
656 )
657 {
658  saf_rfft_data *h = (saf_rfft_data*)(hFFT);
659 
660 #if defined(SAF_USE_FFTW)
661  cblas_scopy(h->N, inputTD, 1, h->fwd_bufferTD, 1);
662  fftwf_execute(h->p_fwd);
663  cblas_ccopy(h->N/2+1, h->fwd_bufferFD, 1, outputFD, 1);
664 #elif defined(SAF_USE_INTEL_IPP)
665  if(h->useIPPfft_FLAG)
666  ippsFFTFwd_RToCCS_32f((Ipp32f*)inputTD, (Ipp32f*)outputFD, h->hFFTspec, h->buffer);
667  else
668  ippsDFTFwd_RToCCS_32f((Ipp32f*)inputTD, (Ipp32f*)outputFD, h->hDFTspec, h->buffer);
669 #elif defined(SAF_USE_APPLE_ACCELERATE)
670  if(!h->useKissFFT_FLAG){
671 # ifdef SAF_USE_INTERLEAVED_VDSP
672  saf_print_error("Not implemented yet");
673 # else
674  vDSP_ctoz((DSPComplex*)inputTD, 2, &(h->VDSP_split_tmp), 1, (h->N)/2);
675  vDSP_DFT_Execute(h->DFT_fwd, h->VDSP_split_tmp.realp, h->VDSP_split_tmp.imagp, h->VDSP_split.realp, h->VDSP_split.imagp);
676  /* DC */
677  outputFD[0] = cmplxf(h->VDSP_split.realp[0], 0.0f);
678  cblas_scopy(h->N/2-1, &h->VDSP_split.realp[1], 1, &((float*)(outputFD))[2], 2);
679  cblas_scopy(h->N/2-1, &h->VDSP_split.imagp[1], 1, &((float*)(outputFD))[3], 2);
680  /* the real part of the Nyquist value is the imaginary part of DC. */
681  /* https://stackoverflow.com/questions/43289265/implementing-an-fft-using-vdsp */
682  outputFD[h->N/2] = cmplxf(h->VDSP_split.imagp[0], 0.0f);
683  /* Note: the output is scaled by 2, because vDSP_fft automatically compensates for the loss of energy
684  * when removing the symmetric/conjugate (N/2+2:N) bins. However, this is dumb... so the 2x scaling
685  * is removed here; so it has parity with the other FFT implementations supported by SAF. */
686 # endif
687  cblas_sscal(2*(h->N/2+1), 0.5f, (float*)outputFD, 1);
688  }
689 #elif defined(SAF_USE_INTEL_MKL_LP64) || defined(SAF_USE_INTEL_MKL_ILP64)
690  h->Status = DftiComputeForward(h->MKL_FFT_Handle, inputTD, outputFD);
691 #endif
692  if(h->useKissFFT_FLAG)
693  kiss_fftr(h->kissFFThandle_fwd, inputTD, (kiss_fft_cpx*)outputFD);
694 }

696 void saf_rfft_backward
697 (
698  void * const hFFT,
699  float_complex* inputFD,
700  float* outputTD
701 )
702 {

708  cblas_scopy(h->N, h->bwd_bufferTD, 1, outputTD, 1);

719 # else

721  h->VDSP_split_tmp.imagp[0] = crealf(inputFD[h->N/2]);
722  cblas_scopy(h->N/2-1, &((float*)(inputFD))[2], 2, &h->VDSP_split_tmp.realp[1], 1);
723  cblas_scopy(h->N/2-1, &((float*)(inputFD))[3], 2, &h->VDSP_split_tmp.imagp[1], 1);
724  vDSP_DFT_Execute(h->DFT_bwd, h->VDSP_split_tmp.realp, h->VDSP_split_tmp.imagp, h->VDSP_split.realp, h->VDSP_split.imagp);

731 #endif
732  if(h->useKissFFT_FLAG){
733  kiss_fftri(h->kissFFThandle_bkw, (kiss_fft_cpx*)inputFD, outputTD);
734  cblas_sscal(h->N, 1.0f/(float)(h->N), outputTD, 1);
735  }

743 void saf_fft_create
744 (
745  void ** const phFFT,
746  int N
747 )
748 {
749  *phFFT = malloc1d(sizeof(saf_fft_data));
750  saf_fft_data *h = (saf_fft_data*)(*phFFT);
751 
752  h->N = N;
753  h->Scale = 1.0f/(float)N; /* output scaling after ifft */
754  saf_assert(N>=2, "Only even (non zero) FFT sizes are supported");
755 #if defined(SAF_USE_FFTW)
756  h->fwd_bufferTD = malloc1d(h->N*sizeof(fftwf_complex));
757  h->bwd_bufferTD = malloc1d(h->N*sizeof(fftwf_complex));
758  h->fwd_bufferFD = malloc1d(h->N*sizeof(fftwf_complex));
759  h->bwd_bufferFD = malloc1d(h->N*sizeof(fftwf_complex));
760  h->p_fwd = fftwf_plan_dft_1d(h->N, h->fwd_bufferTD, h->fwd_bufferFD, FFTW_FORWARD, FFTW_ESTIMATE);
761  h->p_bwd = fftwf_plan_dft_1d(h->N, h->bwd_bufferFD, h->bwd_bufferTD, FFTW_BACKWARD, FFTW_ESTIMATE);
762 #elif defined(SAF_USE_INTEL_IPP)
763  /* Use ippsFFT if N is 2^x, otherwise, use ippsDFT */
764  if(ceilf(log2f(N)) == floorf(log2f(N))){
765  h->useIPPfft_FLAG = 1;
766  h->log2n = (int)(log2f((float)N)+0.1f);
767  ippsFFTGetSize_C_32f(h->log2n, IPP_FFT_DIV_INV_BY_N, ippAlgHintNone, &(h->specSize), &(h->specBufferSize), &(h->bufferSize));
768  h->hFFTspec = NULL;
769  h->memSpec = (Ipp8u*) ippsMalloc_8u(h->specSize);
770  h->buffer = (Ipp8u*) ippsMalloc_8u(h->bufferSize);
771  h->memInit = (Ipp8u*) ippsMalloc_8u(h->specBufferSize);
772  ippsFFTInit_C_32fc(&(h->hFFTspec), h->log2n, IPP_FFT_DIV_INV_BY_N, ippAlgHintNone, h->memSpec, h->memInit);
773  }
774  else{
775  h->useIPPfft_FLAG = 0;
776  ippsDFTGetSize_C_32f(N, IPP_FFT_DIV_INV_BY_N, ippAlgHintNone, &(h->specSize), &(h->specBufferSize), &(h->bufferSize));
777  h->hDFTspec = (IppsDFTSpec_C_32fc*) ippsMalloc_8u(h->specSize);
778  h->buffer = (Ipp8u*) ippsMalloc_8u(h->bufferSize);
779  h->memInit = (Ipp8u*) ippsMalloc_8u(h->specBufferSize);
780  ippsDFTInit_C_32fc(N, IPP_FFT_DIV_INV_BY_N, ippAlgHintNone, h->hDFTspec, h->memInit);
781  }
782  if (h->memInit)
783  ippFree(h->memInit);
784 #elif defined(SAF_USE_APPLE_ACCELERATE)
785 # ifdef SAF_USE_INTERLEAVED_VDSP
786  h->DFT_fwd = vDSP_DFT_Interleaved_CreateSetup(0, N, vDSP_DFT_FORWARD, vDSP_DFT_Interleaved_RealtoComplex);
787  h->DFT_bwd = vDSP_DFT_Interleaved_CreateSetup(0, N, vDSP_DFT_INVERSE, vDSP_DFT_Interleaved_RealtoComplex);
788 # else
789  h->DFT_fwd = vDSP_DFT_zop_CreateSetup(0, N, vDSP_DFT_FORWARD);
790  h->DFT_bwd = vDSP_DFT_zop_CreateSetup(0, N, vDSP_DFT_INVERSE);
791 # endif
792  if(h->DFT_fwd==0 || h->DFT_bwd==0) /* specified N not supported by vDSP, so must use the default */
793  h->useKissFFT_FLAG = 1;
794  else{
795  /* Note that DFT lengths must satisfy: f * 2.^g, where f is 1, 3, 5, or 15, and g >=3 */
796  saf_assert(h->DFT_fwd!=0 && h->DFT_bwd!=0, "Failed to create vDSP DFT");
797 # ifndef SAF_USE_INTERLEAVED_VDSP
798  h->VDSP_split_tmp.realp = malloc1d((h->N)*sizeof(float));
799  h->VDSP_split_tmp.imagp = malloc1d((h->N)*sizeof(float));
800  h->VDSP_split.realp = malloc1d((h->N)*sizeof(float));
801  h->VDSP_split.imagp = malloc1d((h->N)*sizeof(float));
802 # endif
803  }
804 #elif defined(SAF_USE_INTEL_MKL_LP64) || defined(SAF_USE_INTEL_MKL_ILP64)
805  h->MKL_FFT_Handle = 0;
806  h->Status = DftiCreateDescriptor( &(h->MKL_FFT_Handle), DFTI_SINGLE,
807  DFTI_COMPLEX, 1, h->N); /* 1-D, single precision, complex_input_td->fft->complex_input_fd->ifft->complex_output_td */
808  h->Status = DftiSetValue(h->MKL_FFT_Handle, DFTI_PLACEMENT, DFTI_NOT_INPLACE); /* Not inplace, i.e. output has its own dedicated memory */
809  /* Configuration parameters for backward-FFT */
810  h->Status = DftiSetValue(h->MKL_FFT_Handle, DFTI_BACKWARD_SCALE, h->Scale); /* scalar applied after ifft */
811  /* commit these chosen parameters */
812  h->Status = DftiCommitDescriptor(h->MKL_FFT_Handle);
813 #else
814  /* Must use default FFT: */
815  h->useKissFFT_FLAG = 1;
816 #endif
817  /* DEFAULT: */
818  if(h->useKissFFT_FLAG){
819  h->kissFFThandle_fwd = kiss_fft_alloc(h->N, 0, NULL, NULL);
820  h->kissFFThandle_bkw = kiss_fft_alloc(h->N, 1, NULL, NULL);
821  }
