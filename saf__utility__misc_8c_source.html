75 void findERBpartitions
76 (
77  float* centerFreq,
78  int nBands,
79  float maxFreqLim,
80  int** erb_idx,
81  float** erb_freqs,
82  int* nERBBands
83 )
84 {
85  int i, band, counter, next_erb_idx;
86  float band_centreFreq, erb, erb_centre, tmp;
87 
88  band_centreFreq = (powf(2.0f, 1.0f/3.0f)+1.0f)/2.0f;
89  free(*erb_idx);
90  free(*erb_freqs);
91  (*erb_idx) = malloc1d(sizeof(int));
92  (*erb_freqs) = malloc1d(sizeof(float));
93  (*erb_idx)[0] = 1;
94  (*erb_freqs)[0] = centerFreq[0];
95  counter = 0;
96  next_erb_idx = 0;
97  while((*erb_freqs)[counter]<maxFreqLim){
98  erb = 24.7f + 0.108f * (*erb_freqs)[counter] * band_centreFreq;
99  (*erb_idx) = realloc1d((*erb_idx), (counter+2)*sizeof(int));
100  (*erb_freqs) = realloc1d((*erb_freqs), (counter+2)*sizeof(float));
101  (*erb_freqs)[counter+1] = (*erb_freqs)[counter] + erb;
102  erb_centre = FLT_MAX;
103  /* find closest band frequency as upper partition limit */
104  for(band=0; band<nBands; band++){
105  tmp =fabsf((*erb_freqs)[counter+1] - centerFreq[band]);
106  if(tmp <erb_centre){
107  erb_centre = tmp;
108  next_erb_idx = band;
109  }
110  }
111  (*erb_idx)[counter+1] = next_erb_idx + 1;
112  if((*erb_idx)[counter+1] == (*erb_idx)[counter])
113  (*erb_idx)[counter+1] = (*erb_idx)[counter+1]+1;
114  (*erb_freqs)[counter+1] = centerFreq[(*erb_idx)[counter+1]-1];
115  counter++;
116  }
117  /* last limit set at last band */
118  (*erb_idx) = realloc1d((*erb_idx), (counter + 2) * sizeof(int));
119  (*erb_freqs) = realloc1d((*erb_freqs), (counter + 2) * sizeof(float));
120  (*erb_idx)[counter+1] = nBands;
121  (*erb_freqs)[counter+1] = centerFreq[nBands-1];
122  (*nERBBands) = counter+2;
123 
124  /* subtract 1 from the indices (the above is a direct port from Matlab...) */
125  for(i=0; i<(*nERBBands); i++)
126  (*erb_idx)[i] = (*erb_idx)[i] - 1;

371 void unique_i
372 (
373  int* input,
374  int nInputs,
375  int** uniqueVals,
376  int** uniqueInds,
377  int* nUnique
378 )
379 {
380  int i, j, k, nDups, foundDups, foundNewDup;
381  int* dups, *nDuplicates_perInput;
382 
383  /* If only 1 input... */
384  if(nInputs == 1){
385  (*nUnique) = 1;
386  if(uniqueVals!=NULL){
387  (*uniqueVals) = malloc1d((*nUnique)*sizeof(int));
388  (*uniqueVals)[0] = input[0];
389  }
390  if(uniqueInds!=NULL){
391  (*uniqueInds) = malloc1d((*nUnique)*sizeof(int));
392  (*uniqueInds)[0] = 0;
393  }
394  }
395 
396  /* prep */
397  dups = malloc1d(nInputs*sizeof(int));
398  nDuplicates_perInput = calloc1d(nInputs, sizeof(int));
399  (*nUnique) = nInputs;
400 
401  /* Find duplicates */
402  nDups = 0;
403  for(i=0; i<nInputs; i++) {
404  foundDups = 0;
405  for(j=i+1; j<nInputs; j++) {
406  if(input[i] == input[j]) {
407  foundNewDup = 1;
408  for(k=0; k<nDups; k++)
409  if(input[i]==dups[k])
410  foundNewDup = 0;
411 
412  /* input value is repeated: */
413  nDuplicates_perInput[i]++;
414  if(foundNewDup){
415  (*nUnique)--; /* one less unique value... */
416  foundDups = 1;
417  }
418  }
419  }
420  /* Store repeated value, so "nUnique" is not decreased more than once for it */
421  if(foundDups){
422  dups[nDups] = input[i];
423  nDups++;
424  }
425  }
426  assert((*nUnique)>-1);
427  free(dups);
428 
429  /* If no unique values were found */
430  if((*nUnique)==0){
431  (*uniqueVals) = NULL;
432  (*uniqueInds) = NULL;
433  (*nUnique) = 0;
434  free(nDuplicates_perInput);
435  return;
436  }
437 
438  /* Save unique values and/or their indices */
439  j=0;
440  if(uniqueVals!=NULL)
441  (*uniqueVals) = malloc1d((*nUnique)*sizeof(int));
442  if(uniqueInds!=NULL)
443  (*uniqueInds) = malloc1d((*nUnique)*sizeof(int));
444  for(i=0; i<nInputs; i++) {
445  if(nDuplicates_perInput[i] == 0) {
446  /* If no duplicate exists, append... */
447  if(uniqueVals!=NULL)
448  (*uniqueVals)[j] = input[i];
449  if(uniqueInds!=NULL)
450  (*uniqueInds)[j] = i;
451  j++;
452  }
453  }
454 
455  /* clean-up */
456  free(nDuplicates_perInput);
457 }

459 /* Based heavily on the Matlab script found here:
460  * https://se.mathworks.com/matlabcentral/fileexchange/50413-generalized-matrix-exponential
461  * Copyright (c) 2015, Kenneth Johnson (BSD-3-clause license) */
462 void gexpm
463 (
464  float* D,
465  int sizeD,
466  int m1,
467  float* Y
468 )
469 {
470  int i, j, k;
471  float tol, s, h2, h, hh, hhh, two;
472  float** D_2, **D_3, **D_4, **D_5, **Dh, **Ym1, **Ym2;
473 
474  tol = FLT_EPSILON;
475 
476  /* Calculate Y = expm(D), Ym1 = Y-I.
477  *
478  * Scale and square: Y = expm(D/n)^n; n = 2^s (non-negative integer s)
479  *
480  * Pade approximation: expm(D/n) = R
481  * = (I-Dh+(2/5)*Dh^2-(1/15)*Dh^3)\‍(I+Dh+(2/5)*Dh^2+(1/15)*Dh^3)
482  * where Dh = D*h; h = 1/(2*n) (h = integration half-step)
483  *
484  * Pade approximation error: R-expm(D/n) = (-2/1575)*(Dh)^7
485  *
486  * Set Y = R^n. Approximate absolute error:
487  * Y-expm(D) = R^n-expm(D/n)^n = n*R^(n-1)*(R-expm(D/n))
488  * = n*Y*R^(-1)*(-2/1575)*(Dh)^7
489  * R^(-1) is close to I, so
490  * Y-expm(D) = n*Y*(-2/1575)*(Dh)^7 (approx)
491  *
492  * Large D (|D|>=1): Bound the relative error magnitude by tol,
493  * n*(2/1575)*|(Dh)^7| <= tol
494  *
495  * Small D (|D|<1): |Y-I| is of order 1 or less; the absolute error
496  * Y-expm(D) is of order n*(-2/1575)*(Dh)^7. Bound the error magnitude
497  * by tol*|D| to preserve relative accuracy of Ym1:
498  * n*(2/1575)*|(Dh)^7| <= tol*|D|
499  *
500  * Combine large/small Dh conditions conjunctively:
501  * n*(2/1575)*|(Dh)^7| <= tol*min(1,|D|)
502  *
503  * Substitute h = 1/(2*n):
504  * (2*n)^6 >= |D^7|/(1575*tol*min(1,|D|))
505  * Substitute n = 2^s:
506  * s >= log2(|D^7|/(1575*tol*min(1,|D|)))/6-1
507  * (Use the Frobenius norm for |...| to preserve symmetry of expm under
508  * matrix transposition.)
509  */
510  D_2 = (float**)malloc2d(sizeD, sizeD, sizeof(float));
511  cblas_sgemm(CblasRowMajor, CblasNoTrans, CblasNoTrans, sizeD, sizeD, sizeD, 1.0f,
512  D, sizeD,
513  D, sizeD, 0.0f,
514  FLATTEN2D(D_2), sizeD);
515  D_3 = (float**)malloc2d(sizeD, sizeD, sizeof(float));
516  cblas_sgemm(CblasRowMajor, CblasNoTrans, CblasNoTrans, sizeD, sizeD, sizeD, 1.0f,
517  FLATTEN2D(D_2), sizeD,
518  D, sizeD, 0.0f,
519  FLATTEN2D(D_3), sizeD);
520  D_4 = (float**)malloc2d(sizeD, sizeD, sizeof(float));
521  cblas_sgemm(CblasRowMajor, CblasNoTrans, CblasNoTrans, sizeD, sizeD, sizeD, 1.0f,
522  FLATTEN2D(D_3), sizeD,
523  FLATTEN2D(D_3), sizeD, 0.0f,
524  FLATTEN2D(D_4), sizeD);
525  D_5 = (float**)malloc2d(sizeD, sizeD, sizeof(float));
526  cblas_sgemm(CblasRowMajor, CblasNoTrans, CblasNoTrans, sizeD, sizeD, sizeD, 1.0f,
527  FLATTEN2D(D_4), sizeD,
528  D, sizeD, 0.0f,
529  FLATTEN2D(D_5), sizeD);
530  s = ceilf(log2f(Frob_norm(FLATTEN2D(D_5), sizeD, sizeD)/
531  (1575.0f*tol*MIN(1.0f,Frob_norm(D, sizeD, sizeD))))/6.0f-1.0f);
532  s = MAX(s, 0.0f);
533 
534  /* Get Pade approximation for expm(D*h2) = Y =
535  * (I-Dh+(2/5)*Dh^2-(1/15)*Dh^3)\‍(I+Dh+(2/5)*Dh^2+(1/15)*Dh^3)
536  * Ym1 = Y-I =
537  * (I-Dh+(2/5)*Dh^2-(1/15)*Dh^3)\‍(2*(Dh+(1/15)*Dh^3))
538  * (Calculate Ym1, not Y, to avoid precision loss from dominant I
539  * terms when Dh is small.) */
540  h2 = powf(2.0f,-s);
541  h = h2/2.0f;
542  hh = h*h;
543  hhh = hh*h;
544  Dh = (float**)malloc2d(sizeD, sizeD, sizeof(float));
545  memcpy(FLATTEN2D(Dh), D, sizeD*sizeD*sizeof(float));
546  utility_svsmul(FLATTEN2D(Dh), &h, sizeD*sizeD, NULL);
547  utility_svsmul(FLATTEN2D(D_2), &hh, sizeD*sizeD, NULL);
548  utility_svsmul(FLATTEN2D(D_3), &hhh, sizeD*sizeD, NULL);
549  Ym1 = (float**)malloc2d(sizeD, sizeD, sizeof(float));
550  for(i=0; i<sizeD; i++)
551  for(j=0; j<sizeD; j++)
552  Ym1[i][j] = Dh[i][j] + (1.0f/15.0f)*D_3[i][j];
553  Ym2 = (float**)malloc2d(sizeD, sizeD, sizeof(float));
554  for(i=0; i<sizeD; i++){
555  for(j=0; j<sizeD; j++){
556  Ym2[i][j] = (2.0f/5.0f)*D_2[i][j]-Ym1[i][j];
557  if(i==j)
558  Ym2[i][j] += 1.0f;
559  }
560  }
561  two = 2.0f;
562  utility_svsmul(FLATTEN2D(Ym1), &two, sizeD*sizeD, NULL);
563  utility_sglslv(FLATTEN2D(Ym2), sizeD, FLATTEN2D(Ym1), sizeD, FLATTEN2D(Ym1));
564 
565  /* Y = Ym1+I = expm(D)
566  * Square Y (i.e., Y <-- Y*Y) s times to get Y = expm(D*2^s). */
567  for (k = 0; k<(int)s; k++){
568  cblas_sgemm(CblasRowMajor, CblasNoTrans, CblasNoTrans, sizeD, sizeD, sizeD, 1.0f,
569  FLATTEN2D(Ym1), sizeD,
570  FLATTEN2D(Ym1), sizeD, 0.0f,
571  FLATTEN2D(Ym2), sizeD);
572  for(i=0; i<sizeD; i++)
573  for(j=0; j<sizeD; j++)
574  Ym1[i][j] = Ym2[i][j] + 2.0f*Ym1[i][j]; /* (Ym1+I) <-- (Ym1+I)*(Ym1+I) */
575  }
576  memcpy(Y, FLATTEN2D(Ym1), sizeD*sizeD*sizeof(float));
577  if (m1){}
578  else{
579  for(i=0; i<sizeD; i++)
580  for(j=0; j<sizeD; j++)
581  if(i==j)
582  Y[i*sizeD+j] += 1.0f;
583  }
584 
585  /* clean-up */
586  free(D_2);
587  free(D_3);
588  free(D_4);
589  free(D_5);
590  free(Dh);
591  free(Ym1);
592  free(Ym2);
593 }
