102void findERBpartitions
103(
104 float* centerFreq,
105 int nBands,
106 float maxFreqLim,
107 int** erb_idx,
108 float** erb_freqs,
109 int* nERBBands
110)
111{
112 int i, band, counter, next_erb_idx;
113 float band_centreFreq, erb, erb_centre, tmp;
114
115 band_centreFreq = (powf(2.0f, 1.0f/3.0f)+1.0f)/2.0f;
116 free(*erb_idx);
117 free(*erb_freqs);
118 (*erb_idx) = malloc1d(sizeof(int));
119 (*erb_freqs) = malloc1d(sizeof(float));
120 (*erb_idx)[0] = 1;
121 (*erb_freqs)[0] = centerFreq[0];
122 counter = 0;
123 next_erb_idx = 0;
124 while((*erb_freqs)[counter]<maxFreqLim){
125 erb = 24.7f + 0.108f * (*erb_freqs)[counter] * band_centreFreq;
126 (*erb_idx) = realloc1d((*erb_idx), (counter+2)*sizeof(int));
127 (*erb_freqs) = realloc1d((*erb_freqs), (counter+2)*sizeof(float));
128 (*erb_freqs)[counter+1] = (*erb_freqs)[counter] + erb;
129 erb_centre = FLT_MAX;
130 /* find closest band frequency as upper partition limit */
131 for(band=0; band<nBands; band++){
132 tmp =fabsf((*erb_freqs)[counter+1] - centerFreq[band]);
133 if(tmp <erb_centre){
134 erb_centre = tmp;
135 next_erb_idx = band;
136 }
137 }
138 (*erb_idx)[counter+1] = next_erb_idx + 1;
139 if((*erb_idx)[counter+1] == (*erb_idx)[counter])
140 (*erb_idx)[counter+1] = (*erb_idx)[counter+1]+1;
141 (*erb_freqs)[counter+1] = centerFreq[(*erb_idx)[counter+1]-1];
142 counter++;
143 }
144 /* last limit set at last band */
145 (*erb_idx) = realloc1d((*erb_idx), (counter + 2) * sizeof(int));
146 (*erb_freqs) = realloc1d((*erb_freqs), (counter + 2) * sizeof(float));
147 (*erb_idx)[counter+1] = nBands;
148 (*erb_freqs)[counter+1] = centerFreq[nBands-1];
149 (*nERBBands) = counter+2;
150
151 /* subtract 1 from the indices (the above is a direct port from Matlab...) */
152 for(i=0; i<(*nERBBands); i++)
153 (*erb_idx)[i] = (*erb_idx)[i] - 1;
154}

193void cxcorr
194(
195 float* a,
196 float* b,
197 float* x_ab,
198 size_t la,
199 size_t lb
200)
201{
202 int m, n, negFLAG, arg, len, lim;
203
204 len = (int)(la + lb) - 1;
205 memset(x_ab, 0, len*sizeof(float));
206 for(m=1; m<=len; m++){
207 arg = m-(int)la;
208 if(arg<0){
209 negFLAG = 1;
210 lim = (int)la + arg;
211 }
212 else{
213 negFLAG = 0;
214 lim = (int)la - arg;
215 }
216 for(n=1; n<=lim; n++){
217 if(negFLAG == 0)
218 x_ab[m-1] += (a[arg+n-1] * b[n-1]);
219 else
220 x_ab[m-1] += (a[n-1] * b[n-arg-1]);
221 }
222 }
223}

401void unique_i
402(
403 int* input,
404 int nInputs,
405 int** uniqueVals,
406 int** uniqueInds,
407 int* nUnique
408)
409{
410 int i, j, k, nDups, foundDups, foundNewDup;
411 int* dups, *nDuplicates_perInput;
412
413 /* If only 1 input... */
414 if(nInputs == 1){
415 (*nUnique) = 1;
416 if(uniqueVals!=NULL){
417 (*uniqueVals) = malloc1d((*nUnique)*sizeof(int));
418 (*uniqueVals)[0] = input[0];
419 }
420 if(uniqueInds!=NULL){
421 (*uniqueInds) = malloc1d((*nUnique)*sizeof(int));
422 (*uniqueInds)[0] = 0;
423 }
424 }
425
426 /* prep */
427 dups = malloc1d(nInputs*sizeof(int));
428 nDuplicates_perInput = calloc1d(nInputs, sizeof(int));
429 (*nUnique) = nInputs;
430
431 /* Find duplicates */
432 nDups = 0;
433 for(i=0; i<nInputs; i++) {
434 foundDups = 0;
435 for(j=i+1; j<nInputs; j++) {
436 if(input[i] == input[j]) {
437 foundNewDup = 1;
438 for(k=0; k<nDups; k++)
439 if(input[i]==dups[k])
440 foundNewDup = 0;
441
442 /* input value is repeated: */
443 nDuplicates_perInput[i]++;
444 if(foundNewDup){
445 (*nUnique)--; /* one less unique value... */
446 foundDups = 1;
447 }
448 }
449 }
450 /* Store repeated value, so "nUnique" is not decreased more than once for it */
451 if(foundDups){
452 dups[nDups] = input[i];
453 nDups++;
454 }
455 }
456 saf_assert((*nUnique)>-1, "Something very bad happened");
457 free(dups);
458
459 /* If no unique values were found */
460 if((*nUnique)==0){
461 (*uniqueVals) = NULL;
462 (*uniqueInds) = NULL;
463 (*nUnique) = 0;
464 free(nDuplicates_perInput);
465 return;
466 }
467
468 /* Save unique values and/or their indices */
469 j=0;
470 if(uniqueVals!=NULL)
471 (*uniqueVals) = malloc1d((*nUnique)*sizeof(int));
472 if(uniqueInds!=NULL)
473 (*uniqueInds) = malloc1d((*nUnique)*sizeof(int));
474 for(i=0; i<nInputs; i++) {
475 if(nDuplicates_perInput[i] == 0) {
476 /* If no duplicate exists, append... */
477 if(uniqueVals!=NULL)
478 (*uniqueVals)[j] = input[i];
479 if(uniqueInds!=NULL)
480 (*uniqueInds)[j] = i;
481 j++;
482 }
483 }
484
485 /* clean-up */
486 free(nDuplicates_perInput);
487}

506/* Based heavily on the Matlab script found here:
507 * https://se.mathworks.com/matlabcentral/fileexchange/50413-generalized-matrix-exponential
508 * Copyright (c) 2015, Kenneth Johnson (BSD-3-clause license) */
509void gexpm
510(
511 float* D,
512 int sizeD,
513 int m1,
514 float* Y
515)
516{
517 int i, j, k;
518 float tol, s, h2, h, hh, hhh, two;
519 float** D_2, **D_3, **D_4, **D_5, **Dh, **Ym1, **Ym2;
520
521 tol = FLT_EPSILON;
522
523 /* Calculate Y = expm(D), Ym1 = Y-I.
524 *
525 * Scale and square: Y = expm(D/n)^n; n = 2^s (non-negative integer s)
526 *
527 * Pade approximation: expm(D/n) = R
528 * = (I-Dh+(2/5)*Dh^2-(1/15)*Dh^3)\‍(I+Dh+(2/5)*Dh^2+(1/15)*Dh^3)
529 * where Dh = D*h; h = 1/(2*n) (h = integration half-step)
530 *
531 * Pade approximation error: R-expm(D/n) = (-2/1575)*(Dh)^7
532 *
533 * Set Y = R^n. Approximate absolute error:
534 * Y-expm(D) = R^n-expm(D/n)^n = n*R^(n-1)*(R-expm(D/n))
535 * = n*Y*R^(-1)*(-2/1575)*(Dh)^7
536 * R^(-1) is close to I, so
537 * Y-expm(D) = n*Y*(-2/1575)*(Dh)^7 (approx)
538 *
539 * Large D (|D|>=1): Bound the relative error magnitude by tol,
540 * n*(2/1575)*|(Dh)^7| <= tol
541 *
542 * Small D (|D|<1): |Y-I| is of order 1 or less; the absolute error
543 * Y-expm(D) is of order n*(-2/1575)*(Dh)^7. Bound the error magnitude
544 * by tol*|D| to preserve relative accuracy of Ym1:
545 * n*(2/1575)*|(Dh)^7| <= tol*|D|
546 *
547 * Combine large/small Dh conditions conjunctively:
548 * n*(2/1575)*|(Dh)^7| <= tol*min(1,|D|)
549 *
550 * Substitute h = 1/(2*n):
551 * (2*n)^6 >= |D^7|/(1575*tol*min(1,|D|))
552 * Substitute n = 2^s:
553 * s >= log2(|D^7|/(1575*tol*min(1,|D|)))/6-1
554 * (Use the Frobenius norm for |...| to preserve symmetry of expm under
555 * matrix transposition.)
556 */
557 D_2 = (float**)malloc2d(sizeD, sizeD, sizeof(float));
558 cblas_sgemm(CblasRowMajor, CblasNoTrans, CblasNoTrans, sizeD, sizeD, sizeD, 1.0f,
559 D, sizeD,
560 D, sizeD, 0.0f,
561 FLATTEN2D(D_2), sizeD);
562 D_3 = (float**)malloc2d(sizeD, sizeD, sizeof(float));
563 cblas_sgemm(CblasRowMajor, CblasNoTrans, CblasNoTrans, sizeD, sizeD, sizeD, 1.0f,
564 FLATTEN2D(D_2), sizeD,
565 D, sizeD, 0.0f,
566 FLATTEN2D(D_3), sizeD);
567 D_4 = (float**)malloc2d(sizeD, sizeD, sizeof(float));
568 cblas_sgemm(CblasRowMajor, CblasNoTrans, CblasNoTrans, sizeD, sizeD, sizeD, 1.0f,
569 FLATTEN2D(D_3), sizeD,
570 FLATTEN2D(D_3), sizeD, 0.0f,
571 FLATTEN2D(D_4), sizeD);
572 D_5 = (float**)malloc2d(sizeD, sizeD, sizeof(float));
573 cblas_sgemm(CblasRowMajor, CblasNoTrans, CblasNoTrans, sizeD, sizeD, sizeD, 1.0f,
574 FLATTEN2D(D_4), sizeD,
575 D, sizeD, 0.0f,
576 FLATTEN2D(D_5), sizeD);
577 s = ceilf(log2f(Frob_norm(FLATTEN2D(D_5), sizeD, sizeD)/
578 (1575.0f*tol*SAF_MIN(1.0f,Frob_norm(D, sizeD, sizeD))))/6.0f-1.0f);
579 s = SAF_MAX(s, 0.0f);
580
581 /* Get Pade approximation for expm(D*h2) = Y =
582 * (I-Dh+(2/5)*Dh^2-(1/15)*Dh^3)\‍(I+Dh+(2/5)*Dh^2+(1/15)*Dh^3)
583 * Ym1 = Y-I =
584 * (I-Dh+(2/5)*Dh^2-(1/15)*Dh^3)\‍(2*(Dh+(1/15)*Dh^3))
585 * (Calculate Ym1, not Y, to avoid precision loss from dominant I
586 * terms when Dh is small.) */
587 h2 = powf(2.0f,-s);
588 h = h2/2.0f;
589 hh = h*h;
590 hhh = hh*h;
591 Dh = (float**)malloc2d(sizeD, sizeD, sizeof(float));
592 memcpy(FLATTEN2D(Dh), D, sizeD*sizeD*sizeof(float));
593 utility_svsmul(FLATTEN2D(Dh), &h, sizeD*sizeD, NULL);
594 utility_svsmul(FLATTEN2D(D_2), &hh, sizeD*sizeD, NULL);
595 utility_svsmul(FLATTEN2D(D_3), &hhh, sizeD*sizeD, NULL);
596 Ym1 = (float**)malloc2d(sizeD, sizeD, sizeof(float));
597 for(i=0; i<sizeD; i++)
598 for(j=0; j<sizeD; j++)
599 Ym1[i][j] = Dh[i][j] + (1.0f/15.0f)*D_3[i][j];
600 Ym2 = (float**)malloc2d(sizeD, sizeD, sizeof(float));
601 for(i=0; i<sizeD; i++){
602 for(j=0; j<sizeD; j++){
603 Ym2[i][j] = (2.0f/5.0f)*D_2[i][j]-Ym1[i][j];
604 if(i==j)
605 Ym2[i][j] += 1.0f;
606 }
607 }
608 two = 2.0f;
609 utility_svsmul(FLATTEN2D(Ym1), &two, sizeD*sizeD, NULL);
610 utility_sglslv(NULL, FLATTEN2D(Ym2), sizeD, FLATTEN2D(Ym1), sizeD, FLATTEN2D(Ym1));
611
612 /* Y = Ym1+I = expm(D)
613 * Square Y (i.e., Y <-- Y*Y) s times to get Y = expm(D*2^s). */
614 for (k = 0; k<(int)s; k++){
615 cblas_sgemm(CblasRowMajor, CblasNoTrans, CblasNoTrans, sizeD, sizeD, sizeD, 1.0f,
616 FLATTEN2D(Ym1), sizeD,
617 FLATTEN2D(Ym1), sizeD, 0.0f,
618 FLATTEN2D(Ym2), sizeD);
619 for(i=0; i<sizeD; i++)
620 for(j=0; j<sizeD; j++)
621 Ym1[i][j] = Ym2[i][j] + 2.0f*Ym1[i][j]; /* (Ym1+I) <-- (Ym1+I)*(Ym1+I) */
622 }
623 memcpy(Y, FLATTEN2D(Ym1), sizeD*sizeD*sizeof(float));
624 if (m1){}
625 else{
626 for(i=0; i<sizeD; i++)
627 for(j=0; j<sizeD; j++)
628 if(i==j)
629 Y[i*sizeD+j] += 1.0f;
630 }
631
632 /* clean-up */
633 free(D_2);
634 free(D_3);
635 free(D_4);
636 free(D_5);
637 free(Dh);
638 free(Ym1);
639 free(Ym2);
640}
