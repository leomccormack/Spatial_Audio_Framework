38void hades_getInterpolatedHRTFs
39(
40 hades_analysis_handle const hAna,
41 HADES_HRTF_INTERP_OPTIONS interpOption,
42 hades_binaural_config* binConfig,
43 float* target_dirs_deg,
44 int nTargetDirs,
45 float_complex* hrtf_interp
46)
47{
48 hades_analysis_data *a = (hades_analysis_data*)(hAna);
49 int band, i, j, ntable, ntri;
50 int* idx;
51 float* itds_s, *interpTable, *w;
52 float_complex*** hrtf_fb; /* nBands x NUM_EARS x N_dirs */
53
54 /* Pass HRIRs through the filterbank */
55 hrtf_fb = (float_complex***)malloc3d(a->nBands, NUM_EARS, binConfig->nHRIR, sizeof(float_complex));
56 switch(a->fbOpt){
57 case HADES_USE_AFSTFT_LD: HRIRs2HRTFs_afSTFT(binConfig->hrirs, binConfig->nHRIR, binConfig->lHRIR, a->hopsize, 1, a->hybridmode, FLATTEN3D(hrtf_fb)); break;
58 case HADES_USE_AFSTFT: HRIRs2HRTFs_afSTFT(binConfig->hrirs, binConfig->nHRIR, binConfig->lHRIR, a->hopsize, 0, a->hybridmode, FLATTEN3D(hrtf_fb)); break;
59 }
60
61 /* Integration weights */
62 if (cblas_sasum(nTargetDirs, target_dirs_deg+1, 2)/(float)nTargetDirs<0.0001)
63 w = NULL;
64 else{
65 w = malloc1d(nTargetDirs*sizeof(float));
66 getVoronoiWeights(target_dirs_deg, nTargetDirs, 0, w);
67 }
68
69 /* estimate the ITDs for each HRIR */
70 itds_s = malloc1d(binConfig->nHRIR*sizeof(float));
71 estimateITDs(binConfig->hrirs, binConfig->nHRIR, binConfig->lHRIR, binConfig->hrir_fs, itds_s);
72
73 /* Apply HRTF interpolation */
74 switch(interpOption){
75 case HADES_HRTF_INTERP_NEAREST:
76 /* Quantise to nearest hrir direction */
77 idx = malloc1d(nTargetDirs*sizeof(int));
78 findClosestGridPoints(binConfig->hrir_dirs_deg, binConfig->nHRIR, target_dirs_deg, nTargetDirs, 1, idx, NULL, NULL);
79 for(band=0; band<a->nBands; band++)
80 for(i=0; i<NUM_EARS; i++)
81 for(j=0; j<nTargetDirs; j++)
82 hrtf_interp[band*NUM_EARS*nTargetDirs + i*nTargetDirs + j] = hrtf_fb[band][i][idx[j]];
83
84 /* Diffuse-field EQ without phase-simplification */
85 diffuseFieldEqualiseHRTFs(nTargetDirs, NULL, NULL, a->nBands, w, 1, 0, hrtf_interp);
86 free(idx);
87 break;
88
89 case HADES_HRTF_INTERP_TRIANGULAR:
90 /* Diffuse-field EQ with phase-simplification */
91 diffuseFieldEqualiseHRTFs(binConfig->nHRIR, itds_s, a->freqVector, a->nBands, w, 1, 1, FLATTEN3D(hrtf_fb));
92
93 /* Interpolation table */
94 interpTable = NULL;
95 generateVBAPgainTable3D_srcs(target_dirs_deg, nTargetDirs, binConfig->hrir_dirs_deg, binConfig->nHRIR, 0, 0, 0.0f, &interpTable, &ntable, &ntri);
96 VBAPgainTable2InterpTable(interpTable, nTargetDirs, binConfig->nHRIR);
97
98 /* Interpolate */
99 interpHRTFs(FLATTEN3D(hrtf_fb), itds_s, a->freqVector, interpTable, binConfig->nHRIR, a->nBands, nTargetDirs, hrtf_interp);
100
101 /* Clean-up */
102 free(interpTable);
103 break;
104 }
105
106 /* Clean-up */
107 free(itds_s);
108 free(hrtf_fb);
109 free(w);
110}

125void hades_sdMUSIC_create
126(
127 void ** const phMUSIC,
128 int nMics,
129 float* grid_dirs_deg,
130 int nDirs
131)
132{
133 *phMUSIC = malloc1d(sizeof(hades_sdMUSIC_data));
134 hades_sdMUSIC_data *h = (hades_sdMUSIC_data*)(*phMUSIC);
135
136 h->nMics = nMics;
137 h->nDirs = nDirs;
138
139 /* store cartesian coords of scanning directions (for optional peak finding) */
140 h->grid_dirs_xyz = malloc1d(h->nDirs * 3 * sizeof(float));
141 unitSph2cart(grid_dirs_deg, h->nDirs, 1, h->grid_dirs_xyz);
142
143 /* for run-time */
144 h->VnA = malloc1d(h->nMics * (h->nDirs) * sizeof(float_complex));
145 h->abs_VnA = malloc1d(h->nMics * (h->nDirs) * sizeof(float));
146 h->pSpec = malloc1d(h->nDirs*sizeof(float));
147 h->pSpecInv = malloc1d(h->nDirs*sizeof(float));
148 h->P_minus_peak = malloc1d(h->nDirs*sizeof(float));
149 h->VM_mask = malloc1d(h->nDirs*sizeof(float));
150}

173void hades_sdMUSIC_compute
174(
175 void* const hMUSIC,
176 float_complex* A_grid, /* nMics x nGrid */
177 float_complex *Vn, /* nMics x (nMics - nSrcs) */
178 int nSrcs,
179 float* P_music,
180 int* peak_inds
181)
182{
183 hades_sdMUSIC_data *h = (hades_sdMUSIC_data*)(hMUSIC);
184 int i, k, VnD2, peak_idx;
185 float kappa, scale;
186 float VM_mean[3];
187 const float_complex calpha = cmplxf(1.0f, 0.0f); const float_complex cbeta = cmplxf(0.0f, 0.0f);
188
189 VnD2 = h->nMics - nSrcs; /* noise subspace second dimension length */
190
191 /* derive the pseudo-spectrum value for each grid direction */
192 cblas_cgemm(CblasRowMajor, CblasConjTrans, CblasNoTrans, h->nDirs, VnD2, h->nMics, &calpha,
193 A_grid, h->nDirs,
194 Vn, VnD2, &cbeta,
195 h->VnA, VnD2);
196 utility_cvabs(h->VnA, (h->nDirs)*VnD2, h->abs_VnA);
197 for (i = 0; i < (h->nDirs); i++)
198 h->pSpecInv[i] = cblas_sdot(VnD2, &(h->abs_VnA[i*VnD2]), 1, &(h->abs_VnA[i*VnD2]), 1);
199 //h->pSpec[i] = 1.0f / (h->pSpecInv[i] + 2.23e-10f);
200 utility_svrecip(h->pSpecInv, h->nDirs, h->pSpec);
201
202 /* Output pseudo-spectrum */
203 if(P_music!=NULL)
204 cblas_scopy(h->nDirs, h->pSpec, 1, P_music, 1);
205
206 /* Peak-finding */
207 if(peak_inds!=NULL){
208 kappa = 50.0f;
209 scale = kappa/(2.0f*SAF_PI*expf(kappa)-expf(-kappa));
210 cblas_scopy(h->nDirs, h->pSpec, 1, h->P_minus_peak, 1);
211
212 /* Loop over the number of sources */
213 for(k=0; k<nSrcs; k++){
214 utility_simaxv(h->P_minus_peak, h->nDirs, &peak_idx);
215 peak_inds[k] = peak_idx;
216 if(k==nSrcs-1)
217 break;
218 VM_mean[0] = h->grid_dirs_xyz[peak_idx*3];
219 VM_mean[1] = h->grid_dirs_xyz[peak_idx*3+1];
220 VM_mean[2] = h->grid_dirs_xyz[peak_idx*3+2];
221
222 /* Apply mask for next iteration */
223 cblas_sgemm(CblasRowMajor, CblasNoTrans, CblasTrans, h->nDirs, 1, 3, 1.0f,
224 h->grid_dirs_xyz, 3,
225 (const float*)VM_mean, 3, 0.0f,
226 h->VM_mask, 1);
227 cblas_sscal(h->nDirs, kappa, h->VM_mask, 1);
228 for(i=0; i<h->nDirs; i++)
229 h->VM_mask[i] = expf(h->VM_mask[i]); /* VM distribution */
230 cblas_sscal(h->nDirs, scale, h->VM_mask, 1);
231 for(i=0; i<h->nDirs; i++)
232 h->VM_mask[i] = 1.0f/(0.00001f+(h->VM_mask[i])); /* inverse VM distribution */
233 utility_svvmul(h->P_minus_peak, h->VM_mask, h->nDirs, h->P_minus_peak);
234 }
235 }

238float hades_comedie
239(
240 float* lambda,
241 int N
242)
243{
244 int i;
245 float Nord, sum, g_0, mean_ev, g, sumAbsDiff;
246
247 Nord = sqrtf((float)N)-1.0f;
248 sum = 0.0f;
249 for(i=0; i<N; i++)
250 sum += lambda[i];
251 if(sum < 0.0001f) /* FLT_EPS*FLT_MIN -/+ range */
252 return 1.0f;
253 else{
254 g_0 = 2.0f*(powf(Nord+1.0f,2.0f)-1.0f);
255 mean_ev = (1.0f/(powf(Nord+1.0f,2.0f)))*sum;
256 sumAbsDiff = 0.0f;
257 for(i=0; i<N; i++)
258 sumAbsDiff += fabsf(lambda[i]-mean_ev);
259 g = (1.0f/mean_ev)*sumAbsDiff;
260 /* due to numerical error small (10e-7) negative numbers were occuring
261 * sometimes for the single plane-wave case; hence bounding it to >=0 */
262 return SAF_MAX(1.0f-g/g_0, 0.0f);
263 }
264}
