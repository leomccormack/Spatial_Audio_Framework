40void dirass_create
41(
42 void ** const phDir
43)
44{
45 dirass_data* pData = (dirass_data*)malloc1d(sizeof(dirass_data));
46 *phDir = (void*)pData;
47 int i;
48
49 /* Default user parameters */
50 pData->inputOrder = pData->new_inputOrder = SH_ORDER_FIRST;
51 pData->beamType = STATIC_BEAM_TYPE_HYPERCARDIOID;
52 pData->DirAssMode = REASS_UPSCALE;
53 pData->upscaleOrder = pData->new_upscaleOrder = UPSCALE_ORDER_TENTH;
54 pData->gridOption = GRID_GEOSPHERE_8;
55 pData->pmapAvgCoeff = 0.666f;
56 pData->minFreq_hz = 100.0f;
57 pData->maxFreq_hz = 8e3f;
58 pData->dispWidth = 120;
59 pData->chOrdering = CH_ACN;
60 pData->norm = NORM_SN3D;
61 pData->HFOVoption = HFOV_360;
62 pData->aspectRatioOption = ASPECT_RATIO_2_1;
63
64 /* codec data */
65 pData->pars = (dirass_codecPars*)malloc1d(sizeof(dirass_codecPars));
66 dirass_codecPars* pars = pData->pars;
67 pars->interp_dirs_deg = NULL;
68 pars->interp_dirs_rad = NULL;
69 pars->Y_up = NULL;
70 pars->interp_table = NULL;
71 pars->w = NULL;
72 pars->Cw = NULL;
73 pars->Uw = NULL;
74 pars->Cxyz = NULL;
75 pars->ss = NULL;
76 pars->ssxyz = NULL;
77 pars->est_dirs = NULL;
78 pars->est_dirs_idx = NULL;
79 pars->prev_intensity = NULL;
80 pars->prev_energy = NULL;
81
82 /* internal */
83 pData->progressBar0_1 = 0.0f;
84 pData->progressBarText = malloc1d(PROGRESSBARTEXT_CHAR_LENGTH*sizeof(char));
85 strcpy(pData->progressBarText,"");
86 pData->codecStatus = CODEC_STATUS_NOT_INITIALISED;
87 pData->procStatus = PROC_STATUS_NOT_ONGOING;
88
89 /* display */
90 pData->pmap = NULL;
91 for(i=0; i<NUM_DISP_SLOTS; i++)
92 pData->pmap_grid[i] = NULL;
93 pData->pmapReady = 0;
94 pData->recalcPmap = 1;
95
96 /* set FIFO buffers */
97 pData->FIFO_idx = 0;
98 memset(pData->inFIFO, 0, MAX_NUM_INPUT_SH_SIGNALS*DIRASS_FRAME_SIZE*sizeof(float));
99}

101void dirass_destroy
102(
103 void ** const phDir
104)
105{
106 dirass_data *pData = (dirass_data*)(*phDir);
107 dirass_codecPars* pars;
108 int i;
109
110 if (pData != NULL) {
111 /* not safe to free memory during intialisation/processing loop */
112 while (pData->codecStatus == CODEC_STATUS_INITIALISING ||
113 pData->procStatus == PROC_STATUS_ONGOING){
114 SAF_SLEEP(10);
115 }
116
117 if(pData->pmap!=NULL)
118 free(pData->pmap);
119 for(i=0; i<NUM_DISP_SLOTS; i++)
120 free(pData->pmap_grid[i]);
121
122 pars = pData->pars;
123 free(pars->interp_dirs_deg);
124 free(pars->interp_dirs_rad);
125 free(pars->Y_up);
126 free(pars->interp_table);
127 free(pars->ss);
128 free(pars->ssxyz);
129 free(pars->Cxyz);
130 free(pars->w);
131 free(pars->Cw);
132 free(pars->Uw);
133 free(pars->est_dirs);
134 free(pars->est_dirs_idx);
135 free(pars->prev_intensity);
136 free(pars->prev_energy);
137
138 free(pData->pars);
139 free(pData->progressBarText);
140 free(pData);
141 pData = NULL;
142 }
143}

146(
147 void * const hDir,
148 float sampleRate
149)
150{
151 dirass_data *pData = (dirass_data*)(hDir);
152 dirass_codecPars* pars = pData->pars;
153
154 pData->fs = sampleRate;
155
156 /* intialise parameters */
157 if(pars->prev_intensity!=NULL)
158 memset(pars->prev_intensity, 0, pars->grid_nDirs*3*sizeof(float));
159 if(pars->prev_energy!=NULL)
160 memset(pars->prev_energy, 0, pars->grid_nDirs*sizeof(float));
161 memset(pData->Wz12_hpf, 0, MAX_NUM_INPUT_SH_SIGNALS*2*sizeof(float));
162 memset(pData->Wz12_lpf, 0, MAX_NUM_INPUT_SH_SIGNALS*2*sizeof(float));
163 pData->pmapReady = 0;
164 pData->dispSlotIdx = 0;
165}

167void dirass_initCodec
168(
169 void* const hDir
170)
171{
172 dirass_data *pData = (dirass_data*)(hDir);
173
174 if (pData->codecStatus != CODEC_STATUS_NOT_INITIALISED)
175 return; /* re-init not required, or already happening */
176 while (pData->procStatus == PROC_STATUS_ONGOING){
177 /* re-init required, but we need to wait for the current processing loop to end */
178 pData->codecStatus = CODEC_STATUS_INITIALISING; /* indicate that we want to init */
179 SAF_SLEEP(10);
180 }
181
182 /* for progress bar */
183 pData->codecStatus = CODEC_STATUS_INITIALISING;
184 strcpy(pData->progressBarText,"Initialising");
185 pData->progressBar0_1 = 0.0f;
186
187 dirass_initAna(hDir);
188
189 /* done! */
190 strcpy(pData->progressBarText,"Done!");
191 pData->progressBar0_1 = 1.0f;
192 pData->codecStatus = CODEC_STATUS_INITIALISED;
193}

195void dirass_analysis
196(
197 void * const hDir,
198 const float *const * inputs,
199 int nInputs,
200 int nSamples,
201 int isPlaying
202)
203{
204 dirass_data *pData = (dirass_data*)(hDir);
205 dirass_codecPars* pars = pData->pars;
206 int s, i, j, k, ch, sec_nSH, secOrder, nSH, up_nSH;
207 float intensity[3];
208
209 /* local copy of user parameters */
210 int inputOrder, DirAssMode, upscaleOrder;
211 float pmapAvgCoeff, minFreq_hz, maxFreq_hz;
212 NORM_TYPES norm;
213 CH_ORDER chOrdering;
214 norm = pData->norm;
215 chOrdering = pData->chOrdering;
216 pmapAvgCoeff = pData->pmapAvgCoeff;
217 DirAssMode = pData->DirAssMode;
218 upscaleOrder = pData->upscaleOrder;
219 minFreq_hz = pData->minFreq_hz;
220 maxFreq_hz = pData->maxFreq_hz;
221 inputOrder = pData->inputOrder;
222 secOrder = inputOrder-1;
223 nSH = (inputOrder+1)*(inputOrder+1);
224 sec_nSH = (secOrder+1)*(secOrder+1);
225 up_nSH = (upscaleOrder+1)*(upscaleOrder+1);
226
227 /* Loop over all samples */
228 for(s=0; s<nSamples; s++){
229 /* Load input signals into inFIFO buffer */
230 for(ch=0; ch<SAF_MIN(nInputs,nSH); ch++)
231 pData->inFIFO[ch][pData->FIFO_idx] = inputs[ch][s];
232 for(; ch<nSH; ch++) /* Zero any channels that were not given */
233 pData->inFIFO[ch][pData->FIFO_idx] = 0.0f;
234
235 /* Increment buffer index */
236 pData->FIFO_idx++;
237
238 /* Process frame if inFIFO is full and codec is ready for it */
239 if (pData->FIFO_idx >= DIRASS_FRAME_SIZE && (pData->codecStatus == CODEC_STATUS_INITIALISED) && isPlaying) {
240 pData->FIFO_idx = 0;
241 pData->procStatus = PROC_STATUS_ONGOING;
242
243 /* Load time-domain data */
244 for(ch=0; ch<nSH; ch++)
245 memcpy(pData->SHframeTD[ch],pData->inFIFO[ch], DIRASS_FRAME_SIZE*sizeof(float));
246
247 /* account for input channel order */
248 switch(chOrdering){
249 case CH_ACN: /* already ACN */ break; /* Otherwise, convert to ACN... */
250 case CH_FUMA: convertHOAChannelConvention((float*)pData->SHframeTD, inputOrder, DIRASS_FRAME_SIZE, HOA_CH_ORDER_FUMA, HOA_CH_ORDER_ACN); break;
251 }
252
253 /* account for input normalisation scheme */
254 switch(norm){
255 case NORM_N3D: /* already in N3D, do nothing */ break; /* Otherwise, convert to N3D... */
256 case NORM_SN3D: convertHOANormConvention((float*)pData->SHframeTD, inputOrder, DIRASS_FRAME_SIZE, HOA_NORM_SN3D, HOA_NORM_N3D); break;
257 case NORM_FUMA: convertHOANormConvention((float*)pData->SHframeTD, inputOrder, DIRASS_FRAME_SIZE, HOA_NORM_FUMA, HOA_NORM_N3D); break;
258 }
259
260 /* update the dirass powermap */
261 if(pData->recalcPmap==1){
262 pData->recalcPmap = 0;
263 pData->pmapReady = 0;
264
265 /* filter input signals */
266 float b[3], a[3];
267 biQuadCoeffs(BIQUAD_FILTER_HPF, minFreq_hz, pData->fs, 0.7071f, 0.0f, b, a);
268 for(i=0; i<nSH; i++)
269 applyBiQuadFilter(b, a, pData->Wz12_hpf[i], pData->SHframeTD[i], DIRASS_FRAME_SIZE);
270 biQuadCoeffs(BIQUAD_FILTER_LPF, maxFreq_hz, pData->fs, 0.7071f, 0.0f, b, a);
271 for(i=0; i<nSH; i++)
272 applyBiQuadFilter(b, a, pData->Wz12_lpf[i], pData->SHframeTD[i], DIRASS_FRAME_SIZE);
273
274 /* DoA estimation for each spatially-localised sector */
275 if(DirAssMode==REASS_UPSCALE || DirAssMode==REASS_NEAREST){
276 /* Beamform using the sector patterns */
277 cblas_sgemm(CblasRowMajor, CblasNoTrans, CblasNoTrans, pars->grid_nDirs, DIRASS_FRAME_SIZE, sec_nSH, 1.0f,
278 pars->Cw, sec_nSH,
279 (const float*)pData->SHframeTD, DIRASS_FRAME_SIZE, 0.0f,
280 pars->ss, DIRASS_FRAME_SIZE);
281
282 for(i=0; i<pars->grid_nDirs; i++){
283 /* beamforming to get velocity patterns */
284 cblas_sgemm(CblasRowMajor, CblasTrans, CblasNoTrans, 3, DIRASS_FRAME_SIZE, nSH, 1.0f,
285 &(pars->Cxyz[i*nSH*3]), 3,
286 (const float*)pData->SHframeTD, DIRASS_FRAME_SIZE, 0.0f,
287 pars->ssxyz, DIRASS_FRAME_SIZE);
288
289 /* take the sum or mean ss.*ssxyz, to get intensity vector */
290 memset(intensity, 0, 3*sizeof(float));
291 for(k=0; k<3; k++){
292 for(j=0; j<DIRASS_FRAME_SIZE; j++)
293 intensity[k] += pars->ssxyz[k*DIRASS_FRAME_SIZE + j] * pars->ss[i*DIRASS_FRAME_SIZE+j];
294 intensity[k] /= (float)DIRASS_FRAME_SIZE;
295
296 /* average over time */
297 intensity[k] = pmapAvgCoeff * (pars->prev_intensity[i*3+k]) + (1.0f-pmapAvgCoeff) * intensity[k];
298 pars->prev_intensity[i*3+k] = intensity[k];
299 }
300
301 /* extract DoA [azi elev] convention */
302 pars->est_dirs[i*2] = atan2f(intensity[1], intensity[0]);
303 pars->est_dirs[i*2+1] = atan2f(intensity[2], sqrtf(powf(intensity[0], 2.0f) + powf(intensity[1], 2.0f)));
304 if(DirAssMode==REASS_UPSCALE)
305 pars->est_dirs[i*2+1] = M_PI/2.0f - pars->est_dirs[i*2+1]; /* convert to inclination */
306 }
307 }
308
309 /* Obtain pmap/upscaled pmap in the case of REASS_MODE_OFF and REASS_UPSCALE modes, respectively.
310 * OR find the nearest display grid indices, corresponding to the DoA estimates, for the REASS_NEAREST mode */
311 switch(DirAssMode) {
312 default:
313 case REASS_MODE_OFF:
314 /* Standard beamformer-based pmap */
315 cblas_sgemm(CblasRowMajor, CblasNoTrans, CblasNoTrans, pars->grid_nDirs, DIRASS_FRAME_SIZE, nSH, 1.0f,
316 pars->w, nSH,
317 (const float*)pData->SHframeTD, DIRASS_FRAME_SIZE, 0.0f,
318 pars->ss, DIRASS_FRAME_SIZE);
319
320 /* sum energy over the length of the frame to obtain the pmap */
321 memset(pData->pmap, 0, pars->grid_nDirs *sizeof(float));
322 for(i=0; i<pars->grid_nDirs; i++)
323 for(j=0; j<DIRASS_FRAME_SIZE; j++)
324 pData->pmap[i] += (pars->ss[i*DIRASS_FRAME_SIZE+j])*(pars->ss[i*DIRASS_FRAME_SIZE+j]);
325
326 /* average energy over time */
327 for(i=0; i<pars->grid_nDirs; i++){
328 pData->pmap[i] = pmapAvgCoeff * (pars->prev_energy[i]) + (1.0f-pmapAvgCoeff) * (pData->pmap[i]);
329 pars->prev_energy[i] = pData->pmap[i];
330 }
331
332 /* interpolate the pmap */
333 cblas_sgemm(CblasRowMajor, CblasNoTrans, CblasNoTrans, pars->interp_nDirs, 1, pars->grid_nDirs, 1.0f,
334 pars->interp_table, pars->grid_nDirs,
335 pData->pmap, 1, 0.0f,
336 pData->pmap_grid[pData->dispSlotIdx], 1);
337 break;
338
339 case REASS_UPSCALE:
340 /* upscale */
341 getSHreal_recur(upscaleOrder, pars->est_dirs, pars->grid_nDirs, pars->Y_up);
342 cblas_sgemm(CblasRowMajor, CblasNoTrans, CblasNoTrans, up_nSH, DIRASS_FRAME_SIZE, pars->grid_nDirs, 1.0f,
343 pars->Y_up, pars->grid_nDirs,
344 pars->ss, DIRASS_FRAME_SIZE, 0.0f,
345 (float*)pData->SHframe_upTD, DIRASS_FRAME_SIZE);
346
347 /* Beamform using the new spatially upscaled frame */
348 cblas_sgemm(CblasRowMajor, CblasNoTrans, CblasNoTrans, pars->grid_nDirs, DIRASS_FRAME_SIZE, up_nSH, 1.0f,
349 pars->Uw, up_nSH,
350 (float*)pData->SHframe_upTD, DIRASS_FRAME_SIZE, 0.0f,
351 pars->ss, DIRASS_FRAME_SIZE);
352
353 /* sum energy over the length of the frame to obtain the pmap */
354 memset(pData->pmap, 0, pars->grid_nDirs *sizeof(float));
355 for(i=0; i<pars->grid_nDirs; i++)
356 for(j=0; j<DIRASS_FRAME_SIZE; j++)
357 pData->pmap[i] += (pars->ss[i*DIRASS_FRAME_SIZE+j])*(pars->ss[i*DIRASS_FRAME_SIZE+j]);
358
359 /* average energy over time */
360 for(i=0; i<pars->grid_nDirs; i++){
361 pData->pmap[i] = pmapAvgCoeff * (pars->prev_energy[i]) + (1.0f-pmapAvgCoeff) * (pData->pmap[i]);
362 pars->prev_energy[i] = pData->pmap[i];
363 }
364
365 /* interpolate the pmap */
366 cblas_sgemm(CblasRowMajor, CblasNoTrans, CblasNoTrans, pars->interp_nDirs, 1, pars->grid_nDirs, 1.0f,
367 pars->interp_table, pars->grid_nDirs,
368 pData->pmap, 1, 0.0f,
369 pData->pmap_grid[pData->dispSlotIdx], 1);
370 break;
371
372 case REASS_NEAREST:
373 /* Assign the sector energies to the nearest display grid point */
374 findClosestGridPoints(pars->interp_dirs_rad, pars->interp_nDirs, pars->est_dirs, pars->grid_nDirs, 0, pars->est_dirs_idx, NULL, NULL);
375 memset(pData->pmap_grid[pData->dispSlotIdx], 0, pars->interp_nDirs * sizeof(float));
376 for(i=0; i< pars->grid_nDirs; i++)
377 for(j=0; j<DIRASS_FRAME_SIZE; j++)
378 pData->pmap[i] = (pars->ss[i*DIRASS_FRAME_SIZE+j])*(pars->ss[i*DIRASS_FRAME_SIZE+j]);
379
380 /* average energy over time, and assign to nearest grid direction */
381 for(i=0; i<pars->grid_nDirs; i++){
382 pData->pmap[i] = pmapAvgCoeff * (pars->prev_energy[i]) + (1.0f-pmapAvgCoeff) * (pData->pmap[i]);
383 pars->prev_energy[i] = pData->pmap[i];
384 pData->pmap_grid[pData->dispSlotIdx][pars->est_dirs_idx[i]] += pData->pmap[i];
385 }
386 break;
387 }
388
389 /* ascertain the minimum and maximum values for pmap colour scaling */
390 int ind;
391 utility_siminv(pData->pmap_grid[pData->dispSlotIdx], pars->interp_nDirs, &ind);
392 pData->pmap_grid_minVal = pData->pmap_grid[pData->dispSlotIdx][ind];
393 utility_simaxv(pData->pmap_grid[pData->dispSlotIdx], pars->interp_nDirs, &ind);
394 pData->pmap_grid_maxVal = pData->pmap_grid[pData->dispSlotIdx][ind];
395
396 /* normalise the pmap to 0..1 */
397 for(i=0; i<pars->interp_nDirs; i++)
398 pData->pmap_grid[pData->dispSlotIdx][i] = (pData->pmap_grid[pData->dispSlotIdx][i]-pData->pmap_grid_minVal)/(pData->pmap_grid_maxVal-pData->pmap_grid_minVal+1e-11f);
399
400 /* signify that the pmap in the current slot is ready for plotting */
401 pData->dispSlotIdx++;
402 if(pData->dispSlotIdx>=NUM_DISP_SLOTS)
403 pData->dispSlotIdx = 0;
404 pData->pmapReady = 1;
405 }
406 }
407 else if(pData->FIFO_idx >= DIRASS_FRAME_SIZE){
408 /* reset FIFO_idx index if codec was not ready */
409 pData->FIFO_idx = 0;
410 }
411 }
412
413 pData->procStatus = PROC_STATUS_NOT_ONGOING;
414}

658{
659 dirass_data *pData = (dirass_data*)(hDir);

662 (*grid_dirs) = pars->interp_dirs_deg;
663 (*pmap) = pData->pmap_grid[pData->dispSlotIdx-1 < 0 ? NUM_DISP_SLOTS-1 : pData->dispSlotIdx-1];
664 (*nDirs) = pars->interp_nDirs;
665 (*pmapWidth) = pData->dispWidth;
666 switch(pData->HFOVoption){
667 default:
668 case HFOV_360: (*hfov) = 360; break;
669 case HFOV_180: (*hfov) = 180; break;
670 case HFOV_90: (*hfov) = 90; break;
671 case HFOV_60: (*hfov) = 60; break;
672 }
673 switch(pData->aspectRatioOption){
674 default:
675 case ASPECT_RATIO_2_1: (*aspectRatio) = 2.0f; break;
676 case ASPECT_RATIO_16_9: (*aspectRatio) = 16.0f/9.0f; break;
677 case ASPECT_RATIO_4_3: (*aspectRatio) = 4.0f/3.0f; break;
678 }
679 }
680 return pData->pmapReady;
