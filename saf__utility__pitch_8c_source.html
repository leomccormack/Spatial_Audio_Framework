39static void smbFft(float *fftBuffer, int fftFrameSize, long sign)
40/*
41 FFT routine, (C)1996 S.M.Bernsee. Sign = -1 is FFT, 1 is iFFT (inverse)
42 Fills fftBuffer[0...2*fftFrameSize-1] with the Fourier transform of the
43 time domain data in fftBuffer[0...2*fftFrameSize-1]. The FFT array takes
44 and returns the cosine and sine parts in an interleaved manner, ie.
45 fftBuffer[0] = cosPart[0], fftBuffer[1] = sinPart[0], asf. fftFrameSize
46 must be a power of 2. It expects a complex input signal (see footnote 2),
47 ie. when working with 'common' audio signals our input signal has to be
48 passed as {in[0],0.,in[1],0.,in[2],0.,...} asf. In that case, the transform
49 of the frequencies of interest is in fftBuffer[0...fftFrameSize].
50*/
51{
52 float wr, wi, arg, *p1, *p2, temp;
53 float tr, ti, ur, ui, *p1r, *p1i, *p2r, *p2i;
54 long i, bitm, j, le, le2, k;
55
56 for (i = 2; i < 2*fftFrameSize-2; i += 2) {
57 for (bitm = 2, j = 0; bitm < 2*fftFrameSize; bitm <<= 1) {
58 if (i & bitm) j++;
59 j <<= 1;
60 }
61 if (i < j) {
62 p1 = fftBuffer+i; p2 = fftBuffer+j;
63 temp = *p1; *(p1++) = *p2;
64 *(p2++) = temp; temp = *p1;
65 *p1 = *p2; *p2 = temp;
66 }
67 }
68 for (k = 0, le = 2; k < (int)(log(fftFrameSize)/log(2.)+.5); k++) {
69 le <<= 1;
70 le2 = le>>1;
71 ur = 1.0;
72 ui = 0.0;
73 arg = SAF_PI / (le2>>1);
74 wr = cos(arg);
75 wi = sign*sin(arg);
76 for (j = 0; j < le2; j += 2) {
77 p1r = fftBuffer+j; p1i = p1r+1;
78 p2r = p1r+le2; p2i = p2r+1;
79 for (i = j; i < 2*fftFrameSize; i += le) {
80 tr = *p2r * ur - *p2i * ui;
81 ti = *p2r * ui + *p2i * ur;
82 *p2r = *p1r - tr; *p2i = *p1i - ti;
83 *p1r += tr; *p1i += ti;
84 p1r += le; p1i += le;
85 p2r += le; p2i += le;
86 }
87 tr = ur*wr - ui*wi;
88 ui = ur*wi + ui*wr;
89 ur = tr;
90 }
91 }
92}
93#endif

123void smb_pitchShift_create
124(
125 void** hSmb,
126 int nCH,
127 int fftFrameSize,
128 int osamp,
129 float sampleRate
130)
131{
132 *hSmb = malloc(sizeof(smb_pitchShift_data));
133 smb_pitchShift_data *h = (smb_pitchShift_data*)(*hSmb);
134 int i;
135
136 h->fftFrameSize = fftFrameSize;
137 h->osamp = osamp;
138 h->sampleRate = sampleRate;
139 h->nCH = nCH;
140 h->pitchShiftFactor = 1.0f;
141
142 /* internals */
143 saf_fft_create(&(h->hFFT), fftFrameSize);
144 h->stepSize = fftFrameSize/h->osamp;
145 h->inFifoLatency = fftFrameSize - (h->stepSize);
146 h->gRover = (int*)malloc1d(nCH * sizeof(int));
147 for(i=0; i<nCH; i++)
148 h->gRover[i] = h->inFifoLatency;
149 h->window = (float*)malloc1d(fftFrameSize*sizeof(float));
150 for (i = 0; i < fftFrameSize; i++)
151 h->window[i] = -0.5f*cosf(2.0f*SAF_PI*(float)i/(float)fftFrameSize)+0.5f;
152 h->gInFIFO = (float**)calloc2d(nCH,fftFrameSize,sizeof(float));
153 h->gOutFIFO = (float**)calloc2d(nCH,fftFrameSize,sizeof(float));
154#ifdef SMB_ENABLE_SAF_FFT
155 h->gFFTworksp_td = (float_complex**)calloc2d(nCH,fftFrameSize,sizeof(float_complex));
156 h->gFFTworksp_fd = (float_complex**)calloc2d(nCH,fftFrameSize,sizeof(float_complex));
157#else
158 h->gFFTworksp = (float**)calloc2d(nCH,2*fftFrameSize,sizeof(float));
159#endif
160 h->gLastPhase = (float**)calloc2d(nCH,fftFrameSize/2+1,sizeof(float));
161 h->gSumPhase = (float**)calloc2d(nCH,fftFrameSize/2+1,sizeof(float));
162 h->gOutputAccum = (float**)calloc2d(nCH,2*(h->fftFrameSize),sizeof(float));
163 h->gAnaFreq = (float**)calloc2d(nCH,fftFrameSize,sizeof(float));
164 h->gAnaMagn = (float**)calloc2d(nCH,fftFrameSize,sizeof(float));
165 h->gSynFreq = (float**)malloc2d(nCH,fftFrameSize,sizeof(float));
166 h->gSynMagn = (float**)malloc2d(nCH,fftFrameSize,sizeof(float));
167}

211void smb_pitchShift_apply
212(
213 void* hSmb,
214 float pitchShift,
215 int frameSize,
216 float *indata,
217 float *outdata
218)
219{
220 smb_pitchShift_data *h = (smb_pitchShift_data*)(hSmb);
221 float magn, phase, tmp, real, imag;
222 float freqPerBin, expct;
223 int ch, i, k, qpd, index, fftFrameSize2;
224
225 /* set up some handy variables */
226 fftFrameSize2 = h->fftFrameSize/2;
227 freqPerBin = h->sampleRate/(float)h->fftFrameSize;
228 expct = 2.0f*SAF_PI*(float)(h->stepSize)/(float)h->fftFrameSize;
229
230 /* flush buffers */
231 if(h->pitchShiftFactor!=pitchShift){
232 h->pitchShiftFactor = pitchShift;
233 for(ch=0; ch<h->nCH; ch++){
234 memset(h->gOutputAccum[ch], 0, h->stepSize*sizeof(float));
235 memset(h->gLastPhase[ch], 0, (h->fftFrameSize/2+1)*sizeof(float));
236 memset(h->gSumPhase[ch], 0, (h->fftFrameSize/2+1)*sizeof(float));
237 }
238 }
239
240 /* main processing loop */
241 for(ch=0; ch<h->nCH; ch++){
242 for (i = 0; i < frameSize; i++){
243 /* As long as we have not yet collected enough data just read in */
244 h->gInFIFO[ch][h->gRover[ch]] = indata[ch*frameSize+i];
245 outdata[ch*frameSize+i] = h->gOutFIFO[ch][h->gRover[ch]-(h->inFifoLatency)];
246 h->gRover[ch]++;
247
248 /* now we have enough data for processing */
249 if (h->gRover[ch] >= h->fftFrameSize) {
250 h->gRover[ch] = h->inFifoLatency;
251
252#ifdef SMB_ENABLE_SAF_FFT
253 for (k = 0; k < h->fftFrameSize;k++)
254 h->gFFTworksp_td[ch][k] = cmplxf(h->gInFIFO[ch][k] * (float)h->window[k], 0.0f);
255 saf_fft_forward(h->hFFT, h->gFFTworksp_td[ch], h->gFFTworksp_fd[ch]);
256#else
257 /* do windowing and re,im interleave */
258 for (k = 0; k < h->fftFrameSize;k++) {
259 h->gFFTworksp[ch][2*k] = h->gInFIFO[ch][k] * h->window[k];
260 h->gFFTworksp[ch][2*k+1] = 0.;
261 }
262
263 /* do transform */
264 smbFft(h->gFFTworksp[ch], h->fftFrameSize, -1);
265#endif
266
267 /* ***************** ANALYSIS ******************* */
268 for (k = 0; k <= fftFrameSize2; k++) {
269#ifdef SMB_ENABLE_SAF_FFT
270 real = crealf(h->gFFTworksp_fd[ch][k]);
271 imag = cimagf(h->gFFTworksp_fd[ch][k]);
272#else
273 /* de-interlace FFT buffer */
274 real = h->gFFTworksp[ch][2*k];
275 imag = h->gFFTworksp[ch][2*k+1];
276#endif
277 /* compute magnitude and phase */
278 magn = 2.0f*sqrtf(real*real + imag*imag);
279 phase = atan2f(imag,real);
280
281 /* compute phase difference */
282 tmp = phase - h->gLastPhase[ch][k];
283 h->gLastPhase[ch][k] = phase;
284
285 /* subtract expected phase difference */
286 tmp -= (float)k*expct;
287
288 /* map delta phase into +/- Pi interval */
289 qpd = (int)(tmp/SAF_PI);
290 if (qpd >= 0) qpd += qpd&1;
291 else qpd -= qpd&1;
292 tmp -= SAF_PI*(float)qpd;
293
294 /* get deviation from bin frequency from the +/- Pi interval */
295 tmp = (float)h->osamp*tmp/(2.0f*SAF_PI);
296
297 /* compute the k-th partials' true frequency */
298 tmp = (float)k*freqPerBin + tmp*freqPerBin;
299
300 /* store magnitude and true frequency in analysis arrays */
301 h->gAnaMagn[ch][k] = magn;
302 h->gAnaFreq[ch][k] = tmp;
303
304 }
305
306 /* ***************** PROCESSING ******************* */
307 /* this does the actual pitch shifting */
308 memset(h->gSynMagn[ch], 0, h->fftFrameSize*sizeof(float));
309 memset(h->gSynFreq[ch], 0, h->fftFrameSize*sizeof(float));
310 for (k = 0; k <= fftFrameSize2; k++) {
311 index = (int)((float)k*(h->pitchShiftFactor));
312 if (index <= fftFrameSize2) {
313 h->gSynMagn[ch][index] += (h->gAnaMagn[ch][k]);
314 h->gSynFreq[ch][index] = h->gAnaFreq[ch][k] * (h->pitchShiftFactor);
315 }
316 }
317
318 /* ***************** SYNTHESIS ******************* */
319 /* this is the synthesis step */
320 for (k = 0; k <= fftFrameSize2; k++) {
321 /* get magnitude and true frequency from synthesis arrays */
322 magn = h->gSynMagn[ch][k];
323 tmp = h->gSynFreq[ch][k];
324
325 /* subtract bin mid frequency */
326 tmp -= (float)k*freqPerBin;
327
328 /* get bin deviation from freq deviation */
329 tmp /= freqPerBin;
330
331 /* take osamp into account */
332 tmp = 2.0f*SAF_PI*tmp/(h->osamp);
333
334 /* add the overlap phase advance back in */
335 tmp += (float)k*expct;
336
337 /* accumulate delta phase to get bin phase */
338 h->gSumPhase[ch][k] += tmp;
339 phase = h->gSumPhase[ch][k];
340
341#ifdef SMB_ENABLE_SAF_FFT
342 h->gFFTworksp_fd[ch][k] = cmplxf(magn*cosf(phase), magn*sinf(phase));
343#else
344 /* get real and imag part and re-interleave */
345 h->gFFTworksp[ch][2*k] = magn*cos(phase);
346 h->gFFTworksp[ch][2*k+1] = magn*sin(phase);
347#endif
348 }
349
350#ifdef SMB_ENABLE_SAF_FFT
351 for (k = fftFrameSize2+1; k < h->fftFrameSize; k++)
352 h->gFFTworksp_fd[ch][k] = cmplxf(0.0f, 0.0f);
353
354 saf_fft_backward(h->hFFT, h->gFFTworksp_fd[ch], h->gFFTworksp_td[ch]);
355
356 for(k=0; k < h->fftFrameSize; k++)
357 h->gOutputAccum[ch][k] += 2.0f*(h->window[k])*crealf(h->gFFTworksp_td[ch][k])/((h->osamp));
358#else
359 /* zero negative frequencies */
360 for (k = h->fftFrameSize+2; k < 2*(h->fftFrameSize); k++)
361 h->gFFTworksp[ch][k] = 0.;
362
363 /* do inverse transform */
364 smbFft(h->gFFTworksp[ch], h->fftFrameSize, 1);
365
366 /* do windowing and add to output accumulator */
367 for(k=0; k < h->fftFrameSize; k++)
368 h->gOutputAccum[ch][k] += 2.*(h->window[k])*(h->gFFTworksp[ch][2*k])/(fftFrameSize2*(h->osamp));
369#endif
370 for (k = 0; k < h->stepSize; k++)
371 h->gOutFIFO[ch][k] = h->gOutputAccum[ch][k];
372
373 /* shift accumulator */
374 memmove(h->gOutputAccum[ch], h->gOutputAccum[ch] + (h->stepSize), h->fftFrameSize*sizeof(float));
375
376 /* move input FIFO */
377 for (k = 0; k < h->inFifoLatency; k++)
378 h->gInFIFO[ch][k] = h->gInFIFO[ch][k+h->stepSize];
379 }
380 }
381 }
