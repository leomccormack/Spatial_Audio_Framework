84void cdf4sap_create
85(
86 void ** const phCdf,
87 int nXcols,
88 int nYcols
89)
90{
91 *phCdf = malloc1d(sizeof(cdf4sap_data));
92 cdf4sap_data *h = (cdf4sap_data*)(*phCdf);
93
94 h->nXcols = nXcols;
95 h->nYcols = nYcols;
96 h->lambda = malloc1d(nYcols * nXcols * sizeof(float));
97
98 /* For the SVD */
99 utility_ssvd_create(&h->hSVD, SAF_MAX(nXcols, nYcols), SAF_MAX(nXcols, nYcols));
100
101 /* For the decomposition of Cy */
102 h->U_Cy = malloc1d(nYcols*nYcols*sizeof(float));
103 h->S_Cy = malloc1d(nYcols*nYcols*sizeof(float));
104 h->Ky = malloc1d(nYcols*nYcols*sizeof(float));
105
106 /* For the decomposition of Cx */
107 h->U_Cx = malloc1d(nXcols*nXcols*sizeof(float));
108 h->S_Cx = malloc1d(nXcols*nXcols*sizeof(float));
109 h->s_Cx = malloc1d(nXcols*sizeof(float));
110 h->Kx = malloc1d(nXcols*nXcols*sizeof(float));
111
112 /* For the formulation of regularised Kx^-1 */
113 h->Kx_reg_inverse = malloc1d(nXcols*nXcols*sizeof(float));
114
115 /* For the formulation of normalisation matrix G_hat */
116 h->G_hat = malloc1d(nYcols*nYcols*sizeof(float));
117 h->Cx_QH = malloc1d(nXcols*nYcols*sizeof(float));
118
119 /* For the formulation of optimal P */
120 h->GhatH_Ky = malloc1d(nYcols*nYcols*sizeof(float));
121 h->QH_GhatH_Ky = malloc1d(nXcols*nYcols*sizeof(float));
122 h->KxH_QH_GhatH_Ky = malloc1d(nXcols*nYcols*sizeof(float));
123 h->U = malloc1d(nXcols*nXcols*sizeof(float));
124 h->V = malloc1d(nYcols*nYcols*sizeof(float));
125 h->lambda_UH = malloc1d(nYcols*nXcols*sizeof(float));
126 h->P = malloc1d(nYcols*nXcols*sizeof(float));
127
128 /* For the formulation of M */
129 h->P_Kxreginverse = malloc1d(nYcols*nXcols*sizeof(float));
130
131 /* For the formulation of the residual covariance matrix */
132 h->Cx_MH = malloc1d(nXcols*nYcols*sizeof(float));
133 h->Cy_tilde = malloc1d(nYcols*nYcols*sizeof(float));
134
135 /* For using energy compensation instead of residuals */
136 h->G_M = malloc1d(nYcols*nXcols*sizeof(float));
137}

268void formulate_M_and_Cr
269(
270 void * const hCdf,
271 float* Cx,
272 float* Cy,
273 float* Q,
274 int useEnergyFLAG,
275 float reg,
276 float* M,
277 float* Cr
278)
279{
280 cdf4sap_data *h = (cdf4sap_data*)(hCdf);
281 int i, j, nXcols, nYcols;
282
283 nXcols = h->nXcols;
284 nYcols = h->nYcols;
285
286 memset(h->lambda, 0, nYcols * nXcols * sizeof(float));
287 for(i = 0; i<SAF_MIN(nXcols,nYcols); i++)
288 h->lambda[i*nXcols + i] = 1.0f;
289
290 /* Decomposition of Cy */
291 utility_ssvd(h->hSVD, Cy, nYcols, nYcols, h->U_Cy, h->S_Cy, NULL, NULL);
292 for(i=0; i< nYcols; i++)
293 h->S_Cy[i*nYcols+i] = sqrtf(SAF_MAX(h->S_Cy[i*nYcols+i], 2.23e-20f));
294 cblas_sgemm(CblasRowMajor, CblasNoTrans, CblasNoTrans, nYcols, nYcols, nYcols, 1.0f,
295 h->U_Cy, nYcols,
296 h->S_Cy, nYcols, 0.0f,
297 h->Ky, nYcols);
298
299 /* Decomposition of Cx */
300 utility_ssvd(h->hSVD, Cx, nXcols, nXcols, h->U_Cx, h->S_Cx, NULL, h->s_Cx);
301 for(i=0; i< nXcols; i++){
302 h->S_Cx[i*nXcols+i] = sqrtf(SAF_MAX(h->S_Cx[i*nXcols+i], 2.23e-20f));
303 h->s_Cx[i] = sqrtf(SAF_MAX(h->s_Cx[i], 2.23e-20f));
304 }
305 cblas_sgemm(CblasRowMajor, CblasNoTrans, CblasNoTrans, nXcols, nXcols, nXcols, 1.0f,
306 h->U_Cx, nXcols,
307 h->S_Cx, nXcols, 0.0f,
308 h->Kx, nXcols);
309
310 /* Regularisation of S_Cx */
311 int ind;
312 float limit, maxVal;
313 utility_simaxv(h->s_Cx, nXcols, &ind);
314 limit = h->s_Cx[ind] * reg + 2.23e-13f;
315 for(i=0; i < nXcols; i++)
316 h->S_Cx[i*nXcols+i] = 1.0f / SAF_MAX(h->S_Cx[i*nXcols+i], limit);
317
318 /* Formulate regularised Kx^-1 */
319 cblas_sgemm(CblasRowMajor, CblasNoTrans, CblasTrans, nXcols, nXcols, nXcols, 1.0f,
320 h->S_Cx, nXcols,
321 h->U_Cx, nXcols, 0.0f,
322 h->Kx_reg_inverse, nXcols);
323
324 /* Formulate normalisation matrix G_hat */
325 cblas_sgemm(CblasRowMajor, CblasNoTrans, CblasTrans, nXcols, nYcols, nXcols, 1.0f,
326 Cx, nXcols,
327 Q, nXcols, 0.0f,
328 h->Cx_QH, nYcols);
329 cblas_sgemm(CblasRowMajor, CblasNoTrans, CblasNoTrans, nYcols, nYcols, nXcols, 1.0f,
330 Q, nXcols,
331 h->Cx_QH, nYcols, 0.0f,
332 h->G_hat, nYcols);
333 maxVal = -2.23e13f;
334 for(i=0; i< nYcols; i++)
335 maxVal = h->G_hat[i*nYcols+i] > maxVal ? h->G_hat[i*nYcols+i] : maxVal;
336 limit = maxVal * 0.001f + 2.23e-13f;
337 for(i=0; i < nYcols; i++)
338 for(j=0; j < nYcols; j++)
339 h->G_hat[i*nYcols+j] = i==j ? sqrtf(SAF_MAX(Cy[i*nYcols+j], 2.23e-13f) / SAF_MAX(h->G_hat[i*nYcols+j], limit)) : 0.0f;
340
341 /* Formulate optimal P */
342 cblas_sgemm(CblasRowMajor, CblasTrans, CblasNoTrans, nYcols, nYcols, nYcols, 1.0f,
343 h->G_hat, nYcols,
344 h->Ky, nYcols, 0.0f,
345 h->GhatH_Ky, nYcols);
346 cblas_sgemm(CblasRowMajor, CblasTrans, CblasNoTrans, nXcols, nYcols, nYcols, 1.0f,
347 Q, nXcols,
348 h->GhatH_Ky, nYcols, 0.0f,
349 h->QH_GhatH_Ky, nYcols);
350 cblas_sgemm(CblasRowMajor, CblasTrans, CblasNoTrans, nXcols, nYcols, nXcols, 1.0f,
351 h->Kx, nXcols,
352 h->QH_GhatH_Ky, nYcols, 0.0f,
353 h->KxH_QH_GhatH_Ky, nYcols);
354 utility_ssvd(h->hSVD, h->KxH_QH_GhatH_Ky, nXcols, nYcols, h->U, NULL, h->V, NULL);
355 cblas_sgemm(CblasRowMajor, CblasNoTrans, CblasTrans, nYcols, nXcols, nXcols, 1.0f,
356 h->lambda, nXcols,
357 h->U, nXcols, 0.0f,
358 h->lambda_UH, nXcols);
359 cblas_sgemm(CblasRowMajor, CblasNoTrans, CblasNoTrans, nYcols, nXcols, nYcols, 1.0f,
360 h->V, nYcols,
361 h->lambda_UH, nXcols, 0.0f,
362 h->P, nXcols);
363
364 /* Formulate M */
365 cblas_sgemm(CblasRowMajor, CblasNoTrans, CblasNoTrans, nYcols, nXcols, nXcols, 1.0f,
366 h->P, nXcols,
367 h->Kx_reg_inverse, nXcols, 0.0f,
368 h->P_Kxreginverse, nXcols);
369 cblas_sgemm(CblasRowMajor, CblasNoTrans, CblasNoTrans, nYcols, nXcols, nYcols, 1.0f,
370 h->Ky, nYcols,
371 h->P_Kxreginverse, nXcols, 0.0f,
372 M, nXcols);
373
374 /* Formulate residual covariance matrix */
375 cblas_sgemm(CblasRowMajor, CblasNoTrans, CblasTrans, nXcols, nYcols, nXcols, 1.0f,
376 Cx, nXcols,
377 M, nXcols, 0.0f,
378 h->Cx_MH, nYcols);
379 cblas_sgemm(CblasRowMajor, CblasNoTrans, CblasNoTrans, nYcols, nYcols, nXcols, 1.0f,
380 M, nXcols,
381 h->Cx_MH, nYcols, 0.0f,
382 h->Cy_tilde, nYcols);
383 if(Cr != NULL)
384 for(i=0; i < nYcols*nYcols; i++)
385 Cr[i] = Cy[i] - h->Cy_tilde[i];
386
387 /* Use energy compensation instead of residuals */
388 if(useEnergyFLAG){
389 for(i=0; i < nYcols; i++)
390 for(j=0; j < nYcols; j++)
391 h->G_hat[i*nYcols+j] = i==j ? sqrtf( SAF_MAX(Cy[i*nYcols+j], 2.23e-20f) / (h->Cy_tilde[i*nYcols+j]+2.23e-7f)) : 0.0f;
392 cblas_sgemm(CblasRowMajor, CblasNoTrans, CblasNoTrans, nYcols, nXcols, nYcols, 1.0f,
393 h->G_hat, nYcols,
394 M, nXcols, 0.0f,
395 h->G_M, nXcols);
396 memcpy(M, h->G_M, nYcols*nXcols*sizeof(float));
397 if(Cr != NULL)
398 memset(Cr, 0, nYcols*nYcols*sizeof(float));
399 }
400}

402void formulate_M_and_Cr_cmplx
403(
404 void * const hCdf,
405 float_complex* Cx,
406 float_complex* Cy,
407 float_complex* Q,
408 int useEnergyFLAG,
409 float reg,
410 float_complex* M,
411 float_complex* Cr
412)
413{
414 cdf4sap_cmplx_data *h = (cdf4sap_cmplx_data*)(hCdf);
415 int i, j, nXcols, nYcols;
416 const float_complex calpha = cmplxf(1.0f, 0.0f); const float_complex cbeta = cmplxf(0.0f, 0.0f);
417
418 nXcols = h->nXcols;
419 nYcols = h->nYcols;
420
421 memset(h->lambda, 0, nYcols * nXcols * sizeof(float_complex));
422 for(i = 0; i<SAF_MIN(nXcols,nYcols); i++)
423 h->lambda[i*nXcols + i] = cmplxf(1.0f, 0.0f);
424
425 /* Decomposition of Cy */
426 utility_csvd(h->hSVD, Cy, nYcols, nYcols, h->U_Cy, h->S_Cy, NULL, NULL);
427 for(i=0; i< nYcols; i++)
428 h->S_Cy[i*nYcols+i] = cmplxf(sqrtf(SAF_MAX(crealf(h->S_Cy[i*nYcols+i]), 2.23e-20f)), 0.0f);
429 cblas_cgemm(CblasRowMajor, CblasNoTrans, CblasNoTrans, nYcols, nYcols, nYcols, &calpha,
430 h->U_Cy, nYcols,
431 h->S_Cy, nYcols, &cbeta,
432 h->Ky, nYcols);
433
434 /* Decomposition of Cx */
435 utility_csvd(h->hSVD, Cx, nXcols, nXcols, h->U_Cx, h->S_Cx, NULL, h->s_Cx);
436 for(i=0; i< nXcols; i++){
437 h->s_Cx[i] = sqrtf(SAF_MAX(h->s_Cx[i], 2.23e-13f));
438 h->S_Cx[i*nXcols+i] = cmplxf(h->s_Cx[i], 0.0f);
439 }
440 cblas_cgemm(CblasRowMajor, CblasNoTrans, CblasNoTrans, nXcols, nXcols, nXcols, &calpha,
441 h->U_Cx, nXcols,
442 h->S_Cx, nXcols, &cbeta,
443 h->Kx, nXcols);
444
445 /* Regularisation of S_Cx */
446 int ind;
447 float limit, maxVal;
448 //utility_simaxv(h->s_Cx, nXcols, &ind);
449 ind = 0; /* utility_csvd returns the singular values in decending order */
450 limit = h->s_Cx[ind] * reg + 2.23e-13f;
451 for(i=0; i < nXcols; i++)
452 h->S_Cx[i*nXcols+i] = cmplxf(1.0f / SAF_MAX(h->s_Cx[i], limit), 0.0f);
453
454 /* Formulate regularised Kx^-1 */
455 cblas_cgemm(CblasRowMajor, CblasNoTrans, CblasConjTrans, nXcols, nXcols, nXcols, &calpha,
456 h->S_Cx, nXcols,
457 h->U_Cx, nXcols, &cbeta,
458 h->Kx_reg_inverse, nXcols);
459
460 /* Formulate normalisation matrix G_hat */
461 cblas_cgemm(CblasRowMajor, CblasNoTrans, CblasConjTrans, nXcols, nYcols, nXcols, &calpha,
462 Cx, nXcols,
463 Q, nXcols, &cbeta,
464 h->Cx_QH, nYcols);
465 cblas_cgemm(CblasRowMajor, CblasNoTrans, CblasNoTrans, nYcols, nYcols, nXcols, &calpha,
466 Q, nXcols,
467 h->Cx_QH, nYcols, &cbeta,
468 h->G_hat, nYcols);
469 /* h->G_hat: imaginary parts along the diagonal are ~0, so it's OK to take
470 * the real instead below: */
471 maxVal = -2.23e13f;
472 for(i=0; i< nYcols; i++)
473 maxVal = cabsf(h->G_hat[i*nYcols+i]) > maxVal ? cabsf(h->G_hat[i*nYcols+i]) : maxVal; // crealf->cabsf
474 limit = maxVal * 0.001f + 2.23e-13f;
475 for(i=0; i < nYcols; i++)
476 for(j=0; j < nYcols; j++)
477 h->G_hat[i*nYcols+j] = i==j ? cmplxf(crealf(csqrtf( ccdivf(Cy[i*nYcols+j], cmplxf(SAF_MAX(cabsf(h->G_hat[i*nYcols+j]), limit), 0.0f)))), 0.0f) : cmplxf(0.0f, 0.0f); // changed crealf->cabsf
478
479 /* Formulate optimal P */
480 cblas_cgemm(CblasRowMajor, CblasConjTrans, CblasNoTrans, nYcols, nYcols, nYcols, &calpha,
481 h->G_hat, nYcols,
482 h->Ky, nYcols, &cbeta,
483 h->GhatH_Ky, nYcols);
484 cblas_cgemm(CblasRowMajor, CblasConjTrans, CblasNoTrans, nXcols, nYcols, nYcols, &calpha,
485 Q, nXcols,
486 h->GhatH_Ky, nYcols, &cbeta,
487 h->QH_GhatH_Ky, nYcols);
488 cblas_cgemm(CblasRowMajor, CblasConjTrans, CblasNoTrans, nXcols, nYcols, nXcols, &calpha,
489 h->Kx, nXcols,
490 h->QH_GhatH_Ky, nYcols, &cbeta,
491 h->KxH_QH_GhatH_Ky, nYcols);
492 utility_csvd(h->hSVD, h->KxH_QH_GhatH_Ky, nXcols, nYcols, h->U, NULL, h->V, NULL);
493 cblas_cgemm(CblasRowMajor, CblasNoTrans, CblasConjTrans, nYcols, nXcols, nXcols, &calpha,
494 h->lambda, nXcols,
495 h->U, nXcols, &cbeta,
496 h->lambda_UH, nXcols);
497 cblas_cgemm(CblasRowMajor, CblasNoTrans, CblasNoTrans, nYcols, nXcols, nYcols, &calpha,
498 h->V, nYcols,
499 h->lambda_UH, nXcols, &cbeta,
500 h->P, nXcols);
501
502 /* Formulate M */
503 cblas_cgemm(CblasRowMajor, CblasNoTrans, CblasNoTrans, nYcols, nXcols, nXcols, &calpha,
504 h->P, nXcols,
505 h->Kx_reg_inverse, nXcols, &cbeta,
506 h->P_Kxreginverse, nXcols);
507 cblas_cgemm(CblasRowMajor, CblasNoTrans, CblasNoTrans, nYcols, nXcols, nYcols, &calpha,
508 h->Ky, nYcols,
509 h->P_Kxreginverse, nXcols, &cbeta,
510 M, nXcols);
511
512 /* Formulate residual covariance matrix */
513 cblas_cgemm(CblasRowMajor, CblasNoTrans, CblasConjTrans, nXcols, nYcols, nXcols, &calpha,
514 Cx, nXcols,
515 M, nXcols, &cbeta,
516 h->Cx_MH, nYcols);
