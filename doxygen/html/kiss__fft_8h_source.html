75 /*
76  * kiss_fft_alloc
77  *
78  * Initialize a FFT (or IFFT) algorithm's cfg/state buffer.
79  *
80  * typical usage: kiss_fft_cfg mycfg=kiss_fft_alloc(1024,0,NULL,NULL);
81  *
82  * The return value from fft_alloc is a cfg buffer used internally
83  * by the fft routine or NULL.
84  *
85  * If lenmem is NULL, then kiss_fft_alloc will allocate a cfg buffer using malloc.
86  * The returned value should be free()d when done to avoid memory leaks.
87  *
88  * The state can be placed in a user supplied buffer 'mem':
89  * If lenmem is not NULL and mem is not NULL and *lenmem is large enough,
90  * then the function places the cfg in mem and the size used in *lenmem
91  * and returns mem.
92  *
93  * If lenmem is not NULL and ( mem is NULL or *lenmem is not large enough),
94  * then the function returns NULL and places the minimum cfg
95  * buffer size in *lenmem.
96  * */
97 
98 kiss_fft_cfg kiss_fft_alloc(int nfft,int inverse_fft,void * mem,size_t * lenmem);

100 /*
101  * kiss_fft(cfg,in_out_buf)
102  *
103  * Perform an FFT on a complex input buffer.
104  * for a forward FFT,
105  * fin should be f[0] , f[1] , ... ,f[nfft-1]
106  * fout will be F[0] , F[1] , ... ,F[nfft-1]
107  * Note that each element is complex and can be accessed like
108  f[k].r and f[k].i
109  * */
110 void kiss_fft(kiss_fft_cfg cfg,const kiss_fft_cpx *fin,kiss_fft_cpx *fout);
