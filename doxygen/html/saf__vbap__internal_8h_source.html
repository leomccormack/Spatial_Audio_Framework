54 /*
55  * Function: findLsTriplets
56  * ------------------------
57  * Computes the 3D convex-hull of a spherical grid of loudspeaker directions
58  *
59  * Input Arguments:
60  * ls_dirs_deg - Loudspeaker directions in DEGREES; FLAT: L x 2
61  * L - number of loudspeakers
62  * omitLargeTriangles - 0: normal triangulation, 1: remove large triangles
63  * Output Arguments:
64  * out_vertices - & loudspeaker directions in cartesian coordinates;
65  * FLAT: L x 3
66  * numOutVertices - & number of loudspeakers
67  * out_faces - & true loudspeaker triangle indices;
68  * FLAT: numOutFaces x 3
69  * numOutFaces - & number of true loudspeaker triangles
70  */
71 void findLsTriplets(/* Input Arguments */
72  float* ls_dirs_deg,
73  int L,
74  int omitLargeTriangles,
75  /* Output Arguments */
76  float** out_vertices,
77  int* numOutVertices,
78  int** out_faces,
79  int* numOutFaces);
80 
81 /*
82  * Function: invertLsMtx3D
83  * ------------------------
84  * Inverts the loudspeaker matrix
85  *
86  * Input Arguments:
87  * U_spkr - loudspeaker directions as cartesian coordinates (unit
88  * length); FLAT: L x 3
89  * ls_groups - true loudspeaker triangle indices; FLAT: N_group x 3
90  * N_group - number of true loudspeaker triangles
91  * Output Arguments:
92  * layoutInvMtx - & inverted 3x3 loudspeaker matrices per group;
93  * FLAT: N_group x 9
94  */
95 void invertLsMtx3D(/* Input Arguments */
96  float* U_spkr,
97  int* ls_groups,
98  int N_group,
99  /* Output Arguments */
100  float** layoutInvMtx);

102 /*
103  * Function: getSpreadSrcDirs3D
104  * ----------------------------
105  * Returns a set of points that surround the source direction with a specific
106  * degree of spread.
107  *
108  * Input Arguments:
109  * src_azi_rad - source azimuth, in RADIANS
110  * src_elev_rad - source elevation, in RADIANS
111  * spread - spread in DEGREES
112  * num_src - number of auxiliary sources to use for spreading
113  * num_rings_3d - number of concentric rings of num_src each to generate
114  * inside the spreading surface
115  * Output Arguments:
116  * U_spread - spread directions Cartesian coordinates;
117  * FLAT: (num_src*num_rings_3d+1) x 3
118  */
119 void getSpreadSrcDirs3D(/* Input Arguments */
120  float src_azi_rad,
121  float src_elev_rad,
122  float spread,
123  int num_src,
124  int num_rings_3d,
125  /* Output Arguments */
126  float* U_spread);

128 /*
129  * Function: vbap3D
130  * ----------------
131  * Calculates 3D VBAP gains for pre-calculated loudspeaker triangles and
132  * predefined source directions
133  *
134  * Input Arguments:
135  * src_dirs - source directions; FLAT: src_num x 2
136  * src_num - number of sources
137  * ls_num - number of loudspeakers
138  * ls_groups - true loudspeaker triangle indices; FLAT: nFaces x 3
139  * nFaces - number of true loudspeaker triangles
140  * spread - spreading in degrees, 0: VBAP, >0: MDAP
141  * layoutInvMtx - inverted 3x3 loudspeaker matrix flattened;
142  * FLAT: nFaces x 9
143  * Output Arguments:
144  * GainMtx - & Loudspeaker VBAP gain table; FLAT: src_num x ls_num
145  */
146 void vbap3D(/* Input Arguments */
147  float* src_dirs,
148  int src_num,
149  int ls_num,
150  int* ls_groups,
151  int nFaces,
152  float spread,
153  float* layoutInvMtx,
154  /* Output Arguments */
155  float** GainMtx);
