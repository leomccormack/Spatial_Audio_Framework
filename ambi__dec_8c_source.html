48void ambi_dec_create
49(
50 void ** const phAmbi
51)
52{
53 ambi_dec_data* pData = (ambi_dec_data*)malloc1d(sizeof(ambi_dec_data));
54 *phAmbi = (void*)pData;
55 int i, j, ch, band;
56
57 /* default user parameters */
58 loadLoudspeakerArrayPreset(LOUDSPEAKER_ARRAY_PRESET_T_DESIGN_24, pData->loudpkrs_dirs_deg, &(pData->new_nLoudpkrs), &(pData->loudpkrs_nDims));
59 pData->masterOrder = pData->new_masterOrder = 1;
60 for (band = 0; band<HYBRID_BANDS; band++)
61 pData->orderPerBand[band] = 1;
62 pData->useDefaultHRIRsFLAG = 1; /* pars->sofa_filepath must be valid to set this to 0 */
63 pData->enableHRIRsPreProc = 1;
64 pData->nLoudpkrs = pData->new_nLoudpkrs;
65 pData->chOrdering = CH_ACN;
66 pData->norm = NORM_SN3D;
67 pData->dec_method[0] = DECODING_METHOD_ALLRAD;
68 pData->dec_method[1] = DECODING_METHOD_ALLRAD;
69 pData->rE_WEIGHT[0] = 1;
70 pData->rE_WEIGHT[1] = 1;
71 pData->diffEQmode[0] = ENERGY_PRESERVING;
72 pData->diffEQmode[1] = ENERGY_PRESERVING;
73 pData->transitionFreq = 800.0f;
74
75 /* afSTFT stuff and audio buffers */
76 pData->hSTFT = NULL;
77 pData->SHFrameTD = (float**)malloc2d(MAX_NUM_SH_SIGNALS, AMBI_DEC_FRAME_SIZE, sizeof(float));
78 pData->outputFrameTD = (float**)malloc2d(SAF_MAX(MAX_NUM_LOUDSPEAKERS, NUM_EARS), AMBI_DEC_FRAME_SIZE, sizeof(float));
79 pData->SHframeTF = (float_complex***)malloc3d(HYBRID_BANDS, MAX_NUM_SH_SIGNALS, TIME_SLOTS, sizeof(float_complex));
80 pData->outputframeTF = (float_complex***)malloc3d(HYBRID_BANDS, MAX_NUM_LOUDSPEAKERS, TIME_SLOTS, sizeof(float_complex));
81 pData->binframeTF = (float_complex***)malloc3d(HYBRID_BANDS, NUM_EARS, TIME_SLOTS, sizeof(float_complex));
82
83 /* codec data */
84 pData->progressBar0_1 = 0.0f;
85 pData->progressBarText = malloc1d(PROGRESSBARTEXT_CHAR_LENGTH*sizeof(char));
86 strcpy(pData->progressBarText,"");
87 pData->codecStatus = CODEC_STATUS_NOT_INITIALISED;
88 pData->pars = (ambi_dec_codecPars*)malloc1d(sizeof(ambi_dec_codecPars));
89 ambi_dec_codecPars* pars = pData->pars;
90 for (i=0; i<NUM_DECODERS; i++){
91 for(j=0; j<MAX_SH_ORDER; j++){
92 pars->M_dec[i][j] = NULL;
93 pars->M_dec_cmplx[i][j] = NULL;
94 pars->M_dec_maxrE[i][j] = NULL;
95 pars->M_dec_cmplx_maxrE[i][j] = NULL;
96 }
97 }
98 pars->sofa_filepath = NULL;
99 pars->hrirs = NULL;
100 pars->hrir_dirs_deg = NULL;
101 pars->hrtf_vbap_gtableIdx = NULL;
102 pars->hrtf_vbap_gtableComp = NULL;
103 pars->itds_s = NULL;
104 pars->hrtf_fb = NULL;
105 pars->hrtf_fb_mag = NULL;
106 pars->weights = NULL;
107
108 /* internal parameters */
109 pData->binauraliseLS = pData->new_binauraliseLS = 0;
110
111 /* flags */
112 pData->procStatus = PROC_STATUS_NOT_ONGOING;
113 pData->reinit_hrtfsFLAG = 1;
114 for(ch=0; ch<MAX_NUM_LOUDSPEAKERS; ch++)
115 pData->recalc_hrtf_interpFLAG[ch] = 1;
116}

121)
122{
123 ambi_dec_data *pData = (ambi_dec_data*)(*phAmbi);
124 ambi_dec_codecPars *pars;

126
127 if (pData != NULL) {
128 /* not safe to free memory during intialisation/processing loop */

133

146 free(pars->hrtf_vbap_gtableIdx);
147 free(pars->hrtf_fb);
148 free(pars->hrtf_fb_mag);
149 free(pars->itds_s);
150 free(pars->sofa_filepath);
151 free(pars->hrirs);
152 free(pars->hrir_dirs_deg);
153 free(pars->weights);
154 for (i=0; i<NUM_DECODERS; i++){
155 for(j=0; j<MAX_SH_ORDER; j++){
156 free(pars->M_dec[i][j]);
157 free(pars->M_dec_cmplx[i][j]);
158 free(pars->M_dec_maxrE[i][j]);
159 free(pars->M_dec_cmplx_maxrE[i][j]);
160 }
161 }
162 free(pData->progressBarText);
163 free(pData);
164 pData = NULL;
165 }

181void ambi_dec_initCodec
182(
183 void* const hAmbi
184)
185{
186 ambi_dec_data *pData = (ambi_dec_data*)(hAmbi);
187 ambi_dec_codecPars* pars = pData->pars;
188 int i, ch, d, j, n, ng, nGrid_dirs, masterOrder, nSH_order, max_nSH, nLoudspeakers;
189 float* grid_dirs_deg, *Y, *M_dec_tmp, *g, *a, *e, *a_n, *hrtf_vbap_gtable;;
190 float a_avg[MAX_SH_ORDER], e_avg[MAX_SH_ORDER], azi_incl[2], sum_elev;
191#ifdef SAF_ENABLE_SOFA_READER_MODULE
192 SAF_SOFA_ERROR_CODES error;
193 saf_sofa_container sofa;
194#endif
195
196 if (pData->codecStatus != CODEC_STATUS_NOT_INITIALISED)
197 return; /* re-init not required, or already happening */
198 while (pData->procStatus == PROC_STATUS_ONGOING){
199 /* re-init required, but we need to wait for the current processing loop to end */
200 pData->codecStatus = CODEC_STATUS_INITIALISING; /* indicate that we want to init */
201 SAF_SLEEP(10);
202 }
203
204 /* for progress bar */
205 pData->codecStatus = CODEC_STATUS_INITIALISING;
206 strcpy(pData->progressBarText,"Initialising");
207 pData->progressBar0_1 = 0.0f;
208
209 /* reinit afSTFT */
210 masterOrder = pData->new_masterOrder;
211 max_nSH = (masterOrder+1)*(masterOrder+1);
212 nLoudspeakers = pData->new_nLoudpkrs;
213 if(pData->hSTFT==NULL){
214 if(pData->new_binauraliseLS)
215 afSTFT_create(&(pData->hSTFT), max_nSH, NUM_EARS, HOP_SIZE, 0, 1, AFSTFT_BANDS_CH_TIME);
216 else
217 afSTFT_create(&(pData->hSTFT), max_nSH, nLoudspeakers, HOP_SIZE, 0, 1, AFSTFT_BANDS_CH_TIME);
218 afSTFT_clearBuffers(pData->hSTFT);
219 }
220 else{
221 if(pData->new_binauraliseLS)
222 afSTFT_channelChange(pData->hSTFT, max_nSH, NUM_EARS);
223 else
224 afSTFT_channelChange(pData->hSTFT, max_nSH, nLoudspeakers);
225 afSTFT_clearBuffers(pData->hSTFT);
226 }
227 pData->binauraliseLS = pData->new_binauraliseLS;
228 pData->nLoudpkrs = nLoudspeakers;
229
230 /* Quick and dirty check to find loudspeaker dimensionality */
231 strcpy(pData->progressBarText,"Computing decoder");
232 pData->progressBar0_1 = 0.2f;
233 sum_elev = 0.0f;
234 for(ch=0; ch < nLoudspeakers; ch++)
235 sum_elev += fabsf(pData->loudpkrs_dirs_deg[ch][1]);
236 if( (((sum_elev < 5.0f) && (sum_elev > -5.0f))) || (nLoudspeakers < 4) )
237 pData->loudpkrs_nDims = 2;
238 else
239 pData->loudpkrs_nDims = 3;
240
241 /* add virtual loudspeakers for 2D case if using AllRAD, so that the triangulation cannot fail. */
242 if (pData->loudpkrs_nDims == 2 && (pData->dec_method[0]==DECODING_METHOD_ALLRAD || pData->dec_method[1]==DECODING_METHOD_ALLRAD)){
243 assert(nLoudspeakers<=MAX_NUM_LOUDSPEAKERS-2);
244 pData->loudpkrs_dirs_deg[nLoudspeakers][0] = 0.0f;
245 pData->loudpkrs_dirs_deg[nLoudspeakers][1] = -90.0f;
246 pData->loudpkrs_dirs_deg[nLoudspeakers+1][0] = 0.0f;
247 pData->loudpkrs_dirs_deg[nLoudspeakers+1][1] = 90.0f;
248 nLoudspeakers += 2;
249 }
250
251 /* prep */
252 nGrid_dirs = 480; /* Minimum t-design of degree 30, has 480 points */
253 g = malloc1d(nLoudspeakers*sizeof(float));
254 a = malloc1d(nGrid_dirs*sizeof(float));
255 e = malloc1d(nGrid_dirs*sizeof(float));
256
257 /* calculate loudspeaker decoding matrices */
258 for( d=0; d<NUM_DECODERS; d++){
259 M_dec_tmp = malloc1d(nLoudspeakers * max_nSH * sizeof(float));
260 switch(pData->dec_method[d]){
261 case DECODING_METHOD_SAD:
262 getLoudspeakerDecoderMtx((float*)pData->loudpkrs_dirs_deg, nLoudspeakers, LOUDSPEAKER_DECODER_SAD, masterOrder, 0, M_dec_tmp);
263 break;
264 case DECODING_METHOD_MMD:
265 getLoudspeakerDecoderMtx((float*)pData->loudpkrs_dirs_deg, nLoudspeakers, LOUDSPEAKER_DECODER_MMD, masterOrder, 0, M_dec_tmp);
266 break;
267 case DECODING_METHOD_EPAD:
268 getLoudspeakerDecoderMtx((float*)pData->loudpkrs_dirs_deg, nLoudspeakers, LOUDSPEAKER_DECODER_EPAD, masterOrder, 0, M_dec_tmp);
269 break;
270 case DECODING_METHOD_ALLRAD:
271 getLoudspeakerDecoderMtx((float*)pData->loudpkrs_dirs_deg, nLoudspeakers, LOUDSPEAKER_DECODER_ALLRAD, masterOrder, 0, M_dec_tmp);
272 break;
273 }
274
275 /* diffuse-field EQ for orders 1..masterOrder */
276 for( n=1; n<=masterOrder; n++){
277 /* truncate M_dec for each order */
278 nSH_order = (n+1)*(n+1);
279 free(pars->M_dec[d][n-1]);
280 pars->M_dec[d][n-1] = malloc1d(nLoudspeakers* nSH_order * sizeof(float));
281 free(pars->M_dec_cmplx[d][n-1]);
282 pars->M_dec_cmplx[d][n-1] = malloc1d(nLoudspeakers * nSH_order * sizeof(float_complex));
283 for(i=0; i<nLoudspeakers; i++){
284 for(j=0; j<nSH_order; j++){
285 pars->M_dec[d][n-1][i*nSH_order+j] = M_dec_tmp[i*max_nSH +j]; /* for applying in the time domain, and... */
286 pars->M_dec_cmplx[d][n-1][i*nSH_order+j] = cmplxf(pars->M_dec[d][n-1][i*nSH_order+j], 0.0f); /* for the time-frequency domain */
287 }
288 }
289
290 /* create dedicated maxrE weighted versions */
291 a_n = malloc1d(nSH_order*nSH_order*sizeof(float));
292 getMaxREweights(n, 1, a_n); /* weights returned as diagonal matrix */
293 free(pars->M_dec_maxrE[d][n-1]);
294 pars->M_dec_maxrE[d][n-1] = malloc1d(nLoudspeakers * nSH_order * sizeof(float));
295 free(pars->M_dec_cmplx_maxrE[d][n-1]);
296 pars->M_dec_cmplx_maxrE[d][n-1] = malloc1d(nLoudspeakers * nSH_order * sizeof(float_complex));
297 cblas_sgemm(CblasRowMajor, CblasNoTrans, CblasNoTrans, nLoudspeakers, nSH_order, nSH_order, 1.0f,
298 pars->M_dec[d][n-1], nSH_order,
299 a_n, nSH_order, 0.0f,
300 pars->M_dec_maxrE[d][n-1], nSH_order); /* for applying in the time domain */
301 for(i=0; i<nLoudspeakers * nSH_order; i++)
302 pars->M_dec_cmplx_maxrE[d][n-1][i] = cmplxf(pars->M_dec_maxrE[d][n-1][i], 0.0f); /* for the time-frequency domain */
303
304 /* fire a plane-wave from each grid direction to find the total energy/amplitude (using non-maxrE weighted versions) */
305 Y = malloc1d(nSH_order*sizeof(float));
306 grid_dirs_deg = (float*)(&__Tdesign_degree_30_dirs_deg[0][0]);
307 for(ng=0; ng<nGrid_dirs; ng++){
308 azi_incl[0] = grid_dirs_deg[ng*2]*SAF_PI/180.0f;
309 azi_incl[1] = SAF_PI/2.0f-grid_dirs_deg[ng*2+1]*SAF_PI/180.0f;
310 getSHreal(n, (float*)azi_incl, 1, Y);
311 cblas_sgemm(CblasRowMajor, CblasNoTrans, CblasTrans, nLoudspeakers, 1, nSH_order, 1.0f,
312 pars->M_dec[d][n-1], nSH_order,
313 Y, nSH_order, 0.0f,
314 g, 1);
315 a[ng] = e[ng] = 0.0f;
316 for(i=0; i<nLoudspeakers; i++){
317 a[ng] += g[i];
318 e[ng] += powf(g[i], 2.0f);
319 }
320 }
321
322 /* determine the order+decoder dependent normalisation factor for energy+amplitude preserving decoding */
323 a_avg[n-1] = e_avg[n-1] = 0.0f;
324 for(ng=0; ng<nGrid_dirs; ng++){
325 a_avg[n-1] += a[ng];
326 e_avg[n-1] += e[ng];
327 }
328 a_avg[n-1] /= (float)nGrid_dirs;
329 e_avg[n-1] /= (float)nGrid_dirs;
330 pars->M_norm[d][n-1][0] = 1.0f/(a_avg[n-1]+2.23e-6f); /* use this to preserve omni amplitude */
331 pars->M_norm[d][n-1][1] = sqrtf(1.0f/(e_avg[n-1]+2.23e-6f)); /* use this to preserve omni energy */
332 free(a_n);
333 free(Y);
334
335 /* remove virtual loudspeakers from the decoder (if needed) */
336 if (pData->loudpkrs_nDims == 2 && (pData->dec_method[0]==DECODING_METHOD_ALLRAD || pData->dec_method[1]==DECODING_METHOD_ALLRAD)){
337 pars->M_dec[d][n-1] = realloc1d(pars->M_dec[d][n-1], pData->nLoudpkrs * nSH_order * sizeof(float));
338 pars->M_dec_cmplx[d][n-1] = realloc1d(pars->M_dec_cmplx[d][n-1], pData->nLoudpkrs * nSH_order * sizeof(float_complex));
339 pars->M_dec_maxrE[d][n-1] = realloc1d(pars->M_dec_maxrE[d][n-1], pData->nLoudpkrs * nSH_order * sizeof(float));
340 pars->M_dec_cmplx_maxrE[d][n-1] = realloc1d(pars->M_dec_cmplx_maxrE[d][n-1], pData->nLoudpkrs * nSH_order * sizeof(float_complex));
341 }
342 }
343 free(M_dec_tmp);
344 }
345
346 /* update order */
347 pData->masterOrder = pData->new_masterOrder;
348
349 /* Binaural-related initialisations */
350 if(pData->reinit_hrtfsFLAG){
351 strcpy(pData->progressBarText,"Computing VBAP gain table");
352 pData->progressBar0_1 = 0.4f;
353
354 /* load sofa file or load default hrir data */
355#ifdef SAF_ENABLE_SOFA_READER_MODULE
356 if(!pData->useDefaultHRIRsFLAG && pars->sofa_filepath!=NULL){
357 /* Load SOFA file */
358 error = saf_sofa_open(&sofa, pars->sofa_filepath, SAF_SOFA_READER_OPTION_DEFAULT);
359
360 /* Load defaults instead */
361 if(error!=SAF_SOFA_OK || sofa.nReceivers!=NUM_EARS){
362 pData->useDefaultHRIRsFLAG = 1;
363 saf_print_warning("Unable to load the specified SOFA file, or it contained something other than 2 channels. Using default HRIR data instead.");
364 }
365 else{
366 /* Copy SOFA data */
367 pars->hrir_fs = (int)sofa.DataSamplingRate;
368 pars->hrir_len = sofa.DataLengthIR;
369 pars->N_hrir_dirs = sofa.nSources;
370 pars->hrirs = realloc1d(pars->hrirs, pars->N_hrir_dirs*NUM_EARS*(pars->hrir_len)*sizeof(float));
371 memcpy(pars->hrirs, sofa.DataIR, pars->N_hrir_dirs*NUM_EARS*(pars->hrir_len)*sizeof(float));
372 pars->hrir_dirs_deg = realloc1d(pars->hrir_dirs_deg, pars->N_hrir_dirs*2*sizeof(float));
373 cblas_scopy(pars->N_hrir_dirs, sofa.SourcePosition, 3, pars->hrir_dirs_deg, 2); /* azi */
374 cblas_scopy(pars->N_hrir_dirs, &sofa.SourcePosition[1], 3, &pars->hrir_dirs_deg[1], 2); /* elev */
375 }
376
377 /* Clean-up */
378 saf_sofa_close(&sofa);
379 }
380#else
381 pData->useDefaultHRIRsFLAG = 1; /* Can only load the default HRIR data */
382#endif
383 if(pData->useDefaultHRIRsFLAG){
384 /* Copy default HRIR data */
385 pars->hrir_fs = __default_hrir_fs;
386 pars->hrir_len = __default_hrir_len;
387 pars->N_hrir_dirs = __default_N_hrir_dirs;
388 pars->hrirs = realloc1d(pars->hrirs, pars->N_hrir_dirs*NUM_EARS*(pars->hrir_len)*sizeof(float));
389 memcpy(pars->hrirs, (float*)__default_hrirs, pars->N_hrir_dirs*NUM_EARS*(pars->hrir_len)*sizeof(float));
390 pars->hrir_dirs_deg = realloc1d(pars->hrir_dirs_deg, pars->N_hrir_dirs*2*sizeof(float));
391 memcpy(pars->hrir_dirs_deg, (float*)__default_hrir_dirs_deg, pars->N_hrir_dirs*2*sizeof(float));
392 }
393
394 /* estimate the ITDs for each HRIR */
395 pars->itds_s = realloc1d(pars->itds_s, pars->N_hrir_dirs*sizeof(float));
396 estimateITDs(pars->hrirs, pars->N_hrir_dirs, pars->hrir_len, pars->hrir_fs, pars->itds_s);
397
398 /* generate VBAP gain table for the hrir_dirs */
399 hrtf_vbap_gtable = NULL;
400 pars->hrtf_vbapTableRes[0] = 2; /* azimuth resolution in degrees */
401 pars->hrtf_vbapTableRes[1] = 5; /* elevation resolution in degrees */
402 generateVBAPgainTable3D(pars->hrir_dirs_deg, pars->N_hrir_dirs, pars->hrtf_vbapTableRes[0], pars->hrtf_vbapTableRes[1], 1, 0, 0.0f,
403 &hrtf_vbap_gtable, &(pars->N_hrtf_vbap_gtable), &(pars->hrtf_nTriangles));
404 if(hrtf_vbap_gtable==NULL){
405 /* if generating vbap gain tabled failed, re-calculate with default HRIR set (which is known to triangulate correctly) */
406 pData->useDefaultHRIRsFLAG = 1;
407 ambi_dec_initCodec(hAmbi);
408 }
409
410 /* compress VBAP table (i.e. remove the zero elements) */
411 pars->hrtf_vbap_gtableComp = realloc1d(pars->hrtf_vbap_gtableComp, pars->N_hrtf_vbap_gtable * 3 * sizeof(float));
412 pars->hrtf_vbap_gtableIdx = realloc1d(pars->hrtf_vbap_gtableIdx, pars->N_hrtf_vbap_gtable * 3 * sizeof(int));
413 compressVBAPgainTable3D(hrtf_vbap_gtable, pars->N_hrtf_vbap_gtable, pars->N_hrir_dirs, pars->hrtf_vbap_gtableComp, pars->hrtf_vbap_gtableIdx);
414
415 /* convert hrirs to filterbank coefficients */
416 strcpy(pData->progressBarText,"Preparing HRIRs");
417 pData->progressBar0_1 = 0.85f;
418 pars->hrtf_fb = realloc1d(pars->hrtf_fb, HYBRID_BANDS * NUM_EARS * (pars->N_hrir_dirs)*sizeof(float_complex));
419 HRIRs2HRTFs_afSTFT(pars->hrirs, pars->N_hrir_dirs, pars->hrir_len, HOP_SIZE, 0, 1, pars->hrtf_fb);
420 /* HRIR pre-processing */
421 if(pData->enableHRIRsPreProc){
422 /* get integration weights */
423 strcpy(pData->progressBarText,"Applying HRIR Pre-Processing");
424 pData->progressBar0_1 = 0.95f;
425 if(pars->N_hrir_dirs<=3600){
426 pars->weights = realloc1d(pars->weights, pars->N_hrir_dirs*sizeof(float));
427 getVoronoiWeights(pars->hrir_dirs_deg, pars->N_hrir_dirs, 0, pars->weights);
428 }
429 else{
430 pars->weights = realloc1d(pars->weights, pars->N_hrir_dirs*sizeof(float));
431 for(int idx=0; idx < pars->N_hrir_dirs; idx++)
432 pars->weights[idx] = 4.f*SAF_PI / (float)pars->N_hrir_dirs;
433 }
434 diffuseFieldEqualiseHRTFs(pars->N_hrir_dirs, pars->itds_s, pData->freqVector, HYBRID_BANDS, pars->weights, 1, 0, pars->hrtf_fb);
435 }
436
437 /* calculate magnitude responses */
438 pars->hrtf_fb_mag = realloc1d(pars->hrtf_fb_mag, HYBRID_BANDS*NUM_EARS*(pars->N_hrir_dirs)*sizeof(float));
439 for(i=0; i<HYBRID_BANDS*NUM_EARS* (pars->N_hrir_dirs); i++)
440 pars->hrtf_fb_mag[i] = cabsf(pars->hrtf_fb[i]);
441
442 /* clean-up */
443 free(hrtf_vbap_gtable);
444 pData->reinit_hrtfsFLAG = 0;
445 }
446
447 /* done! */
448 strcpy(pData->progressBarText,"Done!");
449 pData->progressBar0_1 = 1.0f;
450 pData->codecStatus = CODEC_STATUS_INITIALISED;
451
452 free(g);
453 free(a);
454 free(e);
455}

457void ambi_dec_process
458(
459 void * const hAmbi,
460 const float *const * inputs,
461 float ** const outputs,
462 int nInputs,
463 int nOutputs,
464 int nSamples
465)
466{
467 ambi_dec_data *pData = (ambi_dec_data*)(hAmbi);
468 ambi_dec_codecPars* pars = pData->pars;
469 int ch, ear, i, band, orderBand, nSH_band, decIdx, nSH;
470 const float_complex calpha = cmplxf(1.0f, 0.0f), cbeta = cmplxf(0.0f, 0.0f);
471
472 /* local copies of user parameters */
473 int nLoudspeakers, binauraliseLS, masterOrder;
474 int orderPerBand[HYBRID_BANDS], rE_WEIGHT[NUM_DECODERS];
475 float transitionFreq;
476 AMBI_DEC_DIFFUSE_FIELD_EQ_APPROACH diffEQmode[NUM_DECODERS];
477 NORM_TYPES norm;
478 CH_ORDER chOrdering;
479 masterOrder = pData->masterOrder;
480 nSH = ORDER2NSH(masterOrder);
481 nLoudspeakers = pData->nLoudpkrs;
482 memcpy(orderPerBand, pData->orderPerBand, HYBRID_BANDS*sizeof(int));
483 transitionFreq = pData->transitionFreq;
484 memcpy(diffEQmode, pData->diffEQmode, NUM_DECODERS*sizeof(int));
485 binauraliseLS = pData->binauraliseLS;
486 norm = pData->norm;
487 chOrdering = pData->chOrdering;
488 memcpy(rE_WEIGHT, pData->rE_WEIGHT, NUM_DECODERS*sizeof(int));
489
490 /* Process frame */
491 if (nSamples == AMBI_DEC_FRAME_SIZE && (pData->codecStatus == CODEC_STATUS_INITIALISED) ) {
492 pData->procStatus = PROC_STATUS_ONGOING;
493
494 /* Load time-domain data */
495 for(i=0; i < SAF_MIN(nSH, nInputs); i++)
496 utility_svvcopy(inputs[i], AMBI_DEC_FRAME_SIZE, pData->SHFrameTD[i]);
497 for(; i<nSH; i++)
498 memset(pData->SHFrameTD[i], 0, AMBI_DEC_FRAME_SIZE * sizeof(float)); /* fill remaining channels with zeros */
499
500 /* account for channel order convention */
501 switch(chOrdering){
502 case CH_ACN: /* already ACN, do nothing */ break; /* Otherwise, convert to ACN... */
503 case CH_FUMA: convertHOAChannelConvention(FLATTEN2D(pData->SHFrameTD), masterOrder, AMBI_DEC_FRAME_SIZE, HOA_CH_ORDER_FUMA, HOA_CH_ORDER_ACN); break;
504 }
505
506 /* account for input normalisation scheme */
507 switch(norm){
508 case NORM_N3D: /* already in N3D, do nothing */ break; /* Otherwise, convert to N3D... */
509 case NORM_SN3D: convertHOANormConvention(FLATTEN2D(pData->SHFrameTD), masterOrder, AMBI_DEC_FRAME_SIZE, HOA_NORM_SN3D, HOA_NORM_N3D); break;
510 case NORM_FUMA: convertHOANormConvention(FLATTEN2D(pData->SHFrameTD), masterOrder, AMBI_DEC_FRAME_SIZE, HOA_NORM_FUMA, HOA_NORM_N3D); break;
511 }
512
513 /* Apply time-frequency transform (TFT) */
514 afSTFT_forward_knownDimensions(pData->hSTFT, pData->SHFrameTD, AMBI_DEC_FRAME_SIZE, MAX_NUM_SH_SIGNALS, TIME_SLOTS, pData->SHframeTF);
515
516 /* Decode to loudspeaker set-up */
517 memset(FLATTEN3D(pData->outputframeTF), 0, HYBRID_BANDS*MAX_NUM_LOUDSPEAKERS*TIME_SLOTS*sizeof(float_complex));
518 for(band=0; band<HYBRID_BANDS; band++){
519 orderBand = SAF_MAX(SAF_MIN(orderPerBand[band], masterOrder),1);
520 nSH_band = (orderBand+1)*(orderBand+1);
521
522 /* There is a different decoder for low (0) and high (1) frequencies, and for max_rE weights enabled/disabled */
523 decIdx = pData->freqVector[band] < transitionFreq ? 0 : 1;
524 if(rE_WEIGHT[decIdx]){
525 cblas_cgemm(CblasRowMajor, CblasNoTrans, CblasNoTrans, nLoudspeakers, TIME_SLOTS, nSH_band, &calpha,
526 pars->M_dec_cmplx_maxrE[decIdx][orderBand-1], nSH_band,
527 FLATTEN2D(pData->SHframeTF[band]), TIME_SLOTS, &cbeta,
528 FLATTEN2D(pData->outputframeTF[band]), TIME_SLOTS);
529 }
530 else{
531 cblas_cgemm(CblasRowMajor, CblasNoTrans, CblasNoTrans, nLoudspeakers, TIME_SLOTS, nSH_band, &calpha,
532 pars->M_dec_cmplx[decIdx][orderBand-1], nSH_band,
533 FLATTEN2D(pData->SHframeTF[band]), TIME_SLOTS, &cbeta,
534 FLATTEN2D(pData->outputframeTF[band]), TIME_SLOTS);
535 }
536
537 /* Apply scaling to preserve either the amplitude or energy when the decododing orders are different over frequency */
538 cblas_sscal(/*re+im*/2*nLoudspeakers*TIME_SLOTS, pars->M_norm[decIdx][orderBand-1][diffEQmode[decIdx]==AMPLITUDE_PRESERVING ? 0 : 1],
539 (float*)FLATTEN2D(pData->outputframeTF[band]), 1);
540 }
541
542 /* Binauralise the loudspeaker signals */
543 if(binauraliseLS){
544 /* Initialise the binaural buffer with zeros */
545 memset(FLATTEN3D(pData->binframeTF), 0, HYBRID_BANDS*NUM_EARS*TIME_SLOTS * sizeof(float_complex));
546
547 /* Convolve each loudspeaker signals with the respective HRTFs */
548 for (ch = 0; ch < nLoudspeakers; ch++) {
549 if(pData->recalc_hrtf_interpFLAG[ch]){
550 /* Re-compute the interpolated HRTF (only if loudspeaker direction changed) */
551 ambi_dec_interpHRTFs(hAmbi, pData->loudpkrs_dirs_deg[ch][0], pData->loudpkrs_dirs_deg[ch][1], pars->hrtf_interp[ch]);
552 pData->recalc_hrtf_interpFLAG[ch] = 0;
553 }
554
555 /* Convolve this loudspeaker channel with the interpolated HRTF, and add it to the binaural buffer */
556 for (band = 0; band < HYBRID_BANDS; band++)
557 for (ear = 0; ear < NUM_EARS; ear++)
558 cblas_caxpy(TIME_SLOTS, &pars->hrtf_interp[ch][band][ear], pData->outputframeTF[band][ch], 1, pData->binframeTF[band][ear], 1);
559 }
560
561 /* Scale by sqrt(number of loudspeakers) */
562 cblas_sscal(/*re+im*/2*HYBRID_BANDS*NUM_EARS*TIME_SLOTS, 1.0f/sqrtf((float)nLoudspeakers), (float*)FLATTEN3D(pData->binframeTF), 1);
563 }
564
565 /* inverse-TFT */
566 afSTFT_backward_knownDimensions(pData->hSTFT, binauraliseLS ? pData->binframeTF : pData->outputframeTF,
567 AMBI_DEC_FRAME_SIZE, binauraliseLS ? NUM_EARS : MAX_NUM_LOUDSPEAKERS, TIME_SLOTS, pData->outputFrameTD);
568
569 /* Copy to output buffer */
570 for(ch = 0; ch < SAF_MIN(binauraliseLS==1 ? NUM_EARS : nLoudspeakers, nOutputs); ch++)
571 utility_svvcopy(pData->outputFrameTD[ch], AMBI_DEC_FRAME_SIZE, outputs[ch]);
572 for (; ch < nOutputs; ch++)
573 memset(outputs[ch], 0, AMBI_DEC_FRAME_SIZE*sizeof(float));
574 }
575 else
576 for (ch=0; ch < nOutputs; ch++)
577 memset(outputs[ch], 0, AMBI_DEC_FRAME_SIZE*sizeof(float));
578
579 pData->procStatus = PROC_STATUS_NOT_ONGOING;
580}

711void ambi_dec_setSourcePreset(void* const hAmbi, int newPresetID)
712{
713 ambi_dec_data *pData = (ambi_dec_data*)(hAmbi);
714 int band, rangeIdx, curOrder, reverse;
715
716 rangeIdx = 0;
717 curOrder = 1;
718 reverse = 0;
719 switch(newPresetID){
720 /* Ideal spherical harmonics will have SH_ORDER at all frequencies */
721 case MIC_PRESET_IDEAL:
722 for(band=0; band<HYBRID_BANDS; band++)
723 pData->orderPerBand[band] = pData->masterOrder;
724 break;
725
726 /* For real microphone arrays, the maximum usable spherical harmonic order will depend on frequency */
727 case MIC_PRESET_ZYLIA:
728 for(band=0; band<HYBRID_BANDS; band++){
729 if(rangeIdx<2*(__Zylia_maxOrder-1)){
730 if(pData->freqVector[band]>__Zylia_freqRange[rangeIdx]){
731 if(!reverse)
732 curOrder++;
733 else
734 curOrder--;
735 reverse = (curOrder == __Zylia_maxOrder) || (reverse) ? 1 : 0;
736 rangeIdx++;
737 }
738 }
739 pData->orderPerBand[band] = SAF_MIN(pData->masterOrder,curOrder);
740 }
741 break;
742
743 case MIC_PRESET_EIGENMIKE32:
744 for(band=0; band<HYBRID_BANDS; band++){
745 if(rangeIdx<2*(__Eigenmike32_maxOrder-1)){
746 if(pData->freqVector[band]>__Eigenmike32_freqRange[rangeIdx]){
747 if(!reverse)
748 curOrder++;
749 else
750 curOrder--;
751 reverse = (curOrder == __Eigenmike32_maxOrder) || (reverse) ? 1 : 0;
752 rangeIdx++;
753 }
754 }
755 pData->orderPerBand[band] = SAF_MIN(pData->masterOrder,curOrder);
756 }
757 break;
758
759 case MIC_PRESET_DTU_MIC:
760 for(band=0; band<HYBRID_BANDS; band++){
761 if(rangeIdx<2*(__DTU_mic_maxOrder-1)){
762 if(pData->freqVector[band]>__DTU_mic_freqRange[rangeIdx]){
763 if(!reverse)
764 curOrder++;
765 else
766 curOrder--;
767 reverse = (curOrder == __DTU_mic_maxOrder) || (reverse) ? 1 : 0;
768 rangeIdx++;
769 }
770 }
771 pData->orderPerBand[band] = SAF_MIN(pData->masterOrder,curOrder);
772 }
773 break;
774 }
775}
