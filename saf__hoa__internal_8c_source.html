40void getEPAD
41(
42 int order,
43 float* ls_dirs_deg,
44 int nLS,
45 float* decMtx
46)
47{
48 int i, j, nSH;
49 float scale;
50 float* Y_ls, *U, *V, *U_tr, *V_tr;
51
52 nSH = ORDER2NSH(order);
53 scale = 1.0f/SQRT4PI;
54
55 /* Prep */
56 Y_ls = malloc1d(nSH*nLS*sizeof(float));
57 U = malloc1d(nSH*nSH*sizeof(float));
58 V = malloc1d(nLS*nLS*sizeof(float));
59 getRSH(order, ls_dirs_deg, nLS, Y_ls);
60 cblas_sscal(nLS*nSH, scale, Y_ls, 1);
61 utility_ssvd(NULL, Y_ls, nSH, nLS, U, NULL, V, NULL);
62
63 /* Apply truncation */
64 if(nSH>nLS){
65 /* truncate the U matrix */
66 U_tr = malloc1d(nSH*nLS*sizeof(float));
67 for(i=0; i<nSH; i++)
68 for(j=0; j<nLS; j++)
69 U_tr[i*nLS+j] = U[i*nSH+j];
70 cblas_sgemm(CblasRowMajor, CblasNoTrans, CblasTrans, nLS, nSH, nLS, 1.0f,
71 V, nLS,
72 U_tr, nLS, 0.0f,
73 decMtx, nSH);
74 free(U_tr);
75 }
76 else{
77 /* truncate the V matrix (NOT V^T!) */
78 V_tr = malloc1d(nLS*nSH*sizeof(float));
79 for(i=0; i<nLS; i++)
80 for(j=0; j<nSH; j++)
81 V_tr[i*nSH+j] = V[i*nLS+j];
82 cblas_sgemm(CblasRowMajor, CblasNoTrans, CblasTrans, nLS, nSH, nSH, 1.0f,
83 V_tr, nSH,
84 U, nSH, 0.0f,
85 decMtx, nSH);
86 free(V_tr);
87 }
88
89 /* Apply normalisation, and scale by number of loudspeakers */
90 scale = sqrtf(4.0f*SAF_PI/(float)nLS);
91 utility_svsmul(decMtx, &scale, nLS*nSH, decMtx);
92
93 /* clean-up */
94 free(U);
95 free(V);
96 free(Y_ls);
97}

161void getBinDecoder_LS
162(
163 float_complex* hrtfs, /* the HRTFs; FLAT: N_bands x 2 x N_dirs */
164 float* hrtf_dirs_deg,
165 int N_dirs,
166 int N_bands,
167 int order,
168 float* weights,
169 float_complex* decMtx /* N_bands x 2 x (order+1)^2 */
170)
171{
172 int i, j, nSH, band;
173 float* Y_tmp;
174 float_complex* W, *Y_na, *Yna_W, *Yna_W_Yna, *Yna_W_H, *B;
175 const float_complex calpha = cmplxf(1.0f, 0.0f), cbeta = cmplxf(0.0f, 0.0f);
176
177 nSH = ORDER2NSH(order);
178
179 /* SH */
180 Y_tmp = malloc1d(nSH*N_dirs*sizeof(float));
181 Y_na = malloc1d(nSH*N_dirs*sizeof(float_complex));
182 B = malloc1d(nSH * 2 * sizeof(float_complex));
183 getRSH(order, hrtf_dirs_deg, N_dirs, Y_tmp);
184 for(i=0; i<nSH*N_dirs; i++)
185 Y_na[i] = cmplxf(Y_tmp[i], 0.0f);
186 free(Y_tmp);
187
188 /* compute decoding matrix, incorporating integration weights */
189 W = calloc1d(N_dirs*N_dirs, sizeof(float_complex));
190 if(weights!=NULL)
191 for(i=0; i<N_dirs; i++)
192 W[i*N_dirs+i] = cmplxf(weights[i], 0.0f);
193 else
194 for(i=0; i<N_dirs; i++)
195 W[i*N_dirs+i] = cmplxf(1.0f/(float)N_dirs, 0.0f);
196
197 /* calculate decoding matrix per band */
198 Yna_W = malloc1d(nSH * N_dirs*sizeof(float_complex));
199 Yna_W_Yna = malloc1d(nSH * nSH * sizeof(float_complex));
200 Yna_W_H = malloc1d(nSH * 2 * sizeof(float_complex));
201 cblas_cgemm(CblasRowMajor, CblasNoTrans, CblasNoTrans, nSH, N_dirs, N_dirs, &calpha,
202 Y_na, N_dirs,
203 W, N_dirs, &cbeta,
204 Yna_W, N_dirs);
205 cblas_cgemm(CblasRowMajor, CblasNoTrans, CblasTrans, nSH, nSH, N_dirs, &calpha,
206 Yna_W, N_dirs,
207 Y_na, N_dirs, &cbeta,
208 Yna_W_Yna, nSH);
209 for(band=0; band<N_bands; band++){
210 cblas_cgemm(CblasRowMajor, CblasNoTrans, CblasConjTrans, nSH, 2, N_dirs, &calpha,
211 Yna_W, N_dirs,
212 &hrtfs[band*2*N_dirs], N_dirs, &cbeta,
213 Yna_W_H, 2);
214 utility_cglslv(NULL, Yna_W_Yna, nSH, Yna_W_H, 2, B);
215 for(i=0; i<nSH; i++)
216 for(j=0; j<2; j++)
217 decMtx[band*2*nSH + j*nSH + i] = conjf(B[i*2+j]); /* ^H */
218 }
219
220 /* clean-up */
221 free(W);
222 free(Yna_W);
223 free(Yna_W_Yna);
224 free(Yna_W_H);
225 free(Y_na);
226 free(B);
227}

229void getBinDecoder_LSDIFFEQ
230(
231 float_complex* hrtfs, /* the HRTFs; FLAT: N_bands x 2 x N_dirs */
232 float* hrtf_dirs_deg,
233 int N_dirs,
234 int N_bands,
235 int order,
236 float* weights,
237 float_complex* decMtx /* N_bands x 2 x (order+1)^2 */
238)
239{
240 int i, j, nSH, band;
241 float Gh;
242 float* Y_tmp;
243 float_complex* W, *Y_na, *Yna_W, *Yna_W_Yna, *Yna_W_H, *B_ls, *hrtfs_ls, *H_W;
244 float_complex C_ref[2][2], C_ls[2][2];
245 const float_complex calpha = cmplxf(1.0f, 0.0f), cbeta = cmplxf(0.0f, 0.0f);
246
247 nSH = ORDER2NSH(order);
248
249 /* integration weights */
250 W = calloc1d(N_dirs*N_dirs, sizeof(float_complex));
251 if(weights!=NULL)
252 for(i=0; i<N_dirs; i++)
253 W[i*N_dirs+i] = cmplxf(weights[i], 0.0f);
254 else
255 for(i=0; i<N_dirs; i++)
256 W[i*N_dirs+i] = cmplxf(1.0f/(float)N_dirs, 0.0f);
257
258 /* SH */
259 Y_tmp = malloc1d(nSH*N_dirs*sizeof(float));
260 Y_na = malloc1d(nSH*N_dirs*sizeof(float_complex));
261 getRSH(order, hrtf_dirs_deg, N_dirs, Y_tmp);
262 for(i=0; i<nSH*N_dirs; i++)
263 Y_na[i] = cmplxf(Y_tmp[i], 0.0f);
264 free(Y_tmp);
265
266 /* calculate decoding matrix per band */
267 Yna_W = malloc1d(nSH * N_dirs*sizeof(float_complex));
268 Yna_W_Yna = malloc1d(nSH * nSH * sizeof(float_complex));
269 Yna_W_H = malloc1d(nSH * 2 * sizeof(float_complex));
270 B_ls = malloc1d(nSH * 2 * sizeof(float_complex));
271 hrtfs_ls = malloc1d(2*N_dirs*sizeof(float_complex));
272 H_W = malloc1d(2*N_dirs*sizeof(float_complex));
273 cblas_cgemm(CblasRowMajor, CblasNoTrans, CblasNoTrans, nSH, N_dirs, N_dirs, &calpha,
274 Y_na, N_dirs,
275 W, N_dirs, &cbeta,
276 Yna_W, N_dirs);
277 cblas_cgemm(CblasRowMajor, CblasNoTrans, CblasTrans, nSH, nSH, N_dirs, &calpha,
278 Yna_W, N_dirs,
279 Y_na, N_dirs, &cbeta,
280 Yna_W_Yna, nSH);
281 for(band=0; band<N_bands; band++){
282 /* find least-squares decoding matrix */
283 cblas_cgemm(CblasRowMajor, CblasNoTrans, CblasConjTrans, nSH, 2, N_dirs, &calpha,
284 Yna_W, N_dirs,
285 &hrtfs[band*2*N_dirs], N_dirs, &cbeta,
286 Yna_W_H, 2);
287 utility_cglslv(NULL, Yna_W_Yna, nSH, Yna_W_H, 2, B_ls);
288 cblas_cgemm(CblasRowMajor, CblasConjTrans, CblasNoTrans, 2, N_dirs, nSH, &calpha,
289 B_ls, 2,
290 Y_na, N_dirs, &cbeta,
291 hrtfs_ls, N_dirs);
292
293 /* Diffuse-field responses */
294 cblas_cgemm(CblasRowMajor, CblasNoTrans, CblasNoTrans, 2, N_dirs, N_dirs, &calpha,
295 &hrtfs[band*2*N_dirs], N_dirs,
296 W, N_dirs, &cbeta,
297 H_W, N_dirs);
298 cblas_cgemm(CblasRowMajor, CblasNoTrans, CblasConjTrans, 2, 2, N_dirs, &calpha,
299 H_W, N_dirs,
300 &hrtfs[band*2*N_dirs], N_dirs, &cbeta,
301 (float_complex*)C_ref, 2);
302 cblas_cgemm(CblasRowMajor, CblasNoTrans, CblasNoTrans, 2, N_dirs, N_dirs, &calpha,
303 hrtfs_ls, N_dirs,
304 W, N_dirs, &cbeta,
305 H_W, N_dirs);
306 cblas_cgemm(CblasRowMajor, CblasNoTrans, CblasConjTrans, 2, 2, N_dirs, &calpha,
307 H_W, N_dirs,
308 hrtfs_ls, N_dirs, &cbeta,
309 (float_complex*)C_ls, 2);
310
311 /* Diffuse-Equalisation factor */
312 Gh = (sqrtf(crealf(C_ref[0][0])/(crealf(C_ls[0][0])+2.23e-7f)) +
313 sqrtf(crealf(C_ref[1][1])/(crealf(C_ls[1][1])+2.23e-7f))) /2.0f;
314
315 /* apply diff-EQ */
316 for(i=0; i<nSH; i++)
317 for(j=0; j<2; j++)
318 decMtx[band*2*nSH + j*nSH + i] = crmulf(conjf(B_ls[i*2+j]), Gh); /* ^H */
319 }
320
321 free(W);
322 free(Y_na);
323 free(Yna_W);
324 free(Yna_W_Yna);
325 free(Yna_W_H);
326 free(B_ls);
327 free(hrtfs_ls);
328 free(H_W);
329}

331void getBinDecoder_SPR
332(
333 float_complex* hrtfs, /* the HRTFs; FLAT: N_bands x 2 x N_dirs */
334 float* hrtf_dirs_deg,
335 int N_dirs,
336 int N_bands,
337 int order,
338 float* weights,
339 float_complex* decMtx /* N_bands x 2 x (order+1)^2 */
340)
341{
342 int band, i, j, nSH, nSH_nh, Nh_max, Nh, K_td;
343 float* hrtf_dirs_rad, *W, *cnd_num, *Y_nh, *Y_na, *tdirs_deg, *Y_td, *Ynh_Ytd, *tmp;
344 float_complex* Y_td_cmplx, *W_Ynh_Ytd, *hrtfs_td, *B;
345 const float_complex calpha = cmplxf(1.0f, 0.0f), cbeta = cmplxf(0.0f, 0.0f);
346
347 nSH = ORDER2NSH(order);
348
349 /* integration weights */
350 W = calloc1d(N_dirs*N_dirs, sizeof(float));
351 if(weights!=NULL)
352 for(i=0; i<N_dirs; i++)
353 W[i*N_dirs+i] = weights[i]/(4.0f*SAF_PI);
354 else
355 for(i=0; i<N_dirs; i++)
356 W[i*N_dirs+i] = 1.0f/(float)N_dirs;
357
358 /* find SH-order for interpolation of the HRTF set */
359 Nh_max = SAF_MIN((int)(sqrtf((float)N_dirs)-1.0f), 20); /* Cap to something more sensible if needed... */
360 hrtf_dirs_rad = malloc1d(N_dirs*2*sizeof(float));
361 cnd_num = malloc1d((Nh_max+1)*sizeof(float));
362 for(i=0; i<N_dirs; i++){ /* [azi, elev] degrees, to: [azi, inclination] radians */
363 hrtf_dirs_rad[i*2] = hrtf_dirs_deg[i*2]*(SAF_PI/180.0f);
364 hrtf_dirs_rad[i*2+1] = SAF_PI/2.0f - hrtf_dirs_deg[i*2+1]*(SAF_PI/180.0f);
365 }
366 checkCondNumberSHTReal(Nh_max, hrtf_dirs_rad, N_dirs, weights, cnd_num);
367 for(i=0, Nh=0; i<Nh_max+1; i++)
368 Nh = cnd_num[i] < 100.0f ? i : Nh;
369 saf_assert(Nh>=order, "Input order exceeds the modal order of the spatial grid");
370 nSH_nh = (Nh+1)*(Nh+1);
371 Y_nh = malloc1d(nSH_nh*N_dirs*sizeof(float));
372 getRSH(Nh, hrtf_dirs_deg, N_dirs, Y_nh);
373 Y_na = malloc1d(nSH * N_dirs *sizeof(float));
374 for(i=0; i<nSH; i++)
375 for(j=0; j<N_dirs; j++)
376 Y_na[i*N_dirs+j] = Y_nh[i*N_dirs+j];
377
378 /* Get t-design SH for ambisonic signals */
379 tdirs_deg = (float*)__HANDLES_Tdesign_dirs_deg[2*order-1];
380 K_td = __Tdesign_nPoints_per_degree[2*order-1];
381 Y_td = malloc1d(nSH_nh*K_td*sizeof(float));
382 getRSH(Nh, tdirs_deg, K_td, Y_td);
383 Y_td_cmplx = malloc1d(nSH_nh*K_td*sizeof(float_complex));
384 for(i=0; i<nSH_nh*K_td; i++)
385 Y_td_cmplx[i] = cmplxf(Y_td[i], 0.0f);
386
387 /* calculate decoding matrix per band */
388 Ynh_Ytd = malloc1d(N_dirs * K_td * sizeof(float));
389 tmp = malloc1d(N_dirs * K_td * sizeof(float));
390 W_Ynh_Ytd = malloc1d(N_dirs * K_td * sizeof(float_complex));
391 hrtfs_td = malloc1d(2*K_td*sizeof(float_complex));
392 B = malloc1d(nSH * 2 * sizeof(float_complex));
393 for(band=0; band<N_bands; band++){
394 cblas_sgemm(CblasRowMajor, CblasTrans, CblasNoTrans, N_dirs, K_td, nSH_nh, 1.0f,
395 Y_nh, N_dirs,
396 Y_td, K_td, 0.0f,
397 Ynh_Ytd, K_td);
398 cblas_sgemm(CblasRowMajor, CblasNoTrans, CblasNoTrans, N_dirs, K_td, N_dirs, 1.0f,
399 W, N_dirs,
400 Ynh_Ytd, K_td, 0.0f,
401 tmp, K_td);
402 for(i=0; i<N_dirs*K_td; i++)
403 W_Ynh_Ytd[i] = cmplxf(tmp[i], 0.0f);
404 cblas_cgemm(CblasRowMajor, CblasNoTrans, CblasNoTrans, 2, K_td, N_dirs, &calpha,
405 &hrtfs[band*2*N_dirs], N_dirs,
406 W_Ynh_Ytd, K_td, &cbeta,
407 hrtfs_td, K_td);
408 cblas_cgemm(CblasRowMajor, CblasNoTrans, CblasConjTrans, nSH, 2, K_td, &calpha,
409 Y_td_cmplx, K_td,
410 hrtfs_td, K_td, &cbeta,
411 B, 2);
412 for(i=0; i<nSH; i++)
413 for(j=0; j<2; j++)
414 decMtx[band*2*nSH + j*nSH + i] = crmulf(conjf(B[i*2+j]), 1.0f/(float)K_td); /* ^H */
415 }
416
417 free(hrtf_dirs_rad);
418 free(W);
419 free(cnd_num);
420 free(Y_nh);
421 free(Y_na);
422 free(Y_td);
423 free(Y_td_cmplx);
424 free(Ynh_Ytd);
425 free(tmp);
426 free(W_Ynh_Ytd);
427 free(hrtfs_td);
428 free(B);
429}

431void getBinDecoder_TA
432(
433 float_complex* hrtfs, /* the HRTFs; FLAT: N_bands x 2 x N_dirs */
434 float* hrtf_dirs_deg,
435 int N_dirs,
436 int N_bands,
437 int order,
438 float* freqVector,
439 float* itd_s,
440 float* weights,
441 float_complex* decMtx /* N_bands x 2 x (order+1)^2 */
442)
443{
444 int i, j, nSH, band, band_cutoff;
445 float cutoff, minVal;
446 float* Y_tmp;
447 float_complex* W, *Y_na, *hrtfs_mod, *Yna_W, *Yna_W_Yna, *Yna_W_H, *B;
448 const float_complex calpha = cmplxf(1.0f, 0.0f), cbeta = cmplxf(0.0f, 0.0f);
449
450 nSH = ORDER2NSH(order);
451
452 /* integration weights */
453 W = calloc1d(N_dirs*N_dirs, sizeof(float_complex));
454 if(weights!=NULL)
455 for(i=0; i<N_dirs; i++)
456 W[i*N_dirs+i] = cmplxf(weights[i], 0.0f);
457 else
458 for(i=0; i<N_dirs; i++)
459 W[i*N_dirs+i] = cmplxf(1.0f/(float)N_dirs, 0.0f);
460
461 /* SH */
462 Y_tmp = malloc1d(nSH*N_dirs*sizeof(float));
463 Y_na = malloc1d(nSH*N_dirs*sizeof(float_complex));
464 getRSH(order, hrtf_dirs_deg, N_dirs, Y_tmp);
465 for(i=0; i<nSH*N_dirs; i++)
466 Y_na[i] = cmplxf(Y_tmp[i], 0.0f);
467 free(Y_tmp);
468
469 /* find band index for cutoff frequency */
470 cutoff = 1.5e3f;
471 minVal = 2.23e10f;
472 for(band=0, band_cutoff=0; band<N_bands; band++){
473 if(minVal>fabsf(freqVector[band]-cutoff)){
474 minVal = fabsf(freqVector[band]-cutoff);
475 band_cutoff = band;
476 }
477 }
478
479 /* calculate decoding matrix per band */
480 Yna_W = malloc1d(nSH * N_dirs*sizeof(float_complex));
481 Yna_W_Yna = malloc1d(nSH * nSH * sizeof(float_complex));
482 Yna_W_H = malloc1d(nSH * 2 * sizeof(float_complex));
483 B = malloc1d(nSH * 2 * sizeof(float_complex));
484 hrtfs_mod = malloc1d(2*N_dirs*sizeof(float_complex));
485 cblas_cgemm(CblasRowMajor, CblasNoTrans, CblasNoTrans, nSH, N_dirs, N_dirs, &calpha,
486 Y_na, N_dirs,
487 W, N_dirs, &cbeta,
488 Yna_W, N_dirs);
489 cblas_cgemm(CblasRowMajor, CblasNoTrans, CblasTrans, nSH, nSH, N_dirs, &calpha,
490 Yna_W, N_dirs,
491 Y_na, N_dirs, &cbeta,
492 Yna_W_Yna, nSH);
493 for(band=0; band < N_bands; band++){
494 /* Remove itd from high frequency HRTFs */
495 if(band>=band_cutoff){
496 for(j=0; j<N_dirs; j++){
497 hrtfs_mod[0*N_dirs+j] = ccmulf(hrtfs[band_cutoff*2*N_dirs + 0*N_dirs + j],
498 cexpf( crmulf(cmplxf(0.0f, 0.0f), (itd_s[j]/2.0f))));
499 hrtfs_mod[1*N_dirs+j] = ccmulf(hrtfs[band_cutoff*2*N_dirs + 1*N_dirs + j],
500 cexpf( crmulf(cmplxf(0.0f, 0.0f), (-itd_s[j]/2.0f))));
501 }
502 }
503 else
504 memcpy(hrtfs_mod, &hrtfs[band*2*N_dirs], 2*N_dirs*sizeof(float_complex));
505 cblas_cgemm(CblasRowMajor, CblasNoTrans, CblasConjTrans, nSH, 2, N_dirs, &calpha,
506 Yna_W, N_dirs,
507 hrtfs_mod, N_dirs, &cbeta,
508 Yna_W_H, 2);
509 utility_cglslv(NULL, Yna_W_Yna, nSH, Yna_W_H, 2, B);
510 for(i=0; i<nSH; i++)
511 for(j=0; j<2; j++)

524void getBinDecoder_MAGLS
525(
526 float_complex* hrtfs, /* the HRTFs; FLAT: N_bands x 2 x N_dirs */
527 float* hrtf_dirs_deg,
528 int N_dirs,
529 int N_bands,
530 int order,
531 float* freqVector,
532 float* weights,
533 float_complex* decMtx /* N_bands x 2 x (order+1)^2 */
534)
535{
536 int i, j, nSH, band, band_cutoff;
537 float cutoff, minVal;
538 float* Y_tmp;
539 float_complex* W, *Y_na, *hrtfs_ls, *Yna_W, *Yna_W_Yna, *Yna_W_H, *H_mod, *B_magls;
540 const float_complex calpha = cmplxf(1.0f, 0.0f), cbeta = cmplxf(0.0f, 0.0f);
541
542 nSH = ORDER2NSH(order);
543
544 /* integration weights */
545 W = calloc1d(N_dirs*N_dirs, sizeof(float_complex));
546 if(weights!=NULL)
547 for(i=0; i<N_dirs; i++)
548 W[i*N_dirs+i] = cmplxf(weights[i], 0.0f);
549 else
550 for(i=0; i<N_dirs; i++)
551 W[i*N_dirs+i] = cmplxf(1.0f/(float)N_dirs, 0.0f);
552
553 /* SH */
554 Y_tmp = malloc1d(nSH*N_dirs*sizeof(float));
555 Y_na = malloc1d(nSH*N_dirs*sizeof(float_complex));
556 getRSH(order, hrtf_dirs_deg, N_dirs, Y_tmp);
557 for(i=0; i<nSH*N_dirs; i++)
558 Y_na[i] = cmplxf(Y_tmp[i], 0.0f);
559 free(Y_tmp);
560
561 /* find band index for cutoff frequency */
562 cutoff = 1.5e3f;
563 minVal = 2.23e10f;
564 for(band=0, band_cutoff=0; band<N_bands; band++){
565 if(minVal>fabsf(freqVector[band]-cutoff)){
566 minVal = fabsf(freqVector[band]-cutoff);
567 band_cutoff = band;
568 }
569 }
570
571 /* calculate decoding matrix per band */
572 Yna_W = malloc1d(nSH * N_dirs*sizeof(float_complex));
573 Yna_W_Yna = malloc1d(nSH * nSH * sizeof(float_complex));
574 Yna_W_H = malloc1d(nSH * 2 * sizeof(float_complex));
575 B_magls = malloc1d(nSH * 2 * sizeof(float_complex));
576 hrtfs_ls = malloc1d(2*N_dirs*sizeof(float_complex));
577 H_mod = malloc1d(2*N_dirs*sizeof(float_complex));
578 cblas_cgemm(CblasRowMajor, CblasNoTrans, CblasNoTrans, nSH, N_dirs, N_dirs, &calpha,
579 Y_na, N_dirs,
580 W, N_dirs, &cbeta,
581 Yna_W, N_dirs);
582 cblas_cgemm(CblasRowMajor, CblasNoTrans, CblasTrans, nSH, nSH, N_dirs, &calpha,
583 Yna_W, N_dirs,
584 Y_na, N_dirs, &cbeta,
585 Yna_W_Yna, nSH);
586 for (band=0; band<N_bands; band++){
587 if(band<=band_cutoff){
588 cblas_cgemm(CblasRowMajor, CblasNoTrans, CblasConjTrans, nSH, 2, N_dirs, &calpha,
589 Yna_W, N_dirs,
590 &hrtfs[band*2*N_dirs], N_dirs, &cbeta,
591 Yna_W_H, 2);
592 utility_cglslv(NULL, Yna_W_Yna, nSH, Yna_W_H, 2, B_magls);
593 }
594 else{
595 /* Remove itd from high frequency HRTFs */
596 cblas_cgemm(CblasRowMajor, CblasNoTrans, CblasNoTrans, 2, N_dirs, nSH, &calpha,
597 &decMtx[(band-1)*2*nSH] , nSH,
598 Y_na, N_dirs, &cbeta,
599 H_mod, N_dirs);
600 for(i=0; i<2*N_dirs; i++)
601 H_mod[i] = ccmulf(cmplxf(cabsf(hrtfs[band*2*N_dirs + i]), 0.0f), cexpf(cmplxf(0.0f, atan2f(cimagf(H_mod[i]), crealf(H_mod[i])))));
602 cblas_cgemm(CblasRowMajor, CblasNoTrans, CblasConjTrans, nSH, 2, N_dirs, &calpha,
603 Yna_W, N_dirs,
604 H_mod, N_dirs, &cbeta,
605 Yna_W_H, 2);
606 utility_cglslv(NULL, Yna_W_Yna, nSH, Yna_W_H, 2, B_magls);
607 }
608
609 for(i=0; i<nSH; i++)
610 for(j=0; j<2; j++)
611 decMtx[band*2*nSH + j*nSH + i] = conjf(B_magls[i*2+j]); /* ^H */
612 }
613
614 free(W);
615 free(Y_na);
616 free(Yna_W);
617 free(Yna_W_Yna);
618 free(Yna_W_H);
619 free(B_magls);
620 free(hrtfs_ls);
621 free(H_mod);
622}
