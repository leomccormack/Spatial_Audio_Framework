34 void ambi_roomsim_create
35 (
36  void ** const phAmbi
37 )
38 {
39  ambi_roomsim_data* pData = (ambi_roomsim_data*)malloc1d(sizeof(ambi_roomsim_data));
40  *phAmbi = (void*)pData;
41 
42  /* default user parameters */
43  pData->enableReflections = 1;
44  pData->sh_order = 3;
45  pData->refl_order = 3;
46  pData->nSources = 1;
47  pData->nReceivers = 1;
48  memcpy(pData->abs_wall, default_abs_wall, 6*sizeof(float));
49  memcpy(pData->room_dims, default_room_dims, 3*sizeof(float));
50  pData->chOrdering = CH_ACN;
51  pData->norm = NORM_SN3D;
52  memset(pData->src_pos, 0, ROOM_SIM_MAX_NUM_SOURCES*3*sizeof(float));
53  memset(pData->rec_pos, 0, ROOM_SIM_MAX_NUM_RECEIVERS*3*sizeof(float));
54 
55  /* Internal */
56  pData->hIms = NULL;
57  float src_pos[3] = {5.2f, 1.5f, 1.4f};
58  memcpy(pData->src_pos[0], src_pos, 3*sizeof(float));
59  float src2_pos[3] = {2.1f, 1.0f, 1.3f};
60  memcpy(pData->src_pos[1], src2_pos, 3*sizeof(float));
61  float src3_pos[3] = {3.1f, 5.0f, 2.3f};
62  memcpy(pData->src_pos[2], src3_pos, 3*sizeof(float));
63  float src4_pos[3] = {7.1f, 2.0f, 1.4f};
64  memcpy(pData->src_pos[3], src4_pos, 3*sizeof(float));
65  float rec_pos[3] = {5.2f, 3.5f, 1.4f};
66  memcpy(pData->rec_pos[0], rec_pos, 3*sizeof(float));
67  memcpy(pData->rec_pos[1], rec_pos, 3*sizeof(float));
68  pData->new_sh_order = pData->sh_order;
69  pData->new_nSources = pData->nSources;
70  pData->new_nReceivers = pData->nReceivers;
71 
72  pData->src_sigs = (float**)malloc2d(MAX_NUM_CHANNELS, AMBI_ROOMSIM_FRAME_SIZE, sizeof(float));
73  pData->rec_sh_outsigs = (float***)malloc3d(IMS_MAX_NUM_RECEIVERS, MAX_NUM_SH_SIGNALS, AMBI_ROOMSIM_FRAME_SIZE, sizeof(float));
74 
75  pData->reinit_room = 1;
76 }

104 void ambi_roomsim_process
105 (
106  void * const hAmbi,
107  const float *const * inputs,
108  float ** const outputs,
109  int nInputs,
110  int nOutputs,
111  int nSamples
112 )
113 {
114  ambi_roomsim_data *pData = (ambi_roomsim_data*)(hAmbi);
115  int i, j, rec, ch, nSources, nReceivers, nSH, order;
116  float maxTime_s;
117  CH_ORDER chOrdering;
118  NORM_TYPES norm;
119 
120  /* (ims_shoebox is actually much more flexible than this. So consider this as a minimal example, which will also make things easier when designing a GUI) */
121 
122  /* Reinitialise room if needed */
123  if(pData->reinit_room){
124  ims_shoebox_destroy(&(pData->hIms));
125  ims_shoebox_create(&(pData->hIms), pData->room_dims, (float*)pData->abs_wall, 250.0f, 1, 343.0f, pData->fs);
126  for(i=0; i<pData->new_nSources; i++) /* re-add source objects... */
127  pData->sourceIDs[i] = ims_shoebox_addSource(pData->hIms, (float*)pData->src_pos[i], &(pData->src_sigs[i]));
128  for(i=0; i<pData->new_nReceivers; i++) /* re-add receiver objects... */
129  pData->receiverIDs[i] = ims_shoebox_addReceiverSH(pData->hIms, pData->new_sh_order, (float*)pData->rec_pos[i], &(pData->rec_sh_outsigs[i]));
130  pData->nSources = pData->new_nSources;
131  pData->nReceivers = pData->new_nReceivers;
132  pData->sh_order = pData->new_sh_order;
133  pData->reinit_room = 0;
134  }
135 
136  /* Add/remove source objects */
137  if(pData->new_nSources!=pData->nSources){
138  if(pData->new_nSources>pData->nSources)
139  for(i=pData->nSources; i<pData->new_nSources; i++)
140  pData->sourceIDs[i] = ims_shoebox_addSource(pData->hIms, (float*)pData->src_pos[i], &(pData->src_sigs[i]));
141  else
142  for(i=pData->new_nSources; i<pData->nSources; i++)
143  ims_shoebox_removeSource(pData->hIms, pData->sourceIDs[i]);
144  pData->nSources = pData->new_nSources;
145  }
146 
147  /* Add/remove receiver objects */
148  if(pData->new_nReceivers!=pData->nReceivers){
149  if(pData->new_nReceivers>pData->nReceivers)
150  for(i=pData->nReceivers; i<pData->new_nReceivers; i++)
151  pData->receiverIDs[i] = ims_shoebox_addReceiverSH(pData->hIms, pData->sh_order, (float*)pData->rec_pos[i], &(pData->rec_sh_outsigs[i]));
152  else
153  for(i=pData->new_nReceivers; i<pData->nReceivers; i++)
154  ims_shoebox_removeReceiver(pData->hIms, pData->receiverIDs[i]);
155  pData->nReceivers = pData->new_nReceivers;
156  }
157 
158  /* local copies of user parameters */
159  chOrdering = pData->chOrdering;
160  norm = pData->norm;
161  order = SAF_MIN(pData->sh_order, MAX_SH_ORDER);
162  nSH = ORDER2NSH(order);
163  nSources = pData->nSources;
164  nReceivers = pData->nReceivers;
165  maxTime_s = -0.05f; /* 50ms */
166 
167  /* Process frame */
168  if (nSamples == AMBI_ROOMSIM_FRAME_SIZE) {
169  /* Load time-domain data */
170  for(i=0; i < SAF_MIN(nSources,nInputs); i++)
171  memcpy(pData->src_sigs[i], inputs[i], AMBI_ROOMSIM_FRAME_SIZE * sizeof(float));
172  for(; i < nInputs; i++)
173  memset(pData->src_sigs[i], 0, AMBI_ROOMSIM_FRAME_SIZE * sizeof(float));
174 
175  /* Update source/receiver positions, room dims/coeffs and re-compute echgrams
176  * (note, if nothing has changed since last frame then these calls will be bypassed internally) */
177  for(i=0; i<nSources; i++)
178  ims_shoebox_updateSource(pData->hIms, pData->sourceIDs[i], pData->src_pos[i]);
179  for(i=0; i<nReceivers; i++)
180  ims_shoebox_updateReceiver(pData->hIms, pData->receiverIDs[i], pData->rec_pos[i]);
181  ims_shoebox_setRoomDimensions(pData->hIms, pData->room_dims);
182  ims_shoebox_setWallAbsCoeffs(pData->hIms, (float*)pData->abs_wall);
183  ims_shoebox_computeEchograms(pData->hIms, pData->enableReflections ? pData->refl_order : 0, maxTime_s);
184 
185  /* Render audio for each receiver */
186  for(i=0; i<nReceivers; i++)
187  ims_shoebox_applyEchogramTD(pData->hIms, pData->receiverIDs[i], nSamples, 0);
188 
189  /* Handle output */
190  for(rec=0, i=0; rec<nReceivers; rec++){
191  /* account for output channel order */
192  switch(chOrdering){
193  case CH_ACN: break;
194  case CH_FUMA: convertHOAChannelConvention(FLATTEN2D(pData->rec_sh_outsigs[rec]), order, AMBI_ROOMSIM_FRAME_SIZE, HOA_CH_ORDER_ACN, HOA_CH_ORDER_FUMA); break;
195  }
196 
197  /* account for normalisation scheme */
198  switch(norm){
199  case NORM_N3D: break;
200  case NORM_SN3D: convertHOANormConvention(FLATTEN2D(pData->rec_sh_outsigs[rec]), order, AMBI_ROOMSIM_FRAME_SIZE, HOA_NORM_N3D, HOA_NORM_SN3D); break;
201  case NORM_FUMA: convertHOANormConvention(FLATTEN2D(pData->rec_sh_outsigs[rec]), order, AMBI_ROOMSIM_FRAME_SIZE, HOA_NORM_N3D, HOA_NORM_FUMA); break;
202  }
203 
204  /* Append this receiver's output channels to the master output buffer */
205  for(j=0; (j<SAF_MIN(nSH, MAX_NUM_SH_SIGNALS) && i<SAF_MIN(nOutputs, MAX_NUM_CHANNELS)); j++, i++)
206  memcpy(outputs[i], pData->rec_sh_outsigs[rec][j], AMBI_ROOMSIM_FRAME_SIZE * sizeof(float));
207  }
208  for(; i < nOutputs; i++)
209  memset(outputs[i], 0, AMBI_ROOMSIM_FRAME_SIZE * sizeof(float));
210  }
211  else{
212  for (ch=0; ch < nOutputs; ch++)
213  memset(outputs[ch],0, nSamples*sizeof(float));
214  }
215 }

238 {
