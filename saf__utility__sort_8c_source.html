217 void sortc
218 (
219  float_complex* in_vec,
220  float_complex* out_vec,
221  int len,
222  int descendFLAG
223 )
224 {
225  int i, start_idx, end_idx, sFlag, eFlag;
226  int* ind;
227  const float tol = 0.0001f;
228  float* vec_real, *vec_imag;
229 
230  ind = malloc1d(len*sizeof(int));
231 
232  /* First sort in_vec based on its real part */
233  vec_real = malloc1d(len*sizeof(float));
234  vec_imag = malloc1d(len*sizeof(float));
235  for(i=0; i<len; i++)
236  vec_real[i] = crealf(in_vec[i]);
237  sortf(vec_real, vec_real, ind, len, descendFLAG); /* real parts sorted based on the real parts */
238  for(i=0; i<len; i++)
239  vec_imag[i] = cimagf(in_vec[ind[i]]); /* imaginary parts sorted based on the real parts */
240 
241  /* Then take the values of in_vec that have identical real parts (given some
242  * tolerance), and sort them based on their imaginary parts */
243  sFlag = eFlag = 0;
244  start_idx = end_idx = -1;
245  for(i=0; i<len-1; i++){
246  /* Find duplicate real values (given some tolerance) */
247  if(fabsf(vec_real[i]-vec_real[i+1])<tol){
248  if(!sFlag){
249  start_idx = i;
250  sFlag = 1;
251  }
252  }
253  else if (sFlag){
254  end_idx = i;
255  eFlag = 1;
256  }
257 
258  /* Special case for last one */
259  if(sFlag && i==len-2){
260  end_idx = len-1;
261  eFlag = 1;
262  }
263 
264  /* Sort imaginary parts */
265  if( (sFlag) && (eFlag) ){
266  sortf(&vec_imag[start_idx], &vec_imag[start_idx], NULL, end_idx-start_idx+1, descendFLAG);
267  sFlag = 0; eFlag = 0;
268  }
269  }
270 
271  /* output */
272  for(i=0; i<len; i++)
273  out_vec[i] = cmplxf(vec_real[i], vec_imag[i]);
274 
275  /* clean-up */
276  free(ind);
277  free(vec_real);
278  free(vec_imag);
279 }

345 void cmplxPairUp
346 (
347  double_complex* in_vec,
348  double_complex* out_vec,
349  int len
350 )
351 {
352  int i, realCount;
353  double_complex tmp;
354 
355  /* First sort input vector in ascending order. The complex conjugate pairs
356  * are now in the correct order. */
357  sortz(in_vec, out_vec, len, 0);
358 
359  /* Now identify any purely real values, and push them to the end of the
360  * vector */
361  realCount = 0;
362  for(i=0; i<len-1-realCount; i++){
363  if(fabs(cimag(out_vec[i])) < 0.00001 ){
364  tmp = out_vec[i];
365  /* Push this value to the end */
366  memmove(&out_vec[i], &out_vec[i+1], (len-i-1)*sizeof(double_complex));
367  out_vec[len-1] = tmp;
368  realCount++;
369  }
370  }
371 }

373 void findClosestGridPoints
374 (
375  float* grid_dirs,
376  int nGrid,
377  float* target_dirs,
378  int nTarget,
379  int degFLAG,
380  int* idx_closest,
381  float* dirs_closest,
382  float* angle_diff
383 )
384 {
385  int i, j;
386  float* grid_xyz, *target_xyz;
387  float rcoselev, max_val, current_val;
388 
389  /* convert sph coords into Cartesian coords */
390  grid_xyz = malloc1d(nGrid*3*sizeof(float));
391  target_xyz = malloc1d(nTarget*3*sizeof(float));
392  if(degFLAG){
393  for(i=0; i<nGrid; i++){
394  grid_xyz[i*3+2] = sinf(grid_dirs[i*2+1]* SAF_PI/180.0f);
395  rcoselev = cosf( grid_dirs[i*2+1]* SAF_PI/180.0f);
396  grid_xyz[i*3] = rcoselev * cosf(grid_dirs[i*2]* SAF_PI/180.0f);
397  grid_xyz[i*3+1] = rcoselev * sinf(grid_dirs[i*2]* SAF_PI/180.0f);
398  }
399  for(i=0; i<nTarget; i++){
400  target_xyz[i*3+2] = sinf(target_dirs[i*2+1]* SAF_PI/180.0f);
401  rcoselev = cosf(target_dirs[i*2+1]* SAF_PI/180.0f);
402  target_xyz[i*3] = rcoselev * cosf(target_dirs[i*2]* SAF_PI/180.0f);
403  target_xyz[i*3+1] = rcoselev * sinf(target_dirs[i*2]* SAF_PI/180.0f);
404  }
405  }
406  else{
407  for(i=0; i<nGrid; i++){
408  grid_xyz[i*3+2] = sinf(grid_dirs[i*2+1]);
409  rcoselev = cosf(grid_dirs[i*2+1]);
410  grid_xyz[i*3] = rcoselev * cosf(grid_dirs[i*2]);
411  grid_xyz[i*3+1] = rcoselev * sinf(grid_dirs[i*2]);
412  }
413  for(i=0; i<nTarget; i++){
414  target_xyz[i*3+2] = sinf(target_dirs[i*2+1]);
415  rcoselev = cosf(target_dirs[i*2+1]);
416  target_xyz[i*3] = rcoselev * cosf(target_dirs[i*2]);
417  target_xyz[i*3+1] = rcoselev * sinf(target_dirs[i*2]);
418  }
419  }
420 
421  /* determine which 'grid_dirs' indices are the closest to 'target_dirs' */
422  for(i=0; i<nTarget; i++){
423  max_val = -2.23e10f;
424  for(j=0; j<nGrid; j++){
425  current_val = grid_xyz[j*3] * target_xyz[i*3] +
426  grid_xyz[j*3+1] * target_xyz[i*3+1] +
427  grid_xyz[j*3+2] * target_xyz[i*3+2];
428  if(current_val > max_val)
429  idx_closest[i] = j;
430  if(current_val>max_val){
431  idx_closest[i] = j;
432  max_val = current_val;
433  if(angle_diff!=NULL)
434  angle_diff[i] = acosf(max_val);
435  }
436  }
437  }
438 
439  /* optional output of directions */
440  if(dirs_closest!=NULL){
441  for(i=0; i<nTarget; i++){
442  dirs_closest[i*2] = grid_dirs[idx_closest[i]*2];
443  dirs_closest[i*2+1] = grid_dirs[idx_closest[i]*2+1];
444  }
445  }
446 
447  free(grid_xyz);
448  free(target_xyz);
449 }
