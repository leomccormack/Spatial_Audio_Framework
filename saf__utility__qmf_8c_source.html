105 void qmf_create
106 (
107  void ** const phQMF,
108  int nCHin,
109  int nCHout,
110  int hopsize,
111  int hybridmode,
112  QMF_FDDATA_FORMAT format
113 )
114 {
115  *phQMF = malloc1d(sizeof(qmf_data));
116  qmf_data *h = (qmf_data*)(*phQMF);
117  int i,j,K,N;
118  float scale;
119  float* k_tmp, *n_tmp;
120 
121  assert(hopsize==4 || hopsize==8 || hopsize==16 || hopsize==32 || hopsize==64 || hopsize==128);
122 
123  h->nCHin = nCHin;
124  h->nCHout = nCHout;
125  h->hopsize = hopsize;
126  h->hybridmode = hybridmode;
127  h->nBands = hybridmode ? hopsize+7 : hopsize; /* hybrid mode incurs an additional 7 bands */
128  h->format = format;
129 
130  K = hopsize;
131  N = 2*hopsize;
132  k_tmp = malloc1d(K*sizeof(float));
133  n_tmp = malloc1d(N*sizeof(float));
134 
135  /* QMF Analysis filters */
136  h->h_a = (float_complex**)malloc2d(K, N, sizeof(float_complex));
137  scale = (float)QMF_MAX_HOP_SIZE / (2.0f*(float)hopsize); /* (Used to balance the levels between different hopsizes) */
138  for(i=0; i<K; i++)
139  k_tmp[i] = SAF_PI/2.0f/(float)K * ((float)i+0.5f);
140  for(i=0; i<N; i++)
141  n_tmp[i] = 2.0f*(float)i - 2.0f*(float)K/(float)QMF_MAX_HOP_SIZE;
142  for(i=0; i<K; i++)
143  for(j=0; j<N; j++)
144  h->h_a[i][j] = crmulf(cexpf(cmplxf(0.0f, k_tmp[i]*n_tmp[j])), scale);
145 
146  /* QMF Synthesis filters */
147  h->h_s_real = (float**)malloc2d(N, K, sizeof(float));
148  h->h_s_imag = (float**)malloc2d(N, K, sizeof(float));
149  scale = 2.0f / QMF_MAX_HOP_SIZE; /* (Used to balance the levels between different hopsizes) */
150  for(i=0; i<N; i++)
151  n_tmp[i] = 2.0f*(float)i - (2.0f*(float)QMF_MAX_HOP_SIZE-1.0f)*(float)K/((float)QMF_MAX_HOP_SIZE/2.0f);
152  for(i=0; i<N; i++){
153  for(j=0; j<K; j++){
154  h->h_s_real[i][j] = scale * cosf(k_tmp[j]*n_tmp[i]);
155  h->h_s_imag[i][j] = scale * sinf(k_tmp[j]*n_tmp[i]);
156  }
157  }
158 
159  /* Prototype filter */
160  h->h_p = malloc1d(10*hopsize*sizeof(float));
161  for(i=0,j=0; i<10*QMF_MAX_HOP_SIZE; i+=QMF_MAX_HOP_SIZE/hopsize, j++)
162  h->h_p[j] = (float)__qmf_protofilter[i];
163 
164  /* Run-time buffers */
165  h->buffer_ana = (float**)malloc1d(nCHin*sizeof(float*)); /* actually faster to go with non-contiguous allocated memory here due to memmove... */
166  for(i=0; i<nCHin; i++)
167  h->buffer_ana[i] = calloc1d(hopsize * 10, sizeof(float));
168  h->buffer_syn = (float**)malloc1d(nCHout*sizeof(float*));
169  for(i=0; i<nCHout; i++)
170  h->buffer_syn[i] = calloc1d(hopsize * 20, sizeof(float));
171  h->buffer_win = malloc1d(hopsize * 10 * sizeof(float));
172  h->win_sum = malloc1d(hopsize * 2 * sizeof(float));
173  h->win_sum_cmplx_dummy = calloc1d(hopsize * 4, sizeof(float)); /* ca */
174  h->qmfTF_frame = malloc1d(hopsize * sizeof(float_complex));
175  h->qmfTF_frame_tmp = malloc1d(hopsize * sizeof(float));
176  h->tmp_real_frame = malloc1d(hopsize * 2 * sizeof(float));
177  h->tmp_imag_frame = malloc1d(hopsize * 2 * sizeof(float));
178 
179  /* Init hybrid filtering coefficients: */
180  if(hybridmode){
181  /* Coefficients to subdivide the lowest QMF band into 8 subbands. The
182  * 1st, 2nd, 7th and 8th subbands then become the 1st, 2nd, 5th and 6th
183  * hybrid bands directly. Hybrid bands 3 and 4 are then obtained by
184  * summing subbands 3 and 6, and 4 and 5, respectively. Therefore, this
185  * raises the total number of bands by 5.
186  * (Note: __fb8bandCoeffs = 0.125*kaiser(13, 4.6) ) */
187  for(i=0; i<8; i++)
188  for(j=0; j<QMF_HYBRID_FILTER_LENGTH; j++)
189  h->fb8bandCoeffs[i][j] = crmulf(cexpf(cmplxf(0.0f, -1.0f*SAF_PI*((float)j-((float)QMF_HYBRID_FILTER_LENGTH-1.0f)/2.0f)/8.0f*(1.0f+2.0f*(float)i))),
190  (float)__qmf_fb8bandCoeffs[j]);
191 
192  /* Coefficients to subdivide the 2nd and 3rd lowest QMF bands into 4
193  * subbands each. The 1st and 4th subdivided bands, and 2nd and 3rd
194  * subdivided bands are then summed to obtain the 2 hybrid bands for
195  * each QMF band . Therefore, this raises the total number of bands by
196  * 2 */
197  for(i=0; i<2; i++)
198  for(j=0; j<QMF_HYBRID_FILTER_LENGTH; j++)
199  h->fb4bandCoeffs[i][j] = cmplxf((float)__qmf_fb4bandCoeffs[j] *
200  cosf(2.0f*SAF_PI*(float)i*((float)j-(((float)QMF_HYBRID_FILTER_LENGTH-1.0f)/2.0f))/2.0f), 0.0f);
201 
202  /* For run-time */
203  h->qmfDelayBuffer = (float_complex***)calloc3d(nCHin, hopsize-QMF_NBANDS_2_SUBDIVIDE, (QMF_HYBRID_FILTER_LENGTH-1)/2 + 1, sizeof(float_complex)); /* ca */
204  h->hybBuffer = (float_complex***)calloc3d(nCHin, QMF_NBANDS_2_SUBDIVIDE, QMF_HYBRID_FILTER_LENGTH, sizeof(float_complex));
205  h->hybQmfTF_frame = malloc1d(h->nBands * sizeof(float_complex));
206 
207  /* Processing delay */
208  h->procDelay = hopsize*15+1;
209  }
210  else{
211  /* Processing delay */
212  h->procDelay = hopsize*9+1;
213  }
214 
215  /* clean-up */
216  free(k_tmp);
217  free(n_tmp);
218 }

339  cblas_cgemm(CblasRowMajor, CblasNoTrans, CblasNoTrans, 2, 1, QMF_HYBRID_FILTER_LENGTH, &calpha,
340  h->fb4bandCoeffs, QMF_HYBRID_FILTER_LENGTH,

342  subBands2, 1);
343  h->hybQmfTF_frame[6] = subBands2[1]; /* Flipped! */
344  h->hybQmfTF_frame[7] = subBands2[0];

428 #endif

464  utility_svvmul(&(h->buffer_syn[ch][h->hopsize*4]), &(h->h_p[h->hopsize*2]), h->hopsize, &(h->buffer_win[h->hopsize*2]));

473  /* Sum all 1:hopsizes to get output frame */
474  utility_svvadd(h->buffer_win, &(h->buffer_win[h->hopsize]), h->hopsize, &dataTD[ch][t*(h->hopsize)]);
475  utility_svvadd(&dataTD[ch][t*(h->hopsize)], &(h->buffer_win[h->hopsize*2]), h->hopsize, &dataTD[ch][t*(h->hopsize)]);
476  utility_svvadd(&dataTD[ch][t*(h->hopsize)], &(h->buffer_win[h->hopsize*3]), h->hopsize, &dataTD[ch][t*(h->hopsize)]);
477  utility_svvadd(&dataTD[ch][t*(h->hopsize)], &(h->buffer_win[h->hopsize*4]), h->hopsize, &dataTD[ch][t*(h->hopsize)]);
478  utility_svvadd(&dataTD[ch][t*(h->hopsize)], &(h->buffer_win[h->hopsize*5]), h->hopsize, &dataTD[ch][t*(h->hopsize)]);
479  utility_svvadd(&dataTD[ch][t*(h->hopsize)], &(h->buffer_win[h->hopsize*6]), h->hopsize, &dataTD[ch][t*(h->hopsize)]);
480  utility_svvadd(&dataTD[ch][t*(h->hopsize)], &(h->buffer_win[h->hopsize*7]), h->hopsize, &dataTD[ch][t*(h->hopsize)]);
