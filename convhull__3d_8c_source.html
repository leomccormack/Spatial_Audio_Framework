141 static void sort_float
142 (
143  CH_FLOAT* in_vec, /* vector[len] to be sorted */
144  CH_FLOAT* out_vec, /* if NULL, then in_vec is sorted "in-place" */
145  int* new_idices, /* set to NULL if you don't need them */
146  int len, /* number of elements in vectors, must be consistent with the input data */
147  int descendFLAG /* !1:ascending, 1:descending */
148 )
149 {
150  int i;
151  struct float_w_idx *data;
152 
153  data = (float_w_idx*)malloc(len*sizeof(float_w_idx));
154  for(i=0;i<len;i++) {
155  data[i].val=in_vec[i];
156  data[i].idx=i;
157  }
158  if(descendFLAG)
159  qsort(data,len,sizeof(data[0]),cmp_desc_float);
160  else
161  qsort(data,len,sizeof(data[0]),cmp_asc_float);
162  for(i=0;i<len;i++){
163  if (out_vec!=NULL)
164  out_vec[i] = data[i].val;
165  else
166  in_vec[i] = data[i].val; /* overwrite input vector */
167  if(new_idices!=NULL)
168  new_idices[i] = data[i].idx;
169  }
170  free(data);
171 }

205 static ch_vec3 cross(ch_vec3* v1, ch_vec3* v2)
206 {
207  ch_vec3 cross;
208  cross.x = v1->y * v2->z - v1->z * v2->y;
209  cross.y = v1->z * v2->x - v1->x * v2->z;
210  cross.z = v1->x * v2->y - v1->y * v2->x;
211  return cross;
212 }

215 static CH_FLOAT det_4x4(CH_FLOAT* m) {
216  return
217  m[3] * m[6] * m[9] * m[12] - m[2] * m[7] * m[9] * m[12] -
218  m[3] * m[5] * m[10] * m[12] + m[1] * m[7] * m[10] * m[12] +
219  m[2] * m[5] * m[11] * m[12] - m[1] * m[6] * m[11] * m[12] -
220  m[3] * m[6] * m[8] * m[13] + m[2] * m[7] * m[8] * m[13] +
221  m[3] * m[4] * m[10] * m[13] - m[0] * m[7] * m[10] * m[13] -
222  m[2] * m[4] * m[11] * m[13] + m[0] * m[6] * m[11] * m[13] +
223  m[3] * m[5] * m[8] * m[14] - m[1] * m[7] * m[8] * m[14] -

231 /* Calculates the coefficients of the equation of a PLANE in 3D.
232  * Original Copyright (c) 2014, George Papazafeiropoulos
233  * Distributed under the BSD (2-clause) license
234  */
235 static void plane_3d
236 (
237  CH_FLOAT* p,
238  CH_FLOAT* c,
239  CH_FLOAT* d
240 )
241 {
242  int i, j, k, l;
243  int r[3];
244  CH_FLOAT sign, det, norm_c;
245  CH_FLOAT pdiff[2][3], pdiff_s[2][2];
246 
247  for(i=0; i<2; i++)
248  for(j=0; j<3; j++)
249  pdiff[i][j] = p[(i+1)*3+j] - p[i*3+j];
250  memset(c, 0, 3*sizeof(CH_FLOAT));
251  sign = 1.0;
252  for(i=0; i<3; i++)
253  r[i] = i;
254  for(i=0; i<3; i++){
255  for(j=0; j<2; j++){
256  for(k=0, l=0; k<3; k++){
257  if(r[k]!=i){
258  pdiff_s[j][l] = pdiff[j][k];
259  l++;
260  }
261  }
262  }
263  det = pdiff_s[0][0]*pdiff_s[1][1] - pdiff_s[1][0]*pdiff_s[0][1];
264  c[i] = sign * det;
265  sign *= -1.0;
266  }
267  norm_c = (CH_FLOAT)0.0;
268  for(i=0; i<3; i++)
269  norm_c += (pow(c[i], 2.0));
270  norm_c = sqrt(norm_c);
271  for(i=0; i<3; i++)
272  c[i] /= norm_c;
273  (*d) = (CH_FLOAT)0.0;
274  for(i=0; i<3; i++)
275  (*d) += -p[i] * c[i];
276 }

295 /* A C version of the 3D quickhull matlab implementation from here:
296  * https://www.mathworks.com/matlabcentral/fileexchange/48509-computational-geometry-toolbox?focused=3851550&tab=example
297  * (*out_faces) is returned as NULL, if triangulation fails *
298  * Original Copyright (c) 2014, George Papazafeiropoulos
299  * Distributed under the BSD (2-clause) license
300  * Reference: "The Quickhull Algorithm for Convex Hull, C. Bradford Barber, David P. Dobkin
301  * and Hannu Huhdanpaa, Geometry Center Technical Report GCG53, July 30, 1993"
302  */
303 void convhull_3d_build
304 (
305  ch_vertex* const in_vertices,
306  const int nVert,
307  int** out_faces,
308  int* nOut_faces
309 )
310 {
311  int i, j, k, l, h;
312  int nFaces, p, d;
313  int* aVec, *faces;
314  CH_FLOAT dfi, v, max_p, min_p;
315  CH_FLOAT* points, *cf, *cfi, *df, *p_s, *span;
316 
317  if(nVert<3 || in_vertices==NULL){
318  (*out_faces) = NULL;
319  (*nOut_faces) = 0;
320  return;
321  }
322 
323  /* 3 dimensions. The code should theoretically work for >=2 dimensions, but "plane_3d" and "det_4x4" are hardcoded for 3,
324  * so would need to be rewritten */
325  d = 3;
326  span = (CH_FLOAT*)malloc(d*sizeof(CH_FLOAT));
327  for(j=0; j<d; j++){
328  max_p = 2.23e-13; min_p = 2.23e+13;
329  for(i=0; i<nVert; i++){
330  max_p = MAX(max_p, in_vertices[i].v[j]);
331  min_p = MIN(min_p, in_vertices[i].v[j]);
332  }
333  span[j] = max_p - min_p;
334  }
335  points = (CH_FLOAT*)malloc(nVert*(d+1)*sizeof(CH_FLOAT));
336  for(i=0; i<nVert; i++){
337  for(j=0; j<d; j++)
338  points[i*(d+1)+j] = in_vertices[i].v[j] + CH_NOISE_VAL*rand()/(float)RAND_MAX; /* noise mitigates duplicates */
339  points[i*(d+1)+d] = 1.0f; /* add a last column of ones. Used only for determinant calculation */
340  }
341 
342  /* The initial convex hull is a simplex with (d+1) facets, where d is the number of dimensions */
343  nFaces = (d+1);
344  faces = (int*)calloc(nFaces*d, sizeof(int));
345  aVec = (int*)malloc(nFaces*sizeof(int));
346  for(i=0; i<nFaces; i++)
347  aVec[i] = i;
348 
349  /* Each column of cf contains the coefficients of a plane */
350  cf = (CH_FLOAT*)malloc(nFaces*d*sizeof(CH_FLOAT));
351  cfi = (CH_FLOAT*)malloc(d*sizeof(CH_FLOAT));
352  df = (CH_FLOAT*)malloc(nFaces*sizeof(CH_FLOAT));
353  p_s = (CH_FLOAT*)malloc(d*d*sizeof(CH_FLOAT));
354  for(i=0; i<nFaces; i++){
355  /* Set the indices of the points defining the face */
356  for(j=0, k=0; j<(d+1); j++){
357  if(aVec[j]!=i){
358  faces[i*d+k] = aVec[j];
359  k++;
360  }
361  }
362 
363  /* Calculate and store the plane coefficients of the face */
364  for(j=0; j<d; j++)
365  for(k=0; k<d; k++)
366  p_s[j*d+k] = points[(faces[i*d+j])*(d+1) + k];
367 
368  /* Calculate and store the plane coefficients of the face */
369  plane_3d(p_s, cfi, &dfi);
370  for(j=0; j<d; j++)
371  cf[i*d+j] = cfi[j];
372  df[i] = dfi;
373  }
374  CH_FLOAT *A;
375  int *bVec, *fVec, *asfVec, *face_tmp;
376 
377  /* Check to make sure that faces are correctly oriented */
378  bVec = (int*)malloc(4*sizeof(int));
379  for(i=0; i<d+1; i++)
380  bVec[i] = i;
381 
382  /* A contains the coordinates of the points forming a simplex */
383  A = (CH_FLOAT*)calloc((d+1)*(d+1), sizeof(CH_FLOAT));
384  face_tmp = (int*)malloc((d+1)*sizeof(int));
385  fVec = (int*)malloc((d+1)*sizeof(int));
386  asfVec = (int*)malloc((d+1)*sizeof(int));
387  for(k=0; k<(d+1); k++){
388  /* Get the point that is not on the current face (point p) */
389  for(i=0; i<d; i++)
390  fVec[i] = faces[k*d+i];
391  sort_int(fVec, NULL, NULL, d, 0); /* sort accending */
392  p=k;
393  for(i=0; i<d; i++)
394  for(j=0; j<(d+1); j++)
395  A[i*(d+1)+j] = points[(faces[k*d+i])*(d+1) + j];
396  for(; i<(d+1); i++)
397  for(j=0; j<(d+1); j++)
398  A[i*(d+1)+j] = points[p*(d+1)+j];
399 
400  /* det(A) determines the orientation of the face */
401  v = det_4x4(A);
402 
403  /* Orient so that each point on the original simplex can't see the opposite face */
404  if(v<0){
405  /* Reverse the order of the last two vertices to change the volume */
406  for(j=0; j<d; j++)
407  face_tmp[j] = faces[k*d+j];
408  for(j=0, l=d-2; j<d-1; j++, l++)
409  faces[k*d+l] = face_tmp[d-j-1];
410 
411  /* Modify the plane coefficients of the properly oriented faces */
412  for(j=0; j<d; j++)
413  cf[k*d+j] = -cf[k*d+j];
414  df[k] = -df[k];
415  for(i=0; i<d; i++)
416  for(j=0; j<(d+1); j++)
417  A[i*(d+1)+j] = points[(faces[k*d+i])*(d+1) + j];
418  for(; i<(d+1); i++)
419  for(j=0; j<(d+1); j++)
420  A[i*(d+1)+j] = points[p*(d+1)+j];
421  }
422  }
423 
424  /* Coordinates of the center of the point set */
425  CH_FLOAT* meanp, *absdist, *reldist, *desReldist;
426  meanp = (CH_FLOAT*)calloc(d, sizeof(CH_FLOAT));
427  for(i=d+1; i<nVert; i++)
428  for(j=0; j<d; j++)
429  meanp[j] += points[i*(d+1)+j];
430  for(j=0; j<d; j++)
431  meanp[j] = meanp[j]/(CH_FLOAT)(nVert-d-1);
432 
433  /* Absolute distance of points from the center */
434  absdist = (CH_FLOAT*)malloc((nVert-d-1)*d * sizeof(CH_FLOAT));
435  for(i=d+1, k=0; i<nVert; i++, k++)
436  for(j=0; j<d; j++)
437  absdist[k*d+j] = (points[i*(d+1)+j] - meanp[j])/span[j];
438 
439  /* Relative distance of points from the center */
440  reldist = (CH_FLOAT*)calloc((nVert-d-1), sizeof(CH_FLOAT));
441  desReldist = (CH_FLOAT*)malloc((nVert-d-1) * sizeof(CH_FLOAT));
442  for(i=0; i<(nVert-d-1); i++)
443  for(j=0; j<d; j++)
444  reldist[i] += pow(absdist[i*d+j], 2.0);
445 
446  /* Sort from maximum to minimum relative distance */
447  int num_pleft, cnt;
448  int* ind, *pleft;
449  ind = (int*)malloc((nVert-d-1) * sizeof(int));
450  pleft = (int*)malloc((nVert-d-1) * sizeof(int));
451  sort_float(reldist, desReldist, ind, (nVert-d-1), 1);
452 
453  /* Initialize the vector of points left. The points with the larger relative
454  distance from the center are scanned first. */
455  num_pleft = (nVert-d-1);
456  for(i=0; i<num_pleft; i++)
457  pleft[i] = ind[i]+d+1;
458 
459  /* Loop over all remaining points that are not deleted. Deletion of points
460  occurs every #iter2del# iterations of this while loop */
461  memset(A, 0, (d+1)*(d+1) * sizeof(CH_FLOAT));
462 
463  /* cnt is equal to the points having been selected without deletion of
464  nonvisible points (i.e. points inside the current convex hull) */
465  cnt=0;
466 
467  /* The main loop for the quickhull algorithm */
468  CH_FLOAT detA;
469  CH_FLOAT* points_cf, *points_s;
470  int* visible_ind, *visible, *nonvisible_faces, *f0, *face_s, *u, *gVec, *horizon, *hVec, *pp, *hVec_mem_face;
471  int num_visible_ind, num_nonvisible_faces, n_newfaces, count, vis;
472  int f0_sum, u_len, start, num_p, index, horizon_size1;
473  int FUCKED;
474  FUCKED = 0;
475  u = horizon = NULL;
476  nFaces = d+1;
477  visible_ind = (int*)malloc(nFaces*sizeof(int));
478  points_cf = (CH_FLOAT*)malloc(nFaces*sizeof(CH_FLOAT));
479  points_s = (CH_FLOAT*)malloc(d*sizeof(CH_FLOAT));
480  face_s = (int*)malloc(d*sizeof(int));
481  gVec = (int*)malloc(d*sizeof(int));
482  while( (num_pleft>0) ){
483  /* i is the first point of the points left */
484  i = pleft[0];
485 
486  /* Delete the point selected */
487  for(j=0; j<num_pleft-1; j++)
488  pleft[j] = pleft[j+1];
489  num_pleft--;
490  if(num_pleft == 0)
491  free(pleft);
492  else
493  pleft = (int*)realloc(pleft, num_pleft*sizeof(int));
494 
495  /* Update point selection counter */
496  cnt++;
497 
498  /* find visible faces */
499  for(j=0; j<d; j++)
500  points_s[j] = points[i*(d+1)+j];
501  points_cf = (CH_FLOAT*)realloc(points_cf, nFaces*sizeof(CH_FLOAT));
502  visible_ind = (int*)realloc(visible_ind, nFaces*sizeof(int));
503 #ifdef CONVHULL_3D_USE_CBLAS
504  #ifdef CONVHULL_3D_USE_FLOAT_PRECISION
505  cblas_sgemm(CblasRowMajor, CblasNoTrans, CblasTrans, 1, nFaces, d, 1.0f,
506  points_s, d,
507  cf, d, 0.0f,
508  points_cf, nFaces);
509  #else
510  cblas_dgemm(CblasRowMajor, CblasNoTrans, CblasTrans, 1, nFaces, d, 1.0,
511  points_s, d,
512  cf, d, 0.0,
513  points_cf, nFaces);
514  #endif
515 #else
516  for (j = 0; j < nFaces; j++) {
517  points_cf[j] = 0;
518  for (k = 0; k < d; k++)
519  points_cf[j] += points_s[k]*cf[j*d+k];
520  }
521 #endif
522  num_visible_ind = 0;
523  for(j=0; j<nFaces; j++){
524  if(points_cf[j] + df[j] > 0.0){
525  num_visible_ind++; /* will sum to 0 if none are visible */
526  visible_ind[j] = 1;
527  }
528  else
529  visible_ind[j] = 0;
530  }
531  num_nonvisible_faces = nFaces - num_visible_ind;
532 
533  /* proceed if there are any visible faces */
534  if(num_visible_ind!=0){
535  /* Find visible face indices */
536  visible = (int*)malloc(num_visible_ind*sizeof(int));
537  for(j=0, k=0; j<nFaces; j++){
538  if(visible_ind[j]==1){
539  visible[k]=j;
540  k++;
541  }
542  }
543 
544  /* Find nonvisible faces */
545  nonvisible_faces = (int*)malloc(num_nonvisible_faces*d*sizeof(int));
546  f0 = (int*)malloc(num_nonvisible_faces*d*sizeof(int));
