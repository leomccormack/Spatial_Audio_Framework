32 static void applyWindowingFunction
33 (
34  WINDOWING_FUNCTION_TYPES type,
35  int winlength,
36  float* x
37 )
38 {
39  int i, N;
40 
41  /* if winlength is odd -> symmetric window (mid index has value=1) */
42  if ( !(winlength % 2 == 0) )
43  N = winlength-1;
44  /* otherwise, if winlength is even (index: winlength/2+1 = 1.0, but first
45  * value != last value) */
46  else
47  N = winlength;
48 
49  switch(type){
50  case WINDOWING_FUNCTION_RECTANGULAR:
51  break;
52 
53  case WINDOWING_FUNCTION_HAMMING:
54  for(i=0; i<winlength; i++)
55  x[i] *= 0.54f - 0.46f * (cosf(2.0f*SAF_PI*(float)i/(float)N)); /* more wide-spread coefficient values */
56  /* optimal equiripple coefficient values: */
57  /*x[i] *= 0.53836f - 0.46164f * (cosf(2.0f*M_PI*(float)i/(float)N));*/
58  break;
59 
60  case WINDOWING_FUNCTION_HANN:
61  for(i=0; i<winlength; i++)
62  x[i] *= 0.5f - 0.5f * (cosf(2.0f*SAF_PI*(float)i/(float)N));
63  break;
64 
65  case WINDOWING_FUNCTION_BARTLETT:
66  for(i=0; i<winlength; i++)
67  x[i] *= 1.0f - 2.0f * fabsf((float)i-((float)N/2.0f))/(float)N;
68  break;
69 
70  case WINDOWING_FUNCTION_BLACKMAN:
71  for(i=0; i<winlength; i++){
72  x[i] *= 0.42659f -
73  0.49656f *cosf(2.0f*SAF_PI*(float)i/(float)N) +
74  0.076849f*cosf(4.0f*SAF_PI*(float)i/(float)N);
75  }
76  break;
77 
78  case WINDOWING_FUNCTION_NUTTALL:
79  for(i=0; i<winlength; i++){
80  x[i] *= 0.355768f -
81  0.487396f*cosf(2.0f*SAF_PI*(float)i/(float)N) +
82  0.144232f*cosf(4.0f*SAF_PI*(float)i/(float)N) -
83  0.012604f*cosf(6.0f*SAF_PI*(float)i/(float)N);
84  }
85  break;
86 
87  case WINDOWING_FUNCTION_BLACKMAN_NUTTALL:
88  for(i=0; i<winlength; i++){
89  x[i] *= 0.3635819f -
90  0.4891775f*cosf(2.0f*SAF_PI*(float)i/(float)N) +
91  0.1365995f*cosf(4.0f*SAF_PI*(float)i/(float)N) +
92  0.0106411f*cosf(4.0f*SAF_PI*(float)i/(float)N);
93  }
94  break;
95 
96  case WINDOWING_FUNCTION_BLACKMAN_HARRIS:
97  for(i=0; i<winlength; i++){
98  x[i] *= 0.35875f -
99  0.48829f*cosf(2.0f*SAF_PI*(float)i/(float)N) +
100  0.14128f*cosf(4.0f*SAF_PI*(float)i/(float)N) +
101  0.01168f*cosf(4.0f*SAF_PI*(float)i/(float)N);
102  }
103  break;
104  }
105 }

224 void flattenMinphase
225 (
226  float* x,
227  int len
228 )
229 {
230  int i;
231  float_complex* ctd_tmp, *tdi_f, *tdi_f_labs, *dt_min_f;
232  void* hFFT;
233 
234  /* prep */
235  ctd_tmp = malloc1d(len*sizeof(float_complex));
236  tdi_f = malloc1d(len*sizeof(float_complex));
237  tdi_f_labs = malloc1d(len*sizeof(float_complex));
238  dt_min_f = malloc1d(len*sizeof(float_complex));
239  saf_fft_create(&hFFT, len);
240 
241  /* fft */
242  for(i=0; i<len; i++)
243  ctd_tmp[i] = cmplxf(x[i], 0.0f);
244  saf_fft_forward(hFFT, (float_complex*)ctd_tmp, (float_complex*)tdi_f);
245 
246  /* take log(cabs()) */
247  for(i=0; i<len; i++)
248  tdi_f_labs[i] = cmplxf(logf(cabsf(tdi_f[i])), 0.0f);
249 
250  /* Hilbert to acquire discrete-time analytic signal */
251  hilbert(tdi_f_labs, len, dt_min_f);
252 
253  /* compute minimum-phase response, and apply to tdi_f to flatten it to unity magnitude */
254  for(i=0; i<len; i++)
255  dt_min_f[i] = ccdivf(tdi_f[i], cexpf(conjf(dt_min_f[i])));
256 
257  /* ifft */
258  saf_fft_backward(hFFT, dt_min_f, ctd_tmp);
259 
260  /* overwrite input with EQ'd version */
261  for(i=0; i<len; i++)
262  x[i] = crealf(ctd_tmp[i]);
263 
264  /* tidy up */
265  saf_fft_destroy(&hFFT);
266  free(ctd_tmp);
267  free(tdi_f);
268  free(tdi_f_labs);
269  free(dt_min_f);
270 }

277 void biQuadCoeffs
278 (
279  BIQUAD_FILTER_TYPES filterType,
280  float fc,
281  float fs,
282  float Q,
283  float gain_dB,
284  float b[3],
285  float a[3]
286 )
287 {
288  float K, KK, D, V0;
289 
290  a[0] = 1.0f;
291 
292  /* calculate the IIR filter coefficients */
293  switch (filterType){
294  case BIQUAD_FILTER_LPF:
295  /* Filter design equations - DAFX (2nd ed) p50 */
296  K = tanf(SAF_PI * fc/fs);
297  KK = K * K;
298  D = KK * Q + K + Q;
299  b[0] = (KK * Q)/D;
300  b[1] = (2.0f * KK * Q)/D;
301  b[2] = b[0];
302  a[1] = (2.0f * Q * (KK - 1.0f))/D;
303  a[2] = (KK * Q - K + Q)/D;
304  break;
305 
306  case BIQUAD_FILTER_HPF:
307  /* Filter design equations - DAFX (2nd ed) p50 */
308  K = tanf(SAF_PI * fc/fs);
309  KK = K * K;
310  D = KK * Q + K + Q;
311  b[0] = (Q)/D;
312  b[1] = -(2.0f * Q)/D;
313  b[2] = b[0];
314  a[1] = (2.0f * Q * (KK - 1.0f))/D;
315  a[2] = (KK * Q - K + Q)/D;
316  break;
317 
318  case BIQUAD_FILTER_LOW_SHELF:
319  /* Filter design equations - DAFX (2nd ed) p64 */
320  K = tanf(SAF_PI * fc/fs);
321  V0 = powf(10.0f, (gain_dB/20.0f));
322  if (V0 < 1.0f)
323  V0 = 1.0f/V0;
324  KK = K * K;
325  if (gain_dB > 0.0f){
326  D = 1.0f + sqrtf(2.0f) * K + KK;
327  b[0] = (1.0f + sqrtf(2.0f * V0) * K + V0 * KK)/D;
328  b[1] = (2.0f*(V0*KK - 1.0f))/D;
329  b[2] = (1.0f - sqrtf(2.0f * V0) * K + V0 * KK)/D;
330  a[1] = (2.0f * (KK - 1.0f))/D;
331  a[2] = (1.0f - sqrtf(2.0f) * K + KK)/D;
332  }
333  else{
334  D = V0 + sqrtf(2.0f*V0)*K + KK;
335  b[0] = (V0*(1.0f + sqrtf(2.0f)*K + KK))/D;
336  b[1] = (2.0f*V0*(KK - 1.0f))/D;
337  b[2] = (V0*(1.0f - sqrtf(2.0f)*K + KK))/D;
338  a[1] = (2.0f * (KK - V0))/D;
339  a[2] = (V0 - sqrtf(2.0f*V0)*K + KK)/D;
340  }
341  break;
342 
343  case BIQUAD_FILTER_HI_SHELF:
344  /* Filter design equations - DAFX (2nd ed) p64 */
345  K = tanf(SAF_PI * fc/fs);
346  V0 = powf(10.0f, (gain_dB/20.0f));
347  if (V0 < 1.0f)
348  V0 = 1.0f/V0;
349  KK = K * K;
350  if (gain_dB > 0.0f){
351  D = 1.0f + sqrtf(2.0f) * K + KK;
352  b[0] = (V0 + sqrtf(2.0f * V0) * K + KK)/D;
353  b[1] = (2.0f*(KK - V0))/D;
354  b[2] = (V0 - sqrtf(2.0f * V0) * K + KK)/D;
355  a[1] = (2.0f*(KK - 1.0f))/D;
356  a[2] = (1.0f - sqrtf(2.0f) * K + KK)/D;
357  }
358  else{
359  D = 1.0f + sqrtf(2.0f*V0) * K + V0*KK;
360  b[0] = (V0*(1.0f + sqrtf(2.0f)*K + KK))/D;
361  b[1] = (2.0f*V0*(KK - 1.0f))/D;
362  b[2] = (V0*(1.0f - sqrtf(2.0f)*K + KK))/D;
363  a[1] = (2.0f * (V0*KK - 1.0f))/D;
364  a[2] = (1.0f - sqrtf(2.0f*V0)*K + V0*KK)/D;
365  }
366  break;
367 
368  case BIQUAD_FILTER_PEAK:
369  /* Filter design equations - DAFX (2nd ed) p66 */
370  K = tanf(SAF_PI * fc/fs);
371  V0 = powf(10.0f, (gain_dB/20.0f));
372  KK = K * K;
373  if (gain_dB > 0.0f){
374  D = 1.0f + (K/Q) + KK;
375  b[0] = (1.0f + (V0/Q) * K + KK)/D;
376  b[1] = (2.0f*(KK - 1.0f))/D;
377  b[2] = (1.0f - (V0/Q) * K + KK)/D;
378  a[1] = b[1];
379  a[2] = (1.0f - (K/Q) + KK)/D;
380  }
381  else {
382  D = 1.0f + (K/(V0*Q)) + KK;
383  b[0] = (1.0f + (K/Q) + KK)/D;
384  b[1] = (2.0f*(KK - 1.0f))/D;
385  b[2] = (1.0f - (K/Q) + KK)/D;
386  a[1] = b[1];
387  a[2] = (1.0f - (K/(V0*Q)) + KK)/D;
388  }
389  break;
390  }
391 }

415 void evalBiQuadTransferFunction
416 (
417  float b[3],
418  float a[3],
419  float* freqs,
420  int nFreqs,
421  float fs,
422  int mag2dB,
423  float* magnitude,
424  float* phase_rad
425 )
426 {
427  int ff;
428  float w, denom_real, denom_imag, num_real, num_imag;
429 
430  for(ff=0; ff<nFreqs; ff++){
431  w = tanf(SAF_PI * freqs[ff]/fs);
432  /* substituting Euler, z = e^(-jwn) = cos(wn) + j*sin(wn), into:
433  * H(z) = (b0 + b1*z^(-1) + b2*z^(-2)) / (1 + a1*z^(-1) + a2*z^(-2)): */
434  denom_real = 1.0f + a[1]*cosf(w) + a[2]*cosf(2.0f*w);
435  denom_imag = a[1]*sinf(w) + a[2]*sinf(2*w);
436  num_real = b[0] + b[1]*cosf(w) + b[2]*cosf(2.0f*w);
437  num_imag = b[1]*sinf(w) + b[2]*sinf(2.0f*w);
438 
439  if(magnitude!=NULL){
440  magnitude[ff] = sqrtf( (powf(num_real, 2.0f) + powf(num_imag, 2.0f)) / (powf(denom_real, 2.0f) + powf(denom_imag, 2.0f) + 2.23e-7f) );
441  if(mag2dB)
442  magnitude[ff] = 20.0f*log10f(magnitude[ff]);
443  }
444  if(phase_rad!=NULL)
445  phase_rad[ff] = atan2f(num_imag,num_real) - atan2f(denom_imag, denom_real);
446  }
447 }

449 void applyIIR
450 (
451  float* in_signal,
452  int nSamples,
453  int nCoeffs,
454  float* b,
455  float* a,
456  float* wz,
457  float* out_signal
458 )
459 {
460  int n, i;
461  float wn;
462 
463  /* For compiler speed-ups */
464  switch(nCoeffs){
465  case 1: assert(0); /* just divide in_signal by b[0]... */
466  case 2: applyIIR_1(in_signal, nSamples, b, a, wz, out_signal); return;
467  case 3: applyIIR_2(in_signal, nSamples, b, a, wz, out_signal); return;
468  case 4: applyIIR_3(in_signal, nSamples, b, a, wz, out_signal); return;
469  }
470 
471  /* difference equation (Direct form 2) */
472  for (n=0; n<nSamples; n++){
473  /* numerator */
474  wn = in_signal[n];
475  for (i=1; i<nCoeffs;i++)
476  wn = wn - (a[i] * wz[i-1]);
477 
478  /* denominator */
479  out_signal[n] = b[0] * wn;
480  for (i=1; i<nCoeffs; i++)
481  out_signal[n] = out_signal[n] + (b[i] * wz[i-1]);
482 
483  /* shuffle delays */
484  switch(nCoeffs-1){
485  case 10: wz[9] = wz[8]; /* fall through */
486  case 9: wz[8] = wz[7]; /* fall through */
487  case 8: wz[7] = wz[6]; /* fall through */
488  case 7: wz[6] = wz[5]; /* fall through */
489  case 6: wz[5] = wz[4]; /* fall through */
490  case 5: wz[4] = wz[3]; /* fall through */
491  case 4: wz[3] = wz[2]; /* fall through */
492  case 3: wz[2] = wz[1]; /* fall through */
493  case 2: wz[1] = wz[0]; /* fall through */
494  case 1: wz[0] = wn; break;
495  default: assert(0); /* A 5th order BPF/BSF or 10th order LPF/HPF?
496  * Sorry, I gotta put a stop to that... */
497  }
498  }

501 void butterCoeffs
502 (
503  BUTTER_FILTER_TYPES filterType,
504  int order,
505  float cutoff1,
506  float cutoff2,
507  float sampleRate,
508  double* b_coeffs,
509  double* a_coeffs
510 )
511 {
512  int i, j, k, np, tmp_len, numStates, oddPolesFLAG, nCoeffs;
513  double wlow, whi, w0, wl, w1, BW, Wn1, q;
514  double den[3];
515  double* c_state, *r, *b_coeffs_real;
516  double** a_state, **bf_ss, **tmp1, **tmp2, **a_bili;
517  double_complex kaT, kbT;
518  double_complex den_cmplx[3];
519  double_complex* proto, *proto_tmp, *a_coeffs_cmplx, *kern, *rcmplx, *b_coeffs_cmplx;
520 
521  wlow = (double)cutoff1/((double)sampleRate/2.0);
522  whi = (double)cutoff2/((double)sampleRate/2.0);
523  w0 = 4.0 * tan(SAF_PI*wlow/2.0);
524  Wn1 = 0.0;
525 
526  /* Compute prototype for Nth order Butterworth analogue lowpass filter */
527  if (order%2 != 0){/* ISODD */
528  tmp_len = (int)((float)order/2.0f); /* floor */
529  np = 2*tmp_len+1;
530  proto = malloc1d(np*sizeof(double_complex));
531  proto[np-1] = cmplx(-1.0,0.0);
532  }
533  else{ /* ISEVEN */
534  tmp_len = order/2;
535  np = 2*tmp_len;
536  proto = malloc1d(np*sizeof(double_complex));
537  }
538  proto_tmp = malloc1d(np*sizeof(double_complex));
539  for(i=1, j=0; i<=order-1; i+=2, j++)
540  proto_tmp[j] = cexp(cmplx(0.0, SAF_PI*(double)i/(2.0*(double)order) + SAF_PI/2.0) );
541  for (i=0; i<tmp_len; i++){
542  proto[2*i] = proto_tmp[i];
543  proto[2*i+1] = conj(proto_tmp[i]);
544  }
545 
546  /* Transform prototype into state space */
547  numStates = np;
548  cmplxPairUp(proto, proto_tmp, np);
549  memcpy(proto, proto_tmp, np*sizeof(double_complex));
550  free(proto_tmp);
551  a_state = (double**)calloc2d(numStates,numStates,sizeof(double));
552  c_state = malloc1d(numStates*sizeof(double));
553  if (np%2 != 0){/* ISODD */
554  a_state[0][0] = creal(proto[np-1]);
555  c_state[0] = 1.0;
556  np--;
557  oddPolesFLAG = 1;
558  }
559  else
560  oddPolesFLAG = 0;
561 
562  /* Adjust indices as needed */
563  for(i=1; i<np; i+=2){
564  polyz_v(&proto[i-1], den_cmplx, 2);
565  for(j=0; j<3; j++)
566  den[j] = creal(den_cmplx[j]);
567  j = oddPolesFLAG ? i-1 : i-2;
568 
569  if(j==-1){
570  a_state[0][0] = -den[1];
571  a_state[0][1] = -den[2];
572  a_state[1][0] = 1.0;
573  a_state[1][1] = 0.0;
574  c_state[0] = 0.0;
575  c_state[1] = 1.0;
576  }
577  else{
578  for(k=0; k<j+1; k++)
579  a_state[j+1][k] = c_state[k];
580  a_state[j+1][j+1] = -den[1];
581  a_state[j+1][j+2] = -den[2];
582  a_state[j+2][j+1] = 1.0;
583  a_state[j+2][j+2] = 0.0;
584 
585  for(k=0; k<j+1; k++)
586  c_state[k] = 0.0;
587  c_state[j+1] = 0.0;
588  c_state[j+2] = 1.0;
589  }
590  }
591 
592  /* Transform lowpass filter into the desired filter (while in state space) */
593  bf_ss = NULL;
594  switch(filterType){
595  case BUTTER_FILTER_HPF:
596  utility_dinv(FLATTEN2D(a_state), FLATTEN2D(a_state), numStates);
597  /* fall through */
598  case BUTTER_FILTER_LPF:
599  bf_ss = (double**)malloc2d(numStates,numStates,sizeof(double));
600  for(i=0; i<numStates; i++)
601  for(j=0; j<numStates; j++)
602  bf_ss[i][j] = w0*(a_state[i][j]);
603  break;
604  case BUTTER_FILTER_BSF:
605  utility_dinv(FLATTEN2D(a_state), FLATTEN2D(a_state), numStates);
606  /* fall through */
607  case BUTTER_FILTER_BPF:
608  numStates = numStates*2;
609  w1 = 4.0*tan(SAF_PI*whi/2.0);
610  BW = w1 - w0;
611  Wn1 = sqrt(w0*w1);
612  q = Wn1/BW;
613  bf_ss = (double**)calloc2d(numStates,numStates,sizeof(double));
614  for(i=0; i<numStates/2; i++)
615  for(j=0; j<numStates/2; j++)
616  bf_ss[i][j] = Wn1 * (a_state[i][j]) /q;
617  for(i=numStates/2; i<numStates; i++)
618  for(j=0; j<numStates/2; j++)
619  bf_ss[i][j] = (i-numStates/2) == j ? -Wn1 : 0.0;
620  for(i=0; i<numStates/2; i++)
621  for(j=numStates/2; j<numStates; j++)
622  bf_ss[i][j] = i == (j-numStates/2) ? Wn1 : 0.0;
623  break;
624  }
625  assert(bf_ss != NULL);
626  nCoeffs = numStates+1;
627 
628  /* Bilinear transformation to find the discrete equivalent of the filter */
629  tmp1 = (double**)malloc2d(numStates,numStates,sizeof(double));
630  tmp2 = (double**)malloc2d(numStates,numStates,sizeof(double));
631  a_bili = (double**)malloc2d(numStates,numStates,sizeof(double));
632  for(i=0; i<numStates; i++){
633  for(j=0; j<numStates; j++){
634  tmp1[i][j] = (i==j ? 1.0f : 0.0f) + bf_ss[i][j]*0.25;
635  tmp2[i][j] = (i==j ? 1.0f : 0.0f) - bf_ss[i][j]*0.25;
636  }
637  }
638  utility_dglslv(FLATTEN2D(tmp2), numStates, FLATTEN2D(tmp1), numStates, FLATTEN2D(a_bili));
639 
640  /* Compute the filter coefficients for the numerator and denominator */
641  a_coeffs_cmplx = malloc1d(nCoeffs*sizeof(double_complex));
642  polyd_m(FLATTEN2D(a_bili), a_coeffs_cmplx, numStates);
643  rcmplx = NULL;
644  r = NULL;
645  switch(filterType){
646  default:
647  /* fall through */
648  case BUTTER_FILTER_LPF:
649  r = malloc1d(numStates*sizeof(double));
650  for(i=0; i<numStates; i++)
651  r[i] = -1.0;
652  wl = 0.0;
653  break;
654  case BUTTER_FILTER_HPF:
655  r = malloc1d(numStates*sizeof(double));
656  for(i=0; i<numStates; i++)
657  r[i] = 1.0;
658  wl = SAF_PI;
659  break;
660  case BUTTER_FILTER_BPF:
661  r = malloc1d(numStates*sizeof(double));
662  wl = 2.0*atan2(Wn1, 4.0);
663  for(i=0; i<order;i++)
664  r[i] = 1.0;
665  for(; i<2*order;i++)
666  r[i] = -1.0;
667  break;
668  case BUTTER_FILTER_BSF:
669  rcmplx = malloc1d(numStates*sizeof(double_complex));
670  Wn1 = 2.0*atan2(Wn1,4.0);
671  wl = 0.0;
672  for(i=0; i<numStates;i++)
673  rcmplx[i] = cexp(cmplx(0.0, Wn1*pow(-1.0,(double)i)));
674  break;
675  }
676  b_coeffs_real = malloc1d(nCoeffs*sizeof(double));
677  if(filterType == BUTTER_FILTER_BSF){
678  b_coeffs_cmplx = malloc1d(nCoeffs*sizeof(double_complex));
679  polyz_v(rcmplx, b_coeffs_cmplx, numStates);
680  for(i=0; i<nCoeffs; i++)
681  b_coeffs_real[i] = creal(b_coeffs_cmplx[i]);
682  free(b_coeffs_cmplx);
683  }
684  else
685  polyd_v(r, b_coeffs_real, numStates);
686  kern = calloc1d(nCoeffs,sizeof(double_complex));
687  kaT = cmplx(0.0,0.0);
688  kbT = cmplx(0.0,0.0);
689  for(i=0; i<nCoeffs; i++){
690  kern[i] = cexp(cmplx(0.0,-wl*(double)i));
691  kaT = ccadd(kaT, crmul(kern[i],creal(a_coeffs_cmplx[i])));
692  kbT = ccadd(kbT, crmul(kern[i],b_coeffs_real[i]));
693  }
694 
695  /* output */
696  for(i=0; i<nCoeffs; i++){
697  b_coeffs[i] = creal(crmul(ccdiv(kaT,kbT), b_coeffs_real[i]));
698  a_coeffs[i] = creal(a_coeffs_cmplx[i]);
699  }
700 
701  /* clean-up */
702  free(proto);
703  free(a_state);
704  free(c_state);
705  free(bf_ss);
706  free(tmp1);
707  free(tmp2);
708  free(a_bili);
709  free(a_coeffs_cmplx);
710  free(b_coeffs_real);
711  free(kern);
712 }

735 void faf_IIRFilterbank_create
736 (
737  void** phFaF,
738  int order,
739  float* fc,
740  int nCutoffFreq,
741  float sampleRate,
742  int maxNumSamples
743 )
744 {
745  *phFaF = malloc1d(sizeof(faf_IIRFB_data));
746  faf_IIRFB_data *fb = (faf_IIRFB_data*)(*phFaF);
747  double b_lpf[4], a_lpf[4], b_hpf[4], a_hpf[4], r[7], revb[4], reva[4], q[4];
748  double tmp[7], tmp2[7];
749  double_complex d1[3], d2[3], d1_num[3], d2_num[3];
750  double_complex z[3], A[3][3], ztmp[7], ztmp2[7];
751  int i, j, f, filtLen, d1_len, d2_len;
752 
753  assert( (order==1) || (order==3) );
754  assert(nCutoffFreq>1);
755  filtLen = order + 1;
756  fb->filtOrder = order;
757  fb->filtLen = filtLen;
758 
759  /* Number of bands is always one more than the number of cut-off
760  * frequencies */
761  fb->nFilters = nCutoffFreq;
762  fb->nBands = nCutoffFreq + 1;
763 
764  /* Allocate memory for filter coefficients and delay buffers */
765  fb->b_hpf = (float**)malloc2d(nCutoffFreq, filtLen, sizeof(float));
766  fb->a_hpf = (float**)malloc2d(nCutoffFreq, filtLen, sizeof(float));
767  fb->b_lpf = (float**)malloc2d(nCutoffFreq, filtLen, sizeof(float));
768  fb->a_lpf = (float**)malloc2d(nCutoffFreq, filtLen, sizeof(float));
769  fb->wz_hpf = (float***)calloc3d(fb->nBands, nCutoffFreq, order, sizeof(float));
770  fb->wz_lpf = (float***)calloc3d(fb->nBands, nCutoffFreq, order, sizeof(float));
771  fb->wz_apf1 = (float***)calloc3d(fb->nBands, nCutoffFreq, order, sizeof(float));
772  fb->wz_apf2 = (float***)calloc3d(fb->nBands, nCutoffFreq, order, sizeof(float));
773  fb->maxNSamplesToExpect = maxNumSamples;
774  fb->tmp = malloc1d(maxNumSamples*sizeof(float));
775  fb->tmp2 = malloc1d(maxNumSamples*sizeof(float));
776 
777  /* Compute low-pass and complementary high-pass filter coefficients for each
778  * cut-off frequency */
779  for(f=0; f<nCutoffFreq; f++){
780  /* Low-pass filter */
781  butterCoeffs(BUTTER_FILTER_LPF, order, fc[f], 0.0f, sampleRate, (double*)b_lpf, (double*)a_lpf);
782 
783  /* IIR power complementary filter design (i.e. High-pass) */
784  for(i=0; i<filtLen; i++){
785  reva[i] = a_lpf[filtLen-i-1];
786  revb[i] = b_lpf[filtLen-i-1];
787  }
788  convd(revb, b_lpf, filtLen, filtLen, tmp);
789  convd(a_lpf, reva, filtLen, filtLen, tmp2);
790  for(i=0; i<2*filtLen-1; i++)
791  r[i] = tmp[i] - tmp2[i];
792  q[0] = sqrt(-r[0]/-1.0);
793  q[1] = -r[1]/(2.0*-1.0*q[0]);
794  if(order==3){
795  //q[3]=conj(-1.0*q[0]);
796  //q[2]=conj(-1.0*q[1]);
797  q[3] = -1.0*q[0];
798  q[2] = -1.0*q[1];
799  }
800  for(i=0; i<filtLen; i++)
801  q[i] = b_lpf[i] - q[i];
802 
803  /* Find roots of polynomial */
804  if(order==1)
805  z[0] = cmplx(-q[1]/q[0], 0.0);
806  else{ /* 3rd order */
807  memset(A, 0, 9*sizeof(double_complex));
808  A[0][0] = cmplx(-q[1]/q[0], 0.0);
809  A[0][1] = cmplx(-q[2]/q[0], 0.0);
810  A[0][2] = cmplx(-q[3]/q[0], 0.0);
811  A[1][0] = cmplx(1.0, 0.0);
812  A[2][1] = cmplx(1.0, 0.0);
813  utility_zeig((double_complex*)A, 3, NULL, NULL, NULL, (double_complex*)z);
814  }
815 
816  /* Separate the zeros inside the unit circle and the ones outside to
817  * form the allpass functions */
818  d1[0] = cmplx(1.0, 0.0);
819  d2[0] = cmplx(1.0, 0.0);
820  d1_len = d2_len = 1;
821  for(i=0; i<order; i++){
822  if (cabs(z[i]) < 1.0){
823  ztmp[0] = cmplx(1.0, 0.0);
824  ztmp[1] = crmul(z[i], -1.0);
825  convz(d2,ztmp,d2_len,2,ztmp2);
826  d2_len++;
827  for(j=0; j<d2_len; j++)
828  d2[j] = ztmp2[j];
829  }
830  else{
831  ztmp[0] = cmplx(1.0, 0.0);
832  ztmp[1] = ccdiv(cmplx(-1.0, 0.0), conj(z[i]));
833  convz(d1,ztmp,d1_len,2,ztmp2);
834  d1_len++;
835  for(j=0; j<d1_len; j++)
836  d1[j] = ztmp2[j];
837  }
838  }
839 
840  /* Convert coupled allpass filter to transfer function form (code from:
841  * https://github.com/nsk1001/Scilab-functions written by Nagma Samreen
842  * Khan) */
843  for(i=0; i<d1_len; i++)
844  d1_num[i] = conj(d1[d1_len-i-1]);
845  for(i=0; i<d2_len; i++)
846  d2_num[i] = conj(d2[d2_len-i-1]);
847  convz(d1_num, d2, d1_len, d2_len, ztmp);
848  convz(d2_num, d1, d2_len, d1_len, ztmp2);
849  for(i=0; i<filtLen; i++){
850  b_hpf[i] = -0.5 * creal(ccsub(ztmp[filtLen-i-1], ztmp2[filtLen-i-1]));
851  a_hpf[i] = a_lpf[i];
852  }
853 
854  /* Store in single precision for run-time */
855  for(i=0; i<filtLen; i++){
856  fb->b_hpf[f][i] = (float)b_hpf[i];
857  fb->a_hpf[f][i] = (float)a_hpf[i];
858  fb->b_lpf[f][i] = (float)b_lpf[i];
859  fb->a_lpf[f][i] = (float)a_lpf[i];
860  }
861  }
862 }

958 void FIRCoeffs
959 (
960  FIR_FILTER_TYPES filterType,
961  int order,
962  float fc1,
963  float fc2, /* only needed for band-pass/stop */
964  float fs,
965  WINDOWING_FUNCTION_TYPES windowType,
966  int scalingFLAG,
967  float* h_filt
968 )
969 {
970  int i, h_len;
971  float ft1, ft2, h_sum, f0;
972  float_complex h_z_sum;
973 
974  h_len = order + 1;
975  ft1 = fc1/(fs*2.0f);
976 
977  /* compute filter weights */
978  if(order % 2 == 0){
979  /* if order is multiple of 2 */
980  switch(filterType){
981  case FIR_FILTER_LPF:
982  for(i=0; i<h_len; i++)
983  h_filt[i] = i==order/2 ? 2.0f*ft1 : sinf(2.0f*SAF_PI*ft1*(float)(i-order/2)) / (SAF_PI*(float)(i-order/2));
984  break;
985 
986  case FIR_FILTER_HPF:
987  for(i=0; i<h_len; i++)
988  h_filt[i] = i==order/2 ? 1.0f - 2.0f*ft1 : -sinf(2.0f*ft1*SAF_PI*(float)(i-order/2)) / (SAF_PI*(float)(i-order/2));
989  break;
990 
991  case FIR_FILTER_BPF:
992  ft2 = fc2/(fs*2.0f);
993  for(i=0; i<h_len; i++){
994  h_filt[i] = i==order/2 ? 2.0f*(ft2-ft1) :
995  sinf(2.0f*SAF_PI*ft2*(float)(i-order/2)) / (SAF_PI*(float)(i-order/2)) - sinf(2.0f*SAF_PI*ft1*(float)(i-order/2)) / (SAF_PI*(float)(i-order/2));
996  }
997  break;
998 
999  case FIR_FILTER_BSF:
1000  ft2 = fc2/(fs*2.0f);
1001  for(i=0; i<h_len; i++){
1002  h_filt[i] = i==order/2 ? 1.0f - 2.0f*(ft2-ft1) :
1003  sinf(2.0f*SAF_PI*ft1*(float)(i-order/2)) / (SAF_PI*(float)(i-order/2)) - sinf(2.0f*SAF_PI*ft2*(float)(i-order/2)) / (SAF_PI*(float)(i-order/2));
1004  }
1005  break;
1006  }
1007  }
1008  else
1009  assert(0); /* please specify an even value for the filter 'order' argument */
1010 
1011  /* Apply windowing function */
1012  applyWindowingFunction(windowType, h_len, h_filt);
1013 
1014  /* Scaling, to ensure pass-band is truely at 1 (0dB).
1015  * [1] "Programs for Digital Signal Processing", IEEE Press John Wiley &
1016  * Sons, 1979, pg. 5.2-1.
1017  */
1018  if(scalingFLAG){
1019  switch(filterType){
1020  case FIR_FILTER_LPF:
1021  case FIR_FILTER_BSF:
1022  h_sum = 0.0f;
1023  for(i=0; i<h_len; i++)
1024  h_sum += h_filt[i];
1025  for(i=0; i<h_len; i++)
1026  h_filt[i] /= h_sum;
1027  break;
1028 
1029  case FIR_FILTER_HPF:
1030  f0 = 1.0f;
1031  h_z_sum = cmplxf(0.0f, 0.0f);
1032  for(i=0; i<h_len; i++)
1033  h_z_sum = ccaddf(h_z_sum, crmulf(cexpf(cmplxf(0.0f, -2.0f*SAF_PI*(float)i*f0/2.0f)), h_filt[i]));
1034  h_sum = cabsf(h_z_sum);
1035  for(i=0; i<h_len; i++)
1036  h_filt[i] /= h_sum;
1037  break;
1038 
1039  case FIR_FILTER_BPF:
1040  f0 = (fc1/fs+fc2/fs)/2.0f;
1041  h_z_sum = cmplxf(0.0f, 0.0f);
1042  for(i=0; i<h_len; i++)
1043  h_z_sum = ccaddf(h_z_sum, crmulf(cexpf(cmplxf(0.0f, -2.0f*SAF_PI*(float)i*f0/2.0f)), h_filt[i]));
1044  h_sum = cabsf(h_z_sum);
1045  for(i=0; i<h_len; i++)
1046  h_filt[i] /= h_sum;
1047  break;
1048  }
1049  }
1050 }
