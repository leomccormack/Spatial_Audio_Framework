53void unnorm_legendreP
54(
55 int n,
56 double* x,
57 int lenX,
58 double* y /* FLAT: (n+1) x lenX */
59)
60{
61 int i, m;
62 double s, norm, scale;
63 double* P, *s_n, *tc, *sqrt_n;
64
65 if(n==0){
66 for(i=0; i<lenX; i++)
67 y[i] = 1.0;
68 return;
69 }
70
71 /* alloc */
72 P = calloc1d((n+3)*lenX,sizeof(double));
73 s_n = malloc1d(lenX*sizeof(double));
74 tc = malloc1d(lenX*sizeof(double));
75 sqrt_n = malloc1d((2*n+1)*sizeof(double));
76
77 /* init */
78 for(i=0; i<lenX; i++){
79 s = sqrt(1.0-pow(x[i],2.0)) + 2.23e-20;
80 s_n[i] = pow(-s, (double)n);
81 tc[i] = -2.0 * x[i]/s;
82 }
83 for(i=0; i<2*n+1; i++)
84 sqrt_n[i] = sqrt((double)i);
85 norm = 1.0;
86 for(i=1; i<=n; i++)
87 norm *= 1.0 - 1.0/(2.0*(double)i);
88
89 /* Starting values for downwards recursion */
90 for(i=0; i<lenX; i++){
91 P[(n)*lenX+i] = sqrt(norm)*s_n[i];
92 P[(n-1)*lenX+i] = P[(n)*lenX+i] * tc[i] * (double)n/sqrt_n[2*n];
93 }
94
95 /* 3-step downwards recursion to m == 0 */
96 for(m=n-2; m>=0; m--)
97 for(i=0; i<lenX; i++)
98 P[(m)*lenX+i] = (P[(m+1)*lenX+i]*tc[i]*((double)m+1.0) - P[(m+2)*lenX+i] * sqrt_n[n+m+2] * sqrt_n[n-m-1])/(sqrt_n[n+m+1]*sqrt_n[n-m]);
99
100 /* keep up to the last 3 elements in P */
101 for(i=0; i<n+1; i++)
102 memcpy(&(y[i*lenX]), &(P[i*lenX]), lenX*sizeof(double));
103
104 /* Account for polarity when x == -/+1 for first value of P */
105 for(i=0; i<lenX; i++)
106 if(sqrt(1.0-pow(x[i],2.0))==0)
107 y[i] = pow(x[i],(double)n);
108
109 /* scale each row by: sqrt((n+m)!/(n-m)!) */
110 for(m=1; m<n; m++){
111 scale = 1.0;
112 for(i=n-m+1; i<n+m+1; i++)
113 scale*=sqrt_n[i];
114 for(i=0; i<lenX; i++)
115 y[m*lenX+i] *= scale;
116 }
117 scale = 1.0;
118 for(i=1; i<2*n+1; i++)
119 scale*=sqrt_n[i];
120 for(i=0; i<lenX; i++)
121 y[n*lenX+i] *= scale;
122
123 free(P);
124 free(s_n);
125 free(tc);
126 free(sqrt_n);
127}

129void unnorm_legendreP_recur
130(
131 int n,
132 float* x,
133 int lenX,
134 float* Pnm_minus1,
135 float* Pnm_minus2,
136 float* Pnm
137)
138{
139 int i, m, k, kk;
140 float x2, one_min_x2, dfact_k;
141
142 if(n==0){
143 for(i=0; i<lenX; i++)
144 Pnm[i] = 1.0f;
145 return;
146 }
147
148 for(i=0; i<lenX; i++){
149 x2 = (x[i])*(x[i]);
150 switch(n) {
151 case 1:
152 Pnm[0*lenX+i] = x[i];
153 Pnm[1*lenX+i] = sqrtf(1.0f-x2);
154 break;
155 case 2:
156 Pnm[0*lenX+i] = (3.0f*x2-1.0f)/2.0f;
157 Pnm[1*lenX+i] = (x[i])*3.0f*sqrtf(1.0f-x2);
158 Pnm[2*lenX+i] = 3.0f*(1.0f-x2);
159 break;
160 default:
161 one_min_x2 = 1.0f-x2;
162 /* last term m=n */
163 k = 2*n-1;
164 dfact_k = 1.0f;
165 if ((k % 2) == 0)
166 for (kk=1; kk<k/2+1; kk++)
167 dfact_k *= 2.0f*(float)kk;
168 else
169 for (kk=1; kk<(k+1)/2+1; kk++)
170 dfact_k *= (2.0f*(float)kk-1.0f);
171
172 Pnm[n*lenX+i] = dfact_k * powf(one_min_x2, (float)n/2.0f);
173 /* before last term */
174 /* P_{n(n-1)} = (2*n-1)*x*P_{(n-1)(n-1)} */
175 Pnm[(n-1)*lenX+i] = (float)k * (x[i]) *Pnm_minus1[(n-1)*lenX+i];
176 /* three term recurence for the rest */
177 for (m=0; m<n-1; m++)
178 /* P_l = ( (2l-1)xP_(l-1) - (l+m-1)P_(l-2) )/(l-m) */
179 Pnm[m*lenX+i] = ( ((float)k * (x[i]) *Pnm_minus1[m*lenX+i]) - ((float)(n+m-1) * Pnm_minus2[m*lenX+i])) / (float)(n-m);
180 break;
181 }
182 }
183}

190void getSHreal
191(
192 int order,
193 float* dirs_rad,
194 int nDirs,
195 float* Y /* the SH weights: (order+1)^2 x nDirs */
196)
197{
198 int dir, j, n, m, idx_Y;
199 double* Lnm, *CosSin;
200 double *p_nm, *cos_incl;
201 double *norm_real;
202
203 if(nDirs<1)
204 return;
205
206 Lnm = malloc1d((2*order+1)*nDirs*sizeof(double));
207 norm_real = malloc1d((2*order+1)*sizeof(double));
208 CosSin = malloc1d((2*order+1)*sizeof(double));
209 cos_incl = malloc1d(nDirs*sizeof(double));
210 p_nm = malloc1d((order+1)*nDirs * sizeof(double));
211 for (dir = 0; dir<nDirs; dir++)
212 cos_incl[dir] = cos((double)dirs_rad[dir*2+1]);
213
214 idx_Y = 0;
215 for(n=0; n<=order; n++){
216 /* vector of unnormalised associated Legendre functions of current order */
217 unnorm_legendreP(n, cos_incl, nDirs, p_nm); /* includes Condon-Shortley phase term */
218
219 for(dir=0; dir<nDirs; dir++){
220 /* cancel the Condon-Shortley phase from the definition of the Legendre functions to result in signless real SH */
221 if (n != 0)
222 for(m=-n, j=0; m<=n; m++, j++)
223 Lnm[j*nDirs+dir] = pow(-1.0, (double)abs(m)) * p_nm[abs(m)*nDirs+dir];
224 else
225 Lnm[dir] = p_nm[dir];
226 }
227
228 /* normalisation */
229 for(m=-n, j=0; m<=n; m++, j++)
230 norm_real[j] = sqrt( (2.0*(double)n+1.0) * (double)factorial(n-abs(m)) / (4.0*SAF_PId*(double)factorial(n+abs(m))) );
231
232 /* norm_real * Lnm_real .* CosSin; */
233 for(dir=0; dir<nDirs; dir++){
234 for(m=-n, j=0; m<=n; m++, j++){
235 if(j<n)
236 Y[(j+idx_Y)*nDirs+dir] = (float)(norm_real[j] * Lnm[j*nDirs+dir] * sqrt(2.0)*sin((double)(n-j)*(double)dirs_rad[dir*2]));
237 else if(j==n)
238 Y[(j+idx_Y)*nDirs+dir] = (float)(norm_real[j] * Lnm[j*nDirs+dir]);
239 else /* (j>n) */
240 Y[(j+idx_Y)*nDirs+dir] = (float)(norm_real[j] * Lnm[j*nDirs+dir] * sqrt(2.0)*cos((double)(abs(m))*(double)dirs_rad[dir*2]));
241 }
242 }
243
244 /* increment */
245 idx_Y = idx_Y + (2*n+1);
246 }
247
248 free(p_nm);
249 free(Lnm);
250 free(norm_real);
251 free(CosSin);
252 free(cos_incl);
253}

255void getSHreal_recur
256(
257 int N,
258 float* dirs_rad,
259 int nDirs,
260 float* Y
261)
262{
263 int n, m, i, dir, index_n;
264 float Nn0, Nnm;
265 float sleg_n[8], sleg_n_1[8], sleg_n_2[8], scos_incl, sfactorials_n[15];
266 float* leg_n, *leg_n_1, *leg_n_2, *cos_incl, *factorials_n;
267
268 if(nDirs<1)
269 return;
270
271 if(N<=7 && nDirs == 1){
272 /* Single direction optimisation for up to 7th order */
273 leg_n = sleg_n;
274 leg_n_1 = sleg_n_1;
275 leg_n_2 = sleg_n_2;
276 cos_incl = &scos_incl;
277 factorials_n = sfactorials_n;
278 }
279 else{
280 factorials_n = malloc1d((2*N+1)*sizeof(float));
281 leg_n = malloc1d((N+1)*nDirs * sizeof(float));
282 leg_n_1 = malloc1d((N+1)*nDirs * sizeof(float));
283 leg_n_2 = malloc1d((N+1)*nDirs * sizeof(float));
284 cos_incl = malloc1d(nDirs * sizeof(float));
285 }
286 index_n = 0;
287
288 /* precompute factorials */
289 for (i = 0; i < 2*N+1; i++)
290 factorials_n[i] = (float)factorial(i);
291
292 /* cos(inclination) = sin(elevation) */
293 for (dir = 0; dir<nDirs; dir++)
294 cos_incl[dir] = cosf(dirs_rad[dir*2+1]);
295
296 /* compute SHs with the recursive Legendre function */
297 for (n = 0; n<N+1; n++) {
298 if (n==0) {
299 for (dir = 0; dir<nDirs; dir++)
300 Y[n*nDirs+dir] = 1.0f/SQRT4PI;
301 index_n = 1;
302 }
303 else {
304 unnorm_legendreP_recur(n, cos_incl, nDirs, leg_n_1, leg_n_2, leg_n); /* does NOT include Condon-Shortley phase term */
305
306 Nn0 = sqrtf(2.0f*(float)n+1.0f);
307 for (dir = 0; dir<nDirs; dir++){
308 for (m = 0; m<n+1; m++) {
309 if (m==0)
310 Y[(index_n+n)*nDirs+dir] = Nn0/SQRT4PI * leg_n[m*nDirs+dir];
311 else {
312 Nnm = Nn0* sqrtf( 2.0f * factorials_n[n-m]/factorials_n[n+m] );
313 Y[(index_n+n-m)*nDirs+dir] = Nnm/SQRT4PI * leg_n[m*nDirs+dir] * sinf((float)m * (dirs_rad[dir*2]));
314 Y[(index_n+n+m)*nDirs+dir] = Nnm/SQRT4PI * leg_n[m*nDirs+dir] * cosf((float)m * (dirs_rad[dir*2]));
315 }
316 }
317 }
318 index_n += 2*n+1;
319 }
320 utility_svvcopy(leg_n_1, (N+1)*nDirs, leg_n_2);
321 utility_svvcopy(leg_n, (N+1)*nDirs, leg_n_1);
322 }
323
324 if(N>7 || nDirs > 1){
325 free(factorials_n);
326 free(leg_n);
327 free(leg_n_1);
328 free(leg_n_2);
329 free(cos_incl);
330 }
331}

333void getSHcomplex
334(
335 int order,
336 float* dirs_rad,
337 int nDirs,
338 float_complex* Y
339)
340{
341 int dir, j, n, m, idx_Y;
342 double *norm_real;
343 double *Lnm, *cos_incl;
344 double_complex Ynm;
345
346 Lnm = malloc1d((order+1)*nDirs*sizeof(double));
347 norm_real = malloc1d((order+1)*sizeof(double));
348 cos_incl = malloc1d(nDirs*sizeof(double));
349 for (dir = 0; dir<nDirs; dir++)
350 cos_incl[dir] = cos((double)dirs_rad[dir*2+1]);
351
352 idx_Y = 0;
353 for(n=0; n<=order; n++){
354 /* vector of unnormalised associated Legendre functions of current order */
355 unnorm_legendreP(n, cos_incl, nDirs, Lnm); /* includes Condon-Shortley phase term */
356
357 /* normalisation */
358 for(m=0; m<=n; m++)
359 norm_real[m] = sqrt( (2.0*(double)n+1.0)*(double)factorial(n-m) / (4.0*SAF_PId*(double)factorial(n+m)) );
360
361 /* norm_real .* Lnm_real .* CosSin; */
362 for(dir=0; dir<nDirs; dir++){
363 for(m=-n, j=0; m<=n; m++, j++){
364 if(m<0){
365 Ynm = crmul(conj(crmul(cexp(cmplx(0.0, (double)abs(m)*(double)dirs_rad[dir*2])), norm_real[abs(m)] * Lnm[abs(m)*nDirs+dir])), pow(-1.0, (double)abs(m)));
366 Y[(j+idx_Y)*nDirs+dir] = cmplxf((float)creal(Ynm), (float)cimag(Ynm));
367 }
368 else {/* (m>=0) */
369 Ynm = crmul(cexp(cmplx(0.0, (double)abs(m)*(double)dirs_rad[dir*2])), norm_real[m] * Lnm[m*nDirs+dir]);
370 Y[(j+idx_Y)*nDirs+dir] = cmplxf((float)creal(Ynm), (float)cimag(Ynm));
371 }
372 }
373 }
374
375 /* increment */
376 idx_Y = idx_Y + (2*n+1);
377 }
378
379 free(Lnm);
380 free(norm_real);
381 free(cos_incl);
382}

477/* Ivanic, J., Ruedenberg, K. (1998). Rotation Matrices for Real Spherical Harmonics. Direct Determination
478 * by Recursion Page: Additions and Corrections. Journal of Physical Chemistry A, 102(45), 9099?9100. */
479void getSHrotMtxReal
480(
481 float Rxyz[3][3],
482 float* RotMtx/*(L+1)^2 x (L+1)^2 */,
483 int L
484)
485{
486 int i, j, M, l, m, n, d, bandIdx, denom;
487 float u, v, w;
488 float R_1[3][3], _R_lm1[64*64], _R_l[64*64];
489 float* R_lm1, *R_l;
490
491 /* Prep */
492 M = (L+1) * (L+1);
493 if(L<=7){
494 R_lm1 = _R_lm1;
495 R_l = _R_l;
496 }
497 else{
498 R_lm1 = malloc1d(M*M*sizeof(float));
499 R_l = malloc1d(M*M*sizeof(float));
500 }
501 memset(RotMtx, 0, M*M*sizeof(float));
502
503 /* zeroth-band (l=0) is invariant to rotation */
504 RotMtx[0] = 1;
505
506 /* the first band (l=1) is directly related to the rotation matrix */
507 R_1[0][0] = Rxyz[1][1];
508 R_1[0][1] = Rxyz[1][2];
509 R_1[0][2] = Rxyz[1][0];
510 R_1[1][0] = Rxyz[2][1];
511 R_1[1][1] = Rxyz[2][2];
512 R_1[1][2] = Rxyz[2][0];
513 R_1[2][0] = Rxyz[0][1];
514 R_1[2][1] = Rxyz[0][2];
515 R_1[2][2] = Rxyz[0][0];
516 for (i=1; i<4; i++){
517 R_lm1[(i-1)*M+0] = R_1[i-1][0];
518 R_lm1[(i-1)*M+1] = R_1[i-1][1];
519 R_lm1[(i-1)*M+2] = R_1[i-1][2];
520 for (j=1; j<4; j++)
521 RotMtx[i*M+j] = R_1[i-1][j-1];
522 }
523
524 /* compute rotation matrix of each subsequent band recursively */
525 bandIdx = 4;
526 for(l = 2; l<=L; l++){
527 for(i=0; i<2*l+1; i++)
528 memset(R_l + i*M, 0, (2*l+1) * sizeof(float));
529 for(m=-l; m<=l; m++){
530 for(n=-l; n<=l; n++){
531 /* compute u,v,w terms of Eq.8.1 (Table I) */
532 d = m == 0 ? 1 : 0; /* the delta function d_m0 */
533 denom = abs(n) == l ? (2*l)*(2*l-1) : (l*l-n*n);
534 u = sqrtf( (float)((l*l-m*m)) / (float)denom);
535 v = sqrtf( (float)((1+d)*(l+abs(m)-1)*(l+abs(m))) / (float)denom) * (float)(1-2*d)*0.5f;
536 w = sqrtf( (float)((l-abs(m)-1)*(l-abs(m))) / (float)denom) * (float)(1-d)*(-0.5f);
537
538 /* computes Eq.8.1 */
539 if (u!=0)
540 u = u* getU(M,l,m,n,R_1,R_lm1);
541 if (v!=0)
542 v = v* getV(M,l,m,n,R_1,R_lm1);
543 if (w!=0)
544 w = w* getW(M,l,m,n,R_1,R_lm1);
545
546 R_l[(m+l)*M+(n+l)] = u+v+w;
547 }
548 }
549
550 for(i=0; i<2*l+1; i++)
551 for(j=0; j<2*l+1; j++)
552 RotMtx[(bandIdx + i)*M + (bandIdx + j)] = R_l[i*M+j];
553 for(i=0; i<2*l+1; i++)
554 memcpy(R_lm1+i*M, R_l + i*M, (2*l+1) * sizeof(float));
555 bandIdx += 2*l+1;
556 }
557
558 /* clean-up */
559 if(L>7){
560 free(R_lm1);
561 free(R_l);
562 }
563}

565void computeVelCoeffsMtx
566(
567 int sectorOrder,
568 float_complex* A_xyz
569)
570{
571 int i, j, Nxyz, Ns, nC_xyz, nC_s;
572 float x1, x3, z2, y1, y3;
573 float* G_mtx;
574
575 Ns = sectorOrder;
576 Nxyz = Ns+1;
577 nC_xyz = (Nxyz+1)*(Nxyz+1);
578 nC_s = (Ns+1)*(Ns+1);
579 x1 = sqrtf(2.0f*SAF_PI/3.0f);
580 x3 = -x1;
581 y1 = y3 = sqrtf(2.0f*SAF_PI/3.0f);
582 z2 = sqrtf(4.0f*SAF_PI/3.0f);
583 G_mtx = malloc1d(nC_s*4*nC_xyz*sizeof(float));
584 gaunt_mtx(Ns, 1, Nxyz, G_mtx);

586 for (i=0; i<nC_xyz; i++){
587 for (j=0; j<nC_s; j++){

589 A_xyz[i*nC_s*3 + j*3 + 1] = cmplxf(0.0f, y1*G_mtx[j*4*nC_xyz + 1*nC_xyz + i] + y3*G_mtx[j*4*nC_xyz + 3*nC_xyz + i]);
590 A_xyz[i*nC_s*3 + j*3 + 2] = cmplxf(z2*G_mtx[j*4*nC_xyz + 2*nC_xyz + i], 0.0f);
591 }
592 }
593
594 free(G_mtx);
595}

597float computeSectorCoeffsEP
598(
599 int orderSec,
600 float_complex* A_xyz, /* FLAT: (sectorOrder+2)^2 x (sectorOrder+1)^2 x 3 */
601 SECTOR_PATTERNS pattern,
602 float* sec_dirs_deg,
603 int nSecDirs,
604 float* sectorCoeffs /* FLAT: (nSecDirs*4) x (orderSec+2)^2 */
605)
606{
607 int i, j, ns, orderVel, nSH;
608 float normSec, azi_sec, elev_sec, Q;
609 float* b_n, *c_nm, *xyz_nm;
610
611 if(orderSec==0){
612 memcpy(sectorCoeffs, wxyzCoeffs, 16*sizeof(float)); /* ACN/N3D to WXYZ */
613 normSec = 1.0f;
614 }
615 else{
616 orderVel = orderSec+1;
617 nSH = (orderSec+2)*(orderSec+2);
618 b_n = malloc1d((orderSec+1)*sizeof(float));
619 c_nm = calloc1d((orderVel+1)*(orderVel+1), sizeof(float)); /* pad with zeros */
620 xyz_nm = malloc1d((orderVel+1)*(orderVel+1)*3*sizeof(float));
621 switch(pattern){
622 case SECTOR_PATTERN_PWD:
623 beamWeightsHypercardioid2Spherical(orderSec, b_n);
624 Q = (float)((orderSec+1) * (orderSec+1));
625 break;
626 case SECTOR_PATTERN_MAXRE:
627 beamWeightsMaxEV(orderSec, b_n);
628 cblas_sgemm(CblasRowMajor, CblasTrans, CblasNoTrans, 1, 1, orderSec+1, 1.0f,
629 b_n, 1,
630 b_n, 1, 0.0f,
631 &Q, 1);
632 Q = 4.0f*SAF_PI/(Q);
633 break;
634 case SECTOR_PATTERN_CARDIOID:
635 beamWeightsCardioid2Spherical(orderSec, b_n);
636 Q = 2.0f*(float)orderSec + 1.0f;
637 break;
638 }
639 normSec = Q/(float)nSecDirs; /* directivity factor / number of sectors */
640
641 for(ns=0; ns<nSecDirs; ns++){
642 /* rotate the pattern by rotating the coefficients */
643 azi_sec = sec_dirs_deg[ns*2] * SAF_PI/180.0f;
644 elev_sec = sec_dirs_deg[ns*2+1] * SAF_PI/180.0f; /* from elevation to inclination */
645 rotateAxisCoeffsReal(orderSec, b_n, SAF_PI/2.0f-elev_sec, azi_sec, c_nm);
646 beamWeightsVelocityPatternsReal(orderSec, b_n, azi_sec, elev_sec, A_xyz, xyz_nm);
647
648 /* store coefficients */
649 for(j=0; j<nSH; j++){
650 sectorCoeffs[ns*4*nSH + 0*nSH +j] = sqrtf(normSec) * c_nm[j];
651 for(i=0; i<3; i++)
652 sectorCoeffs[ns*4*nSH + (i+1)*nSH +j] = sqrtf(normSec) * xyz_nm[j*3+i];
653 }
654 }
655
656 free(b_n);
657 free(c_nm);
658 free(xyz_nm);
659 }
660 return normSec;
661}

717(
718 int N,
719 float* b_n
720)
721{
722 int n;
723
724 /* The coefficients can be derived by the binomial expansion of the cardioid function */
725 for(n=0; n<N+1; n++) {
726 b_n[n] = sqrtf(4.0f*SAF_PI*(2.0f*(float)n+1.0f)) *
727 (float)factorial(N)* (float)factorial(N+1)/

729 ((float)N+1.0f);
730 }

884void checkCondNumberSHTReal
885(
886 int order,
887 float* dirs_rad,
888 int nDirs,
889 float* w,
890 float* cond_N
891)
892{
893 int n, i, j, nSH, nSH_n, ind;
894 float minVal, maxVal;
895 float *YY_n, *W, *W_Yn, *s;
896 float** Y_N, **Y_n;
897
898 /* get SH */
899 nSH = ORDER2NSH(order);
900 Y_N = (float**)malloc2d(nSH, nDirs, sizeof(float));
901 Y_n = (float**)malloc2d(nDirs, nSH, sizeof(float));
902 YY_n = malloc1d(nSH*nSH*sizeof(float));
903 getSHreal(order, dirs_rad, nDirs, FLATTEN2D(Y_N));
904
905 /* diagonalise the integration weights, if available */
906 if(w!=NULL){
907 W = calloc1d(nDirs*nDirs, sizeof(float));
908 W_Yn = malloc1d(nDirs*nSH*sizeof(float));
909 for(i=0; i<nDirs; i++)
910 W[i*nDirs+i] = w[i];
911 }
912 else{
913 W = NULL;
914 W_Yn = NULL;
915 }
916
917 /* compute the condition number for each order up to N */
918 s = malloc1d(nSH*sizeof(float));
919 for(n=0; n<=order; n++){
920 nSH_n = (n+1)*(n+1);
921 for(i=0; i<nDirs; i++)
922 for(j=0; j<nSH_n; j++)
923 Y_n[i][j] = Y_N[j][i]; /* truncate to current order and transpose */
924 if(w==NULL){
925 cblas_sgemm(CblasRowMajor, CblasTrans, CblasNoTrans, nSH_n, nSH_n, nDirs, 1.0f,
926 FLATTEN2D(Y_n), nSH,
927 FLATTEN2D(Y_n), nSH, 0.0f,
928 YY_n, nSH_n);
929 }
930 else{
931 cblas_sgemm(CblasRowMajor, CblasNoTrans, CblasNoTrans, nDirs, nSH_n, nDirs, 1.0f,
932 W, nDirs,
933 FLATTEN2D(Y_n), nSH, 0.0f,
934 W_Yn, nSH_n);
935 cblas_sgemm(CblasRowMajor, CblasTrans, CblasNoTrans, nSH_n, nSH_n, nDirs, 1.0f,
936 FLATTEN2D(Y_n), nSH,
937 W_Yn, nSH_n, 0.0f,
938 YY_n, nSH_n);
939 }
940
941 /* condition number = max(singularValues)/min(singularValues) */
942 utility_ssvd(NULL, YY_n, nSH_n, nSH_n, NULL, NULL, NULL, s);
943 utility_simaxv(s, nSH_n, &ind);
944 maxVal = s[ind];
945 utility_siminv(s, nSH_n, &ind);
946 minVal = s[ind];
947 cond_N[n] = maxVal/(minVal+2.23e-7f);
948 }
949
950 free(Y_N);
951 free(Y_n);
952 free(YY_n);
953 free(W);
954 free(W_Yn);
955 free(s);
956}

959int calculateGridWeights
960(
961 float* dirs_rad,
962 int nDirs,
963 int order,
964 float* w
965)
966{
967 int i, j, nSH;
968 float sumW;
969 float** Y_N, **Y_N_T, **Y_leftinv;
970
971 if(order<0){
972 int nSH, ind;
973 float minVal, maxVal, cond_N;
974
975 float *YY_N, *s;
976
977 /* get SH */
978 s = NULL;
979 Y_N = NULL;
980 YY_N = NULL;
981
982 for(int n=1; n<100; n++){
983 /* compute the condition number for order N */
984 nSH = ORDER2NSH(n);
985 Y_N = (float**)realloc2d((void**)Y_N, nSH, nDirs, sizeof(float));
986 YY_N = (float*)realloc1d(YY_N, nSH*nSH*sizeof(float));
987 s = (float*) realloc1d(s, nSH*sizeof(float));
988 getSHreal(n, dirs_rad, nDirs, FLATTEN2D(Y_N));
989
990 cblas_sgemm(CblasRowMajor, CblasNoTrans, CblasTrans, nSH, nSH, nDirs, 1.0f,
991 FLATTEN2D(Y_N), nDirs,
992 FLATTEN2D(Y_N), nDirs, 0.0f,
993 YY_N, nSH);
994
995 /* condition number = max(singularValues)/min(singularValues) */
996 utility_ssvd(NULL, YY_N, nSH, nSH, NULL, NULL, NULL, s);
997 utility_simaxv(s, nSH, &ind);
998 maxVal = s[ind];
999 utility_siminv(s, nSH, &ind);
1000 minVal = s[ind];
1001 cond_N = maxVal/(minVal+2.23e-7f);
1002
1003 if(cond_N > 2 * (n + 1)){ // experimental condition
1004 order = n-1;
1005 break;
1006 }
1007 }
1008 }
1009 if(order<1) // could not find order
1010 order=0;
1011
1012 nSH = ORDER2NSH(order);
1013 Y_N = (float**)malloc2d(nSH, nDirs, sizeof(float));
1014 Y_N_T = (float**)malloc2d(nDirs, nSH, sizeof(float));
1015 Y_leftinv = (float**)malloc2d(nSH, nDirs, sizeof(float));
1016
1017 getSHreal(order, dirs_rad, nDirs, FLATTEN2D(Y_N));
1018
1019 for(i=0; i<nDirs; i++)
1020 for(j=0; j<nSH; j++)
1021 Y_N_T[i][j] = Y_N[j][i]; /* truncate to current order and transpose */
1022
1023 utility_spinv(NULL, FLATTEN2D(Y_N_T), nDirs, nSH, FLATTEN2D(Y_leftinv));
1024
1025 sumW=0.f;
1026 for(int idx=0; idx<nDirs; idx++){
1027 w[idx] = sqrtf(FOURPI)*Y_leftinv[0][idx];
1028 sumW += w[idx];
1029 }
1030
1031 if(fabs(sumW - FOURPI) > 0.001){
1032 order=0;
1033 saf_print_warning("Grid weights no bueno!");
1034 }
1035 return order;
1036}

1109void sphPWD_compute
1110(
1111 void* const hPWD,
1112 float_complex* Cx,
1113 int nSrcs,
1114 float* P_map,
1115 int* peak_inds
1116)
1117{
1118 sphPWD_data *h = (sphPWD_data*)(hPWD);
1119 int i, k, peak_idx;
1120 float kappa, scale;
1121 float VM_mean[3];
1122 float_complex A_Cx_A;
1123 const float_complex calpha = cmplxf(1.0f, 0.0f); const float_complex cbeta = cmplxf(0.0f, 0.0f);
1124
1125 /* derive the power-map value for each grid direction */
1126 for (i = 0; i < (h->nDirs); i++){
1127 cblas_cgemv(CblasRowMajor, CblasNoTrans, h->nSH, h->nSH, &calpha,
1128 Cx, h->nSH,
1129 &(h->grid_svecs[i*(h->nSH)]), 1, &cbeta,
1130 h->A_Cx, 1);
1131 cblas_cdotu_sub(h->nSH, h->A_Cx, 1, &(h->grid_svecs[i*(h->nSH)]), 1, &A_Cx_A);
1132 h->pSpec[i] = crealf(A_Cx_A);
1133 }
1134
1135 /* Output power-map */
1136 if(P_map!=NULL)
1137 cblas_scopy(h->nDirs, h->pSpec, 1, P_map, 1);
1138
1139 /* Peak-finding */
1140 if(peak_inds!=NULL){
1141 kappa = 50.0f;
1142 scale = kappa/(2.0f*SAF_PI*expf(kappa)-expf(-kappa));
1143 cblas_scopy(h->nDirs, h->pSpec, 1, h->P_minus_peak, 1);
1144
1145 /* Loop over the number of sources */
1146 for(k=0; k<nSrcs; k++){
1147 utility_simaxv(h->P_minus_peak, h->nDirs, &peak_idx);
1148 peak_inds[k] = peak_idx;
1149 if(k==nSrcs-1)
1150 break;
1151 VM_mean[0] = h->grid_dirs_xyz[peak_idx*3];
1152 VM_mean[1] = h->grid_dirs_xyz[peak_idx*3+1];
1153 VM_mean[2] = h->grid_dirs_xyz[peak_idx*3+2];
1154
1155 /* Apply mask for next iteration */

1243void sphMUSIC_compute
1244(
1245 void* const hMUSIC,
1246 float_complex *Vn, /* nSH x (nSH - nSrcs) */
1247 int nSrcs,
1248 float* P_music,
1249 int* peak_inds
1250)
1251{
1252 sphMUSIC_data *h = (sphMUSIC_data*)(hMUSIC);
1253 int i, k, VnD2, peak_idx;
1254 float kappa, scale;
1255 float VM_mean[3];
1256 const float_complex calpha = cmplxf(1.0f, 0.0f); const float_complex cbeta = cmplxf(0.0f, 0.0f);
1257
1258 VnD2 = h->nSH - nSrcs; /* noise subspace second dimension length */
1259
1260 /* derive the pseudo-spectrum value for each grid direction */
1261 cblas_cgemm(CblasRowMajor, CblasTrans, CblasNoTrans, h->nDirs, VnD2, h->nSH, &calpha,
1262 h->grid_svecs, h->nDirs,
1263 Vn, VnD2, &cbeta,
1264 h->VnA, VnD2);
1265 utility_cvabs(h->VnA, (h->nDirs)*VnD2, h->abs_VnA);
1266 for (i = 0; i < (h->nDirs); i++)
1267 h->pSpecInv[i] = cblas_sdot(VnD2, &(h->abs_VnA[i*VnD2]), 1, &(h->abs_VnA[i*VnD2]), 1);
1268 //h->pSpec[i] = 1.0f / (h->pSpecInv[i] + 2.23e-10f);
1269 utility_svrecip(h->pSpecInv, h->nDirs, h->pSpec);
1270
1271 /* Output pseudo-spectrum */
1272 if(P_music!=NULL)
1273 cblas_scopy(h->nDirs, h->pSpec, 1, P_music, 1);
1274
1275 /* Peak-finding */
1276 if(peak_inds!=NULL){
1277 kappa = 50.0f;
1278 scale = kappa/(2.0f*SAF_PI*expf(kappa)-expf(-kappa));
1279 cblas_scopy(h->nDirs, h->pSpec, 1, h->P_minus_peak, 1);
1280
1281 /* Loop over the number of sources */
1282 for(k=0; k<nSrcs; k++){
1283 utility_simaxv(h->P_minus_peak, h->nDirs, &peak_idx);
1284 peak_inds[k] = peak_idx;
1285 if(k==nSrcs-1)
1286 break;
1287 VM_mean[0] = h->grid_dirs_xyz[peak_idx*3];
1288 VM_mean[1] = h->grid_dirs_xyz[peak_idx*3+1];
1289 VM_mean[2] = h->grid_dirs_xyz[peak_idx*3+2];
1290
1291 /* Apply mask for next iteration */
1292 cblas_sgemm(CblasRowMajor, CblasNoTrans, CblasTrans, h->nDirs, 1, 3, 1.0f,
1293 h->grid_dirs_xyz, 3,
1294 (const float*)VM_mean, 3, 0.0f,
1295 h->VM_mask, 1);
1296 cblas_sscal(h->nDirs, kappa, h->VM_mask, 1);
1297 for(i=0; i<h->nDirs; i++)
1298 h->VM_mask[i] = expf(h->VM_mask[i]); /* VM distribution */
1299 cblas_sscal(h->nDirs, scale, h->VM_mask, 1);
1300 for(i=0; i<h->nDirs; i++)
1301 h->VM_mask[i] = 1.0f/(0.00001f+(h->VM_mask[i])); /* inverse VM distribution */
1302 utility_svvmul(h->P_minus_peak, h->VM_mask, h->nDirs, h->P_tmp);
1303 cblas_scopy(h->nDirs, h->P_tmp, 1, h->P_minus_peak, 1);
1304 }
1305 }

1586void generateMVDRmap
1587(
1588 int order,
1589 float_complex* Cx,
1590 float_complex* Y_grid,
1591 int nGrid_dirs,
1592 float regPar,
1593 float* pmap,
1594 float_complex* w_MVDR_out
1595)
1596{
1597 int i, j, nSH;
1598 float Cx_trace;
1599 float_complex *Cx_d, *invCx_Ygrid, *w_MVDR, *invCx_Ygrid_s, *Y_grid_s;
1600 float_complex denum;
1601
1602 nSH = ORDER2NSH(order);
1603 w_MVDR = malloc1d(nSH * nGrid_dirs*sizeof(float_complex));
1604 Cx_d = malloc1d(nSH*nSH*sizeof(float_complex));
1605 invCx_Ygrid = malloc1d(nSH*nGrid_dirs*sizeof(float_complex));
1606 invCx_Ygrid_s = malloc1d(nSH*sizeof(float_complex));
1607 Y_grid_s = malloc1d(nSH*sizeof(float_complex));
1608
1609 /* apply diagonal loading */
1610 Cx_trace = 0.0f;
1611 for(i=0; i<nSH; i++)
1612 Cx_trace += crealf(Cx[i*nSH+i]);
1613 Cx_trace /= (float)nSH;
1614 memcpy(Cx_d, Cx, nSH*nSH*sizeof(float_complex));
1615 for(i=0; i<nSH; i++)
1616 Cx_d[i*nSH+i] = craddf(Cx_d[i*nSH+i], regPar*Cx_trace);
1617
1618 /* solve the numerator part of the MVDR weights for all grid directions: Cx^-1 * Y */
1619 utility_cslslv(NULL, Cx_d, nSH, Y_grid, nGrid_dirs, invCx_Ygrid);
1620 for(i=0; i<nGrid_dirs; i++){
1621 /* solve the denumerator part of the MVDR weights for each grid direction: Y^T * Cx^-1 * Y */
1622 for(j=0; j<nSH; j++){
1623 invCx_Ygrid_s[j] = conjf(invCx_Ygrid[j*nGrid_dirs+i]);
1624 Y_grid_s[j] = Y_grid[j*nGrid_dirs+i];
1625 }
1626 /* faster to perform the dot-product for each vector seperately */
1627 utility_cvvdot(Y_grid_s, invCx_Ygrid_s, nSH, NO_CONJ, &denum);
1628
1629 /* calculate the MVDR weights per grid direction: (Cx^-1 * Y) * (Y^T * Cx^-1 * Y)^-1 */
1630 for(j=0; j<nSH; j++)
1631 w_MVDR[j*nGrid_dirs +i] = ccdivf(invCx_Ygrid[j*nGrid_dirs +i], denum);
1632 }
1633
1634 /* generate MVDR powermap, by using the generatePWDmap function with the MVDR weights instead */
1635 generatePWDmap(order, Cx, w_MVDR, nGrid_dirs, pmap);
1636
1637 /* optional output of the beamforming weights */
1638 if (w_MVDR_out!=NULL)
1639 memcpy(w_MVDR_out, w_MVDR, nSH * nGrid_dirs*sizeof(float_complex));
1640
1641 free(w_MVDR);
1642 free(Cx_d);
1643 free(invCx_Ygrid);
1644 free(invCx_Ygrid_s);
1645 free(Y_grid_s);
1646}

1650 * weighted-orthogonal beamformers in the spherical harmonic domain. IEEE/ACM Transactions on Audio,
1651 * Speech and Language Processing (TASLP), 24(9), 1507-1519. */
1652void generateCroPaCLCMVmap
1653(
1654 int order,
1655 float_complex* Cx,
1656 float_complex* Y_grid,
1657 int nGrid_dirs,
1658 float regPar,
1659 float lambda,

1661)

1667 float_complex b[2];
1668 const float_complex calpha = cmplxf(1.0f, 0.0f), cbeta = cmplxf(0.0f, 0.0f);
1669 float_complex A_invCxd_A[2][2];
1670 float_complex Y_wo_xspec;
1671
1672 b[0] = cmplxf(1.0f, 0.0f);
1673 b[1] = cmplxf(0.0f, 0.0f);
1674 nSH = ORDER2NSH(order);
1675 Cx_Y = malloc1d(nSH * nGrid_dirs * sizeof(float_complex));
1676 Cx_d = malloc1d(nSH*nSH*sizeof(float_complex));
1677 A = malloc1d(nSH*2*sizeof(float_complex));
1678 invCxd_A = malloc1d(nSH*2*sizeof(float_complex));
1679 invCxd_A_tmp = malloc1d(nSH*2*sizeof(float_complex));
1680 w_LCMV_s = malloc1d(2*nGrid_dirs*sizeof(float_complex));
1681 w_CroPaC = malloc1d(nSH*nGrid_dirs*sizeof(float_complex));
1682 wo = malloc1d(nSH*sizeof(float_complex));
1683 mvdr_map = malloc1d(nGrid_dirs*sizeof(float));
1684 Cx_Y_s = malloc1d(nSH*sizeof(float_complex));
1685
1686 /* generate MVDR map and weights to use as a basis */
1687 generateMVDRmap(order, Cx, Y_grid, nGrid_dirs, regPar, mvdr_map, w_CroPaC);
1688
1689 /* first half of the cross-spectrum */
1690 cblas_cgemm(CblasRowMajor, CblasNoTrans, CblasNoTrans, nSH, nGrid_dirs, nSH, &calpha,
1691 Cx, nSH,
1692 Y_grid, nGrid_dirs, &cbeta,
1693 Cx_Y, nGrid_dirs);
1694
1695 /* apply diagonal loading to cov matrix */
1696 Cx_trace = 0.0f;
1697 for(i=0; i<nSH; i++)
1698 Cx_trace += crealf(Cx[i*nSH+i]);
1699 Cx_trace /= (float)nSH;
1700 memcpy(Cx_d, Cx, nSH*nSH*sizeof(float_complex));
1701 for(i=0; i<nSH; i++)
1702 Cx_d[i*nSH+i] = craddf(Cx_d[i*nSH+i], regPar*Cx_trace);
1703
1704 /* calculate CroPaC beamforming weights for each grid direction */
1705 for(i=0; i<nGrid_dirs; i++){
1706 for(j=0; j<nSH; j++){
1707 A[j*2] = Y_grid[j*nGrid_dirs+i];
1708 A[j*2+1] = ccmulf(A[j*2], Cx[j*nSH+j]);
1709 }
1710
1711 /* solve for minimisation problem for LCMV weights: (Cx^-1 * A) * (A^H * Cx^-1 * A)^-1 * b */
1712 utility_cslslv(NULL, Cx_d, nSH, A, 2, invCxd_A);
1713 for(j=0; j<nSH*2; j++)
1714 invCxd_A_tmp[j] = conjf(invCxd_A[j]);
1715 cblas_cgemm(CblasRowMajor, CblasConjTrans, CblasNoTrans, 2, 2, nSH, &calpha,
1716 A, 2,
1717 invCxd_A_tmp, 2, &cbeta,
1718 A_invCxd_A, 2);
1719 for(j=0; j<nSH; j++)
1720 for(k=0; k<2; k++)
1721 invCxd_A_tmp[k*nSH+j] = invCxd_A[j*2+k];
1722 utility_cglslv(NULL, (float_complex*)A_invCxd_A, 2, invCxd_A_tmp, nSH, w_LCMV_s);
1723 cblas_cgemm(CblasRowMajor, CblasTrans, CblasNoTrans, nSH, 1, 2, &calpha,
1724 w_LCMV_s, nSH,
1725 b, 1, &cbeta,
1726 wo, 1);
1727
1728 /* calculate the cross-spectrum between static beam Y, and adaptive beam wo (LCMV) */
1729 for(j=0; j<nSH; j++)
1730 Cx_Y_s[j] = Cx_Y[j*nGrid_dirs+i];
1731 utility_cvvdot(wo, Cx_Y_s, nSH, NO_CONJ, &Y_wo_xspec);
1732
1733 /* derive CroPaC weights */
1734 S = SAF_MIN(cabsf(Y_wo_xspec), mvdr_map[i]); /* ensures distortionless response */
1735 G = sqrtf(S/(mvdr_map[i]+2.23e-10f));
1736 G = SAF_MAX(lambda, G); /* optional spectral floor parameter, to control harshness of attenuation (good for demos) */
1737 for(j=0; j<nSH; j++)
1738 w_CroPaC[j*nGrid_dirs + i] = crmulf(w_CroPaC[j*nGrid_dirs + i], G);
1739 }
1740
1741 /* generate CroPaC powermap, by using the generatePWDmap function with the CroPaC weights instead */
1742 generatePWDmap(order, Cx, w_CroPaC, nGrid_dirs, pmap);
1743
1744 free(mvdr_map);
1745 free(Cx_d);
1746 free(A);
1747 free(invCxd_A);
1748 free(invCxd_A_tmp);
1749 free(w_LCMV_s);
1750 free(w_CroPaC);
1751 free(wo);
1752 free(Cx_Y);
1753 free(Cx_Y_s);
1754}
