50void ambi_bin_create
51(
52 void ** const phAmbi
53)
54{
55 ambi_bin_data* pData = (ambi_bin_data*)malloc1d(sizeof(ambi_bin_data));
56 *phAmbi = (void*)pData;
57 int band;
58
59 /* default user parameters */
60 for (band = 0; band<HYBRID_BANDS; band++)
61 pData->EQ[band] = 1.0f;
62 pData->useDefaultHRIRsFLAG = 1; /* pars->sofa_filepath must be valid to set this to 0 */
63 pData->preProc = HRIR_PREPROC_EQ;
64 pData->chOrdering = CH_ACN;
65 pData->norm = NORM_SN3D;
66 pData->enableMaxRE = 1;
67 pData->enableDiffuseMatching = 0;
68 pData->enableTruncationEQ = 1;
69 pData->enableRotation = 0;
70 pData->yaw = 0.0f;
71 pData->pitch = 0.0f;
72 pData->roll = 0.0f;
73 pData->bFlipYaw = 0;
74 pData->bFlipPitch = 0;
75 pData->bFlipRoll = 0;
76 pData->useRollPitchYawFlag = 0;
77 pData->method = DECODING_METHOD_MAGLS;
78 pData->order = pData->new_order = 1;
79 pData->nSH = (pData->order+1)*(pData->order+1);
80
81 /* afSTFT and audio buffers */
82 pData->hSTFT = NULL;
83 pData->SHFrameTD = (float**)malloc2d(MAX_NUM_SH_SIGNALS, AMBI_BIN_FRAME_SIZE, sizeof(float));
84 pData->binFrameTD = (float**)malloc2d(NUM_EARS, AMBI_BIN_FRAME_SIZE, sizeof(float));
85 pData->SHframeTF = (float_complex***)malloc3d(HYBRID_BANDS, MAX_NUM_SH_SIGNALS, TIME_SLOTS, sizeof(float_complex));
86 pData->binframeTF = (float_complex***)malloc3d(HYBRID_BANDS, NUM_EARS, TIME_SLOTS, sizeof(float_complex));
87
88 /* codec data */
89 pData->progressBar0_1 = 0.0f;
90 pData->progressBarText = malloc1d(PROGRESSBARTEXT_CHAR_LENGTH*sizeof(char));
91 strcpy(pData->progressBarText,"");
92 pData->pars = (ambi_bin_codecPars*)malloc1d(sizeof(ambi_bin_codecPars));
93 ambi_bin_codecPars* pars = pData->pars;
94 pars->sofa_filepath = NULL;
95 pars->hrirs = NULL;
96 pars->hrir_dirs_deg = NULL;
97 pars->itds_s = NULL;
98 pars->hrtf_fb = NULL;
99 pars->weights = NULL;
100
101 /* flags */
102 pData->procStatus = PROC_STATUS_NOT_ONGOING;
103 pData->codecStatus = CODEC_STATUS_NOT_INITIALISED;
104 pData->recalc_M_rotFLAG = 1;
105 pData->reinit_hrtfsFLAG = 1;
106}

108void ambi_bin_destroy
109(
110 void ** const phAmbi
111)
112{
113 ambi_bin_data *pData = (ambi_bin_data*)(*phAmbi);
114 ambi_bin_codecPars *pars;
115
116 if (pData != NULL) {
117 /* not safe to free memory during intialisation/processing loop */
118 while (pData->codecStatus == CODEC_STATUS_INITIALISING ||
119 pData->procStatus == PROC_STATUS_ONGOING){
120 SAF_SLEEP(10);
121 }
122
123 /* free afSTFT and buffers */
124 afSTFT_destroy(&(pData->hSTFT));
125 free(pData->SHFrameTD);
126 free(pData->binFrameTD);
127 free(pData->SHframeTF);
128 free(pData->binframeTF);
129
130 pars = pData->pars;
131 free(pars->weights);
132 free(pars->hrtf_fb);
133 free(pars->itds_s);
134 free(pars->hrirs);
135 free(pars->hrir_dirs_deg);
136 free(pars);
137 free(pData->progressBarText);
138
139 free(pData);
140 pData = NULL;
141 *phAmbi = NULL;
142 }
143}

165void ambi_bin_initCodec
166(
167 void* const hAmbi
168)
169{
170 ambi_bin_data *pData = (ambi_bin_data*)(hAmbi);
171 ambi_bin_codecPars* pars = pData->pars;
172 int i, j, nSH, order, band;
173#ifdef SAF_ENABLE_SOFA_READER_MODULE
174 SAF_SOFA_ERROR_CODES error;
175 saf_sofa_container sofa;
176#endif
177
178 if (pData->codecStatus != CODEC_STATUS_NOT_INITIALISED)
179 return; /* re-init not required, or already happening */
180 while (pData->procStatus == PROC_STATUS_ONGOING){
181 /* re-init required, but we need to wait for the current processing loop to end */
182 pData->codecStatus = CODEC_STATUS_INITIALISING; /* indicate that we want to init */
183 SAF_SLEEP(10);
184 }
185
186 /* for progress bar */
187 pData->codecStatus = CODEC_STATUS_INITIALISING;
188 strcpy(pData->progressBarText,"Preparing HRIRs");
189 pData->progressBar0_1 = 0.0f;
190
191 /* (Re)Initialise afSTFT */
192 order = pData->new_order;
193 nSH = (order+1)*(order+1);
194 if(pData->hSTFT==NULL)
195 afSTFT_create(&(pData->hSTFT), nSH, NUM_EARS, HOP_SIZE, 0, 1, AFSTFT_BANDS_CH_TIME);
196 else if(pData->nSH != nSH) {/* Or change the number of channels */
197 afSTFT_channelChange(pData->hSTFT, nSH, NUM_EARS);
198 afSTFT_clearBuffers(pData->hSTFT);
199 }
200 pData->nSH = nSH;
201
202 if(pData->reinit_hrtfsFLAG){
203 /* load sofa file or default hrir data */
204 strcpy(pData->progressBarText,"Preparing HRIRs");
205 pData->progressBar0_1 = 0.15f;
206 /* load sofa file or load default hrir data */
207#ifdef SAF_ENABLE_SOFA_READER_MODULE
208 if(!pData->useDefaultHRIRsFLAG && pars->sofa_filepath!=NULL){
209 /* Load SOFA file */
210 error = saf_sofa_open(&sofa, pars->sofa_filepath);
211
212 /* Load defaults instead */
213 if(error!=SAF_SOFA_OK || sofa.nReceivers!=NUM_EARS){
214 pData->useDefaultHRIRsFLAG = 1;
215 saf_print_warning("Unable to load the specified SOFA file, or it contained something other than 2 channels. Using default HRIR data instead.");
216 }
217 else{
218 /* Copy SOFA data */
219 pars->hrir_fs = (int)sofa.DataSamplingRate;
220 pars->hrir_len = sofa.DataLengthIR;
221 pars->N_hrir_dirs = sofa.nSources;
222 pars->hrirs = realloc1d(pars->hrirs, pars->N_hrir_dirs*NUM_EARS*(pars->hrir_len)*sizeof(float));
223 memcpy(pars->hrirs, sofa.DataIR, pars->N_hrir_dirs*NUM_EARS*(pars->hrir_len)*sizeof(float));
224 pars->hrir_dirs_deg = realloc1d(pars->hrir_dirs_deg, pars->N_hrir_dirs*2*sizeof(float));
225 cblas_scopy(pars->N_hrir_dirs, sofa.SourcePosition, 3, pars->hrir_dirs_deg, 2); /* azi */
226 cblas_scopy(pars->N_hrir_dirs, &sofa.SourcePosition[1], 3, &pars->hrir_dirs_deg[1], 2); /* elev */
227 }
228
229 /* Clean-up */
230 saf_sofa_close(&sofa);
231 }
232#else
233 pData->useDefaultHRIRsFLAG = 1; /* Can only load the default HRIR data */
234#endif
235 if(pData->useDefaultHRIRsFLAG){
236 /* Copy default HRIR data */
237 pars->hrir_fs = __default_hrir_fs;
238 pars->hrir_len = __default_hrir_len;
239 pars->N_hrir_dirs = __default_N_hrir_dirs;
240 pars->hrirs = realloc1d(pars->hrirs, pars->N_hrir_dirs*NUM_EARS*(pars->hrir_len)*sizeof(float));
241 memcpy(pars->hrirs, (float*)__default_hrirs, pars->N_hrir_dirs*NUM_EARS*(pars->hrir_len)*sizeof(float));
242 pars->hrir_dirs_deg = realloc1d(pars->hrir_dirs_deg, pars->N_hrir_dirs*2*sizeof(float));
243 memcpy(pars->hrir_dirs_deg, (float*)__default_hrir_dirs_deg, pars->N_hrir_dirs*2*sizeof(float));
244 }
245
246 /* estimate the ITDs for each HRIR */
247 pData->progressBar0_1 = 0.3f;
248 pars->itds_s = realloc1d(pars->itds_s, pars->N_hrir_dirs*sizeof(float));
249 estimateITDs(pars->hrirs, pars->N_hrir_dirs, pars->hrir_len, pars->hrir_fs, pars->itds_s);
250
251 /* convert hrirs to filterbank coefficients */
252 pData->progressBar0_1 = 0.4f;
253 pars->hrtf_fb = realloc1d(pars->hrtf_fb, HYBRID_BANDS * NUM_EARS * (pars->N_hrir_dirs)*sizeof(float_complex));
254 HRIRs2HRTFs_afSTFT(pars->hrirs, pars->N_hrir_dirs, pars->hrir_len, HOP_SIZE, 0, 1, pars->hrtf_fb);
255 /* get integration weights */
256 pData->progressBar0_1 = 0.6f;
257 if(pars->N_hrir_dirs<=1000){
258 pars->weights = realloc1d(pars->weights, pars->N_hrir_dirs*sizeof(float));
259 getVoronoiWeights(pars->hrir_dirs_deg, pars->N_hrir_dirs, 0, pars->weights);
260 }
261 else{
262 free(pars->weights);
263 pars->weights = NULL;
264 }
265 /* HRIR pre-processing */
266 pData->progressBar0_1 = 0.75f;
267 diffuseFieldEqualiseHRTFs(pars->N_hrir_dirs, pars->itds_s, pData->freqVector, HYBRID_BANDS, pars->weights,
268 pData->preProc == HRIR_PREPROC_EQ || pData->preProc == HRIR_PREPROC_ALL ? 1 : 0, /* Apply Diffuse-field EQ? */
269 pData->preProc == HRIR_PREPROC_PHASE || pData->preProc == HRIR_PREPROC_ALL ? 1 : 0, /* Apply phase simplification EQ? */
270 pars->hrtf_fb);
271 pData->reinit_hrtfsFLAG = 0;
272 }
273
274 /* get new decoder */
275 strcpy(pData->progressBarText,"Computing Decoder");
276 pData->progressBar0_1 = 0.95f;
277 float_complex* decMtx;
278 decMtx = calloc1d(HYBRID_BANDS*NUM_EARS*nSH, sizeof(float_complex));
279 switch(pData->method){
280 default:
281 case DECODING_METHOD_LS:
282 getBinauralAmbiDecoderMtx(pars->hrtf_fb, pars->hrir_dirs_deg, pars->N_hrir_dirs, HYBRID_BANDS,
283 BINAURAL_DECODER_LS, order, pData->freqVector, pars->itds_s, pars->weights,
284 pData->enableDiffuseMatching, pData->enableMaxRE, decMtx);
285 break;
286 case DECODING_METHOD_LSDIFFEQ:
287 getBinauralAmbiDecoderMtx(pars->hrtf_fb, pars->hrir_dirs_deg, pars->N_hrir_dirs, HYBRID_BANDS,
288 BINAURAL_DECODER_LSDIFFEQ, order, pData->freqVector, pars->itds_s, pars->weights,
289 pData->enableDiffuseMatching, pData->enableMaxRE, decMtx);
290 break;
291 case DECODING_METHOD_SPR:
292 getBinauralAmbiDecoderMtx(pars->hrtf_fb, pars->hrir_dirs_deg, pars->N_hrir_dirs, HYBRID_BANDS,
293 BINAURAL_DECODER_SPR, order, pData->freqVector, pars->itds_s, pars->weights,
294 pData->enableDiffuseMatching, pData->enableMaxRE, decMtx);
295 break;
296 case DECODING_METHOD_TA:
297 getBinauralAmbiDecoderMtx(pars->hrtf_fb, pars->hrir_dirs_deg, pars->N_hrir_dirs, HYBRID_BANDS,
298 BINAURAL_DECODER_TA, order, pData->freqVector, pars->itds_s, pars->weights,
299 pData->enableDiffuseMatching, pData->enableMaxRE, decMtx);
300 break;
301 case DECODING_METHOD_MAGLS:
302 getBinauralAmbiDecoderMtx(pars->hrtf_fb, pars->hrir_dirs_deg, pars->N_hrir_dirs, HYBRID_BANDS,
303 BINAURAL_DECODER_MAGLS, order, pData->freqVector, pars->itds_s, pars->weights,
304 pData->enableDiffuseMatching, pData->enableMaxRE, decMtx);
305 break;
306 }
307
308 /* Apply Truncation EQ */
309 if(pData->enableTruncationEQ &&
310 pData->method==DECODING_METHOD_LS &&
311 pData->preProc!=HRIR_PREPROC_PHASE &&
312 pData->preProc!=HRIR_PREPROC_ALL)
313 {
314 double *kr;
315 float *w_n, *eqGain;
316 const int order_truncated = order;
317 const int order_target = 42; /* Equalizing diffuse field to 42nd order equivalent. */
318 const float softThreshold = 9.0; /* results in +9 dB max */
319 const double r = 0.085; /* spherical scatterer radius (approx. size of human head) */
320 const int numBands = HYBRID_BANDS;
321 const double c = 343.;
322
323 /* Prep */
324 kr = malloc1d(numBands * sizeof(double));
325 w_n = calloc1d((order_truncated+1), sizeof(float));
326 eqGain = calloc1d(numBands, sizeof(float));
327 for (int k=0; k<numBands; k++)
328 kr[k] = 2.0*SAF_PId / c * (double)pData->freqVector[k] * r;
329
330 if (pData->enableMaxRE) {
331 /* maxRE as order weighting */
332 float *maxRECoeffs = malloc1d((order_truncated+1) * sizeof(float));
333 beamWeightsMaxEV(order_truncated, maxRECoeffs);
334 for (int idx_n=0; idx_n<order_truncated+1; idx_n++) {
335 w_n[idx_n] = maxRECoeffs[idx_n];
336 w_n[idx_n] /= sqrtf((float)(2*idx_n+1) / (4.0f*SAF_PI));
337 }
338 float w_0 = w_n[0];
339 for (int idx_n=0; idx_n<order_truncated+1; idx_n++)
340 w_n[idx_n] /= w_0;
341 free(maxRECoeffs);
342 }
343 else {
344 /* just truncation, no tapering */
345 for (int idx_n=0; idx_n<order_truncated+1; idx_n++)
346 w_n[idx_n] = 1.0f;
347 }
348 truncationEQ(w_n, order_truncated, order_target, kr, numBands, softThreshold, eqGain);
349
350 /* apply to decoding matrix */
351 for (int idxBand=0; idxBand<numBands; idxBand++){
352 for (int idxSH=0; idxSH<pData->nSH; idxSH++){
353 decMtx[idxBand*NUM_EARS*nSH+0*nSH+idxSH] = crmulf(decMtx[idxBand*NUM_EARS*nSH+0*nSH+idxSH], eqGain[idxBand]); /* left ear */
354 decMtx[idxBand*NUM_EARS*nSH+1*nSH+idxSH] = crmulf(decMtx[idxBand*NUM_EARS*nSH+1*nSH+idxSH], eqGain[idxBand]); /* right ear */
355 }
356 }
357
358 /* clean-up */
359 free(kr);
360 free(w_n);
361 free(eqGain);
362 }
363
364 /* replace current decoder */
365 memset(pars->M_dec, 0, HYBRID_BANDS*NUM_EARS*MAX_NUM_SH_SIGNALS*sizeof(float_complex));
366 for(band=0; band<HYBRID_BANDS; band++)
367 for(i=0; i<NUM_EARS; i++)
368 for(j=0; j<nSH; j++)
369 pars->M_dec[band][i][j] = decMtx[band*NUM_EARS*nSH + i*nSH + j];
370 free(decMtx);
371
372 pData->order = order;
373
374 /* done! */
375 strcpy(pData->progressBarText,"Done!");
376 pData->progressBar0_1 = 1.0f;
377 pData->codecStatus = CODEC_STATUS_INITIALISED;
378}

380void ambi_bin_process
381(
382 void * const hAmbi,
383 const float *const * inputs,
384 float ** const outputs,
385 int nInputs,
386 int nOutputs,
387 int nSamples
388)
389{
390 ambi_bin_data *pData = (ambi_bin_data*)(hAmbi);
391 ambi_bin_codecPars* pars = pData->pars;
392 int ch, i, j, band;
393 const float_complex calpha = cmplxf(1.0f,0.0f), cbeta = cmplxf(0.0f, 0.0f);
394 float Rxyz[3][3];
395 float M_rot_tmp[MAX_NUM_SH_SIGNALS*MAX_NUM_SH_SIGNALS];
396
397 /* local copies of user parameters */
398 int order, nSH, enableRot;
399 NORM_TYPES norm;
400 CH_ORDER chOrdering;
401 norm = pData->norm;
402 chOrdering = pData->chOrdering;
403 order = pData->order;
404 nSH = (order+1)*(order+1);
405 enableRot = pData->enableRotation;
406
407 /* Process frame */
408 if (nSamples == AMBI_BIN_FRAME_SIZE && (pData->codecStatus == CODEC_STATUS_INITIALISED) ) {
409 pData->procStatus = PROC_STATUS_ONGOING;
410
411 /* Load time-domain data */
412 for(i=0; i < SAF_MIN(nSH, nInputs); i++)
413 utility_svvcopy(inputs[i], AMBI_BIN_FRAME_SIZE, pData->SHFrameTD[i]);
414 for(; i<nSH; i++)
415 memset(pData->SHFrameTD[i], 0, AMBI_BIN_FRAME_SIZE * sizeof(float)); /* fill remaining channels with zeros */
416
417 /* account for channel order convention */
418 switch(chOrdering){
419 case CH_ACN: /* already in ACN, do nothing */ break; /* Otherwise, convert to ACN... */
420 case CH_FUMA: convertHOAChannelConvention(FLATTEN2D(pData->SHFrameTD), order, AMBI_BIN_FRAME_SIZE, HOA_CH_ORDER_FUMA, HOA_CH_ORDER_ACN); break;
421 }
422
423 /* account for input normalisation scheme */
424 switch(norm){
425 case NORM_N3D: /* already in N3D, do nothing */ break; /* Otherwise, convert to N3D... */
426 case NORM_SN3D: convertHOANormConvention(FLATTEN2D(pData->SHFrameTD), order, AMBI_BIN_FRAME_SIZE, HOA_NORM_SN3D, HOA_NORM_N3D); break;
427 case NORM_FUMA: convertHOANormConvention(FLATTEN2D(pData->SHFrameTD), order, AMBI_BIN_FRAME_SIZE, HOA_NORM_FUMA, HOA_NORM_N3D); break;
428 }
429
430 /* Apply time-frequency transform (TFT) */
431 afSTFT_forward_knownDimensions(pData->hSTFT, pData->SHFrameTD, AMBI_BIN_FRAME_SIZE, MAX_NUM_SH_SIGNALS, TIME_SLOTS, pData->SHframeTF);
432
433 /* Main processing: */
434 if(order > 0 && enableRot) {
435 /* Apply rotation */
436 if(pData->recalc_M_rotFLAG){
437 /* Compute the new SH rotation matrix */
438 memset(pData->M_rot, 0, MAX_NUM_SH_SIGNALS*MAX_NUM_SH_SIGNALS*sizeof(float_complex));
439 yawPitchRoll2Rzyx(pData->yaw, pData->pitch, pData->roll, pData->useRollPitchYawFlag, Rxyz);
440 getSHrotMtxReal(Rxyz, (float*)M_rot_tmp, order);
441 for (i = 0; i < nSH; i++)
442 for (j = 0; j < nSH; j++)
443 pData->M_rot[i][j] = cmplxf(M_rot_tmp[i*nSH + j], 0.0f);
444
445 /* Bake the rotation into the decoding matrix */
446 for(band = 0; band < HYBRID_BANDS; band++) {
447 cblas_cgemm(CblasRowMajor, CblasNoTrans, CblasNoTrans, NUM_EARS, nSH, nSH, &calpha,
448 pars->M_dec[band], MAX_NUM_SH_SIGNALS,
449 pData->M_rot, MAX_NUM_SH_SIGNALS, &cbeta,
450 pars->M_dec_rot[band], MAX_NUM_SH_SIGNALS);
451 }
452 pData->recalc_M_rotFLAG = 0;
453 }
454 }
455
456 /* Apply the decoder to go from SH input to binaural output */
457 for(band = 0; band < HYBRID_BANDS; band++) {
458 cblas_cgemm(CblasRowMajor, CblasNoTrans, CblasNoTrans, NUM_EARS, TIME_SLOTS, nSH, &calpha,
459 enableRot ? pars->M_dec_rot[band] : pars->M_dec[band], MAX_NUM_SH_SIGNALS,
460 FLATTEN2D(pData->SHframeTF[band]), TIME_SLOTS, &cbeta,
461 FLATTEN2D(pData->binframeTF[band]), TIME_SLOTS);
462 }
463
464 /* inverse-TFT */
465 afSTFT_backward_knownDimensions(pData->hSTFT, pData->binframeTF, AMBI_BIN_FRAME_SIZE, NUM_EARS, TIME_SLOTS, pData->binFrameTD);
466
467 /* Copy to output */
468 for (ch = 0; ch < SAF_MIN(NUM_EARS, nOutputs); ch++)
469 utility_svvcopy(pData->binFrameTD[ch], AMBI_BIN_FRAME_SIZE, outputs[ch]);
470 for (; ch < nOutputs; ch++)
471 memset(outputs[ch], 0, AMBI_BIN_FRAME_SIZE*sizeof(float));
472 }
473 else
474 for (ch=0; ch < nOutputs; ch++)
475 memset(outputs[ch],0, AMBI_BIN_FRAME_SIZE*sizeof(float));
476
477 pData->procStatus = PROC_STATUS_NOT_ONGOING;
478}
