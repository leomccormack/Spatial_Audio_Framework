29void test__saf_example_ambi_bin(void){
30 int nSH, i, ch, framesize;
31 void* hAmbi;
32 float leftEarEnergy, rightEarEnergy, direction_deg[2];
33 float* inSig, *y;
34 float** shSig, **binSig, **shSig_frame, **binSig_frame;
35
36 /* Config */
37 const int order = 4;
38 const int fs = 48000;
39 const int signalLength = fs*2;
40
41 /* Create and initialise an instance of ambi_bin */
42 ambi_bin_create(&hAmbi);
43
44 /* Configure and initialise the ambi_bin codec */
45 ambi_bin_setNormType(hAmbi, NORM_N3D);
46 ambi_bin_setInputOrderPreset(hAmbi, (SH_ORDERS)order);
47 ambi_bin_initCodec(hAmbi); /* Can be called whenever (thread-safe) */
48 /* "initCodec" should be called after calling any of the "set" functions.
49 * It should be noted that intialisations are only conducted if they are
50 * needed, so calling this function periodically with a timer on a separate
51 * thread is perfectly safe and viable. Also, if the intialisations take
52 * longer than it takes to "process" the current block of samples, then the
53 * output is simply muted/zeroed during this time. */
54
55 ambi_bin_init(hAmbi, fs); /* Should be called before calling "process"
56 * Cannot be called while "process" is on-going */
57 ambi_bin_initCodec(hAmbi); /* Can be called whenever (thread-safe) */
58 ambi_bin_setEnableRotation(hAmbi, 1);
59 ambi_bin_setYaw(hAmbi, 180.0f); /* turn the listener around */
60
61 /* Define input mono signal */
62 nSH = ORDER2NSH(order);
63 inSig = malloc1d(signalLength*sizeof(float));
64 shSig = (float**)malloc2d(nSH,signalLength,sizeof(float));
65 rand_m1_1(inSig, signalLength); /* Mono white-noise signal */
66
67 /* Encode to get input spherical harmonic (Ambisonic) signal */
68 direction_deg[0] = -90.0f; /* encode hard-right */
69 direction_deg[1] = 0.0f;
70 y = malloc1d(nSH*sizeof(float));
71 getRSH(order, (float*)direction_deg, 1, y); /* SH plane-wave weights */
72 cblas_sgemm(CblasRowMajor, CblasNoTrans, CblasNoTrans, nSH, signalLength, 1, 1.0f,
73 y, 1,
74 inSig, signalLength, 0.0f,
75 FLATTEN2D(shSig), signalLength);
76
77 /* Decode to binaural */
78 framesize = ambi_bin_getFrameSize();
79 binSig = (float**)calloc2d(NUM_EARS,signalLength,sizeof(float));
80 shSig_frame = (float**)malloc1d(nSH*sizeof(float*));
81 binSig_frame = (float**)malloc1d(NUM_EARS*sizeof(float*));
82 for(i=0; i<(int)((float)signalLength/(float)framesize); i++){
83 for(ch=0; ch<nSH; ch++)
84 shSig_frame[ch] = &shSig[ch][i*framesize];
85 for(ch=0; ch<NUM_EARS; ch++)
86 binSig_frame[ch] = &binSig[ch][i*framesize];
87
88 ambi_bin_process(hAmbi, (const float* const*)shSig_frame, binSig_frame, nSH, NUM_EARS, framesize);
89 }
90
91 /* Assert that left ear energy is higher than the right ear */
92 leftEarEnergy = rightEarEnergy = 0.0f;
93 for(i=0; i<signalLength; i++){
94 leftEarEnergy += powf(fabsf(binSig[0][i]), 2.0f);
95 rightEarEnergy += powf(fabsf(binSig[1][i]), 2.0f);
96 }
97 TEST_ASSERT_TRUE(leftEarEnergy>=rightEarEnergy);
98
99 /* Clean-up */
100 ambi_bin_destroy(&hAmbi);
101 free(inSig);
102 free(shSig);
103 free(y);
104 free(binSig);
105 free(shSig_frame);
106 free(binSig_frame);
107}

131 ambi_dec_initCodec(hAmbi); /* Can be called whenever (thread-safe) */
132 /* "initCodec" should be called after calling any of the "set" functions.
133 * It should be noted that intialisations are only conducted if they are
134 * needed, so calling this function periodically with a timer on a separate
135 * thread is perfectly safe and viable. Also, if the intialisations take
136 * longer than it takes to "process" the current block of samples, then the
137 * output is simply muted/zeroed during this time. */
138

140 * Cannot be called while "process" is on-going */
141
142 /* Define input mono signal */
143 nSH = ORDER2NSH(order);
144 inSig = malloc1d(signalLength*sizeof(float));
145 shSig = (float**)malloc2d(nSH,signalLength,sizeof(float));
146 rand_m1_1(inSig, signalLength); /* Mono white-noise signal */
147

149 direction_deg[0] = 90.0f; /* encode to loudspeaker direction: index 8 */
150 direction_deg[1] = 0.0f;

152 getRSH(order, (float*)direction_deg, 1, y); /* SH plane-wave weights */
153 cblas_sgemm(CblasRowMajor, CblasNoTrans, CblasNoTrans, nSH, signalLength, 1, 1.0f,
154 y, 1,
155 inSig, signalLength, 0.0f,
156 FLATTEN2D(shSig), signalLength);
157
158 /* Decode to loudspeakers */
159 framesize = ambi_dec_getFrameSize();
160 lsSig = (float**)calloc2d(22,signalLength,sizeof(float));
161 shSig_frame = (float**)malloc1d(nSH*sizeof(float*));
162 lsSig_frame = (float**)malloc1d(22*sizeof(float*));
163 for(i=0; i<(int)((float)signalLength/(float)framesize); i++){
164 for(ch=0; ch<nSH; ch++)
165 shSig_frame[ch] = &shSig[ch][i*framesize];
166 for(ch=0; ch<22; ch++)
167 lsSig_frame[ch] = &lsSig[ch][i*framesize];
168
169 ambi_dec_process(hAmbi, (const float* const*)shSig_frame, lsSig_frame, nSH, 22, framesize);
170 }
171
172 /* Assert that channel 8 (corresponding to the loudspeaker where the plane-
173 * wave was encoded to) has the most energy */
174 memset(loudspeakerEnergy, 0, 22*sizeof(float));
175 for(i=0; i<signalLength; i++){
176 for(j=0; j<22; j++)
177 loudspeakerEnergy[j] += powf(fabsf(lsSig[j][i]), 2.0f);
178 }
179 utility_simaxv(loudspeakerEnergy, 22, &max_ind);
180 TEST_ASSERT_TRUE(max_ind==7);
181
182 /* Clean-up */
183 ambi_dec_destroy(&hAmbi);
184 free(inSig);
185 free(shSig);
186 free(y);
187 free(lsSig);
188 free(shSig_frame);
189 free(lsSig_frame);
190}

192void test__saf_example_ambi_enc(void){
193 int nSH, i, ch, framesize, j, delay;
194 void* hAmbi;
195 float direction_deg[2][2];
196 float** inSig, *y;
197 float** shSig, **shSig_ref, **inSig_frame, **shSig_frame;
198
199 /* Config */
200 const float acceptedTolerance = 0.000001f;
201 const int order = 4;
202 const int fs = 48000;
203 const int signalLength = fs*2;
204 direction_deg[0][0] = 90.0f; /* encode to loudspeaker direction: index 8 */
205 direction_deg[0][1] = 0.0f;
206 direction_deg[1][0] = 20.0f; /* encode to loudspeaker direction: index 8 */
207 direction_deg[1][1] = -45.0f;
208 delay = ambi_enc_getProcessingDelay();
209
210 /* Create and initialise an instance of ambi_enc */
211 ambi_enc_create(&hAmbi);
212 ambi_enc_init(hAmbi, fs); /* Cannot be called while "process" is on-going */
213
214 /* Configure ambi_enc */
215 ambi_enc_setOutputOrder(hAmbi, (SH_ORDERS)order);
216 ambi_enc_setNormType(hAmbi, NORM_N3D); /* (The default for all SH-related examples is SN3D) */
217 ambi_enc_setEnablePostScaling(hAmbi, 0); /* Disable scaling output by number of input channels */
218 ambi_enc_setNumSources(hAmbi, 2);
219 ambi_enc_setSourceAzi_deg(hAmbi, 0, direction_deg[0][0]);
220 ambi_enc_setSourceElev_deg(hAmbi, 0, direction_deg[0][1]);
221 ambi_enc_setSourceAzi_deg(hAmbi, 1, direction_deg[1][0]);
222 ambi_enc_setSourceElev_deg(hAmbi, 1, direction_deg[1][1]);
223
224 /* Define input mono signal */
225 nSH = ORDER2NSH(order);
226 inSig = (float**)malloc2d(2,signalLength,sizeof(float));
227 shSig_ref = (float**)malloc2d(nSH,signalLength,sizeof(float));
228 rand_m1_1(FLATTEN2D(inSig), 2*signalLength); /* Mono white-noise signal */
229
230 /* Encode reference */
231 y = malloc1d(nSH*2*sizeof(float));
232 getRSH(order, (float*)direction_deg, 2, y); /* SH plane-wave weights */
233 cblas_sgemm(CblasRowMajor, CblasNoTrans, CblasNoTrans, nSH, signalLength, 2, 1.0f,
234 y, 2,
235 FLATTEN2D(inSig), signalLength, 0.0f,
236 FLATTEN2D(shSig_ref), signalLength);
237
238 /* Encode via ambi_enc */
239 framesize = ambi_enc_getFrameSize();
240 shSig = (float**)calloc2d(nSH,signalLength,sizeof(float));
241 inSig_frame = (float**)malloc1d(2*sizeof(float*));
242 shSig_frame = (float**)malloc1d(nSH*sizeof(float*));
243 for(i=0; i<(int)((float)signalLength/(float)framesize); i++){
244 for(ch=0; ch<2; ch++)
245 inSig_frame[ch] = &inSig[ch][i*framesize];
246 for(ch=0; ch<nSH; ch++)
247 shSig_frame[ch] = &shSig[ch][i*framesize];
248
249 ambi_enc_process(hAmbi, (const float* const*)inSig_frame, shSig_frame, 2, nSH, framesize);
250 }
251
252 /* ambi_enc should be equivalent to the reference */
253 for(i=0; i<nSH; i++)
254 for(j=0; j<signalLength-delay; j++)
255 TEST_ASSERT_FLOAT_WITHIN(acceptedTolerance, shSig_ref[i][j], shSig[i][j+delay]);
256
257 /* Clean-up */
258 ambi_enc_destroy(&hAmbi);
259 free(inSig);
260 free(shSig);
261 free(shSig_ref);
262 free(y);
263}

265void test__saf_example_array2sh(void){
266 int nSH, i, j, framesize, ch;
267 void* hA2sh, *safFFT, *hMC;
268 float direction_deg[2], radius;
269 float* inSig, *f;
270 float** shSig, **inSig_32, **micSig, **h_array, **micSig_frame, **shSig_frame;
271 double* kr;
272 float_complex* tmp_H;
273 float_complex*** H_array;
274
275 /* Config */
276 const int order = 4;
277 const int fs = 48000;
278 const int signalLength = fs*2;
279 const int nFFT = 1024;
280 const int nBins = nFFT/2+1;
281
282 /* Create and initialise an instance of array2sh for the Eigenmike32 */
283 array2sh_create(&hA2sh);
284 array2sh_init(hA2sh, fs); /* Cannot be called while "process" is on-going */
285 array2sh_setPreset(hA2sh, MICROPHONE_ARRAY_PRESET_EIGENMIKE32);
286 array2sh_setNormType(hA2sh, NORM_N3D);
287
288 /* Define input mono signal */
289 nSH = ORDER2NSH(order);
290 inSig = malloc1d(signalLength*sizeof(float));
291 rand_m1_1(inSig, signalLength); /* Mono white-noise signal */
292
293 /* Simulate an Eigenmike in a free-field with a single plane-wave */
294 f = malloc1d(nBins*sizeof(float));
295 kr = malloc1d(nBins*sizeof(double));
296 getUniformFreqVector(nFFT, (float)fs, f);
297 f[0] = f[1]/4.0f; /* To avoid NaNs at DC */
298 radius = 0.042f;
299 for(i=0; i<nBins; i++)
300 kr[i] = 2.0*SAF_PId*(f[i])*(radius)/343.0f;
301 direction_deg[0] = 90.0f;
302 direction_deg[1] = 0.0f;
303 H_array = (float_complex***)malloc3d(nBins, 32, 1, sizeof(float_complex));
304 simulateSphArray(order, kr, kr, nBins, (float*)__Eigenmike32_coords_rad, 32,
305 (float*)direction_deg, 1, ARRAY_CONSTRUCTION_RIGID, 1.0f, FLATTEN3D(H_array));
306
307 /* Inverse FFT to get the time-domain filters */
308 tmp_H = malloc1d(nBins*sizeof(float_complex));
309 h_array = (float**)malloc2d(32, nFFT, sizeof(float));
310 saf_rfft_create(&safFFT, nFFT);
311 for(i=0; i<32; i++){
312 for(j=0; j<nBins; j++)
313 tmp_H[j] = H_array[j][i][0];
314 saf_rfft_backward(safFFT, tmp_H, h_array[i]);
315 }
316
317 /* Simulate the Eigenmike time-domain signals by convolving the mono signal
318 * with each sensor transfer function */
319 micSig = (float**)calloc2d(32, signalLength, sizeof(float));
320 inSig_32 = (float**)malloc2d(32, signalLength, sizeof(float));
321 for(i=0; i<32; i++) /* Replicate inSig for all 32 channels */
322 memcpy(inSig_32[i], inSig, signalLength* sizeof(float));
323 saf_multiConv_create(&hMC, 256, FLATTEN2D(h_array), nFFT, 32, 0);
324 for(i=0; i<(int)((float)signalLength/256.0f); i++)
325 saf_multiConv_apply(hMC, FLATTEN2D(inSig_32), FLATTEN2D(micSig));
326
327 /* Encode simulated Eigenmike signals into spherical harmonic signals */
328 framesize = array2sh_getFrameSize();
329 shSig = (float**)malloc2d(nSH,signalLength,sizeof(float));
330 micSig_frame = (float**)malloc1d(32*sizeof(float*));
331 shSig_frame = (float**)malloc1d(nSH*sizeof(float*));
332 for(i=0; i<(int)((float)signalLength/(float)framesize); i++){
333 for(ch=0; ch<32; ch++)
334 micSig_frame[ch] = &micSig[ch][i*framesize];
335 for(ch=0; ch<nSH; ch++)
336 shSig_frame[ch] = &shSig[ch][i*framesize];
337
338 array2sh_process(hA2sh, (const float* const*)micSig_frame, shSig_frame, 32, nSH, framesize);
339 }
340
341 /* Clean-up */
342 array2sh_destroy(&hA2sh);
343 saf_rfft_destroy(&safFFT);
344 saf_multiConv_destroy(&hMC);
345 free(inSig);
346 free(shSig);
347 free(inSig_32);
348 free(f);
349 free(kr);
350 free(H_array);
351 free(h_array);
352 free(tmp_H);
353 free(micSig_frame);
354 free(shSig_frame);
355}

357void test__saf_example_rotator(void){
358 int ch, nSH, i, j, delay, framesize;
359 void* hRot;
360 float direction_deg[2], ypr[3], Rzyx[3][3];
361 float** inSig, *y, **shSig_frame, **shSig_rot_frame;
362 float** shSig, **shSig_rot, **shSig_rot_ref, **Mrot;
363
364 /* Config */
365 const float acceptedTolerance = 0.000001f;
366 const int order = 4;
367 const int fs = 48000;
368 const int signalLength = fs*2;
369 direction_deg[0] = 90.0f; /* encode to loudspeaker direction: index 8 */
370 direction_deg[1] = 0.0f;
371 ypr[0] = -0.4f;
372 ypr[1] = -1.4f;
373 ypr[2] = 2.1f;
374 delay = rotator_getProcessingDelay();
375
376 /* Create and initialise an instance of rotator */
377 rotator_create(&hRot);
378 rotator_init(hRot, fs); /* Cannot be called while "process" is on-going */
379
380 /* Configure rotator codec */
381 rotator_setOrder(hRot, (SH_ORDERS)order);
382 rotator_setNormType(hRot, NORM_N3D);
383 rotator_setYaw(hRot, ypr[0]*180.0f/SAF_PI); /* rad->degrees */
384 rotator_setPitch(hRot, ypr[1]*180.0f/SAF_PI);
385 rotator_setRoll(hRot, ypr[2]*180.0f/SAF_PI);
386
387 /* Define input mono signal */
388 nSH = ORDER2NSH(order);
389 inSig = (float**)malloc2d(1,signalLength,sizeof(float));
390 shSig = (float**)malloc2d(nSH,signalLength,sizeof(float));

392
393 /* Encode */
394 y = malloc1d(nSH*sizeof(float));
395 getRSH(order, (float*)direction_deg, 1, y); /* SH plane-wave weights */
396 cblas_sgemm(CblasRowMajor, CblasNoTrans, CblasNoTrans, nSH, signalLength, 1, 1.0f,
397 y, 1,
398 FLATTEN2D(inSig), signalLength, 0.0f,
399 FLATTEN2D(shSig), signalLength);
400
401 /* Rotated version reference */
402 Mrot = (float**)malloc2d(nSH, nSH, sizeof(float));
403 yawPitchRoll2Rzyx(ypr[0], ypr[1], ypr[2], 0, Rzyx);
404 getSHrotMtxReal(Rzyx, FLATTEN2D(Mrot), order);
405 shSig_rot_ref = (float**)malloc2d(nSH,signalLength,sizeof(float));
406 cblas_sgemm(CblasRowMajor, CblasNoTrans, CblasNoTrans, nSH, signalLength, nSH, 1.0f,
407 FLATTEN2D(Mrot), nSH,
408 FLATTEN2D(shSig), signalLength, 0.0f,
409 FLATTEN2D(shSig_rot_ref), signalLength);
410
411 /* Rotate with rotator */
412 framesize = rotator_getFrameSize();
413 shSig_rot = (float**)malloc2d(nSH,signalLength,sizeof(float));
414 shSig_frame = (float**)malloc1d(nSH*sizeof(float*));
415 shSig_rot_frame = (float**)malloc1d(nSH*sizeof(float*));
416 for(i=0; i<(int)((float)signalLength/(float)framesize); i++){
417 for(ch=0; ch<nSH; ch++)
418 shSig_frame[ch] = &shSig[ch][i*framesize];
419 for(ch=0; ch<nSH; ch++)
420 shSig_rot_frame[ch] = &shSig_rot[ch][i*framesize];
421
422 rotator_process(hRot, (const float* const*)shSig_frame, shSig_rot_frame, nSH, nSH, framesize);
423 }
424
425 /* ambi_enc should be equivalent to the reference, except delayed due to the
426 * temporal interpolation employed in ambi_enc */
427 for(i=0; i<nSH; i++)
428 for(j=0; j<signalLength-delay; j++)
429 TEST_ASSERT_FLOAT_WITHIN(acceptedTolerance, shSig_rot_ref[i][j], shSig_rot[i][j+delay]);
430
431 /* Clean-up */
432 rotator_destroy(&hRot);
433 free(inSig);
434 free(shSig);
435 free(shSig_rot_ref);
436 free(Mrot);
437 free(y);
438 free(shSig_frame);
439 free(shSig_rot_frame);
440}
