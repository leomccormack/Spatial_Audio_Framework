135void test__saf_stft_50pc_overlap(void){
136 int frame, winsize, hopsize, nFrames, ch, i, nBands, nTimesSlots, band;
137 void* hSTFT;
138 float** insig, **outsig, **inframe, **outframe;
139 float_complex*** inspec, ***outspec;
140
141 /* prep */
142 const float acceptedTolerance = 0.000001f;
143 const int fs = 48000;
144 const int signalLength = 1*fs;
145 const int framesize = 512;
146 const int nCHin = 62;
147 const int nCHout = 64;
148 insig = (float**)malloc2d(nCHin,signalLength,sizeof(float)); /* One second long */
149 outsig = (float**)malloc2d(nCHout,signalLength,sizeof(float));
150 inframe = (float**)malloc2d(nCHin,framesize,sizeof(float));
151 outframe = (float**)malloc2d(nCHout,framesize,sizeof(float));
152 rand_m1_1(FLATTEN2D(insig), nCHin*signalLength); /* populate with random numbers */
153
154 /* Set-up STFT for 50% overlapping */
155 winsize = 128;
156 hopsize = winsize/2;
157 nBands = winsize+1;
158 nTimesSlots = framesize/hopsize;
159 inspec = (float_complex***)malloc3d(nBands, nCHin, nTimesSlots, sizeof(float_complex));
160 outspec = (float_complex***)malloc3d(nBands, nCHout, nTimesSlots, sizeof(float_complex));
161 saf_stft_create(&hSTFT, winsize, hopsize, nCHin, nCHout, SAF_STFT_BANDS_CH_TIME);
162 saf_stft_channelChange(hSTFT, 123, 7); /* messing about */
163 saf_stft_flushBuffers(hSTFT); /* messing about */
164 saf_stft_channelChange(hSTFT, nCHin, nCHout); /* change back */
165
166 /* Pass insig through STFT, block-wise processing */
167 nFrames = (int)((float)signalLength/(float)framesize);
168 for(frame = 0; frame<nFrames; frame++){
169 /* Forward */
170 for(ch=0; ch<nCHin; ch++)
171 memcpy(inframe[ch], &insig[ch][frame*framesize], framesize*sizeof(float));
172 saf_stft_forward(hSTFT, inframe, framesize, inspec);
173
174 /* Copy first channel of inspec to all outspec channels */
175 for(band=0; band<nBands; band++)
176 for(ch=0; ch<nCHout; ch++)
177 memcpy(outspec[band][ch], inspec[band][0], nTimesSlots*sizeof(float_complex));
178
179 /* Backward */
180 saf_stft_backward(hSTFT, outspec, framesize, outframe);
181 for(ch=0; ch<nCHout; ch++)
182 memcpy(&outsig[ch][frame*framesize], outframe[ch], framesize*sizeof(float));
183 }
184
185 /* Check that input==output (given some numerical precision) */
186 for(i=0; i<signalLength-framesize; i++)
187 TEST_ASSERT_TRUE( fabsf(insig[0][i] - outsig[0][i+hopsize]) <= acceptedTolerance );
188
189 /* Clean-up */
190 saf_stft_destroy(&hSTFT);
191 free(insig);
192 free(outsig);
193 free(inframe);
194 free(outframe);
195 free(inspec);
196 free(outspec);
197}
