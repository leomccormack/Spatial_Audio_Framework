35void binauraliser_create
36(
37 void ** const phBin
38)
39{
40 binauraliser_data* pData = (binauraliser_data*)malloc1d(sizeof(binauraliser_data));
41 *phBin = (void*)pData;
42 int ch, dummy;
43
44 /* user parameters */
45 binauraliser_loadPreset(SOURCE_CONFIG_PRESET_DEFAULT, pData->src_dirs_deg, &(pData->new_nSources), &(dummy)); /*check setStateInformation if you change default preset*/
46 pData->useDefaultHRIRsFLAG = 1; /* pars->sofa_filepath must be valid to set this to 0 */
47 pData->enableHRIRsDiffuseEQ = 1;
48 pData->nSources = pData->new_nSources;
49 pData->interpMode = INTERP_TRI;
50 pData->yaw = 0.0f;
51 pData->pitch = 0.0f;
52 pData->roll = 0.0f;
53 pData->bFlipYaw = 0;
54 pData->bFlipPitch = 0;
55 pData->bFlipRoll = 0;
56 pData->useRollPitchYawFlag = 0;
57 pData->enableRotation = 0;
58
59 /* time-frequency transform + buffers */
60 pData->hSTFT = NULL;
61 pData->inputFrameTD = (float**)malloc2d(MAX_NUM_INPUTS, BINAURALISER_FRAME_SIZE, sizeof(float));
62 pData->outframeTD = (float**)malloc2d(NUM_EARS, BINAURALISER_FRAME_SIZE, sizeof(float));
63 pData->inputframeTF = (float_complex***)malloc3d(HYBRID_BANDS, MAX_NUM_INPUTS, TIME_SLOTS, sizeof(float_complex));
64 pData->outputframeTF = (float_complex***)malloc3d(HYBRID_BANDS, NUM_EARS, TIME_SLOTS, sizeof(float_complex));
65
66 /* hrir data */
67 pData->hrirs = NULL;
68 pData->hrir_dirs_deg = NULL;
69 pData->sofa_filepath = NULL;
70 pData->weights = NULL;
71 pData->N_hrir_dirs = pData->hrir_loaded_len = pData->hrir_runtime_len = 0;
72 pData->hrir_loaded_fs = pData->hrir_runtime_fs = -1; /* unknown */
73
74 /* vbap (amplitude normalised) */
75 pData->hrtf_vbap_gtableIdx = NULL;
76 pData->hrtf_vbap_gtableComp = NULL;
77 pData->nTriangles = pData->N_hrtf_vbap_gtable = 0;
78
79 /* HRTF filterbank coefficients */
80 pData->itds_s = NULL;
81 pData->hrtf_fb = NULL;
82 pData->hrtf_fb_mag = NULL;
83
84 /* flags/status */
85 pData->progressBar0_1 = 0.0f;
86 pData->progressBarText = malloc1d(PROGRESSBARTEXT_CHAR_LENGTH*sizeof(char));
87 strcpy(pData->progressBarText,"");
88 pData->codecStatus = CODEC_STATUS_NOT_INITIALISED;
89 pData->procStatus = PROC_STATUS_NOT_ONGOING;
90 pData->reInitHRTFsAndGainTables = 1;
91 for(ch=0; ch<MAX_NUM_INPUTS; ch++) {
92 pData->recalc_hrtf_interpFLAG[ch] = 1;
93 pData->src_gains[ch] = 1.f;
94 }
95 pData->recalc_M_rotFLAG = 1;
96}

98void binauraliser_destroy
99(
100 void ** const phBin
101)
102{
103 binauraliser_data *pData = (binauraliser_data*)(*phBin);
104
105 if (pData != NULL) {
106 /* not safe to free memory during intialisation/processing loop */
107 while (pData->codecStatus == CODEC_STATUS_INITIALISING ||
108 pData->procStatus == PROC_STATUS_ONGOING){
109 SAF_SLEEP(10);
110 }
111
112 /* free afSTFT and buffers */
113 if(pData->hSTFT !=NULL)
114 afSTFT_destroy(&(pData->hSTFT));
115 free(pData->inputFrameTD);
116 free(pData->outframeTD);
117 free(pData->inputframeTF);
118 free(pData->outputframeTF);
119 free(pData->hrtf_vbap_gtableComp);
120 free(pData->hrtf_vbap_gtableIdx);
121 free(pData->hrtf_fb);
122 free(pData->hrtf_fb_mag);
123 free(pData->itds_s);
124 free(pData->hrirs);
125 free(pData->hrir_dirs_deg);
126 free(pData->weights);
127 free(pData->progressBarText);
128
129 free(pData);
130 pData = NULL;
131 }
132}

134void binauraliser_init
135(
136 void * const hBin,
137 int sampleRate
138)
139{
140 binauraliser_data *pData = (binauraliser_data*)(hBin);
141
142 /* define frequency vector */
143 pData->fs = sampleRate;
144 afSTFT_getCentreFreqs(pData->hSTFT, (float)sampleRate, HYBRID_BANDS, pData->freqVector);
145 if(pData->hrir_runtime_fs!=pData->fs){
146 pData->reInitHRTFsAndGainTables = 1;
147 binauraliser_setCodecStatus(hBin, CODEC_STATUS_NOT_INITIALISED);
148 }
149
150 /* defaults */
151 pData->recalc_M_rotFLAG = 1;
152}

154void binauraliser_initCodec
155(
156 void* const hBin
157)
158{
159 binauraliser_data *pData = (binauraliser_data*)(hBin);
160
161 if (pData->codecStatus != CODEC_STATUS_NOT_INITIALISED)
162 return; /* re-init not required, or already happening */
163 while (pData->procStatus == PROC_STATUS_ONGOING){
164 /* re-init required, but we need to wait for the current processing loop to end */
165 pData->codecStatus = CODEC_STATUS_INITIALISING; /* indicate that we want to init */
166 SAF_SLEEP(10);
167 }
168
169 /* for progress bar */
170 pData->codecStatus = CODEC_STATUS_INITIALISING;
171 strcpy(pData->progressBarText,"Initialising");
172 pData->progressBar0_1 = 0.0f;
173
174 /* check if TFT needs to be reinitialised */
175 binauraliser_initTFT(hBin);
176
177 /* reinit HRTFs and interpolation tables */
178 if(pData->reInitHRTFsAndGainTables){
179 binauraliser_initHRTFsAndGainTables(hBin);
180 pData->reInitHRTFsAndGainTables = 0;
181 }
182
183 /* done! */
184 strcpy(pData->progressBarText,"Done!");
185 pData->progressBar0_1 = 1.0f;
186 pData->codecStatus = CODEC_STATUS_INITIALISED;
187
188}

190void binauraliser_process
191(
192 void * const hBin,
193 const float *const * inputs,
194 float ** const outputs,
195 int nInputs,
196 int nOutputs,
197 int nSamples
198)
199{
200 binauraliser_data *pData = (binauraliser_data*)(hBin);
201 int ch, ear, i, band, nSources;
202 float src_dirs[MAX_NUM_INPUTS][2], Rxyz[3][3], hypotxy;
203 int enableRotation;
204
205 /* copy user parameters to local variables */
206 nSources = pData->nSources;
207 enableRotation = pData->enableRotation;
208 memcpy(src_dirs, pData->src_dirs_deg, MAX_NUM_INPUTS*2*sizeof(float));
209
210 /* apply binaural panner */
211 if ((nSamples == BINAURALISER_FRAME_SIZE) && (pData->hrtf_fb!=NULL) && (pData->codecStatus==CODEC_STATUS_INITIALISED) ){
212 pData->procStatus = PROC_STATUS_ONGOING;
213
214 /* Load time-domain data */
215 for(i=0; i < SAF_MIN(nSources,nInputs); i++)
216 utility_svvcopy(inputs[i], BINAURALISER_FRAME_SIZE, pData->inputFrameTD[i]);
217 for(; i<nSources; i++)
218 memset(pData->inputFrameTD[i], 0, BINAURALISER_FRAME_SIZE * sizeof(float));
219
220 /* Apply source gains */
221 for (ch = 0; ch < nSources; ch++) {
222 if(fabsf(pData->src_gains[ch] - 1.f) > 1e-6f)
223 utility_svsmul(pData->inputFrameTD[ch], &(pData->src_gains[ch]), BINAURALISER_FRAME_SIZE, NULL);
224 }
225
226 /* Apply time-frequency transform (TFT) */
227 afSTFT_forward_knownDimensions(pData->hSTFT, pData->inputFrameTD, BINAURALISER_FRAME_SIZE, MAX_NUM_INPUTS, TIME_SLOTS, pData->inputframeTF);
228
229 /* Rotate source directions */
230 if(enableRotation && pData->recalc_M_rotFLAG){
231 yawPitchRoll2Rzyx (pData->yaw, pData->pitch, pData->roll, pData->useRollPitchYawFlag, Rxyz);
232 for(i=0; i<nSources; i++){
233 pData->src_dirs_xyz[i][0] = cosf(DEG2RAD(pData->src_dirs_deg[i][1])) * cosf(DEG2RAD(pData->src_dirs_deg[i][0]));
234 pData->src_dirs_xyz[i][1] = cosf(DEG2RAD(pData->src_dirs_deg[i][1])) * sinf(DEG2RAD(pData->src_dirs_deg[i][0]));
235 pData->src_dirs_xyz[i][2] = sinf(DEG2RAD(pData->src_dirs_deg[i][1]));
236 pData->recalc_hrtf_interpFLAG[i] = 1;
237 }
238 cblas_sgemm(CblasRowMajor, CblasNoTrans, CblasNoTrans, nSources, 3, 3, 1.0f,
239 (float*)(pData->src_dirs_xyz), 3,
240 (float*)Rxyz, 3, 0.0f,
241 (float*)(pData->src_dirs_rot_xyz), 3);
242 for(i=0; i<nSources; i++){
243 hypotxy = sqrtf(powf(pData->src_dirs_rot_xyz[i][0], 2.0f) + powf(pData->src_dirs_rot_xyz[i][1], 2.0f));
244 pData->src_dirs_rot_deg[i][0] = RAD2DEG(atan2f(pData->src_dirs_rot_xyz[i][1], pData->src_dirs_rot_xyz[i][0]));
245 pData->src_dirs_rot_deg[i][1] = RAD2DEG(atan2f(pData->src_dirs_rot_xyz[i][2], hypotxy));
246 }
247 pData->recalc_M_rotFLAG = 0;
248 }
249
250 /* interpolate hrtfs and apply to each source */
251 memset(FLATTEN3D(pData->outputframeTF), 0, HYBRID_BANDS*NUM_EARS*TIME_SLOTS * sizeof(float_complex));
252 for (ch = 0; ch < nSources; ch++) {
253 if(pData->recalc_hrtf_interpFLAG[ch]){
254 if(enableRotation)
255 binauraliser_interpHRTFs(hBin, pData->interpMode, pData->src_dirs_rot_deg[ch][0], pData->src_dirs_rot_deg[ch][1], pData->hrtf_interp[ch]);
256 else
257 binauraliser_interpHRTFs(hBin, pData->interpMode, pData->src_dirs_deg[ch][0], pData->src_dirs_deg[ch][1], pData->hrtf_interp[ch]);
258 pData->recalc_hrtf_interpFLAG[ch] = 0;
259 }
260
261 /* Convolve this channel with the interpolated HRTF, and add it to the binaural buffer */
262 for (band = 0; band < HYBRID_BANDS; band++)
263 for (ear = 0; ear < NUM_EARS; ear++)
264 cblas_caxpy(TIME_SLOTS, &pData->hrtf_interp[ch][band][ear], pData->inputframeTF[band][ch], 1, pData->outputframeTF[band][ear], 1);
265 }
266
267 /* scale by number of sources */
268 cblas_sscal(/*re+im*/2*HYBRID_BANDS*NUM_EARS*TIME_SLOTS, 1.0f/sqrtf((float)nSources), (float*)FLATTEN3D(pData->outputframeTF), 1);
269
270 /* inverse-TFT */
271 afSTFT_backward_knownDimensions(pData->hSTFT, pData->outputframeTF, BINAURALISER_FRAME_SIZE, NUM_EARS, TIME_SLOTS, pData->outframeTD);
272
273 /* Copy to output buffer */
274 for (ch = 0; ch < SAF_MIN(NUM_EARS, nOutputs); ch++)
275 utility_svvcopy(pData->outframeTD[ch], BINAURALISER_FRAME_SIZE, outputs[ch]);
276 for (; ch < nOutputs; ch++)
277 memset(outputs[ch], 0, BINAURALISER_FRAME_SIZE*sizeof(float));
278 }
279 else{
280 for (ch=0; ch < nOutputs; ch++)
281 memset(outputs[ch],0, BINAURALISER_FRAME_SIZE*sizeof(float));
282 }
283
284 pData->procStatus = PROC_STATUS_NOT_ONGOING;
285}

289void binauraliser_refreshSettings(void* const hBin)
290{
291 binauraliser_data *pData = (binauraliser_data*)(hBin);
292 int ch;
293 pData->reInitHRTFsAndGainTables = 1;
294 for(ch=0; ch<MAX_NUM_INPUTS; ch++)
295 pData->recalc_hrtf_interpFLAG[ch] = 1;
296 binauraliser_setCodecStatus(hBin, CODEC_STATUS_NOT_INITIALISED);
297}

299void binauraliser_setSourceAzi_deg(void* const hBin, int index, float newAzi_deg)
300{
301 binauraliser_data *pData = (binauraliser_data*)(hBin);
302 if(newAzi_deg>180.0f)
303 newAzi_deg = -360.0f + newAzi_deg;
304 newAzi_deg = SAF_MAX(newAzi_deg, -180.0f);
305 newAzi_deg = SAF_MIN(newAzi_deg, 180.0f);
306 if(pData->src_dirs_deg[index][0]!=newAzi_deg){
307 pData->src_dirs_deg[index][0] = newAzi_deg;
308 pData->recalc_hrtf_interpFLAG[index] = 1;
309 pData->recalc_M_rotFLAG = 1;
310 }
311}

317 newElev_deg = SAF_MIN(newElev_deg, 90.0f);
318 if(pData->src_dirs_deg[index][1] != newElev_deg){
319 pData->src_dirs_deg[index][1] = newElev_deg;
320 pData->recalc_hrtf_interpFLAG[index] = 1;
