28void decorrelator_create
29(
30 void ** const phDecor
31)
32{
33 decorrelator_data* pData = (decorrelator_data*)malloc1d(sizeof(decorrelator_data));
34 *phDecor = (void*)pData;
35
36 /* default user parameters */
37 pData->nChannels = 1;
38 pData->enableTransientDucker = 0;
39 pData->decorAmount = 1.0f;
40 pData->compensateLevel = 0;
41
42 /* afSTFT stuff */
43 pData->hSTFT = NULL;
44 pData->InputFrameTD = (float**)malloc2d(MAX_NUM_CHANNELS, DECORRELATOR_FRAME_SIZE, sizeof(float));
45 pData->OutputFrameTD = (float**)malloc2d(MAX_NUM_CHANNELS, DECORRELATOR_FRAME_SIZE, sizeof(float));
46 pData->InputFrameTF = (float_complex***)malloc3d(HYBRID_BANDS, MAX_NUM_CHANNELS, TIME_SLOTS, sizeof(float_complex));
47 pData->OutputFrameTF = (float_complex***)malloc3d(HYBRID_BANDS, MAX_NUM_CHANNELS, TIME_SLOTS, sizeof(float_complex));
48 pData->transientFrameTF = (float_complex***)malloc3d(HYBRID_BANDS, MAX_NUM_CHANNELS, TIME_SLOTS, sizeof(float_complex));
49
50 /* codec data */
51 pData->hDecor = NULL;
52 pData->hDucker = NULL;
53 pData->new_nChannels = pData->nChannels;
54 pData->progressBar0_1 = 0.0f;
55 pData->progressBarText = malloc1d(PROGRESSBARTEXT_CHAR_LENGTH*sizeof(char));
56 strcpy(pData->progressBarText,"");
57
58 /* flags */
59 pData->procStatus = PROC_STATUS_NOT_ONGOING;
60 pData->codecStatus = CODEC_STATUS_NOT_INITIALISED;
61}

63void decorrelator_destroy
64(
65 void ** const phDecor
66)
67{
68 decorrelator_data *pData = (decorrelator_data*)(*phDecor);
69
70 if (pData != NULL) {
71 /* not safe to free memory during intialisation/processing loop */
72 while (pData->codecStatus == CODEC_STATUS_INITIALISING ||
73 pData->procStatus == PROC_STATUS_ONGOING){
74 SAF_SLEEP(10);
75 }
76
77 /* free afSTFT and buffers */
78 if(pData->hSTFT!=NULL)
79 afSTFT_destroy(&(pData->hSTFT));
80 free(pData->InputFrameTD);
81 free(pData->OutputFrameTD);
82 free(pData->InputFrameTF);
83 free(pData->OutputFrameTF);
84 free(pData->transientFrameTF);
85 free(pData->progressBarText);
86
87 transientDucker_destroy(&(pData->hDucker));
88 latticeDecorrelator_destroy(&(pData->hDecor));
89
90 free(pData);
91 pData = NULL;
92 *phDecor = NULL;
93 }
94}

111void decorrelator_initCodec
112(
113 void* const hDecor
114)
115{
116 decorrelator_data *pData = (decorrelator_data*)(hDecor);
117 int nChannels;
118
119 if (pData->codecStatus != CODEC_STATUS_NOT_INITIALISED)
120 return; /* re-init not required, or already happening */
121 while (pData->procStatus == PROC_STATUS_ONGOING){
122 /* re-init required, but we need to wait for the current processing loop to end */
123 pData->codecStatus = CODEC_STATUS_INITIALISING; /* indicate that we want to init */
124 SAF_SLEEP(10);
125 }
126
127 /* for progress bar */
128 pData->codecStatus = CODEC_STATUS_INITIALISING;
129 strcpy(pData->progressBarText,"Preparing Decorrelators");
130 pData->progressBar0_1 = 0.0f;
131
132 /* (Re)Initialise afSTFT */
133 nChannels = pData->new_nChannels;
134 if(pData->hSTFT==NULL)
135 afSTFT_create(&(pData->hSTFT), nChannels, nChannels, HOP_SIZE, 0, 1, AFSTFT_BANDS_CH_TIME);
136 else if(pData->nChannels != nChannels) {/* Or change the number of channels */
137 afSTFT_channelChange(pData->hSTFT, nChannels, nChannels);
138 afSTFT_clearBuffers(pData->hSTFT);
139 }
140 pData->nChannels = nChannels;
141
142 /* Init transient ducker */
143 transientDucker_destroy(&(pData->hDucker));
144 transientDucker_create(&(pData->hDucker), nChannels, HYBRID_BANDS);
145
146 /* Init decorrelator */
147 const int orders[4] = {20, 15, 6, 3}; /* 20th order up to 700Hz, 15th->2.4kHz, 6th->4kHz, 3rd->12kHz, NONE(only delays)->Nyquist */
148 const float freqCutoffs[4] = {600.0f, 2.4e3f, 4.0e3f, 12e3f};
149 //const float freqCutoffs[4] = {900.0f, 6.8e3f, 12e3f, 16e3f};
150 const int maxDelay = 8;
151 latticeDecorrelator_destroy(&(pData->hDecor));
152 latticeDecorrelator_create(&(pData->hDecor), pData->fs, HOP_SIZE, pData->freqVector, HYBRID_BANDS, pData->nChannels, (int*)orders, (float*)freqCutoffs, 4, maxDelay, 0, 0.75f);
153
154 /* done! */
155 strcpy(pData->progressBarText,"Done!");
156 pData->progressBar0_1 = 1.0f;
157 pData->codecStatus = CODEC_STATUS_INITIALISED;
158}

160void decorrelator_process
161(
162 void * const hDecor,
163 const float *const * inputs,
164 float ** const outputs,
165 int nInputs,
166 int nOutputs,
167 int nSamples
168)
169{
170 decorrelator_data *pData = (decorrelator_data*)(hDecor);
171 int ch, i, band, enableTransientDucker, compensateLevel;
172 float decorAmount;
173
174 /* local copies of user parameters */
175 int nCH;
176 nCH = pData->nChannels;
177 decorAmount = pData->decorAmount;
178 enableTransientDucker = pData->enableTransientDucker;
179 compensateLevel = pData->compensateLevel;
180
181 /* Process frame */
182 if (nSamples == DECORRELATOR_FRAME_SIZE && (pData->codecStatus == CODEC_STATUS_INITIALISED) ) {
183 pData->procStatus = PROC_STATUS_ONGOING;
184
185 /* Load time-domain data */
186 for(i=0; i < SAF_MIN(nCH, nInputs); i++)
187 utility_svvcopy(inputs[i], DECORRELATOR_FRAME_SIZE, pData->InputFrameTD[i]);
188 for(; i<nCH; i++)
189 memset(pData->InputFrameTD[i], 0, DECORRELATOR_FRAME_SIZE * sizeof(float)); /* fill remaining channels with zeros */
190
191 /* Apply time-frequency transform (TFT) */
192 afSTFT_forward_knownDimensions(pData->hSTFT, pData->InputFrameTD, DECORRELATOR_FRAME_SIZE, MAX_NUM_CHANNELS, TIME_SLOTS, pData->InputFrameTF);
193
194 /* Apply decorrelation */
195 if(enableTransientDucker){
196 /* remove transients */
197 transientDucker_apply(pData->hDucker, pData->InputFrameTF, TIME_SLOTS, 0.95f, 0.995f, pData->OutputFrameTF, pData->transientFrameTF);
198 /* decorrelate only the residual */
199 latticeDecorrelator_apply(pData->hDecor, pData->OutputFrameTF, TIME_SLOTS, pData->OutputFrameTF);
200 }
201 else
202 latticeDecorrelator_apply(pData->hDecor, pData->InputFrameTF, TIME_SLOTS, pData->OutputFrameTF);
203
204 /* Optionally compensate for the level (as they channels wll no longer sum coherently) */
205 if(compensateLevel){
206 for(band=0; band<HYBRID_BANDS; band++)
207 cblas_sscal(/*re+im*/2*nCH*TIME_SLOTS, 0.75f*(float)nCH/(sqrtf((float)nCH)), (float*)FLATTEN2D(pData->OutputFrameTF[band]), 1);
208 }
209
210 /* re-introduce the transient part */
211 if(enableTransientDucker){
212 //scalec = cmplxf(1.0f, 0.0f);//!compensateLevel ? cmplxf(1.25f*(sqrtf((float)nCH)/(float)nCH), 0.0f) : cmplxf(1.0f, 0.0f);
213 for(band=0; band<HYBRID_BANDS; band++)
214 cblas_saxpy(/*re+im*/2*nCH*TIME_SLOTS, 1.0f, (float*)FLATTEN2D(pData->transientFrameTF[band]), 1, (float*)FLATTEN2D(pData->OutputFrameTF[band]), 1);
215 }
216
217 /* Mix thedecorrelated audio with the input non-decorrelated audio */
218 for(band=0; band<HYBRID_BANDS; band++){
219 cblas_sscal(/*re+im*/2*nCH*TIME_SLOTS, decorAmount, (float*)FLATTEN2D(pData->OutputFrameTF[band]), 1);
220 cblas_saxpy(/*re+im*/2*nCH*TIME_SLOTS, 1.0f-decorAmount, (float*)FLATTEN2D(pData->InputFrameTF[band]), 1, (float*)FLATTEN2D(pData->OutputFrameTF[band]), 1);
221 }
222
223 /* inverse-TFT */
224 afSTFT_backward_knownDimensions(pData->hSTFT, pData->OutputFrameTF, DECORRELATOR_FRAME_SIZE, MAX_NUM_CHANNELS, TIME_SLOTS, pData->OutputFrameTD);
225
226 /* Copy to output buffer */
227 for (ch = 0; ch < SAF_MIN(nCH, nOutputs); ch++)
228 utility_svvcopy(pData->OutputFrameTD[ch], DECORRELATOR_FRAME_SIZE, outputs[ch]);
229 for (; ch < nOutputs; ch++)
230 memset(outputs[ch], 0, DECORRELATOR_FRAME_SIZE*sizeof(float));
231 }
232 else
233 for (ch=0; ch < nOutputs; ch++)
234 memset(outputs[ch],0, DECORRELATOR_FRAME_SIZE*sizeof(float));
235
236 pData->procStatus = PROC_STATUS_NOT_ONGOING;
237}
